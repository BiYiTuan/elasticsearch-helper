001    package org.xbib.elasticsearch.support;
002    
003    import org.elasticsearch.common.logging.ESLogger;
004    import org.elasticsearch.common.logging.ESLoggerFactory;
005    import org.testng.ITestContext;
006    import org.testng.ITestListener;
007    import org.testng.ITestResult;
008    
009    public class Listener implements ITestListener {
010    
011        private final static ESLogger logger = ESLoggerFactory.getLogger(Listener.class.getSimpleName());
012    
013        @Override
014        public void onTestStart(ITestResult result) {
015            logger.info("------------------------------------------------------");
016            logger.info("starting test method {}", result.getName());
017            logger.info("------------------------------------------------------");
018        }
019    
020        @Override
021        public void onTestSuccess(ITestResult result) {
022            logger.info("------------------------------------------------------");
023            logger.info("success of test method {}", result.getName());
024            logger.info("------------------------------------------------------");
025        }
026    
027        @Override
028        public void onTestFailure(ITestResult result) {
029            logger.info("------------------------------------------------------");
030            logger.info("failure of test method {}", result.getName());
031            logger.info("------------------------------------------------------");
032        }
033    
034        @Override
035        public void onTestSkipped(ITestResult result) {
036            logger.info("skipped test {}", result.getMethod().getMethodName());
037            result.setStatus(ITestResult.FAILURE);
038        }
039    
040        @Override
041        public void onTestFailedButWithinSuccessPercentage(ITestResult result) {
042        }
043    
044        @Override
045        public void onStart(ITestContext context) {
046            logger.info("------------------------------------------------------");
047            logger.info("starting test {}", context.getName());
048            logger.info("------------------------------------------------------");
049        }
050    
051        @Override
052        public void onFinish(ITestContext context) {
053            logger.info("------------------------------------------------------");
054            logger.info("finished test {}", context.getName());
055            logger.info("------------------------------------------------------");
056        }
057    
058    }
