002    package org.xbib.elasticsearch.support;
003    
004    import org.xbib.elasticsearch.support.client.AbstractIngestClient;
005    import org.xbib.elasticsearch.support.client.IngestClient;
006    import org.elasticsearch.client.transport.NoNodeAvailableException;
007    import org.elasticsearch.common.logging.ESLogger;
008    import org.elasticsearch.common.logging.Loggers;
009    import org.testng.annotations.Test;
010    
011    import java.io.IOException;
012    
013    public class ReplicaLevelTests extends AbstractNodeTest {
014    
015        private final static ESLogger logger = Loggers.getLogger(ReplicaLevelTests.class);
016    
017        @Test
018        public void testReplicaLevel() throws IOException {
019    
020            int numberOfShards = 5;
021            int replicaLevel = 4;
022            int shardsAfterReplica = 0;
023    
024            final AbstractIngestClient es = new IngestClient()
025                    .newClient(ADDRESS)
026                    .setIndex("replicatest")
027                    .setType("replicatest")
028                    .numberOfShards(numberOfShards)
029                    .numberOfReplicas(0)
030                    .dateDetection(false)
031                    .timeStampFieldEnabled(false)
032                    .newIndex();
033    
034            try {
035                for (int i = 0; i < 12345; i++) {
036                    es.indexDocument("replicatest", "replicatest", null, "{ \"name\" : \"" + randomString(32) + "\"}");
037                }
038                es.flush();
039                shardsAfterReplica = es.updateReplicaLevel(replicaLevel);
040                logger.info("shardsAfterReplica={}", shardsAfterReplica);
041            } catch (NoNodeAvailableException e) {
042                logger.warn("skipping, no node available");
043            } finally {
044                //assertEquals(shardsAfterReplica, numberOfShards * (replicaLevel + 1));
045                es.shutdown();
046            }
047        }
048    
049    }
