002package org.xbib.elasticsearch.support;
003
004import org.elasticsearch.client.transport.NoNodeAvailableException;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.ESLoggerFactory;
007import org.testng.annotations.Test;
008
009import org.xbib.elasticsearch.support.client.IngestClient;
010
011import java.io.IOException;
012
013public class ReplicaLevelTests extends AbstractNodeRandomTest {
014
015    private final static ESLogger logger = ESLoggerFactory.getLogger(ReplicaLevelTests.class.getSimpleName());
016
017    @Test
018    public void testReplicaLevel() throws IOException {
019
020        // we need 3 nodes for replica level 3
021        startNode("2");
022        startNode("3");
023
024        int numberOfShards = 2;
025        int replicaLevel = 3;
026        int shardsAfterReplica;
027
028        final IngestClient es = new IngestClient()
029                .newClient(getAddress())
030                .setIndex("replicatest")
031                .setType("replicatest")
032                .numberOfShards(numberOfShards)
033                .numberOfReplicas(0)
034                .newIndex();
035
036        try {
037            for (int i = 0; i < 12345; i++) {
038                es.indexDocument("replicatest", "replicatest", null, "{ \"name\" : \"" + randomString(32) + "\"}");
039            }
040            es.flush();
041            shardsAfterReplica = es.updateReplicaLevel(replicaLevel);
042            assertEquals(shardsAfterReplica, numberOfShards * (replicaLevel + 1));
043        } catch (NoNodeAvailableException e) {
044            logger.warn("skipping, no node available");
045        } finally {
046            es.shutdown();
047            if (es.hasErrors()) {
048                logger.error("error", es.getThrowable());
049            }
050            assertFalse(es.hasErrors());
051        }
052
053        stopNode("3");
054        stopNode("2");
055    }
056
057}
