002    package org.xbib.elasticsearch.support;
003    
004    import org.elasticsearch.action.search.SearchResponse;
005    import org.elasticsearch.client.Client;
006    import org.elasticsearch.client.transport.TransportClient;
007    import org.elasticsearch.common.logging.ESLogger;
008    import org.elasticsearch.common.logging.ESLoggerFactory;
009    import org.elasticsearch.common.settings.ImmutableSettings;
010    import org.elasticsearch.common.settings.Settings;
011    import org.elasticsearch.common.transport.InetSocketTransportAddress;
012    import org.elasticsearch.discovery.MasterNotDiscoveredException;
013    import org.elasticsearch.node.Node;
014    import org.testng.annotations.Test;
015    
016    import static org.elasticsearch.index.query.QueryBuilders.*;
017    import static org.elasticsearch.node.NodeBuilder.*;
018    
019    public class QueryTest {
020    
021        private static final ESLogger logger = ESLoggerFactory.getLogger(QueryTest.class.getName());
022    
023        public void testQuery() {
024            Settings settings = ImmutableSettings.settingsBuilder()
025                    .put("cluster.name", "test").build();
026    
027            String index = "test";
028            String type = "test";
029            Node node = null;
030            try {
031                node = nodeBuilder().settings(settings).client(true).node();
032                Client client = node.client();
033                SearchResponse response = client.prepareSearch().setIndices(index).
034                        setTypes(type).
035                        setFrom(0).setSize(10).setQuery(matchQuery("_all", "test")).execute().actionGet();
036            } catch (Exception e) {
037                logger.warn(e.getMessage());
038            } finally {
039                if (node !=null){
040                    node.stop();
041                    node.close();
042                }
043            }
044        }
045    
046        public void testSniff() {
047            Settings settings = ImmutableSettings.settingsBuilder()
048                    .put("cluster.name", "test")
049                    .put("client.transport.sniff", false).build();
050            try {
051                TransportClient client = new TransportClient(settings);
052                InetSocketTransportAddress address = new InetSocketTransportAddress("localhost", 9300);
053                client.addTransportAddress(address);
054                client.close();
055            } catch (MasterNotDiscoveredException e) {
056                logger.warn(e.getMessage());
057            }
058        }
059    }
