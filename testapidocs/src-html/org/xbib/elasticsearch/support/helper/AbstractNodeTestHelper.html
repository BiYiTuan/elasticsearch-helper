001package org.xbib.elasticsearch.support.helper;
002
003import java.util.HashMap;
004import java.util.Map;
005import java.util.concurrent.atomic.AtomicInteger;
006
007import org.elasticsearch.action.admin.cluster.node.info.NodesInfoRequest;
008import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
009import org.elasticsearch.client.support.AbstractClient;
010import org.elasticsearch.common.logging.ESLogger;
011import org.elasticsearch.common.logging.ESLoggerFactory;
012import org.elasticsearch.common.network.NetworkUtils;
013import org.elasticsearch.common.settings.Settings;
014import org.elasticsearch.common.transport.InetSocketTransportAddress;
015import org.elasticsearch.node.Node;
016
017import static org.elasticsearch.common.settings.Settings.settingsBuilder;
018import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
019
020import org.junit.After;
021import org.junit.Before;
022
023public abstract class AbstractNodeTestHelper {
024
025    protected final static ESLogger logger = ESLoggerFactory.getLogger("test");
026
027    private Map<String, Node> nodes = new HashMap<>();
028
029    private Map<String, AbstractClient> clients = new HashMap<>();
030
031    private AtomicInteger counter = new AtomicInteger();
032
033    private String cluster;
034
035    private String host;
036
037    private int port;
038
039    protected void setClusterName() {
040        this.cluster = "test-support-cluster-" + NetworkUtils.getLocalAddress().getHostName() + "-" + counter.incrementAndGet();
041    }
042
043    protected String getClusterName() {
044        return cluster;
045    }
046
047    protected String getHome() {
048        return System.getProperty("path.home");
049    }
050
051    protected Settings getSettings() {
052        return settingsBuilder()
053                .put("host", host)
054                .put("port", port)
055                .put("cluster.name", cluster)
056                .put("path.home", getHome())
057                .build();
058    }
059
060    protected Settings getNodeSettings() {
061        return settingsBuilder()
062                .put("cluster.name", cluster)
063                .put("cluster.routing.schedule", "50ms")
064                .put("cluster.routing.allocation.disk.threshold_enabled", false)
065                .put("discovery.zen.multicast.enabled", false)
066                .put("discovery.zen.multicast.enabled", true)
067                .put("discovery.zen.multicast.ping_timeout", "5s")
068                .put("gateway.type", "none")
069                .put("http.enabled", false)
070                .put("index.store.type", "org.xbib.elasticsearch.support.store.mockfs")
071                .put("threadpool.bulk.size", 2 * Runtime.getRuntime().availableProcessors())
072                .put("threadpool.bulk.queue_size", 16 * Runtime.getRuntime().availableProcessors()) // default is 50, too low
073                .put("path.home", getHome())
074                .build();
075    }
076
077    @Before
078    public void startNodes() throws Exception {
079        setClusterName();
080        startNode("1");
081        // find node address
082        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
083        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
084        Object obj = response.iterator().next().getTransport().getAddress()
085                .publishAddress();
086        if (obj instanceof InetSocketTransportAddress) {
087            InetSocketTransportAddress address = (InetSocketTransportAddress) obj;
088            host = address.address().getHostName();
089            port = address.address().getPort();
090        }
091    }
092
093    @After
094    public void stopNodes() throws Exception {
095        try {
096            // delete all indices
097            client("1").admin().indices().prepareDelete("_all").execute().actionGet();
098        } catch (Exception e) {
099            logger.error("can not delete indexes", e);
100        }
101        closeAllNodes();
102    }
103
104    protected Node startNode(String id) {
105        return buildNode(id).start();
106    }
107
108    public AbstractClient client(String id) {
109        return clients.get(id);
110    }
111
112    private Node buildNode(String id) {
113        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
114        Settings finalSettings = settingsBuilder()
115                .loadFromClasspath(settingsSource)
116                .put(getNodeSettings())
117                .put("name", id)
118                .put("path.home", getHome())
119                .build();
120        logger.info("settings={}", finalSettings.getAsMap());
121        Node node = nodeBuilder()
122                .settings(finalSettings).build();
123        AbstractClient client = (AbstractClient)node.client();
124        nodes.put(id, node);
125        clients.put(id, client);
126        return node;
127    }
128
129    protected void stopNode(String id) {
130        AbstractClient client = clients.remove(id);
131        if (client != null) {
132            client.close();
133        }
134        Node node = nodes.remove(id);
135        if (node != null) {
136            node.close();
137        }
138    }
139
140
141    public void closeAllNodes() {
142        for (AbstractClient client : clients.values()) {
143            client.close();
144        }
145        clients.clear();
146        for (Node node : nodes.values()) {
147            if (node != null) {
148                node.close();
149            }
150        }
151        nodes.clear();
152        logger.info("all nodes closed");
153    }
154
155}
