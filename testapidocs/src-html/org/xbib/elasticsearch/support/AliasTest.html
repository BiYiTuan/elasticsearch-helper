002    package org.xbib.elasticsearch.support;
003    
004    import org.elasticsearch.action.admin.indices.alias.IndicesAliasesRequest;
005    import org.elasticsearch.action.admin.indices.alias.get.GetAliasesRequest;
006    import org.elasticsearch.action.admin.indices.alias.get.GetAliasesResponse;
007    import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
008    import org.elasticsearch.client.Client;
009    import org.elasticsearch.cluster.metadata.AliasAction;
010    import org.elasticsearch.common.Strings;
011    import org.elasticsearch.common.logging.ESLogger;
012    import org.elasticsearch.common.logging.ESLoggerFactory;
013    import org.elasticsearch.common.settings.ImmutableSettings;
014    import org.elasticsearch.common.settings.Settings;
015    import org.elasticsearch.node.Node;
016    import org.testng.annotations.Test;
017    
018    import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
019    
020    public class AliasTest {
021    
022        private static final ESLogger logger = ESLoggerFactory.getLogger(AliasTest.class.getName());
023    
024        public void testAlias() {
025            Node node = null;
026            try {
027                Settings settings = ImmutableSettings.settingsBuilder()
028                        .put("cluster.name", "test")
029                        .put("gateway.type", "none")
030                        .build();
031                node = nodeBuilder().settings(settings).local(true).node();
032                Client client = node.client();
033    
034                // create index
035                CreateIndexRequest indexRequest = new CreateIndexRequest("test");
036                client.admin().indices().create(indexRequest).actionGet();
037    
038                // put alias
039                IndicesAliasesRequest indicesAliasesRequest = new IndicesAliasesRequest();
040                String[] indices = new String[]{"test"};
041                String[] aliases = new String[]{"test_alias"};
042                IndicesAliasesRequest.AliasActions aliasAction = new IndicesAliasesRequest.AliasActions(AliasAction.Type.ADD, indices, aliases);
043                indicesAliasesRequest.addAliasAction(aliasAction);
044                client.admin().indices().aliases(indicesAliasesRequest).actionGet();
045    
046                // get alias
047                GetAliasesRequest getAliasesRequest = new GetAliasesRequest(Strings.EMPTY_ARRAY);
048                long t0 = System.nanoTime();
049                GetAliasesResponse getAliasesResponse = client.admin().indices().getAliases(getAliasesRequest).actionGet();
050                long t1 = System.nanoTime() - t0;
051    
052                logger.info("{} time(ms) = {}", getAliasesResponse.getAliases(), t1 / 1000000);
053    
054            } catch (Exception e) {
055                logger.warn(e.getMessage());
056            } finally {
057                if (node !=null){
058                    node.stop();
059                    node.close();
060                }
061            }
062        }
063    
064    }
