001    package org.xbib.elasticsearch.support;
002    
003    import org.elasticsearch.action.bulk.BulkRequestBuilder;
004    import org.elasticsearch.action.index.IndexRequest;
005    import org.elasticsearch.action.index.IndexRequestBuilder;
006    import org.elasticsearch.client.Client;
007    import org.elasticsearch.client.Requests;
008    import org.junit.Test;
009    import org.xbib.elasticsearch.support.helper.AbstractNodeTestHelper;
010    
011    public class NPETest extends AbstractNodeTestHelper {
012    
013        @Test(expected = NullPointerException.class)
014        public void testNPE1() throws NullPointerException {
015            Client client = client("1");
016            BulkRequestBuilder builder = new BulkRequestBuilder(client)
017                    .add(Requests.indexRequest());
018            client.bulk(builder.request()).actionGet();
019        }
020    
021        @Test(expected = NullPointerException.class)
022        public void testNPE2() {
023            Client client = client("1");
024            BulkRequestBuilder builder = new BulkRequestBuilder(client)
025                    .add((IndexRequest)null);
026            client.bulk(builder.request()).actionGet();
027        }
028    
029        @Test(expected = NullPointerException.class)
030        public void testNPE3() {
031            Client client = client("1");
032            IndexRequestBuilder r = new IndexRequestBuilder(client);
033            BulkRequestBuilder builder = new BulkRequestBuilder(client)
034                    .add(r.request());
035            client.bulk(builder.request()).actionGet();
036        }
037    
038        @Test(expected = NullPointerException.class)
039        public void testNPE4() {
040            Client client = client("1");
041            client.prepareBulk()
042                    .add(Requests.indexRequest().index(null).type(null).id(null))
043                    .execute().actionGet();
044        }
045    
046    }
