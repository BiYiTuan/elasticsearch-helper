001package org.xbib.elasticsearch.support.client.ingest;
002
003import org.elasticsearch.action.admin.indices.stats.CommonStats;
004import org.elasticsearch.action.admin.indices.stats.IndexShardStats;
005import org.elasticsearch.action.admin.indices.stats.IndexStats;
006import org.elasticsearch.action.admin.indices.stats.IndicesStatsResponse;
007import org.elasticsearch.client.transport.NoNodeAvailableException;
008import org.elasticsearch.common.logging.ESLogger;
009import org.elasticsearch.common.logging.ESLoggerFactory;
010import org.elasticsearch.common.unit.TimeValue;
011import org.elasticsearch.index.indexing.IndexingStats;
012import org.elasticsearch.index.query.QueryBuilders;
013import org.junit.Test;
014import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
015
016import java.util.Map;
017
018import static org.junit.Assert.assertEquals;
019import static org.junit.Assert.assertFalse;
020
021public class IngestReplicaTest extends AbstractNodeRandomTestHelper {
022
023    private final static ESLogger logger = ESLoggerFactory.getLogger(IngestReplicaTest.class.getSimpleName());
024
025    @Test
026    public void testReplicaLevel() throws Exception {
027
028        // we need nodes for quorum. 4 nodes => quorum = 2
029        startNode("2");
030        startNode("3");
031        //startNode("4");
032
033        final IngestTransportClient ingest = new IngestTransportClient()
034                .newClient(getAddress())
035                .shards(2)
036                .replica(3)
037                .newIndex("test1")
038                .shards(2)
039                .replica(1)
040                .newIndex("test2");
041        try {
042            for (int i = 0; i < 1234; i++) {
043                ingest.index("test1", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
044            }
045            for (int i = 0; i < 1234; i++) {
046                ingest.index("test2", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
047            }
048            ingest.flushIngest();
049            ingest.waitForResponses(TimeValue.timeValueSeconds(60));
050        } catch (NoNodeAvailableException e) {
051            logger.warn("skipping, no node available");
052        } finally {
053            logger.info("refreshing");
054            ingest.refresh("test1");
055            ingest.refresh("test2");
056            long hits = ingest.client().prepareSearch("test1","test2")
057                    .setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getHits().getTotalHits();
058            logger.info("query total hits={}", hits);
059            assertEquals(2468, hits);
060            IndicesStatsResponse response = ingest.client().admin().indices().prepareStats().all().execute().actionGet();
061            for (Map.Entry<String,IndexStats> m : response.getIndices().entrySet()) {
062                IndexStats indexStats = m.getValue();
063                CommonStats commonStats = indexStats.getTotal();
064                IndexingStats indexingStats = commonStats.getIndexing();
065                IndexingStats.Stats stats = indexingStats.getTotal();
066                logger.info("index {}: count = {}", m.getKey(), stats.getIndexCount());
067                for (Map.Entry<Integer,IndexShardStats> me : indexStats.getIndexShards().entrySet()) {
068                    IndexShardStats indexShardStats = me.getValue();
069                    CommonStats commonShardStats = indexShardStats.getTotal();
070                    logger.info("shard {} count = {}", me.getKey(),
071                            commonShardStats.getIndexing().getTotal().getIndexCount());
072                }
073            }
074            try {
075                ingest.deleteIndex("test1")
076                        .deleteIndex("test2");
077            } catch (Exception e) {
078                logger.error("delete index failed, ignored. Reason:", e);
079            }
080            if (ingest.hasThrowable()) {
081                logger.error("error", ingest.getThrowable());
082            }
083            assertFalse(ingest.hasThrowable());
084            ingest.shutdown();
085        }
086
087        //stopNode("4");
088        stopNode("3");
089        stopNode("2");
090    }
091
092}
