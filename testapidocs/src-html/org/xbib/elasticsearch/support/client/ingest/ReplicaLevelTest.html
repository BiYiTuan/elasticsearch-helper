002    package org.xbib.elasticsearch.support.client.ingest;
003    
004    import org.elasticsearch.client.transport.NoNodeAvailableException;
005    import org.elasticsearch.common.logging.ESLogger;
006    import org.elasticsearch.common.logging.ESLoggerFactory;
007    
008    import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
009    
010    import java.io.IOException;
011    
012    import org.junit.Test;
013    import static org.junit.Assert.assertEquals;
014    import static org.junit.Assert.assertFalse;
015    
016    public class ReplicaLevelTest extends AbstractNodeRandomTestHelper {
017    
018        private final static ESLogger logger = ESLoggerFactory.getLogger(ReplicaLevelTest.class.getSimpleName());
019    
020        @Test
021        public void testReplicaLevel() throws IOException {
022    
023            // we need 3 nodes for replica level 3
024            startNode("2");
025            startNode("3");
026    
027            int numberOfShards = 2;
028            int replicaLevel = 3;
029            int shardsAfterReplica;
030    
031            final IngestClient es = new IngestClient()
032                    .newClient(getAddress())
033                    .setIndex("replicatest")
034                    .setType("replicatest")
035                    .numberOfShards(numberOfShards)
036                    .numberOfReplicas(0)
037                    .newIndex();
038    
039            try {
040                for (int i = 0; i < 12345; i++) {
041                    es.index("replicatest", "replicatest", null, "{ \"name\" : \"" + randomString(32) + "\"}");
042                }
043                es.flush();
044                shardsAfterReplica = es.updateReplicaLevel(replicaLevel);
045                assertEquals(shardsAfterReplica, numberOfShards * (replicaLevel + 1));
046            } catch (NoNodeAvailableException e) {
047                logger.warn("skipping, no node available");
048            } finally {
049                es.shutdown();
050                if (es.hasErrors()) {
051                    logger.error("error", es.getThrowable());
052                }
053                assertFalse(es.hasErrors());
054            }
055    
056            stopNode("3");
057            stopNode("2");
058        }
059    
060    }
