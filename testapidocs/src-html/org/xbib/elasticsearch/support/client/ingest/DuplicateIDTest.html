001package org.xbib.elasticsearch.support.client.ingest;
002
003import org.elasticsearch.client.transport.NoNodeAvailableException;
004import org.elasticsearch.common.logging.ESLogger;
005import org.elasticsearch.common.logging.ESLoggerFactory;
006import org.junit.Test;
007import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
008
009import static org.elasticsearch.index.query.QueryBuilders.matchAllQuery;
010import static org.junit.Assert.assertEquals;
011import static org.junit.Assert.assertFalse;
012import static org.junit.Assert.assertTrue;
013
014public class DuplicateIDTest extends AbstractNodeRandomTestHelper {
015
016    private final static ESLogger logger = ESLoggerFactory.getLogger(DuplicateIDTest.class.getSimpleName());
017
018    @Test
019    public void testDuplicateDocIDs() {
020        final IngestTransportClient es = new IngestTransportClient()
021                .maxActionsPerBulkRequest(1000)
022                .newClient(getAddress())
023                .newIndex("test");
024        try {
025            for (int i = 0; i < 12345; i++) {
026                es.index("test", "test", randomString(1), "{ \"name\" : \"" + randomString(32) + "\"}");
027            }
028            es.refresh("test");
029            long hits = es.client().prepareSearch("test").setTypes("test")
030                    .setQuery(matchAllQuery())
031                    .execute().actionGet().getHits().getTotalHits();
032            logger.info("hits = {}", hits);
033            assertTrue(hits < 12345);
034        } catch (NoNodeAvailableException e) {
035            logger.warn("skipping, no node available");
036        } finally {
037            es.shutdown();
038            assertEquals(es.getState().getTotalIngest().count(), 13);
039            if (es.hasThrowable()) {
040                logger.error("error", es.getThrowable());
041            }
042            assertFalse(es.hasThrowable());
043        }
044    }
045}
