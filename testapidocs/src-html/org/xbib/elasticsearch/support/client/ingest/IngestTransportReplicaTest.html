001package org.xbib.elasticsearch.support.client.ingest;
002
003import org.elasticsearch.action.admin.indices.stats.CommonStats;
004import org.elasticsearch.action.admin.indices.stats.IndexShardStats;
005import org.elasticsearch.action.admin.indices.stats.IndexStats;
006import org.elasticsearch.action.admin.indices.stats.IndicesStatsResponse;
007import org.elasticsearch.client.transport.NoNodeAvailableException;
008import org.elasticsearch.common.logging.ESLogger;
009import org.elasticsearch.common.logging.ESLoggerFactory;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.common.unit.TimeValue;
012import org.elasticsearch.index.indexing.IndexingStats;
013import org.elasticsearch.index.query.QueryBuilders;
014import org.junit.Test;
015import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
016
017import java.util.Map;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertFalse;
021
022public class IngestTransportReplicaTest extends AbstractNodeRandomTestHelper {
023
024    private final static ESLogger logger = ESLoggerFactory.getLogger(IngestTransportReplicaTest.class.getSimpleName());
025
026    @Test
027    public void testReplicaLevel() throws Exception {
028
029        // we need nodes for quorum. 4 nodes => quorum = 2
030        startNode("2");
031        startNode("3");
032        //startNode("4");
033
034        Settings settingsTest1 = Settings.settingsBuilder()
035                .put("index.number_of_shards", 2)
036                .put("index.number_of_replicas", 3)
037                .build();
038
039        Settings settingsTest2 = Settings.settingsBuilder()
040                .put("index.number_of_shards", 2)
041                .put("index.number_of_replicas", 1)
042                .build();
043
044        final IngestTransportClient ingest = new IngestTransportClient()
045                .newClient(getSettings())
046                .newIndex("test1", settingsTest1, null)
047                .newIndex("test2", settingsTest2, null);
048        try {
049            for (int i = 0; i < 1234; i++) {
050                ingest.index("test1", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
051            }
052            for (int i = 0; i < 1234; i++) {
053                ingest.index("test2", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
054            }
055            ingest.flushIngest();
056            ingest.waitForResponses(TimeValue.timeValueSeconds(60));
057        } catch (NoNodeAvailableException e) {
058            logger.warn("skipping, no node available");
059        } finally {
060            logger.info("refreshing");
061            ingest.refreshIndex("test1");
062            ingest.refreshIndex("test2");
063            long hits = ingest.client().prepareSearch("test1","test2")
064                    .setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getHits().getTotalHits();
065            logger.info("query total hits={}", hits);
066            assertEquals(2468, hits);
067            IndicesStatsResponse response = ingest.client().admin().indices().prepareStats().all().execute().actionGet();
068            for (Map.Entry<String,IndexStats> m : response.getIndices().entrySet()) {
069                IndexStats indexStats = m.getValue();
070                CommonStats commonStats = indexStats.getTotal();
071                IndexingStats indexingStats = commonStats.getIndexing();
072                IndexingStats.Stats stats = indexingStats.getTotal();
073                logger.info("index {}: count = {}", m.getKey(), stats.getIndexCount());
074                for (Map.Entry<Integer,IndexShardStats> me : indexStats.getIndexShards().entrySet()) {
075                    IndexShardStats indexShardStats = me.getValue();
076                    CommonStats commonShardStats = indexShardStats.getTotal();
077                    logger.info("shard {} count = {}", me.getKey(),
078                            commonShardStats.getIndexing().getTotal().getIndexCount());
079                }
080            }
081            try {
082                ingest.deleteIndex("test1")
083                        .deleteIndex("test2");
084            } catch (Exception e) {
085                logger.error("delete index failed, ignored. Reason:", e);
086            }
087            if (ingest.hasThrowable()) {
088                logger.error("error", ingest.getThrowable());
089            }
090            assertFalse(ingest.hasThrowable());
091            ingest.shutdown();
092        }
093
094        //stopNode("4");
095        stopNode("3");
096        stopNode("2");
097    }
098
099}
