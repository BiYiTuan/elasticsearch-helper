001package org.xbib.elasticsearch.support.client.ingest;
002
003import org.elasticsearch.action.admin.indices.mapping.get.GetMappingsRequest;
004import org.elasticsearch.action.admin.indices.mapping.get.GetMappingsResponse;
005import org.elasticsearch.cluster.metadata.MappingMetaData;
006import org.elasticsearch.common.settings.ImmutableSettings;
007import org.elasticsearch.common.settings.Settings;
008import org.junit.Test;
009import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
010
011import java.util.Map;
012
013import static org.elasticsearch.common.collect.Maps.newHashMap;
014import static org.junit.Assert.assertEquals;
015
016public class IngestIndexCreationTest extends AbstractNodeRandomTestHelper {
017
018    @Test
019    public void testIngestCreation() throws Exception {
020
021        Settings settings = ImmutableSettings.settingsBuilder()
022                .put("index.number_of_shards", 1)
023                .build();
024
025        Map<String,String> mappings = newHashMap();
026        mappings.put("typename","{\"properties\":{\"message\":{\"type\":\"string\"}}}");
027
028        final IngestTransportClient ingest = new IngestTransportClient()
029                .newClient(getAddress());
030        ingest.newIndex("test", settings, mappings);
031
032        GetMappingsRequest getMappingsRequest = new GetMappingsRequest().indices("test");
033        GetMappingsResponse getMappingsResponse = ingest.client().admin().indices()
034                .getMappings(getMappingsRequest).actionGet();
035        MappingMetaData md = getMappingsResponse.getMappings().get("test").get("typename");
036        assertEquals("{properties={message={type=string}}}", md.getSourceAsMap().toString());
037    }
038
039}
