001package org.xbib.elasticsearch.support.client.ingest;
002
003import org.elasticsearch.action.admin.indices.mapping.get.GetMappingsRequest;
004import org.elasticsearch.action.admin.indices.mapping.get.GetMappingsResponse;
005import org.elasticsearch.cluster.metadata.MappingMetaData;
006import org.elasticsearch.common.settings.Settings;
007import org.junit.Test;
008import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
009
010import java.util.HashMap;
011import java.util.Map;
012
013import static org.junit.Assert.assertEquals;
014
015public class IngestIndexCreationTest extends AbstractNodeRandomTestHelper {
016
017    @Test
018    public void testIngestCreation() throws Exception {
019        Settings settings = Settings.settingsBuilder()
020                .put("index.number_of_shards", 1)
021                .build();
022        Map<String,String> mappings = new HashMap<>();
023        mappings.put("typename","{\"properties\":{\"message\":{\"type\":\"string\"}}}");
024        final IngestTransportClient ingest = new IngestTransportClient()
025                .newClient(getSettings());
026        ingest.newIndex("test", settings, mappings);
027        GetMappingsRequest getMappingsRequest = new GetMappingsRequest().indices("test");
028        GetMappingsResponse getMappingsResponse = ingest.client().admin().indices()
029                .getMappings(getMappingsRequest).actionGet();
030        MappingMetaData md = getMappingsResponse.getMappings().get("test").get("typename");
031        assertEquals("{properties={message={type=string}}}", md.getSourceAsMap().toString());
032    }
033
034}
