060    @Test
061    public void testSingleDocIngestClient() {
062        final IngestTransportClient ingest = new IngestTransportClient()
063                .flushIngestInterval(TimeValue.timeValueSeconds(600))
064                .newClient(getAddress())
065                .shards(2)
066                .replica(0)
067                .newIndex("test");
068        try {
069            ingest.index("test", "test", "1", "{ \"name\" : \"Hello World\"}"); // single doc ingest
070            ingest.flushIngest();
071            ingest.waitForResponses(TimeValue.timeValueSeconds(30));
072        } catch (NoNodeAvailableException e) {
073            logger.warn("skipping, no node available");
074        } catch (InterruptedException e) {
075            // ignore
076        } finally {
077            logger.info("total bulk requests = {}", ingest.getState().getTotalIngest().count());
078            assertEquals(1, ingest.getState().getTotalIngest().count());
079            if (ingest.hasThrowable()) {
080                logger.error("error", ingest.getThrowable());
081            }
082            assertFalse(ingest.hasThrowable());
083            ingest.shutdown();
084        }
085    }

119    @Test
120    public void testThreadedRandomDocsIngestClient() throws Exception {
121        int maxthreads = Runtime.getRuntime().availableProcessors();
122        int maxactions = 1000;
123        final int maxloop = 12345;
124        final IngestTransportClient ingest = new IngestTransportClient()
125                .flushIngestInterval(TimeValue.timeValueSeconds(600))
126                .maxActionsPerBulkRequest(maxactions)
127                .newClient(getAddress())
128                .shards(2)
129                .replica(0)
130                .newIndex("test")
131                .startBulk("test");
132        try {
133            ThreadPoolExecutor pool = EsExecutors.newFixed(maxthreads, 30,
134                    EsExecutors.daemonThreadFactory("ingest-test"));
135            final CountDownLatch latch = new CountDownLatch(maxthreads);
136            for (int i = 0; i < maxthreads; i++) {
137                pool.execute(new Runnable() {
138                    public void run() {
139                        for (int i = 0; i < maxloop; i++) {
140                            ingest.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
141                        }
142                        latch.countDown();
143                    }
144                });
145            }
146            logger.info("waiting for max 60 seconds...");
147            latch.await(60, TimeUnit.SECONDS);
148            logger.info("client flush ...");
149            ingest.flushIngest();
150            ingest.waitForResponses(TimeValue.timeValueSeconds(30));
151            logger.info("thread pool to be shut down ...");
152            pool.shutdown();
153            logger.info("thread pool shut down");
154        } catch (NoNodeAvailableException e) {
155            logger.warn("skipping, no node available");
156        } finally {
157            ingest.stopBulk("test");
158            logger.info("total requests = {}", ingest.getState().getTotalIngest().count());
159            assertEquals(maxthreads * maxloop / maxactions + 1, ingest.getState().getTotalIngest().count());
160            if (ingest.hasThrowable()) {
161                logger.error("error", ingest.getThrowable());
162            }
163            assertFalse(ingest.hasThrowable());
164            ingest.refresh("test");
165            assertEquals(maxthreads * maxloop,
166                    ingest.client().prepareCount("test").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount()
167            );
168            ingest.shutdown();
169        }
170    }
