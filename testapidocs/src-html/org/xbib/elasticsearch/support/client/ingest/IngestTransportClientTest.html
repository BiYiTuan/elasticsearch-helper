059    @Test
060    public void testSingleDocIngestClient() {
061        final IngestTransportClient es = new IngestTransportClient()
062                .newClient(getAddress())
063                .setIndex("test")
064                .setType("test")
065                .newIndex();
066        try {
067            es.deleteIndex();
068            es.newIndex();
069            es.index("test", "test", "1", "{ \"name\" : \"J?rg Prante\"}"); // single doc ingest
070            es.flush();
071        } catch (NoNodeAvailableException e) {
072            logger.warn("skipping, no node available");
073        } catch (Exception e) {
074            logger.error(e.getMessage(), e);
075        } finally {
076            es.shutdown();
077            logger.info("total bulk requests = {}", es.getState().getTotalIngest().count());
078            assertEquals(es.getState().getTotalIngest().count(), 1);
079            if (es.hasThrowable()) {
080                logger.error("error", es.getThrowable());
081            }
082            assertFalse(es.hasThrowable());
083        }
084    }

111    @Test
112    public void testThreadedRandomDocsIngestClient() throws Exception {
113        int max = Runtime.getRuntime().availableProcessors();
114        final IngestTransportClient client = new IngestTransportClient()
115                .maxActionsPerBulkRequest(10000)
116                .newClient(getAddress())
117                .setIndex("test")
118                .setType("test")
119                .newIndex()
120                .startBulk();
121        try {
122            ThreadPoolExecutor pool = EsExecutors.newFixed(max, 30, EsExecutors.daemonThreadFactory("ingest-test"));
123            final CountDownLatch latch = new CountDownLatch(max);
124            for (int i = 0; i < max; i++) {
125                pool.execute(new Runnable() {
126                    public void run() {
127                        for (int i = 0; i < 12345; i++) {
128                            client.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
129                        }
130                        latch.countDown();
131                    }
132                });
133            }
134            logger.info("waiting for max 30 seconds...");
135            latch.await(30, TimeUnit.SECONDS);
136            pool.shutdown();
137        } catch (NoNodeAvailableException e) {
138            logger.warn("skipping, no node available");
139        } finally {
140            client.stopBulk().shutdown();
141            logger.info("total bulk requests = {}", client.getState().getTotalIngest().count());
142            assertEquals(max * 12345 / 10000 + 1, client.getState().getTotalIngest().count());
143            if (client.hasThrowable()) {
144                logger.error("error", client.getThrowable());
145            }
146            assertFalse(client.hasThrowable());
147        }
148
149    }
