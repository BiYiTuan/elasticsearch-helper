113        @Test
114        public void testThreadedRandomIngestClient() throws Exception {
115            int max = Runtime.getRuntime().availableProcessors();
116            final IngestIndexClient es = new IngestIndexClient()
117                    .maxActionsPerBulkRequest(10000)
118                    .newClient(getAddress())
119                    .setIndex("test")
120                    .setType("test")
121                    .newIndex()
122                    .startBulk();
123            try {
124                ThreadPoolExecutor pool = EsExecutors.newFixed(max, 30,
125                        EsExecutors.daemonThreadFactory("ingest-test"));
126                final CountDownLatch latch = new CountDownLatch(max);
127                for (int i = 0; i < max; i++) {
128                    pool.execute(new Runnable() {
129                        public void run() {
130                            for (int i = 0; i < 12345; i++) {
131                                es.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
132                            }
133                            latch.countDown();
134                        }
135                    });
136                }
137                logger.info("waiting for max 30 seconds...");
138                latch.await(30, TimeUnit.SECONDS);
139                pool.shutdown();
140            } catch (NoNodeAvailableException e) {
141                logger.warn("skipping, no node available");
142            } finally {
143                logger.info("stats={}", es.stats());
144                es.stopBulk().shutdown();
145                logger.info("total bulk requests = {}", es.getTotalBulkRequests());
146                int target = max * 12345 / 10000 + 1;
147                assertEquals(es.getTotalBulkRequests(), target);
148                if (es.hasErrors()) {
149                    logger.error("error", es.getThrowable());
150                }
151                assertFalse(es.hasErrors());
152            }
153        }
