060    @Test
061    public void testSingleDocIngestClient() {
062        final IngestIndexTransportClient es = new IngestIndexTransportClient()
063                .newClient(getAddress())
064                .setIndex("test")
065                .setType("test")
066                .newIndex();
067        try {
068            es.deleteIndex();
069            es.newIndex();
070            es.index("test", "test", "1", "{ \"name\" : \"J?rg Prante\"}"); // single doc ingest
071            es.flush();
072        } catch (NoNodeAvailableException e) {
073            logger.warn("skipping, no node available");
074        } catch (Exception e) {
075            logger.error(e.getMessage(), e);
076        } finally {
077            es.shutdown();
078            logger.info("total bulk requests = {}", es.getState().getTotalIngest().count());
079            assertEquals(1, es.getState().getTotalIngest().count());
080            if (es.hasThrowable()) {
081                logger.error("error", es.getThrowable());
082            }
083            assertFalse(es.hasThrowable());
084        }
085    }

112    @Test
113    public void testThreadedRandomIngestClient() throws Exception {
114        int max = Runtime.getRuntime().availableProcessors();
115        final IngestIndexTransportClient es = new IngestIndexTransportClient()
116                .maxActionsPerBulkRequest(10000)
117                .newClient(getAddress())
118                .setIndex("test")
119                .setType("test")
120                .newIndex()
121                .startBulk();
122        try {
123            ThreadPoolExecutor pool = EsExecutors.newFixed(max, 30,
124                    EsExecutors.daemonThreadFactory("ingest-test"));
125            final CountDownLatch latch = new CountDownLatch(max);
126            for (int i = 0; i < max; i++) {
127                pool.execute(new Runnable() {
128                    public void run() {
129                        for (int i = 0; i < 12345; i++) {
130                            es.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
131                        }
132                        latch.countDown();
133                    }
134                });
135            }
136            logger.info("waiting for max 30 seconds...");
137            latch.await(30, TimeUnit.SECONDS);
138            pool.shutdown();
139        } catch (NoNodeAvailableException e) {
140            logger.warn("skipping, no node available");
141        } finally {
142            es.stopBulk().shutdown();
143            logger.info("total bulk requests = {}", es.getState().getTotalIngest().count());
144            assertEquals( max * 12345 / 10000 + 1, es.getState().getTotalIngest().count());
145            if (es.hasThrowable()) {
146                logger.error("error", es.getThrowable());
147            }
148            assertFalse(es.hasThrowable());
149        }
150    }
