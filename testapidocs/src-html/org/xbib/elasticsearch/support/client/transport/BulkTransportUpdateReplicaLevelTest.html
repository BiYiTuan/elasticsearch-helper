002package org.xbib.elasticsearch.support.client.transport;
003
004import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
005import org.elasticsearch.client.transport.NoNodeAvailableException;
006import org.elasticsearch.common.logging.ESLogger;
007import org.elasticsearch.common.logging.ESLoggerFactory;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.common.unit.TimeValue;
010import org.junit.Test;
011import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
012
013import static org.junit.Assert.assertEquals;
014import static org.junit.Assert.assertFalse;
015
016public class BulkTransportUpdateReplicaLevelTest extends AbstractNodeRandomTestHelper {
017
018    private final static ESLogger logger = ESLoggerFactory.getLogger(BulkTransportUpdateReplicaLevelTest.class.getSimpleName());
019
020    @Test
021    public void testUpdateReplicaLevel() throws Exception {
022
023        int numberOfShards = 2;
024        int replicaLevel = 3;
025
026        // we need 3 nodes for replica level 3
027        startNode("2");
028        startNode("3");
029
030        int shardsAfterReplica;
031
032        Settings settings = Settings.settingsBuilder()
033                .put("index.number_of_shards", numberOfShards)
034                .put("index.number_of_replicas", 0)
035                .build();
036
037        final BulkTransportClient ingest = new BulkTransportClient()
038                .newClient(getSettings())
039//                .shards(numberOfShards)
040//                .replica(0)
041                .newIndex("replicatest", settings, null);
042
043        ingest.waitForCluster(ClusterHealthStatus.GREEN, TimeValue.timeValueSeconds(30));
044
045        try {
046            for (int i = 0; i < 12345; i++) {
047                ingest.index("replicatest", "replicatest", null, "{ \"name\" : \"" + randomString(32) + "\"}");
048            }
049            ingest.flushIngest();
050            ingest.waitForResponses(TimeValue.timeValueSeconds(30));
051            shardsAfterReplica = ingest.updateReplicaLevel("replicatest", replicaLevel);
052            assertEquals(shardsAfterReplica, numberOfShards * (replicaLevel + 1));
053        } catch (NoNodeAvailableException e) {
054            logger.warn("skipping, no node available");
055        } finally {
056            ingest.deleteIndex("replicatest");
057            ingest.shutdown();
058            if (ingest.hasThrowable()) {
059                logger.error("error", ingest.getThrowable());
060            }
061            assertFalse(ingest.hasThrowable());
062        }
063
064        stopNode("3");
065        stopNode("2");
066    }
067
068}
