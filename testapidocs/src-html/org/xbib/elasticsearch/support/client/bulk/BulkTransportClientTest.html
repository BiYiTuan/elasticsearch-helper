023    @Test
024    public void testBulkClient() {
025        final BulkTransportClient es = new BulkTransportClient()
026                .flushInterval(TimeValue.timeValueSeconds(5))
027                .newClient(getAddress())
028                .newIndex("test");
029        es.shutdown();
030        if (es.hasThrowable()) {
031            logger.error("error", es.getThrowable());
032        }
033        assertFalse(es.hasThrowable());
034        try {
035            es.deleteIndex("test")
036              .newIndex("test")
037              .deleteIndex("test");
038        } catch (NoNodeAvailableException e) {
039            logger.error("no node available");
040        } finally {
041            es.shutdown();
042            if (es.hasThrowable()) {
043                logger.error("error", es.getThrowable());
044            }
045            assertFalse(es.hasThrowable());
046        }
047    }
048
049    @Test
050    public void testSingleDocBulkClient() {
051        /**
052         * most of the times, BulkProcessor has difficulties with a single document...
053         */
054        final BulkTransportClient es = new BulkTransportClient()
055                .maxActionsPerBulkRequest(1000)
056                .flushInterval(TimeValue.timeValueSeconds(5))
057                .newClient(getAddress())
058                .newIndex("test");
059        try {
060            es.deleteIndex("test");
061            es.newIndex("test");
062            es.index("test", "test", "1", "{ \"name\" : \"J?rg Prante\"}"); // single doc ingest
063            es.flush();
064        } catch (NoNodeAvailableException e) {
065            logger.warn("skipping, no node available");
066        } finally {
067            es.shutdown();
068            logger.info("bulk requests = {}", es.getState().getTotalIngest().count());
069            assertEquals(1, es.getState().getTotalIngest().count());
070            if (es.hasThrowable()) {
071                logger.error("error", es.getThrowable());
072            }
073            assertFalse(es.hasThrowable());
074        }
075    }

102    @Test
103    public void testThreadedRandomDocsBulkClient() throws Exception {
104        int max = Runtime.getRuntime().availableProcessors();
105        int maxactions = 1000;
106        final int maxloop = 12345;
107
108        final BulkTransportClient client = new BulkTransportClient()
109                .flushInterval(TimeValue.timeValueSeconds(600)) // = disable autoflush for this test
110                .maxActionsPerBulkRequest(maxactions)
111                .newClient(getAddress())
112                .newIndex("test")
113                .startBulk("test");
114        try {
115            ThreadPoolExecutor pool = EsExecutors.newFixed(max, 30,
116                    EsExecutors.daemonThreadFactory("bulkclient-test"));
117            final CountDownLatch latch = new CountDownLatch(max);
118            for (int i = 0; i < max; i++) {
119                pool.execute(new Runnable() {
120                    public void run() {
121                        for (int i = 0; i < maxloop; i++) {
122                            client.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
123                        }
124                        latch.countDown();
125                    }
126                });
127            }
128            logger.info("waiting for 60 seconds...");
129            latch.await(60, TimeUnit.SECONDS);
130            logger.info("flush ...");
131            client.flush();
132            logger.info("pool to be shut down ...");
133            pool.shutdown();
134            logger.info("poot shut down");
135        } catch (NoNodeAvailableException e) {
136            logger.warn("skipping, no node available");
137        } finally {
138            client.stopBulk("test").flush().shutdown();
139            logger.info("bulk requests = {}", client.getState().getTotalIngest().count() );
140            assertEquals(max * maxloop / maxactions + 1, client.getState().getTotalIngest().count());
141            assertFalse(client.hasThrowable());
142        }
143    }
