001package org.xbib.elasticsearch.support.client.node;
002
003import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
004import org.elasticsearch.client.transport.NoNodeAvailableException;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.ESLoggerFactory;
007import org.elasticsearch.common.unit.TimeValue;
008import org.junit.Test;
009import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
010
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertFalse;
013
014public class NodeUpdateReplicaLevelTest extends AbstractNodeRandomTestHelper {
015
016    private final static ESLogger logger = ESLoggerFactory.getLogger(NodeUpdateReplicaLevelTest.class.getSimpleName());
017
018    @Test
019    public void testUpdateReplicaLevel() throws Exception {
020
021        int numberOfShards = 2;
022        int replicaLevel = 3;
023
024        // we need 3 nodes for replica level 3
025        startNode("2");
026        startNode("3");
027
028        int shardsAfterReplica;
029
030        final BulkNodeClient ingest = new BulkNodeClient()
031                .newClient(client("1"))
032                .shards(numberOfShards)
033                .replica(0)
034                .newIndex("replicatest");
035
036        ingest.waitForCluster(ClusterHealthStatus.GREEN, TimeValue.timeValueSeconds(30));
037
038        try {
039            for (int i = 0; i < 12345; i++) {
040                ingest.index("replicatest", "replicatest", null, "{ \"name\" : \"" + randomString(32) + "\"}");
041            }
042            ingest.flushIngest();
043            ingest.waitForResponses(TimeValue.timeValueSeconds(30));
044            shardsAfterReplica = ingest.updateReplicaLevel("replicatest", replicaLevel);
045            assertEquals(shardsAfterReplica, numberOfShards * (replicaLevel + 1));
046        } catch (NoNodeAvailableException e) {
047            logger.warn("skipping, no node available");
048        } finally {
049            ingest.deleteIndex("replicatest");
050            ingest.shutdown();
051            if (ingest.hasThrowable()) {
052                logger.error("error", ingest.getThrowable());
053            }
054            assertFalse(ingest.hasThrowable());
055        }
056
057        stopNode("3");
058        stopNode("2");
059    }
060
061}
