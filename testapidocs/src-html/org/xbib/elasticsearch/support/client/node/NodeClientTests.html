002    package org.xbib.elasticsearch.support.client.node;
003    
004    import org.elasticsearch.common.logging.ESLogger;
005    import org.elasticsearch.common.logging.ESLoggerFactory;
006    import org.elasticsearch.common.unit.TimeValue;
007    import org.testng.annotations.Test;
008    import org.elasticsearch.client.transport.NoNodeAvailableException;
009    
010    import org.xbib.elasticsearch.support.AbstractNodeRandomTest;
011    
012    public class NodeClientTests extends AbstractNodeRandomTest {
013    
014        private final static ESLogger logger = ESLoggerFactory.getLogger(NodeClientTests.class.getSimpleName());
015    
016        @Test
017        public void testNewIndexBulkClient() throws Exception {
018            final NodeClient es = new NodeClient()
019                    .flushInterval(TimeValue.timeValueSeconds(5))
020                    .newClient(client("1"))
021                    .setIndex(INDEX)
022                    .setType("test")
023                    .newIndex();
024            es.shutdown();
025            if (es.hasErrors()) {
026                logger.error("error", es.getThrowable());
027            }
028            assertFalse(es.hasErrors());
029        }
030    
031        @Test
032        public void testRandomDocsNodeClient() throws Exception {
033            final NodeClient es = new NodeClient()
034                    .maxActionsPerBulkRequest(1000)
035                    .maxConcurrentBulkRequests(10)
036                    .flushInterval(TimeValue.timeValueSeconds(5))
037                    .newClient(client("1"))
038                    .setIndex(INDEX)
039                    .setType("test")
040                    .newIndex();
041    
042            try {
043                for (int i = 0; i < 12345; i++) {
044                    es.index(INDEX, "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
045                }
046            } catch (NoNodeAvailableException e) {
047                logger.warn("skipping, no node available");
048            } finally {
049                es.shutdown();
050                assertEquals(es.getTotalBulkRequests(), 13);
051                if (es.hasErrors()) {
052                    logger.error("error", es.getThrowable());
053                }
054                assertFalse(es.hasErrors());
055            }
056        }
057    
058    }
