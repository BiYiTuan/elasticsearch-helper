001package org.xbib.elasticsearch.support.client.node;
002
003import org.elasticsearch.client.transport.NoNodeAvailableException;
004import org.elasticsearch.common.logging.ESLogger;
005import org.elasticsearch.common.logging.ESLoggerFactory;
006import org.elasticsearch.common.unit.TimeValue;
007import org.junit.Test;
008import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
009
010import static org.elasticsearch.index.query.QueryBuilders.matchAllQuery;
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertFalse;
013import static org.junit.Assert.assertTrue;
014
015public class NodeDuplicateIDTest extends AbstractNodeRandomTestHelper {
016
017    private final static ESLogger logger = ESLoggerFactory.getLogger(NodeDuplicateIDTest.class.getSimpleName());
018
019    @Test
020    public void testDuplicateDocIDs() throws Exception {
021        final NodeClient client = new NodeClient()
022                .maxActionsPerBulkRequest(1000)
023                .newClient(client("1"))
024                .newIndex("test");
025        try {
026            for (int i = 0; i < 12345; i++) {
027                client.index("test", "test", randomString(1), "{ \"name\" : \"" + randomString(32) + "\"}");
028            }
029            client.flushIngest();
030            client.waitForResponses(TimeValue.timeValueSeconds(30));
031            client.refresh("test");
032            long hits = client.client().prepareSearch("test").setTypes("test")
033                    .setQuery(matchAllQuery())
034                    .execute().actionGet().getHits().getTotalHits();
035            logger.info("hits = {}", hits);
036            assertTrue(hits < 12345);
037        } catch (NoNodeAvailableException e) {
038            logger.warn("skipping, no node available");
039        } finally {
040            client.shutdown();
041            assertEquals(client.getState().getTotalIngest().count(), 13);
042            if (client.hasThrowable()) {
043                logger.error("error", client.getThrowable());
044            }
045            assertFalse(client.hasThrowable());
046        }
047    }
048}
