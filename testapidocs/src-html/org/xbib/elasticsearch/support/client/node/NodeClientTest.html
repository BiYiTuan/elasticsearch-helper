002    package org.xbib.elasticsearch.support.client.node;
003    
004    import org.elasticsearch.common.logging.ESLogger;
005    import org.elasticsearch.common.logging.ESLoggerFactory;
006    import org.elasticsearch.common.unit.TimeValue;
007    import org.elasticsearch.client.transport.NoNodeAvailableException;
008    
009    import org.xbib.elasticsearch.support.helper.AbstractNodeRandomTestHelper;
010    
011    import org.junit.Test;
012    import static org.junit.Assert.assertEquals;
013    import static org.junit.Assert.assertFalse;
014    
015    public class NodeClientTest extends AbstractNodeRandomTestHelper {
016    
017        private final static ESLogger logger = ESLoggerFactory.getLogger(NodeClientTest.class.getSimpleName());
018    
019        @Test
020        public void testNewIndexBulkClient() throws Exception {
021            final NodeClient es = new NodeClient()
022                    .flushInterval(TimeValue.timeValueSeconds(5))
023                    .newClient(client("1"))
024                    .setIndex(INDEX)
025                    .setType("test")
026                    .newIndex();
027            es.shutdown();
028            if (es.hasErrors()) {
029                logger.error("error", es.getThrowable());
030            }
031            assertFalse(es.hasErrors());
032        }
033    
034        @Test
035        public void testRandomDocsNodeClient() throws Exception {
036            final NodeClient es = new NodeClient()
037                    .maxActionsPerBulkRequest(1000)
038                    .maxConcurrentBulkRequests(10)
039                    .flushInterval(TimeValue.timeValueSeconds(5))
040                    .newClient(client("1"))
041                    .setIndex(INDEX)
042                    .setType("test")
043                    .newIndex();
044    
045            try {
046                for (int i = 0; i < 12345; i++) {
047                    es.index(INDEX, "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
048                }
049            } catch (NoNodeAvailableException e) {
050                logger.warn("skipping, no node available");
051            } finally {
052                es.shutdown();
053                assertEquals(es.getTotalBulkRequests(), 13);
054                if (es.hasErrors()) {
055                    logger.error("error", es.getThrowable());
056                }
057                assertFalse(es.hasErrors());
058            }
059        }
060    
061    }
