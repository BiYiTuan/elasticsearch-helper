041    @Test
042    public void testMappingNodeClient() throws Exception {
043        final NodeClient client = new NodeClient()
044                .flushInterval(TimeValue.timeValueSeconds(5))
045                .newClient(client("1"));
046        client.addMapping("test", "{\"test\":{\"properties\":{\"location\":{\"type\":\"geo_point\"}}}}");
047        client.newIndex("test");
048        GetMappingsRequest getMappingsRequest = new GetMappingsRequest()
049                .indices("test");
050        GetMappingsResponse getMappingsResponse = client.client().admin().indices().getMappings(getMappingsRequest).actionGet();
051        logger.info("mappings={}", getMappingsResponse.getMappings());
052        if (client.hasThrowable()) {
053            logger.error("error", client.getThrowable());
054        }
055        assertFalse(client.hasThrowable());
056        client.shutdown();
057    }

059    @Test
060    public void testRandomDocsNodeClient() throws Exception {
061        final NodeClient client = new NodeClient()
062                .maxActionsPerBulkRequest(1000)
063                .flushInterval(TimeValue.timeValueSeconds(10))
064                .newClient(client("1"))
065                .newIndex("test");
066
067        try {
068            for (int i = 0; i < 12345; i++) {
069                client.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
070            }
071            client.flush();
072            client.waitForResponses(TimeValue.timeValueSeconds(30));
073        } catch (NoNodeAvailableException e) {
074            logger.warn("skipping, no node available");
075        } finally {
076            assertEquals(13, client.getState().getTotalIngest().count());
077            if (client.hasThrowable()) {
078                logger.error("error", client.getThrowable());
079            }
080            assertFalse(client.hasThrowable());
081            client.shutdown();
082        }
083    }

085    @Test
086    public void testThreadedRandomDocsNodeClient() throws Exception {
087        int max = Runtime.getRuntime().availableProcessors();
088        int maxactions = 1000;
089        final int maxloop = 12345;
090        final NodeClient client = new NodeClient()
091                .maxActionsPerBulkRequest(maxactions)
092                .flushInterval(TimeValue.timeValueSeconds(600)) // disable auto flush for this test
093                .newClient(client("1"))
094                .newIndex("test")
095                .startBulk("test");
096        try {
097            ThreadPoolExecutor pool = EsExecutors.newFixed(max, 30,
098                    EsExecutors.daemonThreadFactory("nodeclient-test"));
099            final CountDownLatch latch = new CountDownLatch(max);
100            for (int i = 0; i < max; i++) {
101                pool.execute(new Runnable() {
102                    public void run() {
103                        for (int i = 0; i < maxloop; i++) {
104                            client.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
105                        }
106                        latch.countDown();
107                    }
108                });
109            }
110            logger.info("waiting for max 60 seconds...");
111            latch.await(60, TimeUnit.SECONDS);
112            logger.info("flush...");
113            client.flush();
114            client.waitForResponses(TimeValue.timeValueSeconds(30));
115            logger.info("test thread pool shutdown...");
116            pool.shutdown();
117            logger.info("pool is shut down");
118        } catch (NoNodeAvailableException e) {
119            logger.warn("skipping, no node available");
120        } finally {
121            client.stopBulk("test");
122            logger.info("total bulk requests = {}", client.getState().getTotalIngest().count());
123            assertEquals(max * maxloop / maxactions + 1, client.getState().getTotalIngest().count());
124            if (client.hasThrowable()) {
125                logger.error("error", client.getThrowable());
126            }
127            assertFalse(client.hasThrowable());
128            client.refresh("test");
129            assertEquals(max * maxloop,
130                    client.client().prepareCount("test").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount()
131            );
132            client.shutdown();
133        }
134    }
