001package org.xbib.elasticsearch.support.various;
002
003import java.io.IOException;
004import org.elasticsearch.client.Client;
005import org.elasticsearch.index.query.QueryBuilder;
006
007import static org.elasticsearch.client.Requests.deleteIndexRequest;
008import static org.elasticsearch.client.Requests.indexRequest;
009import static org.elasticsearch.common.settings.Settings.settingsBuilder;
010import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
011import static org.elasticsearch.index.query.QueryBuilders.queryStringQuery;
012import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
013
014public class WildcardTest {
015
016    private static final String INDEX = "test";
017
018    private static final String TYPE = "test";
019
020    private static final String FIELD = "field";
021
022    private static Client client;
023
024    public void test() throws Exception {
025        initializeClient();
026        //deleteIndex();
027        index("1", "010");
028        index("2", "0*0");
029        // exact
030        validateCount(queryStringQuery("010").defaultField(FIELD), 1);
031        validateCount(queryStringQuery("0\\*0").defaultField(FIELD), 1);
032        // pattern
033        validateCount(queryStringQuery("0*0").defaultField(FIELD), 2);
034        validateCount(queryStringQuery("0?0").defaultField(FIELD), 2);
035        validateCount(queryStringQuery("0**0").defaultField(FIELD), 2);
036        validateCount(queryStringQuery("0??0").defaultField(FIELD), 0);
037        validateCount(queryStringQuery("*10").defaultField(FIELD), 1);
038        validateCount(queryStringQuery("*1*").defaultField(FIELD), 1);
039        // failing
040        //validateCount(queryString("*\\*0").defaultField(FIELD), 1);
041        //validateCount(queryString("*\\**").defaultField(FIELD), 1);
042    }
043
044    private void initializeClient() {
045        client = nodeBuilder()
046                .settings(settingsBuilder()
047                .put("cluster.name", "test")
048                .put("index.analysis.analyzer.default.filter", "lowercase")
049                .put("index.analysis.analyzer.default.tokenizer", "keyword"))
050                .local(true).node().client();
051    }
052
053    private void deleteIndex() {
054        client.admin().indices().delete(deleteIndexRequest(INDEX)).actionGet();
055    }
056
057    private void index(String id, String fieldValue) throws IOException {
058        client.index(indexRequest()
059                .index(INDEX).type(TYPE).id(id)
060                .source(jsonBuilder().startObject().field(FIELD, fieldValue).endObject())
061                .refresh(true)).actionGet();
062    }
063
064    private long count(QueryBuilder queryBuilder) {
065        return client.prepareSearch(INDEX).setTypes(TYPE)
066                .setQuery(queryBuilder)
067                .execute().actionGet().getHits().getTotalHits();
068    }
069
070    private void validateCount(QueryBuilder queryBuilder, long expectedHits) {
071        final long actualHits = count(queryBuilder);
072        if (actualHits != expectedHits) {
073            throw new RuntimeException("actualHits=" + actualHits + ", expectedHits=" + expectedHits);
074        }
075    }
076}
