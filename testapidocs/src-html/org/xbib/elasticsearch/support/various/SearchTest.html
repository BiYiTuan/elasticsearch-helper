012
013import static org.elasticsearch.client.Requests.indexRequest;
014import static org.elasticsearch.client.Requests.refreshRequest;
015import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
016
017public class SearchTest extends AbstractNodeTestHelper {
018
019    @Test
020    public void test() throws Exception {
021        Client client = client("1");
022        long t0 = System.currentTimeMillis();
023        BulkRequestBuilder builder = new BulkRequestBuilder(client);
024        for (int i = 0; i < 1000; i++) {
025            builder.add(indexRequest()
026                    .index("pages").type("row")
027                    .source(jsonBuilder()
028                            .startObject()
029                            .field("user1", "kimchy")
030                            .field("user2", "kimchy")
031                            .field("user3", "kimchy")
032                            .field("user4", "kimchy")
033                            .field("user5", "kimchy")
034                            .field("user6", "kimchy")
035                            .field("user7", "kimchy")
036                            .field("user8", "kimchy")
037                            .field("user9", "kimchy")
038                            .field("rowcount", i)
039                            .field("rs", 1234)));
040        }
041        client.bulk(builder.request()).actionGet();
042
043        client.admin().indices().refresh(refreshRequest()).actionGet();
044
045        long t1 = System.currentTimeMillis();
046        logger.info("t1-t0 = {}", t1-t0);
047
048        for (int i = 0; i < 100; i++) {
049            t1 = System.currentTimeMillis();
050            QueryBuilder queryStringBuilder =
051                    QueryBuilders.queryString("rs:" + 1234);
052            SearchRequestBuilder requestBuilder = client.prepareSearch()
053                    .setIndices("pages")
054                    .setTypes("row")
055                    .setQuery(queryStringBuilder)
056                    .addSort("rowcount", SortOrder.DESC)
057                    .setFrom(i*10).setSize(10);
058            SearchResponse response = requestBuilder.execute().actionGet();
059            long t2 = System.currentTimeMillis();
060            logger.info("t2-t1 = {}", t2-t1);
061        }
062    }
