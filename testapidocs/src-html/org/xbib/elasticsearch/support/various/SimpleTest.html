001package org.xbib.elasticsearch.support.various;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.logging.ESLogger;
005import org.elasticsearch.common.logging.ESLoggerFactory;
006import org.junit.Test;
007
008import static org.elasticsearch.client.Requests.deleteIndexRequest;
009import static org.elasticsearch.client.Requests.indexRequest;
010import static org.elasticsearch.common.settings.Settings.settingsBuilder;
011import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
012import static org.elasticsearch.index.query.QueryBuilders.matchQuery;
013import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
014import static org.junit.Assert.assertEquals;
015
016public class SimpleTest {
017
018    private final static ESLogger logger = ESLoggerFactory.getLogger(SimpleTest.class.getName());
019
020    @Test
021    public void test() throws Exception {
022        String INDEX = "test";
023        String TYPE = "test";
024        String FIELD = "field";
025        Client client = nodeBuilder().settings(settingsBuilder()
026                .put("cluster.name", "test")
027                .put("path.home", System.getProperty("path.home"))
028                .put("index.analysis.analyzer.default.filter.0", "lowercase")
029                .put("index.analysis.analyzer.default.filter.1", "trim")
030                .put("index.analysis.analyzer.default.tokenizer", "keyword"))
031                .local(true).node().client();
032        try {
033            client.admin().indices().delete(deleteIndexRequest(INDEX)).actionGet();
034        } catch (Exception e) {
035            // ignore
036        }
037        client.index(indexRequest()
038                .index(INDEX).type(TYPE).id("1")
039                .source(jsonBuilder().startObject().field(FIELD, "1%2fPJJP3JV2C24iDfEu9XpHBaYxXh%2fdHTbmchB35SDznXO2g8Vz4D7GTIvY54iMiX_149c95f02a8").endObject())
040                .refresh(true)).actionGet();
041        String doc = client.prepareSearch(INDEX).setTypes(TYPE)
042                .setQuery(matchQuery("field", "1%2fPJJP3JV2C24iDfEu9XpHBaYxXh%2fdHTbmchB35SDznXO2g8Vz4D7GTIvY54iMiX_149c95f02a8"))
043                .execute().actionGet().getHits().getAt(0).getSourceAsString();
044
045        logger.info("{}", doc);
046        assertEquals(doc, "{\"field\":\"1%2fPJJP3JV2C24iDfEu9XpHBaYxXh%2fdHTbmchB35SDznXO2g8Vz4D7GTIvY54iMiX_149c95f02a8\"}");
047    }
048}
