002package org.xbib.elasticsearch.support.various;
003
004import org.elasticsearch.action.search.SearchResponse;
005import org.elasticsearch.client.Client;
006import org.elasticsearch.client.transport.TransportClient;
007import org.elasticsearch.common.logging.ESLogger;
008import org.elasticsearch.common.logging.ESLoggerFactory;
009import org.elasticsearch.common.settings.Settings;
010import org.elasticsearch.common.transport.InetSocketTransportAddress;
011import org.elasticsearch.discovery.MasterNotDiscoveredException;
012import org.elasticsearch.node.Node;
013
014import static org.elasticsearch.index.query.QueryBuilders.*;
015import static org.elasticsearch.node.NodeBuilder.*;
016
017public class QueryTest {
018
019    private static final ESLogger logger = ESLoggerFactory.getLogger(QueryTest.class.getName());
020
021    public void testQuery() {
022        Settings settings = Settings.settingsBuilder()
023                .put("cluster.name", "test").build();
024        String index = "test";
025        String type = "test";
026        Node node = null;
027        try {
028            node = nodeBuilder().settings(settings).client(true).node();
029            Client client = node.client();
030            SearchResponse response = client.prepareSearch().setIndices(index).
031                    setTypes(type).
032                    setFrom(0).setSize(10).setQuery(matchQuery("_all", "test")).execute().actionGet();
033        } catch (Exception e) {
034            logger.warn(e.getMessage());
035        } finally {
036            if (node !=null){
037                node.close();
038            }
039        }
040    }
041
042    public void testSniff() {
043        Settings settings = Settings.settingsBuilder()
044                .put("cluster.name", "test")
045                .put("client.transport.sniff", false).build();
046        try {
047            TransportClient client = TransportClient.builder().settings(settings).build();
048            InetSocketTransportAddress address = new InetSocketTransportAddress("localhost", 9300);
049            client.addTransportAddress(address);
050            client.close();
051        } catch (MasterNotDiscoveredException e) {
052            logger.warn(e.getMessage());
053        }
054    }
055}
