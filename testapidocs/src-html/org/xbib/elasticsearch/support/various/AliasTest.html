002package org.xbib.elasticsearch.support.various;
003
004import org.elasticsearch.action.admin.indices.alias.IndicesAliasesRequest;
005import org.elasticsearch.action.admin.indices.alias.get.GetAliasesRequest;
006import org.elasticsearch.action.admin.indices.alias.get.GetAliasesResponse;
007import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
008import org.elasticsearch.client.Client;
009import org.elasticsearch.cluster.metadata.AliasAction;
010import org.elasticsearch.common.Strings;
011import org.elasticsearch.common.logging.ESLogger;
012import org.elasticsearch.common.logging.ESLoggerFactory;
013import org.elasticsearch.common.settings.ImmutableSettings;
014import org.elasticsearch.common.settings.Settings;
015import org.elasticsearch.node.Node;
016import org.junit.Test;
017
018import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
019
020public class AliasTest {
021
022    private static final ESLogger logger = ESLoggerFactory.getLogger(AliasTest.class.getName());
023
024    @Test
025    public void testAlias() {
026        Node node = null;
027        try {
028            Settings settings = ImmutableSettings.settingsBuilder()
029                    .put("cluster.name", "test")
030                    .put("gateway.type", "none")
031                    .build();
032            node = nodeBuilder().settings(settings).local(true).node();
033            Client client = node.client();
034
035            // create index
036            CreateIndexRequest indexRequest = new CreateIndexRequest("test");
037            client.admin().indices().create(indexRequest).actionGet();
038
039            // put alias
040            IndicesAliasesRequest indicesAliasesRequest = new IndicesAliasesRequest();
041            String[] indices = new String[]{"test"};
042            String[] aliases = new String[]{"test_alias"};
043            IndicesAliasesRequest.AliasActions aliasAction = new IndicesAliasesRequest.AliasActions(AliasAction.Type.ADD, indices, aliases);
044            indicesAliasesRequest.addAliasAction(aliasAction);
045            client.admin().indices().aliases(indicesAliasesRequest).actionGet();
046
047            // get alias
048            GetAliasesRequest getAliasesRequest = new GetAliasesRequest(Strings.EMPTY_ARRAY);
049            long t0 = System.nanoTime();
050            GetAliasesResponse getAliasesResponse = client.admin().indices().getAliases(getAliasesRequest).actionGet();
051            long t1 = System.nanoTime() - t0;
052
053            logger.info("{} time(ms) = {}", getAliasesResponse.getAliases(), t1 / 1000000);
054
055        } catch (Exception e) {
056            logger.warn(e.getMessage());
057        } finally {
058            if (node !=null){
059                node.stop();
060                node.close();
061            }
062        }
063    }
064
065}
