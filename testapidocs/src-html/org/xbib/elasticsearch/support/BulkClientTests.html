113    @Test
114    public void testThreadedRandomDocsBulkClient() throws Exception {
115        final BulkClient es = new BulkClient()
116                .flushInterval(TimeValue.timeValueSeconds(10))
117                .maxBulkActions(10000)
118                .newClient(getAddress())
119                .setIndex("test")
120                .setType("test")
121                .newIndex();
122        try {
123            int min = 0;
124            int max = 8;
125            ThreadPoolExecutor pool = EsExecutors.newScaling(min, max, 100, TimeUnit.DAYS,
126                    EsExecutors.daemonThreadFactory("bulk-test"));
127            final CountDownLatch latch = new CountDownLatch(max);
128            for (int i = 0; i < max; i++) {
129                pool.execute(new Runnable() {
130                    public void run() {
131                        logger.info("starting");
132                        for (int i = 0; i < 12345; i++) {
133                            es.indexDocument("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
134                        }
135                        latch.countDown();
136                        logger.info("completed");
137                    }
138                });
139            }
140            logger.info("waiting for 30 seconds...");
141            latch.await(30, TimeUnit.SECONDS);
142            pool.shutdown();
143            es.flush();
144        } catch (NoNodeAvailableException e) {
145            logger.warn("skipping, no node available");
146        } finally {
147            es.shutdown();
148            logger.info("bulk counter = {}", es.getBulkCounter());
149            assertEquals(es.getBulkCounter(), 10);
150            if (es.hasErrors()) {
151                logger.error("error", es.getThrowable());
152            }
153            assertFalse(es.hasErrors());
154        }
155    }
