001    package org.xbib.elasticsearch.support;
002    
003    import java.net.URI;
004    import java.util.Random;
005    
006    public class AbstractNodeRandomTest extends AbstractNodeTest {
007    
008        private static Random random = new Random();
009    
010        private static char[] numbersAndLetters = ("0123456789abcdefghijklmnopqrstuvwxyz").toCharArray();
011    
012        protected String randomString(int len) {
013            final char[] buf = new char[len];
014            final int n = numbersAndLetters.length - 1;
015            for (int i = 0; i < buf.length; i++) {
016                buf[i] = numbersAndLetters[random.nextInt(n)];
017            }
018            return new String(buf);
019        }
020    
021    }
