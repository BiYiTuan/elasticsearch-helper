001package org.xbib.elasticsearch.support.network;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.junit.Test;
006
007import java.net.InetAddress;
008import java.net.NetworkInterface;
009import java.util.Collections;
010import java.util.Enumeration;
011
012public class NetworkTest {
013
014    private final static Logger logger = LogManager.getLogger(NetworkTest.class);
015
016    @Test
017    public void testNetwork() throws Exception {
018        Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
019        for (NetworkInterface netint : Collections.list(nets)) {
020            System.out.println("checking network interface = " + netint.getName());
021            Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
022            for (InetAddress addr : Collections.list(inetAddresses)) {
023                logger.info("found address = " + addr.getHostAddress()
024                        + " name = " + addr.getHostName()
025                        + " canicalhostname = " + addr.getCanonicalHostName()
026                        + " loopback = " + addr.isLoopbackAddress()
027                        + " sitelocal = " + addr.isSiteLocalAddress()
028                        + " linklocal = " + addr.isLinkLocalAddress()
029                        + " anylocal = " + addr.isAnyLocalAddress()
030                        + " multicast = " + addr.isMulticastAddress()
031                        + " mcglobal = " + addr.isMCGlobal()
032                        + " mclinklocal = " + addr.isMCLinkLocal()
033                        + " mcnodelocal = " + addr.isMCNodeLocal()
034                        + " mcorglocal = " + addr.isMCOrgLocal()
035                        + " mcsitelocal = " + addr.isMCSiteLocal()
036                        + " mcsitelocal = " + addr.isReachable(1000));
037            }
038        }
039
040    }
041}
