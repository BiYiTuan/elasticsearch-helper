001package org.xbib.elasticsearch.support.store.mockfs;
002
003import org.apache.lucene.store.Directory;
004import org.apache.lucene.store.FilterDirectory;
005import org.apache.lucene.store.LockFactory;
006import org.apache.lucene.store.MockDirectoryWrapper;
007import org.apache.lucene.store.NRTCachingDirectory;
008import org.apache.lucene.store.StoreRateLimiting;
009import org.elasticsearch.common.inject.Inject;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.index.settings.IndexSettings;
012import org.elasticsearch.index.shard.ShardPath;
013import org.elasticsearch.index.store.FsDirectoryService;
014import org.elasticsearch.index.store.IndexStore;
015import org.elasticsearch.index.store.IndexStoreModule;
016
017import java.io.IOException;
018import java.lang.reflect.Field;
019import java.nio.file.Path;
020import java.util.Collection;
021import java.util.Set;
022
023public class MockfsDirectoryService extends FsDirectoryService {
024
025    private final FsDirectoryService delegateService;
026    private final Settings indexSettings;
027
028    @Inject
029    public MockfsDirectoryService(@IndexSettings Settings indexSettings, IndexStore indexStore, final ShardPath path) {
030        super(indexSettings, indexStore, path);
031        this.indexSettings = indexSettings;
032        delegateService = directoryService(indexStore, path);
033    }
034
035    @Override
036    public Directory newDirectory() throws IOException {
037        return wrap(delegateService.newDirectory());
038    }
039
040    @Override
041    protected synchronized Directory newFSDirectory(Path location, LockFactory lockFactory) throws IOException {
042        throw new UnsupportedOperationException();
043    }
044
045    @Override
046    public void onPause(long nanos) {
047        delegateService.onPause(nanos);
048    }
049
050    @Override
051    public StoreRateLimiting rateLimiting() {
052        return delegateService.rateLimiting();
053    }
054
055    @Override
056    public long throttleTimeInNanos() {
057        return delegateService.throttleTimeInNanos();
058    }
059
060    private Directory wrap(Directory dir) {
061        return new ElasticsearchMockDirectoryWrapper(dir);
062    }
063
064    private FsDirectoryService directoryService(IndexStore indexStore, ShardPath path) {
065        Settings.Builder builder = Settings.settingsBuilder()
066                .put(indexSettings)
067                .put(IndexStoreModule.STORE_TYPE, IndexStoreModule.Type.DEFAULT); // TODO memory store, RAM codec?
068        return new FsDirectoryService(builder.build(), indexStore, path);
069    }
070
071    class ElasticsearchMockDirectoryWrapper extends MockDirectoryWrapper {
072
073        private final Set<String> superUnSyncedFiles;
074
075        @SuppressWarnings("unchecked")
076        public ElasticsearchMockDirectoryWrapper(Directory delegate) {
077            super(delegate);
078            try {
079                Field field = MockDirectoryWrapper.class.getDeclaredField("unSyncedFiles");
080                field.setAccessible(true);
081                superUnSyncedFiles = (Set<String>) field.get(this);
082            } catch (ReflectiveOperationException roe) {
083                throw new RuntimeException(roe);
084            }
085        }
086
087        private boolean mustSync() {
088            Directory delegate = in;
089            while (delegate instanceof FilterDirectory) {
090                if (delegate instanceof NRTCachingDirectory) {
091                    return true;
092                }
093                delegate = ((FilterDirectory) delegate).getDelegate();
094            }
095            return false;
096        }
097
098        @Override
099        public synchronized void sync(Collection<String> names) throws IOException {
100            if (mustSync()) {
101                super.sync(names);
102            } else {
103                superUnSyncedFiles.removeAll(names);
104            }
105        }
106
107    }
108}
