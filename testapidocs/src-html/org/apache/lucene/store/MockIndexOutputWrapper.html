020import java.io.Closeable;
021import java.io.IOException;
022
023/**
024 * Used by MockRAMDirectory to create an output stream that
025 * will throw an IOException on fake disk full, track max
026 * disk space actually used, and maybe throw random
027 * IOExceptions.
028 */
029public class MockIndexOutputWrapper extends IndexOutput {
030    final String name;
031    private final IndexOutput delegate;
032    byte[] singleByte = new byte[1];
033    private MockDirectoryWrapper dir;
034    private boolean first = true;
035    private boolean closed;
036
037    /**
038     * Construct an empty output buffer.
039     * @param dir dir
040     * @param delegate delefate
041     * @param name name
042     */
043    public MockIndexOutputWrapper(MockDirectoryWrapper dir, IndexOutput delegate, String name) {
044        super("MockIndexOutputWrapper(" + delegate + ")");
045        this.dir = dir;
046        this.name = name;
047        this.delegate = delegate;
048    }
049
050    private void checkCrashed() throws IOException {
051        // If MockRAMDir crashed since we were opened, then don't write anything
052        if (dir.crashed) {
053            throw new IOException("MockRAMDirectory was crashed; cannot write to " + name);
054        }
055    }
056
057    private void checkDiskFull(byte[] b, int offset, DataInput in, long len) throws IOException {
058        long freeSpace = dir.maxSize == 0 ? 0 : dir.maxSize - dir.sizeInBytes();
059        long realUsage = 0;
060
061        // Enforce disk full:
062        if (dir.maxSize != 0 && freeSpace <= len) {
063            // Compute the real disk free.  This will greatly slow
064            // down our test but makes it more accurate:
065            realUsage = dir.getRecomputedActualSizeInBytes();
066            freeSpace = dir.maxSize - realUsage;
067        }
068
069        if (dir.maxSize != 0 && freeSpace <= len) {
070            if (freeSpace > 0) {
071                realUsage += freeSpace;
072                if (b != null) {
073                    delegate.writeBytes(b, offset, (int) freeSpace);
074                } else {
075                    delegate.copyBytes(in, (int) freeSpace);
076                }
077            }
078            if (realUsage > dir.maxUsedSize) {
079                dir.maxUsedSize = realUsage;
080            }
081            String message = "fake disk full at " + dir.getRecomputedActualSizeInBytes() + " bytes when writing " + name + " (file length=" + delegate.getFilePointer();
082            if (freeSpace > 0) {
083                message += "; wrote " + freeSpace + " of " + len + " bytes";
084            }
085            message += ")";
086            throw new IOException(message);
087        }
088    }
089
090    @Override
091    public void close() throws IOException {
092        if (closed) {
093            delegate.close(); // don't mask double-close bugs
094            return;
095        }
096        closed = true;
097
098        try (Closeable delegate = this.delegate) {
099            assert delegate != null;
100            dir.maybeThrowDeterministicException();
101        } finally {
102            dir.removeIndexOutput(this, name);
103            if (dir.trackDiskUsage) {
104                // Now compute actual disk usage & track the maxUsedSize
105                // in the MockDirectoryWrapper:
106                long size = dir.getRecomputedActualSizeInBytes();
107                if (size > dir.maxUsedSize) {
108                    dir.maxUsedSize = size;
109                }
110            }
111        }
112    }
113
114    private void ensureOpen() {
115        if (closed) {
116            throw new AlreadyClosedException("Already closed: " + this);
117        }
118    }
119
120    @Override
121    public void writeByte(byte b) throws IOException {
122        singleByte[0] = b;
123        writeBytes(singleByte, 0, 1);
124    }
125
126    @Override
127    public void writeBytes(byte[] b, int offset, int len) throws IOException {
128        ensureOpen();
129        checkCrashed();
130        checkDiskFull(b, offset, null, len);
131        delegate.writeBytes(b, offset, len);
132        dir.maybeThrowDeterministicException();
133    }
134
135    @Override
136    public long getFilePointer() {
137        return delegate.getFilePointer();
138    }
139
140    @Override
141    public void copyBytes(DataInput input, long numBytes) throws IOException {
142        ensureOpen();
143        checkCrashed();
144        checkDiskFull(null, 0, input, numBytes);
145        delegate.copyBytes(input, numBytes);
146        dir.maybeThrowDeterministicException();
147    }
148
149    @Override
150    public long getChecksum() throws IOException {
151        return delegate.getChecksum();
152    }
153
154    @Override
155    public String toString() {
156        return "MockIndexOutputWrapper(" + delegate + ")";
157    }
158}
