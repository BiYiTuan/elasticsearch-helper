002package org.xbib.elasticsearch.support.client;
003
004import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
005import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
006import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequest;
007import org.elasticsearch.common.logging.ESLogger;
008import org.elasticsearch.common.logging.ESLoggerFactory;
009import org.elasticsearch.common.settings.ImmutableSettings;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.common.unit.TimeValue;
012
013import java.io.IOException;
014import java.io.InputStream;
015import java.util.Map;
016
017public abstract class AbstractIngestTransportClient extends AbstractTransportClient
018        implements Ingest {
019
020    private final static ESLogger logger = ESLoggerFactory.getLogger(AbstractIngestTransportClient.class.getSimpleName());
021
022    /**
023     * The default index
024     */
025    private String index;
026    /**
027     * The default type
028     */
029    private String type;
030
031    private ConfigHelper configHelper = new ConfigHelper();
032
033    @Override
034    public AbstractIngestTransportClient setIndex(String index) {
035        this.index = index;
036        return this;
037    }
038
039    public String getIndex() {
040        return index;
041    }
042
043    @Override
044    public AbstractIngestTransportClient setType(String type) {
045        this.type = type;
046        return this;
047    }
048
049    public String getType() {
050        return type;
051    }
052
053    @Override
054    public AbstractIngestTransportClient shards(int shards) {
055        return setting("index.number_of_shards", shards);
056    }
057
058    @Override
059    public AbstractIngestTransportClient replica(int replica) {
060        return setting("index.number_of_replicas", replica);
061    }
062
063
064    public ImmutableSettings.Builder settingsBuilder() {
065        return configHelper.settingsBuilder();
066    }
067
068    public AbstractIngestTransportClient resetSettings() {
069        configHelper.reset();
070        return this;
071    }
072
073    public AbstractIngestTransportClient setting(String key, String value) {
074        configHelper.setting(key, value);
075        return this;
076    }
077
078    public AbstractIngestTransportClient setting(String key, Boolean value) {
079        configHelper.setting(key, value);
080        return this;
081    }
082
083    public AbstractIngestTransportClient setting(String key, Integer value) {
084        configHelper.setting(key, value);
085        return this;
086    }
087
088    public AbstractIngestTransportClient setting(InputStream in) throws IOException {
089        configHelper.setting(in);
090        return this;
091    }
092
093    public Settings settings() {
094        return configHelper.settings();
095    }
096
097    public AbstractIngestTransportClient mapping(String type, InputStream in) throws IOException {
098        configHelper.mapping(type, in);
099        return this;
100    }
101
102    public AbstractIngestTransportClient mapping(String type, String mapping) {
103        configHelper.mapping(type, mapping);
104        return this;
105    }
106
107    public AbstractIngestTransportClient timeStampField(String timeStampField) {
108        configHelper.timeStampField(timeStampField);
109        return this;
110    }
111
112    public String timeStampField() {
113        return configHelper.timeStampField();
114    }
115
116    public AbstractIngestTransportClient timeStampFieldEnabled(boolean enable) {
117        configHelper.timeStampFieldEnabled(enable);
118        return this;
119    }
120
121    public AbstractIngestTransportClient kibanaEnabled(boolean enable) {
122        configHelper.kibanaEnabled(enable);
123        return this;
124    }
125
126    public String defaultMapping() throws IOException {
127        return configHelper.defaultMapping();
128    }
129
130    public Map<String,String> mappings() {
131        return configHelper.mappings();
132    }
133
134    @Override
135    public AbstractIngestTransportClient newIndex() {
136        if (client == null) {
137            logger.warn("no client for create index");
138            return this;
139        }
140        if (getIndex() == null) {
141            logger.warn("no index name given to create index");
142            return this;
143        }
144        CreateIndexRequest request = new CreateIndexRequest(getIndex());
145        if (settings() != null) {
146            request.settings(settings());
147        }
148        if (mappings() != null) {
149            for (Map.Entry<String,String> me : mappings().entrySet()) {
150                request.mapping(me.getKey(), me.getValue());
151            }
152        }
153        logger.info("creating index {} with settings = {}, mappings = {}",
154                getIndex(), settings() != null ? settings().getAsMap() : null, mappings());
155        try {
156            client.admin().indices().create(request).actionGet();
157        } catch (Exception e) {
158            logger.error(e.getMessage(), e);
159        }
160        return this;
161    }
162
163    @Override
164    public synchronized AbstractIngestTransportClient deleteIndex() {
165        if (client == null) {
166            logger.warn("no client for delete index");
167            return this;
168        }
169        if (getIndex() == null) {
170            logger.warn("no index name given to delete index");
171            return this;
172        }
173        try {
174            client.admin().indices().delete(new DeleteIndexRequest(getIndex())).actionGet();
175        } catch (Exception e) {
176            logger.error(e.getMessage(), e);
177        }
178        return this;
179    }
180
181    public AbstractIngestTransportClient putMapping(String index) {
182        if (client == null) {
183            logger.warn("no client for put mapping");
184            return this;
185        }
186        configHelper.putMapping(client, index);
187        return this;
188    }
189
190    public AbstractIngestTransportClient deleteMapping(String index, String type) {
191        if (client == null) {
192            logger.warn("no client for delete mapping");
193            return this;
194        }
195        configHelper.deleteMapping(client, index, type);
196        return this;
197    }
198
199    @Override
200    public AbstractIngestTransportClient waitForCluster(ClusterHealthStatus status, TimeValue timeValue) throws IOException {
201        ClientHelper.waitForCluster(client, status, timeValue);
202        return this;
203    }
204
205}
