098        @Override
099        public AbstractIngestClient waitForCluster(ClusterHealthStatus status, TimeValue timeout) throws IOException {
100            try {
101                logger.info("waiting for cluster state {}", status.name());
102                ClusterHealthResponse healthResponse =
103                        client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
104                if (healthResponse.isTimedOut()) {
105                    throw new IOException("cluster state is " + healthResponse.getStatus().name()
106                            + " and not " + status.name()
107                            + ", cowardly refusing to continue with operations");
108                } else {
109                    logger.info("... cluster state ok");
110                }
111            } catch (ElasticSearchTimeoutException e) {
112                throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
113            }
114            return this;
115        }

117        @Override
118        public int waitForRecovery() {
119            if (getIndex() == null) {
120                logger.warn("not waiting for recovery, index not set");
121                return -1;
122            }
123            IndicesStatusResponse response = client.admin().indices()
124                    .status(new IndicesStatusRequest(getIndex()).recovery(true)).actionGet();
125            logger.info("indices status response = {}, failed = {}", response.getTotalShards(), response.getFailedShards());
126            for (ShardStatus status : response.getShards()) {
127                logger.info("shard {} status {}", status.getShardId(), status.getState().name());
128            }
129            return response.getTotalShards();
130        }

132        @Override
133        public int updateReplicaLevel(int level) throws IOException {
134            if (getIndex() == null) {
135                logger.warn("no index name given");
136                return -1;
137            }
138            waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
139            update("number_of_replicas", level);
140            return waitForRecovery();
141        }

237        public String defaultMapping() {
238            try {
239                XContentBuilder b = jsonBuilder()
240                                .startObject()
241                                .startObject("_default_")
242                                .field("date_detection", dateDetection);
243                if (timeStampFieldEnabled) {
244                                b.startObject("_timestamp")
245                                .field("enabled", timeStampFieldEnabled)
246                                .field("path", timeStampField)
247                                .endObject();
248                }
249                if (kibanaEnabled) {
250                                b.startObject("properties")
251                                .startObject("@fields")
252                                .field("type", "object")
253                                .field("dynamic", true)
254                                .field("path", "full")
255                                .endObject()
256                                .startObject("@message")
257                                .field("type", "string")
258                                .field("index", "analyzed")
259                                .endObject()
260                                .startObject("@source")
261                                .field("type", "string")
262                                .field("index", "not_analyzed")
263                                .endObject()
264                                .startObject("@source_host")
265                                .field("type", "string")
266                                .field("index", "not_analyzed")
267                                .endObject()
268                                .startObject("@source_path")
269                                .field("type", "string")
270                                .field("index", "not_analyzed")
271                                .endObject()
272                                .startObject("@tags")
273                                .field("type", "string")
274                                .field("index", "not_analyzed")
275                                .endObject()
276                                .startObject("@timestamp")
277                                .field("type", "date")
278                                .endObject()
279                                .startObject("@type")
280                                .field("type", "string")
281                                .field("index", "not_analyzed")
282                                .endObject()
283                                .endObject();
284                }
285                b.endObject()
286                        .endObject();
287                return b.string();
288            } catch (IOException e) {
289                logger.error(e.getMessage(), e);
290            }
291            return null;
292        }

304        @Override
305        public synchronized AbstractIngestClient newIndex() {
306            if (client == null) {
307                logger.warn("no client for create index");
308                return this;
309            }
310            if (getIndex() == null) {
311                logger.warn("no index name given to create index");
312                return this;
313            }
314            CreateIndexRequest indexRequest = new CreateIndexRequest(getIndex());
315            if (settings() != null && !settings().internalMap().isEmpty()) {
316                indexRequest.settings(settings());
317            }
318            if (!mappings.isEmpty()) {
319                for (Map.Entry<String,String> me : mappings.entrySet()) {
320                    indexRequest.mapping(me.getKey(), me.getValue());
321                }
322            }
323            logger.info("creating index = {} settings = {} mappings = {}",
324                    getIndex(),
325                    settings() != null ? settings().build().getAsMap() : null,
326                    mappings);
327            try {
328                client.admin().indices()
329                        .create(indexRequest)
330                        .actionGet();
331            } catch (Exception e) {
332                logger.error(e.getMessage(), e);
333            }
334            mappings.clear();
335            return this;
336        }

338        @Override
339        public synchronized AbstractIngestClient deleteIndex() {
340            if (client == null) {
341                logger.warn("no client for delete index");
342                return this;
343            }
344            if (getIndex() == null) {
345                logger.warn("no index name given to delete index");
346                return this;
347            }
348            try {
349                client.admin().indices()
350                        .delete(new DeleteIndexRequest(getIndex()))
351                        .actionGet();
352            } catch (Exception e) {
353                logger.error(e.getMessage(), e);
354            }
355            return this;
356        }
357    
358        @Override
359        public synchronized AbstractIngestClient newMappings() {
360            if (client == null) {
361                logger.warn("no client for new mapping");
362                return this;
363            }
364            if (getIndex() == null) {
365                logger.warn("no index name for new mapping");
366                return this;
367            }
368            try {
369                if (!mappings.isEmpty()) {
370                    for (Map.Entry<String,String> me : mappings.entrySet()) {
371                        client.admin().indices()
372                            .putMapping(new PutMappingRequest()
373                                    .indices(new String[]{getIndex()})
374                                    .type(me.getKey())
375                                    .source(me.getValue()))
376                            .actionGet();
377                    }
378                }
379            } catch (Exception e) {
380                logger.error(e.getMessage(), e);
381            }
382            mappings.clear();
383            return this;
384        }

386        @Override
387        public synchronized AbstractIngestClient deleteMapping(String type) {
388            if (client == null) {
389                logger.warn("no client for delete mapping");
390                return this;
391            }
392            if (getIndex() == null) {
393                logger.warn("no index name given for delete mapping");
394                return this;
395            }
396            try {
397                client.admin().indices()
398                        .deleteMapping(new DeleteMappingRequest()
399                                .indices(new String[]{getIndex()})
400                                .type(type))
401                        .actionGet();
402            } catch (Exception e) {
403                logger.error(e.getMessage(), e);
404            }
405            return this;
406        }

409        protected void update(String key, Object value) {
410            if (client == null) {
411                return;
412            }
413            if (value == null) {
414                return;
415            }
416            if (getIndex() == null) {
417                return;
418            }
419            ImmutableSettings.Builder settingsBuilder = ImmutableSettings.settingsBuilder();
420            settingsBuilder.put(key, value.toString());
421            UpdateSettingsRequest updateSettingsRequest = new UpdateSettingsRequest(getIndex())
422                    .settings(settingsBuilder);
423            client.admin().indices()
424                    .updateSettings(updateSettingsRequest)
425                    .actionGet();
426        }
