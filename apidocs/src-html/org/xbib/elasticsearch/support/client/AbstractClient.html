036public abstract class AbstractClient {

038    private final static ESLogger logger = Loggers.getLogger(AbstractClient.class);
039
040    // the default cluster name
041    private final static String DEFAULT_CLUSTER_NAME = "elasticsearch";
042    // the default connection specification
043    private final static URI DEFAULT_URI = URI.create("es://hostname:9300");
044    // the transport addresses
045    private final Set<InetSocketTransportAddress> addresses = new HashSet();

047    // singleton
048    protected TransportClient client;
049    // the settings
050    protected Settings settings;

052    protected abstract Settings initialSettings(URI uri, int poolsize);
053
054    public AbstractClient newClient() {
055        return newClient(findURI());
056    }
057
058    public AbstractClient newClient(URI uri) {
059        return newClient(uri, Runtime.getRuntime().availableProcessors() * 4);
060    }
061
062    public synchronized AbstractClient newClient(URI uri, int poolsize) {
063        if (client != null) {
064            // disconnect
065            client.close();
066            // release thread pool
067            client.threadPool().shutdown();
068            client = null;
069        }
070        this.settings = initialSettings(uri, poolsize);
071        logger.info("settings={}", settings.getAsMap());
072        this.client = new TransportClient(settings);
073        try {
074            connect(uri);
075        } catch (UnknownHostException e) {
076            logger.error(e.getMessage(), e);
077        } catch (SocketException e) {
078            logger.error(e.getMessage(), e);
079        } catch (IOException e) {
080            logger.error(e.getMessage(), e);
081        }
082        return this;
083    }
084
085    public Client client() {
086        return client;
087    }
088
089    public String stats() throws IOException {
090        XContentBuilder builder = jsonBuilder();
091        builder.startObject();
092        client.threadPool().stats().toXContent(builder, ToXContent.EMPTY_PARAMS);
093        builder.endObject();
094        return builder.string();
095    }
096
097    public synchronized void shutdown() {
098        if (client != null) {
099            // close() sends a disconnect to the server by using the threadpool
100            client.close();
101            // threadpool can be closed now
102            client.threadPool().shutdown();
103            client = null;
104        }
105        addresses.clear();
106    }

108    protected static URI findURI() {
109        URI uri = DEFAULT_URI;
110        String hostname = "localhost";
111        try {
112            hostname = InetAddress.getLocalHost().getHostName();
113            logger.debug("the hostname is {}", hostname);
114            URL url = AbstractIngestClient.class.getResource("/org/xbib/elasticsearch/cluster.properties");
115            if (url != null) {
116                InputStream in = url.openStream();
117                Properties p = new Properties();
118                p.load(in);
119                in.close();
120                // the properties contains default URIs per hostname
121                if (p.containsKey(hostname)) {
122                    uri = URI.create(p.getProperty(hostname));
123                    logger.debug("URI found in cluster.properties for hostname {}: {}", hostname, uri);
124                    return uri;
125                }
126            }
127        } catch (UnknownHostException e) {
128            logger.warn("can't resolve host name, probably something wrong with network config: " + e.getMessage(), e);
129        } catch (Exception e) {
130            logger.warn(e.getMessage(), e);
131        }
132        logger.debug("URI for hostname {}: {}", hostname, uri);
133        return uri;
134    }

136    protected String findClusterName(URI uri) {
137        String clustername;
138        try {
139            // look for URI parameters
140            Map<String, String> map = parseQueryString(uri, "UTF-8");
141            clustername = map.get("es.cluster.name");
142            if (clustername != null) {
143                logger.info("cluster name found in URI {}: {}", uri, clustername);
144                return clustername;
145            }
146            clustername = map.get("cluster.name");
147            if (clustername != null) {
148                logger.info("cluster name found in URI {}: {}", uri, clustername);
149                return clustername;
150            }
151        } catch (UnsupportedEncodingException ex) {
152            logger.warn(ex.getMessage(), ex);
153        }
154        logger.info("cluster name not found in URI {}, parameter es.cluster.name", uri);
155        clustername = System.getProperty("es.cluster.name");
156        if (clustername != null) {
157            logger.info("cluster name found in es.cluster.name system property: {}", clustername);
158            return clustername;
159        }
160        clustername = System.getProperty("cluster.name");
161        if (clustername != null) {
162            logger.info("cluster name found in cluster.name system property: {}", clustername);
163            return clustername;
164        }
165        logger.info("cluster name not found, falling back to default: {}", DEFAULT_CLUSTER_NAME);
166        clustername = DEFAULT_CLUSTER_NAME;
167        return clustername;
168    }

170    protected void connect(URI uri) throws IOException {
171        String hostname = uri.getHost();
172        int port = uri.getPort();
173        boolean newaddresses = false;
174        if (!"es".equals(uri.getScheme())) {
175            logger.warn("please specify URI scheme 'es'");
176        }
177        if ("hostname".equals(hostname)) {
178            InetSocketTransportAddress address = new InetSocketTransportAddress(InetAddress.getLocalHost().getHostName(), port);
179            if (!addresses.contains(address)) {
180                logger.info("adding hostname address for transport client: {}", address);
181                client.addTransportAddress(address);
182                addresses.add(address);
183                newaddresses = true;
184            }
185        } else if ("interfaces".equals(hostname)) {
186            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
187            for (NetworkInterface netint : Collections.list(nets)) {
188                logger.info("checking network interface = {}", netint.getName());
189                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
190                for (InetAddress addr : Collections.list(inetAddresses)) {
191                    logger.info("checking address = {}", addr.getHostAddress());
192                    InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
193                    if (!addresses.contains(address)) {
194                        logger.info("adding address to transport client: {}", address);
195                        client.addTransportAddress(address);
196                        addresses.add(address);
197                        newaddresses = true;
198                    }
199                }
200            }
201        } else if ("inet4".equals(hostname)) {
202            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
203            for (NetworkInterface netint : Collections.list(nets)) {
204                logger.info("checking network interface = {}", netint.getName());
205                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
206                for (InetAddress addr : Collections.list(inetAddresses)) {
207                    if (addr instanceof Inet4Address) {
208                        logger.info("checking address = {}", addr.getHostAddress());
209                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
210                        if (!addresses.contains(address)) {
211                            logger.info("adding address for transport client: {}", address);
212                            client.addTransportAddress(address);
213                            addresses.add(address);
214                            newaddresses = true;
215                        }
216                    }
217                }
218            }
219        } else if ("inet6".equals(hostname)) {
220            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
221            for (NetworkInterface netint : Collections.list(nets)) {
222                logger.info("checking network interface = {}", netint.getName());
223                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
224                for (InetAddress addr : Collections.list(inetAddresses)) {
225                    if (addr instanceof Inet6Address) {
226                        logger.info("checking address = {}", addr.getHostAddress());
227                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
228                        if (!addresses.contains(address)) {
229                            logger.info("adding address for transport client: {}", address);
230                            client.addTransportAddress(address);
231                            addresses.add(address);
232                            newaddresses = true;
233                        }
234                    }
235                }
236            }
237        } else {
238            InetSocketTransportAddress address = new InetSocketTransportAddress(hostname, port);
239            if (!addresses.contains(address)) {
240                logger.info("adding custom address for transport client: {}", address);
241                client.addTransportAddress(address);
242                addresses.add(address);
243                newaddresses = true;
244            }
245        }
246        logger.info("configured addresses to connect: {}", addresses);
247        if (newaddresses) {
248            List<DiscoveryNode> nodes = client.connectedNodes().asList();
249            logger.info("connected nodes = {}", nodes);
250            for (DiscoveryNode node : nodes) {
251                logger.info("new connection to {} {}", node.getId(), node.getName());
252            }
253        }
254    }

256    private Map<String, String> parseQueryString(URI uri, String encoding)
257            throws UnsupportedEncodingException {
258        Map<String, String> m = new HashMap<String, String>();
259        if (uri == null) {
260            throw new IllegalArgumentException();
261        }
262        if (uri.getRawQuery() == null) {
263            return m;
264        }
265        // use getRawQuery because we do our decoding by ourselves
266        StringTokenizer st = new StringTokenizer(uri.getRawQuery(), "&");
267        while (st.hasMoreTokens()) {
268            String pair = st.nextToken();
269            int pos = pair.indexOf('=');
270            if (pos < 0) {
271                m.put(pair, null);
272            } else {
273                m.put(pair.substring(0, pos), decode(pair.substring(pos + 1, pair.length()), encoding));
274            }
275        }
276        return m;
277    }

279    private String decode(String s, String encoding) {
280        StringBuilder sb = new StringBuilder();
281        boolean fragment = false;
282        for (int i = 0; i < s.length(); i++) {
283            char ch = s.charAt(i);
284            switch (ch) {
285                case '+':
286                    sb.append(' ');
287                    break;
288                case '#':
289                    sb.append(ch);
290                    fragment = true;
291                    break;
292                case '%':
293                    if (!fragment) {
294                        // fast hex decode
295                        sb.append((char) ((Character.digit(s.charAt(++i), 16) << 4)
296                                | Character.digit(s.charAt(++i), 16)));
297                    } else {
298                        sb.append(ch);
299                    }
300                    break;
301                default:
302                    sb.append(ch);
303                    break;
304            }
305        }
306        try {
307            // URL default encoding is ISO-8859-1
308            return new String(sb.toString().getBytes("ISO-8859-1"), encoding);
309        } catch (UnsupportedEncodingException e) {
310            throw new Error("encoding " + encoding + " not supported");
311        }
312    }
