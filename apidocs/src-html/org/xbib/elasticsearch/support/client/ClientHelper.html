001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.ElasticsearchTimeoutException;
004import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
005import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
006import org.elasticsearch.action.admin.indices.recovery.RecoveryResponse;
007import org.elasticsearch.action.admin.indices.settings.put.UpdateSettingsRequest;
008import org.elasticsearch.client.Client;
009import org.elasticsearch.client.transport.NoNodeAvailableException;
010import org.elasticsearch.client.transport.TransportClient;
011import org.elasticsearch.cluster.node.DiscoveryNode;
012import org.elasticsearch.common.settings.ImmutableSettings;
013import org.elasticsearch.common.unit.TimeValue;
014
015import java.io.IOException;
016import java.util.List;
017
018import static org.elasticsearch.common.collect.Lists.newLinkedList;
019
020public class ClientHelper {
021
022    public static List<String> getConnectedNodes(TransportClient client) {
023        List<String> nodes = newLinkedList();
024        if (client.connectedNodes() != null) {
025            for (DiscoveryNode discoveryNode : client.connectedNodes()) {
026                nodes.add(discoveryNode.toString());
027            }
028        }
029        return nodes;
030    }
031
032    public static void updateIndexSetting(Client client, String index, String key, Object value) throws IOException {
033        if (client == null) {
034            throw new IOException("no client");
035        }
036        if (index == null) {
037            throw new IOException("no index name given");
038        }
039        if (key == null) {
040            throw new IOException("no key given");
041        }
042        if (value == null) {
043            throw new IOException("no value given");
044        }
045        ImmutableSettings.Builder settingsBuilder = ImmutableSettings.settingsBuilder();
046        settingsBuilder.put(key, value.toString());
047        UpdateSettingsRequest updateSettingsRequest = new UpdateSettingsRequest(index)
048                .settings(settingsBuilder);
049        client.admin().indices().updateSettings(updateSettingsRequest).actionGet();
050    }
051
052    public static int waitForRecovery(Client client, String index) throws IOException {
053        if (index == null) {
054            throw new IOException("unable to waitfor recovery, index not set");
055        }
056        RecoveryResponse response = client.admin().indices().prepareRecoveries(index).execute().actionGet();
057        int shards = response.getTotalShards();
058        client.admin().cluster().prepareHealth(index).setWaitForActiveShards(shards).execute().actionGet();
059        return shards;
060    }
061
062    public static void waitForCluster(Client client, ClusterHealthStatus status, TimeValue timeout) throws IOException {
063        try {
064            ClusterHealthResponse healthResponse =
065                    client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
066            if (healthResponse != null && healthResponse.isTimedOut()) {
067                throw new IOException("cluster state is " + healthResponse.getStatus().name()
068                        + " and not " + status.name()
069                        + ", cowardly refusing to continue with operations");
070            }
071        } catch (ElasticsearchTimeoutException e) {
072            throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
073        }
074    }
075
076    public static String healthColor(Client client) {
077        try {
078            ClusterHealthResponse healthResponse =
079                    client.admin().cluster().prepareHealth().setTimeout(TimeValue.timeValueSeconds(30)).execute().actionGet();
080            ClusterHealthStatus status = healthResponse.getStatus();
081            return status.name();
082        } catch (ElasticsearchTimeoutException e) {
083            return "TIMEOUT";
084        } catch (NoNodeAvailableException e) {
085            return "DISCONNECTED";
086        } catch (Throwable t) {
087            return "[" + t.getMessage() + "]";
088        }
089    }
090
091    public static int updateReplicaLevel(Client client, String index, int level) throws IOException {
092        waitForCluster(client, ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
093        updateIndexSetting(client, index, "number_of_replicas", level);
094        return waitForRecovery(client, index);
095    }
096
097    public static void disableRefresh(Client client, String index) throws IOException {
098        updateIndexSetting(client, index, "refresh_interval", -1);
099    }
100
101    public static void enableRefresh(Client client, String index) throws IOException {
102        updateIndexSetting(client, index, "refresh_interval", 1000);
103    }
104
105    public static void flush(Client client, String index) {
106        client.admin().indices().prepareFlush().setIndices(index).execute().actionGet();
107    }
108
109    public static void refresh(Client client, String index) {
110        client.admin().indices().prepareRefresh().setIndices(index).setForce(true).execute().actionGet();
111    }
112
113}
