108        @Override
109        public IngestClient newClient(URI uri, Settings settings) {
110            super.newClient(uri, settings);
111            resetSettings();
112            IngestProcessor.Listener listener = new IngestProcessor.Listener() {
113                @Override
114                public void beforeBulk(long executionId, int concurrency, IngestRequest request) {
115                    int n = request.numberOfActions();
116                    submitted.inc(n);
117                    currentIngestNumDocs.inc(n);
118                    totalIngestSizeInBytes.inc(request.estimatedSizeInBytes());
119                    if (logger.isDebugEnabled()) {
120                        logger.debug("before bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
121                                executionId, n, request.estimatedSizeInBytes(), concurrency);
122                    }
123                }
124    
125                @Override
126                public void afterBulk(long executionId, int concurrency, IngestResponse response) {
127                    succeeded.inc(response.successSize());
128                    failed.inc(response.failureSize());
129                    totalIngest.inc(response.tookInMillis());
130                    if (logger.isDebugEnabled()) {
131                        logger.debug("after bulk [{}] [{} items succeeded] [{} items failed] [{}ms]",
132                                executionId, response.successSize(), response.failureSize(), response.tookInMillis());
133                    }
134                    if (!response.failure().isEmpty()) {
135                        closed = true;
136                        for (IngestItemFailure f: response.failure()) {
137                            logger.error("after bulk [{}] [{}] failure, reason: {}", executionId, f.pos(), f.message());
138                        }
139                    } else {
140                        currentIngestNumDocs.dec(response.successSize());
141                    }
142                }
143    
144                @Override
145                public void afterBulk(long executionId, int concurrency, Throwable failure) {
146                    closed = true;
147                    logger.error("after bulk ["+executionId+"] failure", failure);
148                    throwable = failure;
149                }
150            };
151            this.ingestProcessor = new IngestProcessor(client, maxConcurrentBulkRequests, maxActionsPerBulkRequest, maxVolumePerBulkRequest, maxWaitTime)
152                    .listener(listener);
153            this.closed = false;
154            return this;
155        }

264        @Override
265        public IngestClient refresh() {
266            if (client == null) {
267                logger.warn("no client");
268                return this;
269            }
270            if (getIndex() == null) {
271                logger.warn("no index name given");
272                return this;
273            }
274            client.admin().indices().refresh(new RefreshRequest());
275            return this;
276        }

280            return index(Requests.indexRequest(index).type(type).id(id).create(false).source(source));
281        }
282    
283        @Override
284        public IngestClient index(IndexRequest indexRequest) {
285            if (closed) {
286                throw new ElasticsearchIllegalStateException("client is closed");
287            }
288            try {
289                currentIngest.inc();
290                ingestProcessor.add(indexRequest);
291            } catch (Exception e) {
292                throwable = e;
293                closed = true;
294                logger.error("bulk add of index failed: " + e.getMessage(), e);
295            } finally {
296                currentIngest.dec();
297            }
298            return this;
299        }
300    
301        @Override
302        public IngestClient delete(String index, String type, String id) {
303            return delete(Requests.deleteRequest(index).type(type).id(id));
304        }
305    
306        @Override
307        public IngestClient delete(DeleteRequest deleteRequest) {
308            if (closed) {
309                throw new ElasticsearchIllegalStateException("client is closed");
310            }
311            try {
312                currentIngest.inc();
313                ingestProcessor.add(deleteRequest);
314            } catch (Exception e) {
315                throwable = e;
316                closed = true;
317                logger.error("bulk add of delete failed: " + e.getMessage(), e);
318            } finally {
319                currentIngest.dec();
320            }
321            return this;
322        }

324        public IngestClient numberOfShards(int value) {
325            if (closed) {
326                throw new ElasticsearchIllegalStateException("client is closed, possible reason: ", throwable);
327            }
328            if (getIndex() == null) {
329                logger.warn("no index name given");
330                return this;
331            }
332            setting("index.number_of_shards", value);
333            return this;
334        }

337            if (getIndex() == null) {
338                logger.warn("no index name given");
339                return this;
340            }
341            setting("index.number_of_replicas", value);
342            return this;
343        }
344    
345        public IngestClient flush() {
346            if (closed) {
347                throw new ElasticsearchIllegalStateException("client is closed, possible reason: ", throwable);
348            }
349            if (client == null) {
350                logger.warn("no client");
351                return this;
352            }
353            ingestProcessor.flush();
354            return this;
355        }
356    
357        @Override
358        public synchronized void shutdown() {
359            if (closed) {
360                super.shutdown();
361                throw new ElasticsearchIllegalStateException("client is closed, possible reason: ", throwable);
362            }
363            if (client == null) {
364                logger.warn("no client");
365                return;
366            }
367            try {
368                if (ingestProcessor != null) {
369                    logger.info("closing ingest processor...");
370                    ingestProcessor.close();
371                }
372                enableRefreshInterval();
373                logger.info("shutting down...");
374                super.shutdown();
375                logger.info("shutting down completed");
376            } catch (Exception e) {
377                logger.error(e.getMessage(), e);
378            }
379        }
