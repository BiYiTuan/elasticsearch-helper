082    @Override
083    public IngestTransportClient newClient(Settings settings) throws IOException {
084        super.newClient(settings);
085        resetSettings();
086        IngestProcessor.IngestListener ingestListener = new IngestProcessor.IngestListener() {
087            @Override
088            public void onRequest(int concurrency, IngestRequest request) {
089                metric.getCurrentIngest().inc();
090                int num = request.numberOfActions();
091                metric.getSubmitted().inc(num);
092                metric.getCurrentIngestNumDocs().inc(num);
093                metric.getTotalIngestSizeInBytes().inc(request.estimatedSizeInBytes());
094                logger.debug("before ingest [{}] [actions={}] [bytes={}] [concurrent requests={}]",
095                            request.ingestId(),
096                            num,
097                            request.estimatedSizeInBytes(),
098                            concurrency);
099            }
100
101            @Override
102            public void onResponse(int concurrency, IngestResponse response) {
103                metric.getCurrentIngest().dec();
104                metric.getSucceeded().inc(response.successSize());
105                metric.getFailed().inc(response.getFailures().size());
106                metric.getTotalIngest().inc(response.tookInMillis());
107                logger.debug("after ingest [{}] [succeeded={}] [failed={}] [{}ms] [leader={}] [replica={}] [concurrent requests={}]",
108                            response.ingestId(),
109                            metric.getSucceeded().count(),
110                            metric.getFailed().count(),
111                            response.tookInMillis(),
112                            response.leaderShardResponse(),
113                            response.replicaShardResponses(),
114                            concurrency
115                    );
116                if (!response.getFailures().isEmpty()) {
117                    for (IngestActionFailure f : response.getFailures()) {
118                        logger.error("ingest [{}] has failures, reason: {}", response.ingestId(), f.message());
119                    }
120                    closed = true;
121                } else {
122                    metric.getCurrentIngestNumDocs().dec(response.successSize());
123                }
124            }
125
126            @Override
127            public void onFailure(int concurrency, long executionId, Throwable failure) {
128                metric.getCurrentIngest().dec();
129                logger.error("ingest [" + executionId + "] failure", failure);
130                throwable = failure;
131                closed = true;
132            }
133        };
134        this.ingestProcessor = new IngestProcessor(client)
135                .maxConcurrentRequests(maxConcurrentRequests)
136                .maxActions(maxActionsPerRequest)
137                .maxVolumePerRequest(maxVolumePerRequest)
138                .flushInterval(flushInterval)
139                .listener(ingestListener);
140        this.closed = false;
141        return this;
142    }

220    @Override
221    public IngestTransportClient index(String index, String type, String id, String source) {
222        if (closed) {
223            if (throwable != null) {
224                throw new ElasticsearchException("client is closed, possible reason: ", throwable);
225            } else {
226                throw new ElasticsearchException("client is closed");
227            }
228        }
229        try {
230            metric.getCurrentIngest().inc();
231            ingestProcessor.add(new IndexRequest(index).type(type).id(id).create(false).source(source));
232        } catch (Exception e) {
233            logger.error("add of index request failed: " + e.getMessage(), e);
234            throwable = e;
235            closed = true;
236        } finally {
237            metric.getCurrentIngest().dec();
238        }
239        return this;
240    }

362    @Override
363    public synchronized void shutdown() {
364        if (closed) {
365            super.shutdown();
366            if (throwable != null) {
367                throw new ElasticsearchException("client was closed, possible reason: ", throwable);
368            }
369            throw new ElasticsearchException("client was closed");
370        }
371        closed = true;
372        if (client == null) {
373            logger.warn("no client");
374            return;
375        }
376        try {
377            if (ingestProcessor != null) {
378                logger.debug("closing ingest");
379                ingestProcessor.close();
380            }
381            if (metric != null && metric.indices() != null && !metric.indices().isEmpty()) {
382                logger.debug("stopping ingest mode for indices {}...", metric.indices());
383                for (String index : ImmutableSet.copyOf(metric.indices())) {
384                    stopBulk(index);
385                }
386            }
387            logger.debug("shutting down...");
388            super.shutdown();
389            logger.debug("shutting down completed");
390        } catch (Exception e) {
391            logger.error(e.getMessage(), e);
392        }
393    }
