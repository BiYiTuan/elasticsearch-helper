118    @Override
119    public IngestTransportClient newClient(URI uri, Settings settings) {
120        super.newClient(uri, settings);
121        this.state = new State();
122        resetSettings();
123        IngestProcessor.Listener listener = new IngestProcessor.Listener() {
124            @Override
125            public void beforeBulk(long executionId, int concurrency, IngestRequest request) {
126                int n = request.numberOfActions();
127                if (state != null) {
128                    state.getSubmitted().inc(n);
129                    state.getCurrentIngestNumDocs().inc(n);
130                    state.getTotalIngestSizeInBytes().inc(request.estimatedSizeInBytes());
131                }
132                if (logger.isDebugEnabled()) {
133                    logger.debug("before bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
134                            executionId, n, request.estimatedSizeInBytes(), concurrency);
135                }
136            }
137
138            @Override
139            public void afterBulk(long executionId, int concurrency, IngestResponse response) {
140                if (state != null) {
141                    state.getSucceeded().inc(response.successSize());
142                    state.getFailed().inc(response.failureSize());
143                    state.getTotalIngest().inc(response.tookInMillis());
144                }
145                if (logger.isDebugEnabled()) {
146                    logger.debug("after bulk [{}] [succeeded={}] [failed={}] [{}ms]",
147                            executionId,
148                            state.getSucceeded().count(),
149                            state.getFailed().count(),
150                            response.tookInMillis());
151                }
152                if (response.hasFailures()) {
153                    closed = true;
154                    for (IngestItemFailure f: response.failure()) {
155                        logger.error("after bulk [{}] [{}] failure, reason: {}", executionId, f.pos(), f.message());
156                    }
157                } else {
158                    state.getCurrentIngestNumDocs().dec(response.successSize());
159                }
160            }
161
162            @Override
163            public void afterBulk(long executionId, int concurrency, Throwable failure) {
164                closed = true;
165                logger.error("after bulk ["+executionId+"] failure", failure);
166                throwable = failure;
167            }
168        };
169        this.ingestProcessor = new IngestProcessor(client, maxConcurrentBulkRequests, maxActionsPerBulkRequest, maxVolumePerBulkRequest, maxWaitTime)
170                .listener(listener);
171        this.closed = false;
172        return this;
173    }

230    @Override
231    public IngestTransportClient startBulk(String index) throws IOException {
232        if (state == null) {
233            return this;
234        }
235        if (!state.isBulk()) {
236            state.setBulk(true);
237            ClientHelper.startBulk(client, index);
238        }
239        indices.add(index);
240        return this;
241    }
242
243    @Override
244    public IngestTransportClient stopBulk(String index) throws IOException {
245        if (state == null) {
246            return this;
247        }
248        if (state.isBulk()) {
249            state.setBulk(false);
250            ClientHelper.stopBulk(client, index);
251        }
252        indices.remove(index);
253        return this;
254    }

256    @Override
257    public IngestTransportClient refresh(String index) {
258        if (client == null) {
259            logger.warn("no client");
260            return this;
261        }
262        if (index == null) {
263            logger.warn("no index name given");
264            return this;
265        }
266        client.admin().indices().refresh(new RefreshRequest());
267        return this;
268    }

272        return index(Requests.indexRequest(index).type(type).id(id).create(false).source(source));
273    }
274
275    @Override
276    public IngestTransportClient index(IndexRequest indexRequest) {
277        if (closed) {
278            throw new ElasticsearchIllegalStateException("client is closed");
279        }
280        try {
281            state.getCurrentIngest().inc();
282            ingestProcessor.add(indexRequest);
283        } catch (Exception e) {
284            throwable = e;
285            closed = true;
286            logger.error("bulk add of index request failed: " + e.getMessage(), e);
287        } finally {
288            state.getCurrentIngest().dec();
289        }
290        return this;
291    }
292
293    @Override
294    public IngestTransportClient delete(String index, String type, String id) {
295        return delete(Requests.deleteRequest(index).type(type).id(id));
296    }
297
298    @Override
299    public IngestTransportClient delete(DeleteRequest deleteRequest) {
300        if (closed) {
301            throw new ElasticsearchIllegalStateException("client is closed");
302        }
303        try {
304            state.getCurrentIngest().inc();
305            ingestProcessor.add(deleteRequest);
306        } catch (Exception e) {
307            throwable = e;
308            closed = true;
309            logger.error("bulk add of delete request failed: " + e.getMessage(), e);
310        } finally {
311            state.getCurrentIngest().dec();
312        }
313        return this;
314    }
315
316    @Override
317    public IngestTransportClient flush() {
318        if (closed) {
319            throw new ElasticsearchIllegalStateException("client is closed, possible reason: ", throwable);
320        }
321        if (client == null) {
322            logger.warn("no client");
323            return this;
324        }
325        ingestProcessor.flush();
326        return this;
327    }

345    @Override
346    public synchronized void shutdown() {
347        if (closed) {
348            super.shutdown();
349            throw new ElasticsearchIllegalStateException("client was closed, possible reason: ", throwable);
350        }
351        if (client == null) {
352            logger.warn("no client");
353            return;
354        }
355        try {
356            if (ingestProcessor != null) {
357                logger.info("closing ingest processor...");
358                ingestProcessor.close();
359            }
360            logger.info("stopping bulk mode for indices {}...", indices);
361            for (String index : indices) {
362                stopBulk(index);
363            }
364            logger.info("shutting down...");
365            super.shutdown();
366            logger.info("shutting down completed");
367        } catch (Exception e) {
368            logger.error(e.getMessage(), e);
369        }
370    }

373    public boolean hasThrowable() {
374        return throwable != null;
