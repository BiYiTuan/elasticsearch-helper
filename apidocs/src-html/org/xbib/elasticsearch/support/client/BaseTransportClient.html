054    public BaseTransportClient newClient(URI uri, Settings settings) {
055        if (client != null) {
056            logger.warn("client is open, closing...");
057            client.close();
058            client.threadPool().shutdown();
059            logger.warn("client is closed");
060            client = null;
061        }
062        if (settings != null) {
063            logger.info("creating transport client, java version {}, effective settings {}",
064                    System.getProperty("java.version"), settings.getAsMap());
065            // false = do not load config settings from environment
066            this.client = new TransportClient(settings, false);
067        } else {
068            logger.info("creating transport client, java version {}, using default settings",
069                    System.getProperty("java.version"));
070            this.client = new TransportClient();
071        }
072        try {
073            connect(uri, settings);
074        } catch (UnknownHostException e) {
075            logger.error(e.getMessage(), e);
076        } catch (SocketException e) {
077            logger.error(e.getMessage(), e);
078        } catch (IOException e) {
079            logger.error(e.getMessage(), e);
080        }
081        return this;
082    }

086                .put("cluster.name", findClusterName(uri))
087                .put("network.server", false)
088                .put("node.client", true)

090                .put("client.transport.ignore_cluster_name", false)
091                .put("client.transport.ping_timeout", "30s")
092                .put("client.transport.nodes_sampler_interval", "30s")
093                .put("action.ingest.timeout", maxWaitTime)
094                .build();
095    }

109    public synchronized void shutdown() {
110        if (client != null) {
111            logger.debug("shutdown started");
112            client.close();
113            client.threadPool().shutdown();
114            client = null;
115            logger.debug("shutdown complete");
116        }
117        addresses.clear();
118    }

120    protected URI findURI() {
121        URI uri = null;
122        String hostname = "localhost";
123        try {
124            hostname = InetAddress.getLocalHost().getHostName();
125            logger.debug("the hostname is {}", hostname);
126            uri = URI.create("es://" + hostname + ":9300");
127            // custom?
128            URL url = getClass().getResource("/org/xbib/elasticsearch/cluster.properties");
129            if (url != null) {
130                InputStream in = url.openStream();
131                Properties p = new Properties();
132                p.load(in);
133                in.close();
134                // the properties contains default URIs per hostname
135                if (p.containsKey(hostname)) {
136                    uri = URI.create(p.getProperty(hostname));
137                    logger.debug("custom URI found in cluster.properties for hostname {}: {}", hostname, uri);
138                    return uri;
139                }
140            }
141        } catch (UnknownHostException e) {
142            logger.warn("can't resolve host name, probably something wrong with network config: " + e.getMessage(), e);
143        } catch (Exception e) {
144            logger.warn(e.getMessage(), e);
145        }
146        logger.debug("URI for hostname {}: {}", hostname, uri);
147        return uri;
148    }

150    protected String findClusterName(URI uri) {
151        String clustername;
152        try {
153            Map<String, String> params = parseQueryString(uri, "UTF-8");
154            clustername = params.get("es.cluster.name");
155            if (clustername != null) {
156                logger.info("cluster name found in URI {}: {}", uri, clustername);
157                return clustername;
158            }
159            clustername = params.get("cluster.name");
160            if (clustername != null) {
161                logger.info("cluster name found in URI {}: {}", uri, clustername);
162                return clustername;
163            }
164        } catch (UnsupportedEncodingException ex) {
165            logger.warn(ex.getMessage(), ex);
166        }
167        logger.info("cluster name not found in URI {}, parameter es.cluster.name", uri);
168        clustername = System.getProperty("es.cluster.name");
169        if (clustername != null) {
170            logger.info("cluster name found in es.cluster.name system property: {}", clustername);
171            return clustername;
172        }
173        clustername = System.getProperty("cluster.name");
174        if (clustername != null) {
175            logger.info("cluster name found in cluster.name system property: {}", clustername);
176            return clustername;
177        }
178        logger.info("cluster name not found, falling back to default: {}", DEFAULT_CLUSTER_NAME);
179        clustername = DEFAULT_CLUSTER_NAME;
180        return clustername;
181    }

183    protected void connect(URI uri, Settings settings) throws IOException {
184        String hostname = uri.getHost();
185        int port = uri.getPort();
186        boolean newaddresses = false;
187        if (!"es".equals(uri.getScheme())) {
188            logger.warn("please specify URI scheme 'es'");
189        }
190        if ("hostname".equals(hostname)) {
191            InetSocketTransportAddress address = new InetSocketTransportAddress(InetAddress.getLocalHost().getHostName(), port);
192            if (!addresses.contains(address)) {
193                logger.info("adding hostname address for transport client: {}", address);
194                client.addTransportAddress(address);
195                addresses.add(address);
196                newaddresses = true;
197            }
198        } else if ("interfaces".equals(hostname)) {
199            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
200            for (NetworkInterface netint : Collections.list(nets)) {
201                logger.info("checking network interface = {}", netint.getName());
202                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
203                for (InetAddress addr : Collections.list(inetAddresses)) {
204                    logger.info("checking address = {}", addr.getHostAddress());
205                    InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
206                    if (!addresses.contains(address)) {
207                        logger.info("adding address to transport client: {}", address);
208                        client.addTransportAddress(address);
209                        addresses.add(address);
210                        newaddresses = true;
211                    }
212                }
213            }
214        } else if ("inet4".equals(hostname)) {
215            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
216            for (NetworkInterface netint : Collections.list(nets)) {
217                logger.info("checking network interface = {}", netint.getName());
218                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
219                for (InetAddress addr : Collections.list(inetAddresses)) {
220                    if (addr instanceof Inet4Address) {
221                        logger.info("checking address = {}", addr.getHostAddress());
222                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
223                        if (!addresses.contains(address)) {
224                            logger.info("adding address for transport client: {}", address);
225                            client.addTransportAddress(address);
226                            addresses.add(address);
227                            newaddresses = true;
228                        }
229                    }
230                }
231            }
232        } else if ("inet6".equals(hostname)) {
233            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
234            for (NetworkInterface netint : Collections.list(nets)) {
235                logger.info("checking network interface = {}", netint.getName());
236                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
237                for (InetAddress addr : Collections.list(inetAddresses)) {
238                    if (addr instanceof Inet6Address) {
239                        logger.info("checking address = {}", addr.getHostAddress());
240                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
241                        if (!addresses.contains(address)) {
242                            logger.info("adding address for transport client: {}", address);
243                            client.addTransportAddress(address);
244                            addresses.add(address);
245                            newaddresses = true;
246                        }
247                    }
248                }
249            }
250        } else {
251            InetSocketTransportAddress address = new InetSocketTransportAddress(hostname, port);
252            if (!addresses.contains(address)) {
253                logger.info("adding custom address for transport client: {}", address);
254                client.addTransportAddress(address);
255                addresses.add(address);
256                newaddresses = true;
257            }
258        }
259        logger.info("configured addresses to connect: {}", addresses);
260        if (client.connectedNodes() != null) {
261            List<DiscoveryNode> nodes = client.connectedNodes().asList();
262            logger.info("connected nodes = {}", nodes);
263            if (newaddresses) {
264                for (DiscoveryNode node : nodes) {
265                    logger.info("new connection to {}", node);
266                }
267                if (!nodes.isEmpty()) {
268                    Map<String, String> params = parseQueryString(uri, "UTF-8");
269                    if (params.containsKey("es.sniff")) {
270                        try {
271                            connectMore();
272                        } catch (Exception e) {
273                            logger.error("error while connecting to more nodes", e);
274                        }
275                    }
276                }
277            }
278        }
279    }

329    private void connectMore() throws IOException {
330        logger.info("trying to discover more nodes...");
331        ClusterStateResponse clusterStateResponse = client.admin().cluster().state(new ClusterStateRequest()).actionGet();
332        DiscoveryNodes nodes = clusterStateResponse.getState().getNodes();
333        for (DiscoveryNode node : nodes) {
334            logger.info("adding discovered node {}", node);
335            try {
336                client.addTransportAddress(node.address());
337            } catch (Exception e) {
338                logger.warn("can't add node " + node, e);
339            }
340        }
341        logger.info("... discovery done");
342    }

344    private Map<String, String> parseQueryString(URI uri, String encoding)
345            throws UnsupportedEncodingException {
346        Map<String, String> m = new HashMap<String, String>();
347        if (uri == null) {
348            throw new IllegalArgumentException();
349        }
350        if (uri.getRawQuery() == null) {
351            return m;
352        }
353        // getRawQuery() because we do our decoding by ourselves
354        StringTokenizer st = new StringTokenizer(uri.getRawQuery(), "&");
355        while (st.hasMoreTokens()) {
356            String pair = st.nextToken();
357            int pos = pair.indexOf('=');
358            if (pos < 0) {
359                m.put(pair, null);
360            } else {
361                m.put(pair.substring(0, pos), decode(pair.substring(pos + 1, pair.length()), encoding));
362            }
363        }
364        return m;
365    }
366
367    private String decode(String s, String encoding) {
368        StringBuilder sb = new StringBuilder();
369        boolean fragment = false;
370        for (int i = 0; i < s.length(); i++) {
371            char ch = s.charAt(i);
372            switch (ch) {
373                case '+':
374                    sb.append(' ');
375                    break;
376                case '#':
377                    sb.append(ch);
378                    fragment = true;
379                    break;
380                case '%':
381                    if (!fragment) {
382                        // fast hex decode
383                        sb.append((char) ((Character.digit(s.charAt(++i), 16) << 4)
384                                | Character.digit(s.charAt(++i), 16)));
385                    } else {
386                        sb.append(ch);
387                    }
388                    break;
389                default:
390                    sb.append(ch);
391                    break;
392            }
393        }
394        try {
395            // URL default encoding is ISO-8859-1
396            return new String(sb.toString().getBytes("ISO-8859-1"), encoding);
397        } catch (UnsupportedEncodingException e) {
398            throw new Error("encoding " + encoding + " not supported");
399        }
400    }
