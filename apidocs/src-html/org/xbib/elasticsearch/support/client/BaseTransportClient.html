001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.action.admin.cluster.state.ClusterStateRequest;
004import org.elasticsearch.action.admin.cluster.state.ClusterStateResponse;
005import org.elasticsearch.client.ElasticsearchClient;
006import org.elasticsearch.client.support.AbstractClient;
007import org.elasticsearch.client.transport.NoNodeAvailableException;
008import org.elasticsearch.client.transport.TransportClient;
009import org.elasticsearch.cluster.node.DiscoveryNode;
010import org.elasticsearch.cluster.node.DiscoveryNodes;
011import org.elasticsearch.common.logging.ESLogger;
012import org.elasticsearch.common.logging.ESLoggerFactory;
013import org.elasticsearch.common.settings.Settings;
014import org.elasticsearch.common.transport.InetSocketTransportAddress;
015import org.xbib.elasticsearch.support.network.NetworkUtils;
016
017import java.io.IOException;
018import java.io.InputStream;
019import java.net.InetAddress;
020import java.net.UnknownHostException;
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.List;
024import java.util.Map;
025
026public abstract class BaseTransportClient {
027
028    private final static ESLogger logger = ESLoggerFactory.getLogger(BaseTransportClient.class.getSimpleName());
029
030    protected TransportClient client;
031
032    protected ConfigHelper configHelper = new ConfigHelper();
033
034    private boolean isShutdown;
035
036    protected void createClient(Settings settings) throws IOException {
037        if (client != null) {
038            logger.warn("client is open, closing...");
039            client.close();
040            client.threadPool().shutdown();
041            logger.warn("client is closed");
042            client = null;
043        }
044        if (settings != null) {
045            logger.info("creating transport client, java version {}, effective settings {}",
046                    System.getProperty("java.version"), settings.getAsMap());
047            this.client = TransportClient.builder().settings(settings).build();
048            Collection<InetSocketTransportAddress> addrs = findAddresses(settings);
049            if (!connect(addrs, settings.getAsBoolean("autodiscover", false))) {
050                throw new NoNodeAvailableException("no cluster nodes available, check settings "
051                        + settings.getAsMap());
052            }
053        }
054    }
055
056    public AbstractClient client() {
057        return client;
058    }
059
060    public synchronized void shutdown() {
061        if (client != null) {
062            logger.debug("shutdown started");
063            client.close();
064            client.threadPool().shutdown();
065            client = null;
066            logger.debug("shutdown complete");
067        }
068        isShutdown = true;
069    }
070
071    public boolean isShutdown() {
072        return isShutdown;
073    }
074
075    protected Settings findSettings() {
076        Settings.Builder settingsBuilder = Settings.settingsBuilder();
077        settingsBuilder.put("host", "localhost");
078        try {
079            String hostname = InetAddress.getLocalHost().getHostName();
080            logger.debug("the hostname is {}", hostname);
081            settingsBuilder.put("host", hostname)
082                    .put("port", 9300);
083        } catch (UnknownHostException e) {
084            logger.warn("can't resolve host name, probably something wrong with network config: " + e.getMessage(), e);
085        } catch (Exception e) {
086            logger.warn(e.getMessage(), e);
087        }
088        return settingsBuilder.build();
089    }
090
091    protected Collection<InetSocketTransportAddress> findAddresses(Settings settings) throws IOException {
092        String[] hostnames = settings.getAsArray("host", new String[]{"localhost"});
093        int port = settings.getAsInt("port", 9300);
094        Collection<InetSocketTransportAddress> addresses = new ArrayList<>();
095        for (String hostname : hostnames) {
096            String[] splitHost = hostname.split(":", 2);
097            if (splitHost.length == 2) {
098                String host = splitHost[0];
099                InetAddress inetAddress = NetworkUtils.resolveInetAddress(host, null);
100                try {
101                    port = Integer.parseInt(splitHost[1]);
102                } catch (Exception e) {
103                    // ignore
104                }
105                addresses.add(new InetSocketTransportAddress(inetAddress, port));
106            }
107            if (splitHost.length == 1) {
108                String host = splitHost[0];
109                InetAddress inetAddress = NetworkUtils.resolveInetAddress(host, null);
110                addresses.add(new InetSocketTransportAddress(inetAddress, port));
111            }
112        }
113        return addresses;
114    }
115
116    protected boolean connect(Collection<InetSocketTransportAddress> addresses, boolean autodiscover) {
117        logger.info("trying to connect to {}", addresses);
118        for (InetSocketTransportAddress address : addresses) {
119            client.addTransportAddress(address);
120        }
121        if (client.connectedNodes() != null) {
122            List<DiscoveryNode> nodes = client.connectedNodes();
123            if (!nodes.isEmpty()) {
124                logger.info("connected to {}", nodes);
125                if (autodiscover) {
126                    logger.info("trying to auto-discover all cluster nodes...");
127                    ClusterStateResponse clusterStateResponse = client.admin().cluster().state(new ClusterStateRequest()).actionGet();
128                    DiscoveryNodes discoveryNodes = clusterStateResponse.getState().getNodes();
129                    for (DiscoveryNode node : discoveryNodes) {
130                        logger.info("connecting to auto-discovered node {}", node);
131                        try {
132                            client.addTransportAddress(node.address());
133                        } catch (Exception e) {
134                            logger.warn("can't connect to node " + node, e);
135                        }
136                    }
137                    logger.info("after auto-discovery connected to {}", client.connectedNodes());
138                }
139                return true;
140            }
141            return false;
142        }
143        return false;
144    }
145
146    public Settings.Builder getSettingsBuilder() {
147        return configHelper.settingsBuilder();
148    }
149
150    public void resetSettings() {
151        configHelper.reset();
152    }
153
154    public void setting(InputStream in) throws IOException {
155        configHelper.setting(in);
156    }
157
158    public void addSetting(String key, String value) {
159        configHelper.setting(key, value);
160    }
161
162    public void addSetting(String key, Boolean value) {
163        configHelper.setting(key, value);
164    }
165
166    public void addSetting(String key, Integer value) {
167        configHelper.setting(key, value);
168    }
169
170    public void mapping(String type, String mapping) throws IOException {
171        configHelper.mapping(type, mapping);
172    }
173
174    public void mapping(String type, InputStream in) throws IOException {
175        configHelper.mapping(type, in);
176    }
177
178    public Map<String, String> getMappings() {
179        return configHelper.mappings();
180    }
181
182}
