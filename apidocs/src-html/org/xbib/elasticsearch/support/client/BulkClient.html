094    /**
095     * Create new client with concurrent ingestProcessor processor.
096     * <p/>
097     * The URI describes host and port of the node the client should connect to,
098     * with the parameter <tt>es.cluster.name</tt> for the cluster name.
099     *
100     * @param uri the cluster URI
101     * @return this indexer
102     */
103    @Override
104    public BulkClient newClient(URI uri) {
105        super.newClient(uri);
106        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
107            @Override
108            public void beforeBulk(long executionId, BulkRequest request) {
109                long l = outstandingRequests.getAndIncrement();
110                long v = volumeCounter.addAndGet(request.estimatedSizeInBytes());
111                logger.info("new bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
112                        executionId, request.numberOfActions(), v, l);
113            }
114
115            @Override
116            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
117                long l = outstandingRequests.decrementAndGet();
118
119                logger.info("bulk [{}] [{} items] [{}]  [{}ms]",
120                        executionId,
121                        response.getItems().length,
122                        response.hasFailures() ? "failure" : "ok",
123                        response.getTook().millis());
124                if (response.hasFailures()) {
125                    logger.error("bulk [{}] failure reason: {}",
126                            executionId, response.buildFailureMessage());
127                }
128            }
129
130            @Override
131            public void afterBulk(long executionId, BulkRequest requst, Throwable failure) {
132                long l = outstandingRequests.decrementAndGet();
133                logger.error("bulk ["+executionId+"] error", failure);
134                enabled = false;
135            }
136        };
137        this.bulkProcessor = BulkProcessor.builder(client, listener)
138                .setBulkActions(maxBulkActions-1)  // off-by-one
139                .setConcurrentRequests(maxConcurrentBulkRequests)
140                .setFlushInterval(TimeValue.timeValueSeconds(30))
141                .build();
142        this.enabled = true;
143        return this;
144    }

151    /**
152     * Initial settings tailored for index/ingestProcessor client use. Transport
153     * sniffing, only thread pool is for ingestProcessor/indexing, other thread pools are
154     * minimal, 4 * cpucore Netty connections in parallel.
155     *
156     * @param uri the cluster name URI
157     * @param n the client threadpool size
158     * @return the initial settings
159     */
160    @Override
161    protected Settings initialSettings(URI uri, int n) {
162        return ImmutableSettings.settingsBuilder()
163                .put("cluster.name", findClusterName(uri))
164                .put("network.server", false)
165                .put("node.client", true)
166                .put("client.transport.sniff", false)
167                .build();
168    }

294    @Override
295    public BulkClient refresh() {
296        if (client == null) {
297            logger.warn("no client");
298            return this;
299        }
300        if (getIndex() == null) {
301            logger.warn("no index name given");
302            return this;
303        }
304        client.admin().indices().refresh(new RefreshRequest());
305        return this;
306    }

308    @Override
309    public BulkClient createDocument(String index, String type, String id, String source) {
310        if (!enabled) {
311            return this;
312        }
313        if (logger.isTraceEnabled()) {
314            logger.trace("create: {}/{}/{} source = {}", index, type, id, source);
315        }
316        IndexRequest indexRequest = Requests.indexRequest(index).type(type).id(id).create(true).source(source);
317        try {
318            bulkProcessor.add(indexRequest);
319        } catch (Exception e) {
320            logger.error("bulk add of create failed: " + e.getMessage(), e);
321            enabled = false;
322        }
323        return this;
324    }

395    @Override
396    public BulkClient flush() {
397        if (!enabled) {
398            return this;
399        }
400        if (client == null) {
401            logger.warn("no client");
402            return this;
403        }
404        // we simply wait long enough, because BulkProcessor has a 30 sec flush set
405        try {
406            Thread.sleep(30 * 1000L);
407        } catch (InterruptedException e) {
408            logger.error("interrupted", e);
409        }
410        return this;
411    }
412
413    @Override
414    public synchronized void shutdown() {
415        if (!enabled) {
416            super.shutdown();
417            return;
418        }
419        if (client == null) {
420            logger.warn("no client");
421            return;
422        }
423        try {
424            if (bulkProcessor != null) {
425                logger.info("closing ingest processor...");
426                bulkProcessor.close();
427            }
428            logger.info("enabling refresh interval...");
429            enableRefreshInterval();
430            logger.info("shutting down...");
431            super.shutdown();
432            logger.info("shutting down completed");
433        } catch (Exception e) {
434            logger.error(e.getMessage(), e);
435        }
436    }
