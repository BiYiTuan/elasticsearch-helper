134    /**
135     * Create new client
136     * The URI describes host and port of the node the client should connect to,
137     * with the parameter <tt>es.cluster.name</tt> for the cluster name.
138     *
139     * @param uri the cluster URI
140     * @return this client
141     */
142    @Override
143    public BulkClient newClient(URI uri) {
144        super.newClient(uri);
145        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
146            @Override
147            public void beforeBulk(long executionId, BulkRequest request) {
148                long l = outstandingRequests.getAndIncrement();
149                long v = volumeCounter.addAndGet(request.estimatedSizeInBytes());
150                if (logger.isDebugEnabled()) {
151                    logger.debug("new bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
152                        executionId, request.numberOfActions(), v, l);
153                }
154            }
155
156            @Override
157            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
158                bulkCounter.incrementAndGet();
159                outstandingRequests.decrementAndGet();
160                if (logger.isDebugEnabled()) {
161                    logger.debug("bulk [{}] [{} items] [{}] [{}ms]",
162                        executionId,
163                        response.getItems().length,
164                        response.hasFailures() ? "failure" : "ok",
165                        response.getTook().millis());
166                }
167                if (response.hasFailures()) {
168                    logger.error("bulk [{}] failure reason: {}",
169                            executionId, response.buildFailureMessage());
170                    enabled = false;
171                }
172            }
173
174            @Override
175            public void afterBulk(long executionId, BulkRequest requst, Throwable failure) {
176                outstandingRequests.decrementAndGet();
177                throwable = failure;
178                logger.error("bulk ["+executionId+"] error", failure);
179                enabled = false;
180            }
181        };
182        BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
183                .setBulkActions(maxBulkActions-1)  // off-by-one
184                .setConcurrentRequests(maxConcurrentBulkRequests)
185                .setFlushInterval(flushInterval);
186        if (maxVolume != null) {
187            builder.setBulkSize(maxVolume);
188        }
189        this.bulkProcessor =  builder.build();
190        this.enabled = true;
191        return this;
192    }

316    @Override
317    public BulkClient refresh() {
318        if (client == null) {
319            logger.warn("no client");
320            return this;
321        }
322        if (getIndex() == null) {
323            logger.warn("no index name given");
324            return this;
325        }
326        client.admin().indices().refresh(new RefreshRequest());
327        return this;
328    }

330    @Override
331    public BulkClient createDocument(String index, String type, String id, String source) {
332        if (!enabled) {
333            return this;
334        }
335        if (logger.isTraceEnabled()) {
336            logger.trace("create: {}/{}/{} source = {}", index, type, id, source);
337        }
338        IndexRequest indexRequest = Requests.indexRequest(index).type(type).id(id).create(true).source(source);
339        try {
340            bulkProcessor.add(indexRequest);
341        } catch (Exception e) {
342            this.throwable = e;
343            logger.error("bulk add of create failed: " + e.getMessage(), e);
344            enabled = false;
345        }
346        return this;
347    }

420    @Override
421    public BulkClient flush() {
422        if (!enabled) {
423            return this;
424        }
425        if (client == null) {
426            logger.warn("no client");
427            return this;
428        }
429        // we simply wait long enough for BulkProcessor flush plus one second
430        try {
431            logger.info("flushing bulk processor (forced wait of {} seconds...)", flushInterval.seconds() + 1);
432            Thread.sleep(flushInterval.getMillis() + 1000L);
433        } catch (InterruptedException e) {
434            logger.error("interrupted", e);
435        }
436        return this;
437    }
438
439    @Override
440    public synchronized void shutdown() {
441        if (!enabled) {
442            super.shutdown();
443            return;
444        }
445        if (client == null) {
446            logger.warn("no client");
447            return;
448        }
449        try {
450            if (bulkProcessor != null) {
451                flush();
452                bulkProcessor.close();
453            }
454            logger.info("enabling refresh interval...");
455            enableRefreshInterval();
456            logger.info("shutting down...");
457            super.shutdown();
458            logger.info("shutting down completed");
459        } catch (Exception e) {
460            logger.error(e.getMessage(), e);
461        }
462    }
