098    /**
099     * Create new client with concurrent ingestProcessor processor.
100     * <p/>
101     * The URI describes host and port of the node the client should connect to,
102     * with the parameter <tt>es.cluster.name</tt> for the cluster name.
103     *
104     * @param uri the cluster URI
105     * @return this indexer
106     */
107    @Override
108    public BulkClient newClient(URI uri) {
109        super.newClient(uri);
110        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
111            @Override
112            public void beforeBulk(long executionId, BulkRequest request) {
113                long l = outstandingRequests.getAndIncrement();
114                long v = volumeCounter.addAndGet(request.estimatedSizeInBytes());
115                logger.info("new bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
116                        executionId, request.numberOfActions(), v, l);
117            }
118
119            @Override
120            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
121                long l = outstandingRequests.decrementAndGet();
122
123                logger.info("bulk [{}] [{} items] [{}]  [{}ms]",
124                        executionId,
125                        response.getItems().length,
126                        response.hasFailures() ? "failure" : "ok",
127                        response.getTook().millis());
128                if (response.hasFailures()) {
129                    logger.error("bulk [{}] failure reason: {}",
130                            executionId, response.buildFailureMessage());
131                }
132            }
133
134            @Override
135            public void afterBulk(long executionId, BulkRequest requst, Throwable failure) {
136                long l = outstandingRequests.decrementAndGet();
137                logger.error("bulk ["+executionId+"] error", failure);
138                enabled = false;
139            }
140        };
141        this.bulkProcessor = BulkProcessor.builder(client, listener)
142                .setBulkActions(maxBulkActions-1)  // off-by-one
143                .setConcurrentRequests(maxConcurrentBulkRequests)
144                .setFlushInterval(TimeValue.timeValueSeconds(30))
145                .build();
146        this.enabled = true;
147        return this;
148    }

155    /**
156     * Initial settings tailored for index/ingestProcessor client use. Transport
157     * sniffing, only thread pool is for ingestProcessor/indexing, other thread pools are
158     * minimal, 4 * cpucore Netty connections in parallel.
159     *
160     * @param uri the cluster name URI
161     * @param n the client threadpool size
162     * @return the initial settings
163     */
164    @Override
165    protected Settings initialSettings(URI uri, int n) {
166        return ImmutableSettings.settingsBuilder()
167                .put("cluster.name", findClusterName(uri))
168                .put("network.server", false)
169                .put("node.client", true)
170                .put("client.transport.sniff", false)
171                .build();
172    }

238    @Override
239    public synchronized BulkClient newIndex(boolean ignoreException) {
240        if (!enabled) {
241            return this;
242        }
243        if (client == null) {
244            logger.warn("no client");
245            return this;
246        }
247        if (getIndex() == null) {
248            logger.warn("no index name given to create");
249            return this;
250        }
251        if (getType() == null) {
252            logger.warn("no type name given to create");
253            return this;
254        }
255        CreateIndexRequest request = new CreateIndexRequest(getIndex());
256        if (settings() != null) {
257            request.settings(settings());
258        }
259        if (mapping() == null) {
260            mapping(defaultMapping());
261        }
262        request.mapping(getType(), mapping());
263        logger.info("creating index = {} type = {} settings = {} mapping = {}",
264                getIndex(),
265                getType(),
266                settings() != null ? settings().build().getAsMap() : "",
267                mapping());
268        try {
269            client.admin().indices().create(request).actionGet();
270        } catch (IndexAlreadyExistsException e) {
271            if (!ignoreException) {
272                throw new RuntimeException(e);
273            }
274        }
275        return this;
276    }

282    public BulkClient deleteIndex(boolean ignoreException) {
283        if (!enabled) {
284            return this;
285        }
286        if (client == null) {
287            logger.warn("no client");
288            return this;
289        }
290        if (getIndex() == null) {
291            logger.warn("no index name given to create");
292            return this;
293        }
294        try {
295            client.admin().indices().delete(new DeleteIndexRequest(getIndex()));
296        } catch (Exception e) {
297            if (!ignoreException) {
298                throw new RuntimeException(e);
299            }
300        }
301        return this;
302    }

305    public BulkClient newType() {
306        if (!enabled) {
307            return this;
308        }
309        if (client == null) {
310            logger.warn("no client");
311            return this;
312        }
313        if (getIndex() == null) {
314            logger.warn("no index name given");
315            return this;
316        }
317        if (mapping() == null) {
318            mapping(defaultMapping());
319        }
320        client.admin().indices().putMapping(new PutMappingRequest()
321                .indices(new String[]{getIndex()})
322                .type(getType())
323                .source(mapping()))
324                .actionGet();
325        return this;
326    }

386    @Override
387    public BulkClient createDocument(String index, String type, String id, String source) {
388        if (!enabled) {
389            return this;
390        }
391        if (logger.isTraceEnabled()) {
392            logger.trace("create: {}/{}/{} source = {}", index, type, id, source);
393        }
394        IndexRequest indexRequest = Requests.indexRequest(index).type(type).id(id).create(true).source(source);
395        try {
396            bulkProcessor.add(indexRequest);
397        } catch (Exception e) {
398            logger.error("bulk add of create failed: " + e.getMessage(), e);
399            enabled = false;
400        }
401        return this;
402    }

473    @Override
474    public BulkClient flush() {
475        if (!enabled) {
476            return this;
477        }
478        if (client == null) {
479            logger.warn("no client");
480            return this;
481        }
482        // we simply wait long enough, because BulkProcessor has a 30 sec flush set
483        try {
484            Thread.sleep(30 * 1000L);
485        } catch (InterruptedException e) {
486            logger.error("interrupted", e);
487        }
488        return this;
489    }
490
491    @Override
492    public synchronized void shutdown() {
493        if (!enabled) {
494            super.shutdown();
495            return;
496        }
497        if (client == null) {
498            logger.warn("no client");
499            return;
500        }
501        try {
502            if (bulkProcessor != null) {
503                logger.info("closing ingest processor...");
504                bulkProcessor.close();
505            }
506            logger.info("enabling refresh interval...");
507            enableRefreshInterval();
508            logger.info("shutting down...");
509            super.shutdown();
510            logger.info("shutting down completed");
511        } catch (Exception e) {
512            logger.error(e.getMessage(), e);
513        }
514    }
