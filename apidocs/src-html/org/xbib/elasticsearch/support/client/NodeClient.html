002package org.xbib.elasticsearch.support.client;
003
004import org.elasticsearch.ElasticSearchTimeoutException;
005import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
006import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
007import org.elasticsearch.action.bulk.BulkProcessor;
008import org.elasticsearch.action.bulk.BulkRequest;
009import org.elasticsearch.action.bulk.BulkResponse;
010import org.elasticsearch.action.delete.DeleteRequest;
011import org.elasticsearch.action.index.IndexRequest;
012import org.elasticsearch.client.Client;
013import org.elasticsearch.client.Requests;
014import org.elasticsearch.common.logging.ESLogger;
015import org.elasticsearch.common.logging.Loggers;
016import org.elasticsearch.common.unit.TimeValue;
017
018import java.io.IOException;
019import java.util.concurrent.atomic.AtomicLong;
020
021/**
022 * Node client support. Implements minimal API for node client ingesting.
023 * Useful for river implementations.
024 *
025 */
026public class NodeClient implements DocumentIngest {
027
028    private final static ESLogger logger = Loggers.getLogger(NodeClient.class);
029
030    private Client client;
031
032    private String index;
033
034    private String type;
035
036    /**
037     * The default size of a request
038     */
039    private int maxBulkActions = 100;
040    /**
041     * The default number of maximum concurrent ingestProcessor requests
042     */
043    private int maxConcurrentBulkRequests = 30;
044    /**
045     * The outstanding requests
046     */
047    private final AtomicLong outstandingBulkRequests = new AtomicLong();
048    /**
049     * Count the volume
050     */
051    private final AtomicLong volumeCounter = new AtomicLong();
052    /**
053     * Enabled of not
054     */
055    private boolean enabled = true;
056    /**
057     * The bulk processor
058     */
059    private final BulkProcessor bulkProcessor;
060
061    public NodeClient(Client client, String index, String type,
062                      int maxBulkActions, int maxConcurrentBulkRequests) {
063        this.client = client;
064        this.index = index;
065        this.type = type;
066        this.maxBulkActions = maxBulkActions;
067        this.maxConcurrentBulkRequests = maxConcurrentBulkRequests;
068        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
069            @Override
070            public void beforeBulk(long executionId, BulkRequest request) {
071                long l = outstandingBulkRequests.getAndIncrement();
072                long v = volumeCounter.addAndGet(request.estimatedSizeInBytes());
073                logger.info("new bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
074                        executionId, request.numberOfActions(), v, l);
075            }
076
077            @Override
078            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
079                long l = outstandingBulkRequests.decrementAndGet();
080                logger.info("bulk [{}] success [{} items] [{}ms]",
081                        executionId, response.getItems().length, response.getTook().millis());
082            }
083
084            @Override
085            public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
086                long l = outstandingBulkRequests.decrementAndGet();
087                logger.error("bulk ["+executionId+"] error", failure);
088                enabled = false;
089            }
090        };
091        this.bulkProcessor = BulkProcessor.builder(client, listener)
092                .setBulkActions(maxBulkActions-1) // off-by-one
093                .setConcurrentRequests(maxConcurrentBulkRequests)
094                .setFlushInterval(TimeValue.timeValueSeconds(5))
095                .build();
096        try {
097            waitForHealthyCluster();
098            this.enabled = true;
099        } catch (IOException e) {
100            logger.warn(e.getMessage(), e);
101            this.enabled = false;
102        }
103    }
104
105    @Override
106    public Client client() {
107        return client;
108    }
109
110    @Override
111    public NodeClient setIndex(String index) {
112        this.index = index;
113        return this;
114    }
115
116    @Override
117    public NodeClient setType(String type) {
118        this.type = type;
119        return this;
120    }
121
122    public String getIndex() {
123        return index;
124    }
125
126    public String getType() {
127        return type;
128    }
129
130    @Override
131    public NodeClient createDocument(String index, String type, String id, String source) {
132        if (!enabled) {
133            return this;
134        }
135        IndexRequest indexRequest = Requests.indexRequest(index != null ? index : getIndex())
136                .type(type != null ? type : getType()).id(id).create(true).source(source);
137        try {
138            bulkProcessor.add(indexRequest);
139        } catch (Exception e) {
140            logger.error("bulk add of create failed: " + e.getMessage(), e);
141            enabled = false;
142        }
143        return this;
144    }
145
146    @Override
147    public NodeClient indexDocument(String index, String type, String id, String source) {
148        if (!enabled) {
149            return this;
150        }
151        IndexRequest indexRequest = Requests.indexRequest(index != null ? index : getIndex())
152                .type(type != null ? type : getType()).id(id).create(false).source(source);
153        try {
154            bulkProcessor.add(indexRequest);
155        } catch (Exception e) {
156            logger.error("bulk add of index failed: " + e.getMessage(), e);
157            enabled = false;
158        }
159        return this;
160    }
161
162    @Override
163    public NodeClient deleteDocument(String index, String type, String id) {
164        if (!enabled) {
165            return this;
166        }
167        DeleteRequest deleteRequest = Requests.deleteRequest(index != null ? index : getIndex())
168                .type(type != null ? type : getType()).id(id);
169        try {
170            bulkProcessor.add(deleteRequest);
171        } catch (Exception e) {
172            logger.error("bulk add of delete failed: " + e.getMessage(), e);
173            enabled = false;
174        }
175        return this;
176    }
177
178    @Override
179    public NodeClient flush() {
180        if (!enabled) {
181            return this;
182        }
183        // no nothing
184        return this;
185    }
186
187    @Override
188    public void shutdown() {
189        client.close();
190    }
191
192    public NodeClient waitForHealthyCluster() throws IOException {
193        return waitForHealthyCluster(ClusterHealthStatus.YELLOW, "30s");
194    }
195
196    public NodeClient waitForHealthyCluster(ClusterHealthStatus status, String timeout) throws IOException {
197        try {
198            logger.info("waiting for cluster health...");
199            ClusterHealthResponse healthResponse =
200                    client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
201            if (healthResponse.isTimedOut()) {
202                throw new IOException("cluster not healthy, cowardly refusing to continue with operations");
203            }
204        } catch (ElasticSearchTimeoutException e) {
205            throw new IOException("cluster not healthy, cowardly refusing to continue with operations");
206        }
207        return this;
208    }
209}
