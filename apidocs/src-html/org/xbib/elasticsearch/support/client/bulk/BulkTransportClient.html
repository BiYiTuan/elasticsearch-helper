126    @Override
127    public BulkTransportClient newClient(URI uri, Settings settings) {
128        super.newClient(uri, settings);
129        resetSettings();
130        this.state = new State();
131        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
132            @Override
133            public void beforeBulk(long executionId, BulkRequest request) {
134                long l = outstandingRequests.getAndIncrement();
135                if (state != null) {
136                    int n = request.numberOfActions();
137                    state.getSubmitted().inc(n);
138                    state.getCurrentIngestNumDocs().inc(n);
139                    state.getTotalIngestSizeInBytes().inc(request.estimatedSizeInBytes());
140                }
141                if (logger.isDebugEnabled()) {
142                    logger.debug("before bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
143                            executionId, request.numberOfActions(), request.estimatedSizeInBytes(), l);
144                }
145            }
146
147            @Override
148            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
149                outstandingRequests.decrementAndGet();
150                if (state != null) {
151                    state.getSucceeded().inc(response.getItems().length);
152                    state.getTotalIngest().inc(response.getTookInMillis());
153                }
154                if (response.hasFailures()) {
155                    closed = true;
156                    logger.error("bulk [{}] failed", executionId);
157                    for (BulkItemResponse itemResponse : response.getItems()) {
158                        if (itemResponse.isFailed()) {
159                            state.getSucceeded().dec(1);
160                            state.getFailed().inc(1);
161                        }
162                    }
163                } else {
164                    state.getCurrentIngestNumDocs().dec(response.getItems().length);
165                }
166                if (logger.isDebugEnabled()) {
167                    logger.debug("after bulk [{}] [succeeded={}] [failed={}] [{}ms]",
168                            executionId,
169                            state.getSucceeded().count(),
170                            state.getFailed().count(),
171                            response.getTook().millis());
172                }
173            }
174
175            @Override
176            public void afterBulk(long executionId, BulkRequest requst, Throwable failure) {
177                outstandingRequests.decrementAndGet();
178                throwable = failure;
179                closed = true;
180                logger.error("bulk [" + executionId + "] error", failure);
181            }
182        };
183        BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
184                .setBulkActions(maxActionsPerBulkRequest)
185                .setConcurrentRequests(maxConcurrentBulkRequests)
186                .setFlushInterval(flushInterval);
187        if (maxVolumePerBulkRequest != null) {
188            builder.setBulkSize(maxVolumePerBulkRequest);
189        }
190        this.bulkProcessor = builder.build();
191        this.closed = false;
192        return this;
193    }

228    @Override
229    public BulkTransportClient startBulk(String index) throws IOException {
230        if (state == null) {
231            return this;
232        }
233        if (!state.isBulk(index)) {
234            state.startBulk(index);
235            ClientHelper.startBulk(client, index);
236        }
237        return this;
238    }

240    @Override
241    public BulkTransportClient stopBulk(String index) throws IOException {
242        if (state == null) {
243            return this;
244        }
245        if (state.isBulk(index)) {
246            state.stopBulk(index);
247            ClientHelper.stopBulk(client, index);
248        }
249        return this;
250    }

263    @Override
264    public BulkTransportClient index(IndexRequest indexRequest) {
265        if (closed) {
266            throw new ElasticsearchIllegalStateException("client is closed");
267        }
268        try {
269            state.getCurrentIngest().inc();
270            bulkProcessor.add(indexRequest);
271        } catch (Exception e) {
272            throwable = e;
273            closed = true;
274            logger.error("bulk add of index request failed: " + e.getMessage(), e);
275        } finally {
276            state.getCurrentIngest().dec();
277        }
278        return this;
279    }

283        return delete(Requests.deleteRequest(index).type(type).id(id));
284    }

286    @Override
287    public BulkTransportClient delete(DeleteRequest deleteRequest) {
288        if (closed) {
289            throw new ElasticsearchIllegalStateException("client is closed");
290        }
291        try {
292            state.getCurrentIngest().inc();
293            bulkProcessor.add(deleteRequest);
294        } catch (Exception e) {
295            throwable = e;
296            closed = true;
297            logger.error("bulk add of delete request failed: " + e.getMessage(), e);
298        } finally {
299            state.getCurrentIngest().dec();
300        }
301        return this;
302    }
303
304    @Override
305    public synchronized BulkTransportClient flush() {
306        if (closed) {
307            throw new ElasticsearchIllegalStateException("client is closed");
308        }
309        if (client == null) {
310            logger.warn("no client");
311            return this;
312        }
313        logger.info("flushing bulk processor");
314        // hacked BulkProcessor to execute the submission of remaining docs. Wait always 30 seconds at most.
315        BulkProcessorHelper.flush(bulkProcessor);
316        BulkProcessorHelper.waitFor(bulkProcessor, TimeValue.timeValueSeconds(30));
317        return this;
318    }

336    @Override
337    public synchronized void shutdown() {
338        if (closed) {
339            super.shutdown();
340            throw new ElasticsearchIllegalStateException("client is closed");
341        }
342        if (client == null) {
343            logger.warn("no client");
344            return;
345        }
346        try {
347            if (bulkProcessor != null) {
348                logger.info("closing bulk processor...");
349                bulkProcessor.close();
350            }
351            if (state.indices() != null && !state.indices().isEmpty()) {
352                logger.info("stopping bulk mode for indices {}...", state.indices());
353                for (String index : ImmutableSet.copyOf(state.indices())) {
354                    stopBulk(index);
355                }
356            }
357            logger.info("shutting down...");
358            super.shutdown();
359            logger.info("shutting down completed");
360        } catch (Exception e) {
361            logger.error(e.getMessage(), e);
362        }
363    }
