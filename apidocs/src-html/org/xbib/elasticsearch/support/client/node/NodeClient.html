094        public NodeClient newClient(Client client) {
095            this.client = client;
096            BulkProcessor.Listener listener = new BulkProcessor.Listener() {
097                @Override
098                public void beforeBulk(long executionId, BulkRequest request) {
099                    long l = outstandingBulkRequests.getAndIncrement();
100                    currentIngestNumDocs.inc(request.numberOfActions());
101                    totalIngestSizeInBytes.inc(request.estimatedSizeInBytes());
102                    if (logger.isDebugEnabled()) {
103                        logger.debug("before bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
104                            executionId, request.numberOfActions(), request.estimatedSizeInBytes(), l);
105                    }
106                }
107    
108                @Override
109                public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
110                    outstandingBulkRequests.decrementAndGet();
111                    totalIngest.inc(response.getTookInMillis());
112                    if (logger.isDebugEnabled()) {
113                        logger.debug("after bulk [{}] success [{} items] [{}ms]",
114                                executionId, response.getItems().length, response.getTook().millis());
115                    }
116                    if (response.hasFailures()) {
117                        closed = true;
118                        logger.error("after bulk [{}] failure reason: {}",
119                                executionId, response.buildFailureMessage());
120                    } else {
121                        currentIngestNumDocs.dec(response.getItems().length);
122                        totalIngestNumDocs.inc(response.getItems().length);
123                    }
124                }
125    
126                @Override
127                public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
128                    outstandingBulkRequests.decrementAndGet();
129                    throwable = failure;
130                    closed = true;
131                    logger.error("after bulk ["+executionId+"] error", failure);
132                }
133            };
134            BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
135                    .setBulkActions(maxActionsPerBulkRequest-1)  // off-by-one
136                    .setConcurrentRequests(maxConcurrentBulkRequests)
137                    .setFlushInterval(flushInterval);
138            if (maxVolume != null) {
139                builder.setBulkSize(maxVolume);
140            }
141            this.bulkProcessor =  builder.build();
142            try {
143                waitForHealthyCluster();
144                closed = false;
145            } catch (IOException e) {
146                logger.error(e.getMessage(), e);
147                closed = true;
148            }
149            return this;
150        }

179        @Override
180        public NodeClient create(String index, String type, String id, String source) {
181            if (closed) {
182                throw new ElasticSearchIllegalStateException("client is closed");
183            }
184            IndexRequest indexRequest = Requests.indexRequest(index != null ? index : getIndex())
185                    .type(type != null ? type : getType()).id(id).create(true).source(source);
186            try {
187                currentIngest.inc();
188                bulkProcessor.add(indexRequest);
189            } catch (Exception e) {
190                throwable = e;
191                logger.error("bulk add of create failed: " + e.getMessage(), e);
192                closed = true;
193            } finally {
194                currentIngest.dec();
195            }
196            return this;
197        }

199        @Override
200        public NodeClient index(String index, String type, String id, String source) {
201            if (closed) {
202                throw new ElasticSearchIllegalStateException("client is closed");
203            }
204            IndexRequest indexRequest = Requests.indexRequest(index != null ? index : getIndex())
205                    .type(type != null ? type : getType()).id(id).create(false).source(source);
206            try {
207                currentIngest.inc();
208                bulkProcessor.add(indexRequest);
209            } catch (Exception e) {
210                throwable = e;
211                closed = true;
212                logger.error("bulk add of index failed: " + e.getMessage(), e);
213            } finally {
214                currentIngest.dec();
215            }
216            return this;
217        }
218    
219        @Override
220        public NodeClient delete(String index, String type, String id) {
221            if (closed) {
222                throw new ElasticSearchIllegalStateException("client is closed");
223            }
224            DeleteRequest deleteRequest = Requests.deleteRequest(index != null ? index : getIndex())
225                    .type(type != null ? type : getType()).id(id);
226            try {
227                currentIngest.inc();
228                bulkProcessor.add(deleteRequest);
229            } catch (Exception e) {
230                throwable = e;
231                closed = true;
232                logger.error("bulk add of delete failed: " + e.getMessage(), e);
233            } finally {
234                currentIngest.dec();
235            }
236            return this;
237        }
238    
239        public NodeClient flush() {
240            if (closed) {
241                throw new ElasticSearchIllegalStateException("client is closed");
242            }
243            // we simply wait long enough for BulkProcessor flush
244            try {
245                Thread.sleep(flushInterval.getMillis() + 1000L);
246            } catch (InterruptedException e) {
247                logger.error("interrupted", e);
248            }
249            return this;
250        }
251    
252        public void shutdown() {
253            flush();
254            bulkProcessor.close();
255            client.close();
256        }

258        public NodeClient waitForHealthyCluster() throws IOException {
259            return waitForHealthyCluster(ClusterHealthStatus.YELLOW, "30s");
260        }
261    
262        public NodeClient waitForHealthyCluster(ClusterHealthStatus status, String timeout) throws IOException {
263            try {
264                logger.info("waiting for cluster health {}...", status.name());
265                ClusterHealthResponse healthResponse =
266                        client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
267                if (healthResponse.isTimedOut()) {
268                    throw new IOException("cluster not healthy, cowardly refusing to continue with operations");
269                }
270            } catch (ElasticSearchTimeoutException e) {
271                throw new IOException("cluster not healthy, cowardly refusing to continue with operations");
272            }
273            return this;
274        }

324        public NodeClient newIndex() {
325            if (closed) {
326                throw new ElasticSearchIllegalStateException("client is closed");
327            }
328            if (client == null) {
329                logger.warn("no client");
330                return this;
331            }
332            if (getIndex() == null) {
333                logger.warn("no index name given to create index");
334                return this;
335            }
336            CreateIndexRequest request = new CreateIndexRequest(getIndex());
337            if (settings() != null) {
338                request.settings(settings());
339            }
340            if (mappings() != null) {
341                for (Map.Entry<String,String> me : mappings().entrySet()) {
342                    request.mapping(me.getKey(), me.getValue());
343                }
344            }
345            logger.info("creating index {} with settings = {}, mappings = {}",
346                    getIndex(), settings(), mappings());
347            try {
348                client.admin().indices().create(request).actionGet();
349            } catch (Exception e) {
350                logger.error(e.getMessage(), e);
351            }
352            return this;
353        }
354    
355        public NodeClient deleteIndex() {
356            if (closed) {
357                throw new ElasticSearchIllegalStateException("client is closed");
358            }
359            if (client == null) {
360                logger.warn("no client");
361                return this;
362            }
363            if (getIndex() == null) {
364                logger.warn("no index name given to delete index");
365                return this;
366            }
367            try {
368                client.admin().indices().delete(new DeleteIndexRequest(getIndex())).actionGet();
369            } catch (Exception e) {
370                logger.error(e.getMessage(), e);
371            }
372            return this;
373        }
