095        public NodeClient newClient(Client client) {
096            this.client = client;
097            BulkProcessor.Listener listener = new BulkProcessor.Listener() {
098                @Override
099                public void beforeBulk(long executionId, BulkRequest request) {
100                    long l = outstandingBulkRequests.getAndIncrement();
101                    currentIngestNumDocs.inc(request.numberOfActions());
102                    totalIngestSizeInBytes.inc(request.estimatedSizeInBytes());
103                    if (logger.isDebugEnabled()) {
104                        logger.debug("before bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
105                            executionId, request.numberOfActions(), request.estimatedSizeInBytes(), l);
106                    }
107                }
108    
109                @Override
110                public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
111                    outstandingBulkRequests.decrementAndGet();
112                    totalIngest.inc(response.getTookInMillis());
113                    if (logger.isDebugEnabled()) {
114                        logger.debug("after bulk [{}] success [{} items] [{}ms]",
115                                executionId, response.getItems().length, response.getTook().millis());
116                    }
117                    if (response.hasFailures()) {
118                        closed = true;
119                        logger.error("after bulk [{}] failure reason: {}",
120                                executionId, response.buildFailureMessage());
121                    } else {
122                        currentIngestNumDocs.dec(response.getItems().length);
123                        totalIngestNumDocs.inc(response.getItems().length);
124                    }
125                }
126    
127                @Override
128                public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
129                    outstandingBulkRequests.decrementAndGet();
130                    throwable = failure;
131                    closed = true;
132                    logger.error("after bulk ["+executionId+"] error", failure);
133                }
134            };
135            BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
136                    .setBulkActions(maxActionsPerBulkRequest-1)  // off-by-one
137                    .setConcurrentRequests(maxConcurrentBulkRequests)
138                    .setFlushInterval(flushInterval);
139            if (maxVolume != null) {
140                builder.setBulkSize(maxVolume);
141            }
142            this.bulkProcessor =  builder.build();
143            try {
144                waitForHealthyCluster();
145                closed = false;
146            } catch (IOException e) {
147                logger.error(e.getMessage(), e);
148                closed = true;
149            }
150            return this;
151        }

180        @Override
181        public NodeClient index(String index, String type, String id, BytesReference source) {
182            return index(Requests.indexRequest(index != null ? index : getIndex())
183                    .type(type != null ? type : getType()).id(id).create(false).source(source, false));
184        }
185    
186        @Override
187        public NodeClient index(IndexRequest indexRequest) {
188            if (closed) {
189                throw new ElasticsearchIllegalStateException("client is closed");
190            }
191            try {
192                currentIngest.inc();
193                bulkProcessor.add(indexRequest);
194            } catch (Exception e) {
195                throwable = e;
196                closed = true;
197                logger.error("bulk add of index failed: " + e.getMessage(), e);
198            } finally {
199                currentIngest.dec();
200            }
201            return this;
202        }
203    
204        @Override
205        public NodeClient delete(String index, String type, String id) {
206            return delete(Requests.deleteRequest(index != null ? index : getIndex())
207                    .type(type != null ? type : getType()).id(id));
208        }
209    
210        @Override
211        public NodeClient delete(DeleteRequest deleteRequest) {
212            if (closed) {
213                throw new ElasticsearchIllegalStateException("client is closed");
214            }
215            try {
216                currentIngest.inc();
217                bulkProcessor.add(deleteRequest);
218            } catch (Exception e) {
219                throwable = e;
220                closed = true;
221                logger.error("bulk add of delete failed: " + e.getMessage(), e);
222            } finally {
223                currentIngest.dec();
224            }
225            return this;
226        }
227    
228        public NodeClient flush() {
229            if (closed) {
230                throw new ElasticsearchIllegalStateException("client is closed");
231            }
232            // we simply wait long enough for BulkProcessor flush
233            try {
234                Thread.sleep(flushInterval.getMillis() + 1000L);
235            } catch (InterruptedException e) {
236                logger.error("interrupted", e);
237            }
238            return this;
239        }
240    
241        public void shutdown() {
242            flush();
243            bulkProcessor.close();
244            client.close();
245        }

247        public NodeClient waitForHealthyCluster() throws IOException {
248            return waitForHealthyCluster(ClusterHealthStatus.YELLOW, "30s");
249        }
250    
251        public NodeClient waitForHealthyCluster(ClusterHealthStatus status, String timeout) throws IOException {
252            try {
253                logger.info("waiting for cluster health {}...", status.name());
254                ClusterHealthResponse healthResponse =
255                        client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
256                if (healthResponse.isTimedOut()) {
257                    throw new IOException("cluster not healthy, cowardly refusing to continue with operations");
258                }
259            } catch (ElasticsearchTimeoutException e) {
260                throw new IOException("cluster not healthy, cowardly refusing to continue with operations");
261            }
262            return this;
263        }

313        public NodeClient newIndex() {
314            if (closed) {
315                throw new ElasticsearchIllegalStateException("client is closed");
316            }
317            if (client == null) {
318                logger.warn("no client");
319                return this;
320            }
321            if (getIndex() == null) {
322                logger.warn("no index name given to create index");
323                return this;
324            }
325            CreateIndexRequest request = new CreateIndexRequest(getIndex());
326            if (settings() != null) {
327                request.settings(settings());
328            }
329            if (mappings() != null) {
330                for (Map.Entry<String,String> me : mappings().entrySet()) {
331                    request.mapping(me.getKey(), me.getValue());
332                }
333            }
334            logger.info("creating index {} with settings = {}, mappings = {}",
335                    getIndex(), settings(), mappings());
336            try {
337                client.admin().indices().create(request).actionGet();
338            } catch (Exception e) {
339                logger.error(e.getMessage(), e);
340            }
341            return this;
342        }
343    
344        public NodeClient deleteIndex() {
345            if (closed) {
346                throw new ElasticsearchIllegalStateException("client is closed");
347            }
348            if (client == null) {
349                logger.warn("no client");
350                return this;
351            }
352            if (getIndex() == null) {
353                logger.warn("no index name given to delete index");
354                return this;
355            }
356            try {
357                client.admin().indices().delete(new DeleteIndexRequest(getIndex())).actionGet();
358            } catch (Exception e) {
359                logger.error(e.getMessage(), e);
360            }
361            return this;
362        }
