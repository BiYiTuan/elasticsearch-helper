002    package org.xbib.elasticsearch.support.client;
003    
004    import org.elasticsearch.action.delete.DeleteRequest;
005    import org.elasticsearch.action.index.IndexRequest;
006    import org.elasticsearch.client.Client;
007    import org.elasticsearch.common.bytes.BytesReference;
008    
009    /**
010     * Minimal API for feed
011     */
012    public interface Feeder {
013    
014        Client client();
015    
016        /**
017         * Index document
018         *
019         * @param index the index
020         * @param type the type
021         * @param id the id
022         * @param source the source
023         * @return this
024         */
025        Feeder index(String index, String type, String id, BytesReference source);
026    
027        /**
028         * Index document
029         * @param indexRequest the index request
030         * @return this
031         */
032        Feeder index(IndexRequest indexRequest);
033    
034        /**
035         * Delete document
036         *
037         * @param index the index
038         * @param type the type
039         * @param id the id
040         * @return this
041         */
042        Feeder delete(String index, String type, String id);
043    
044        Feeder delete(DeleteRequest deleteRequest);
045    
046    
047    }
