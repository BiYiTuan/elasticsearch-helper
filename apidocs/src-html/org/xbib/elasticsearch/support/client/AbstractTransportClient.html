059        public AbstractTransportClient newClient(URI uri, Settings settings) {
060            if (client != null) {
061                logger.warn("client is open, closing...");
062                client.close();
063                client.threadPool().shutdown();
064                logger.warn("client is closed");
065                client = null;
066            }
067            if (settings != null) {
068                logger.info("creating transport client, java version {}, effective settings {}",
069                        System.getProperty("java.version"), settings.getAsMap());
070                // false = do not load config settings from environment
071                this.client = new TransportClient(settings, false);
072            } else {
073                logger.info("creating transport client, java version {}, using default settings",
074                        System.getProperty("java.version"));
075                this.client = new TransportClient();
076            }
077            try {
078                connect(uri);
079            } catch (UnknownHostException e) {
080                logger.error(e.getMessage(), e);
081            } catch (SocketException e) {
082                logger.error(e.getMessage(), e);
083            } catch (IOException e) {
084                logger.error(e.getMessage(), e);
085            }
086            return this;
087        }

089        public Settings defaultSettings(URI uri) {
090            return settingsBuilder()
091                    .put("cluster.name", findClusterName(uri))
092                    .put("network.server", false)
093                    .put("node.client", true)
094                    .put("client.transport.sniff", false)
095                    .put("client.transport.ignore_cluster_name", false)
096                    .put("client.transport.ping_timeout", "30s")
097                    .put("client.transport.nodes_sampler_interval", "30s")
098                    .build();
099        }

109        public void waitForCluster(ClusterHealthStatus status, TimeValue timeout) throws IOException {
110            try {
111                logger.info("waiting for cluster state {}", status.name());
112                ClusterHealthResponse healthResponse =
113                        client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
114                if (healthResponse.isTimedOut()) {
115                    throw new IOException("cluster state is " + healthResponse.getStatus().name()
116                            + " and not " + status.name()
117                            + ", cowardly refusing to continue with operations");
118                } else {
119                    logger.info("... cluster state ok");
120                }
121            } catch (ElasticsearchTimeoutException e) {
122                throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
123            }
124        }

130        public String healthColor() {
131            try {
132                ClusterHealthResponse healthResponse =
133                    client.admin().cluster().prepareHealth().setTimeout(TimeValue.timeValueSeconds(30)).execute().actionGet();
134                ClusterHealthStatus status = healthResponse.getStatus();
135                return status.name();
136            } catch (ElasticsearchTimeoutException e) {
137                return "TIMEOUT";
138            } catch (NoNodeAvailableException e) {
139                return "DISCONNECTED";
140            } catch (Throwable t) {
141                return "[" + t.getMessage() + "]";
142            }
143        }

145        public List<String> connectNodes() {
146            List<String> nodes = newLinkedList();
147            if (client.connectedNodes() != null) {
148                for (DiscoveryNode discoveryNode : client.connectedNodes()) {
149                    nodes.add(discoveryNode.toString());
150                }
151            }
152            return nodes;

172        protected URI findURI() {
173            URI uri = null;
174            String hostname = "localhost";
175            try {
176                hostname = InetAddress.getLocalHost().getHostName();
177                logger.debug("the hostname is {}", hostname);
178                uri = URI.create("es://"+hostname+":9300");
179                // custom?
180                URL url = getClass().getResource("/org/xbib/elasticsearch/cluster.properties");
181                if (url != null) {
182                    InputStream in = url.openStream();
183                    Properties p = new Properties();
184                    p.load(in);
185                    in.close();
186                    // the properties contains default URIs per hostname
187                    if (p.containsKey(hostname)) {
188                        uri = URI.create(p.getProperty(hostname));
189                        logger.debug("custom URI found in cluster.properties for hostname {}: {}", hostname, uri);
190                        return uri;
191                    }
192                }
193            } catch (UnknownHostException e) {
194                logger.warn("can't resolve host name, probably something wrong with network config: " + e.getMessage(), e);
195            } catch (Exception e) {
196                logger.warn(e.getMessage(), e);
197            }
198            logger.debug("URI for hostname {}: {}", hostname, uri);
199            return uri;
200        }

202        protected String findClusterName(URI uri) {
203            try {
204                Map<String, String> map = parseQueryString(uri, "UTF-8");
205                clustername = map.get("es.cluster.name");
206                if (clustername != null) {
207                    logger.info("cluster name found in URI {}: {}", uri, clustername);
208                    return clustername;
209                }
210                clustername = map.get("cluster.name");
211                if (clustername != null) {
212                    logger.info("cluster name found in URI {}: {}", uri, clustername);
213                    return clustername;
214                }
215            } catch (UnsupportedEncodingException ex) {
216                logger.warn(ex.getMessage(), ex);
217            }
218            logger.info("cluster name not found in URI {}, parameter es.cluster.name", uri);
219            clustername = System.getProperty("es.cluster.name");
220            if (clustername != null) {
221                logger.info("cluster name found in es.cluster.name system property: {}", clustername);
222                return clustername;
223            }
224            clustername = System.getProperty("cluster.name");
225            if (clustername != null) {
226                logger.info("cluster name found in cluster.name system property: {}", clustername);
227                return clustername;
228            }
229            logger.info("cluster name not found, falling back to default: {}", DEFAULT_CLUSTER_NAME);
230            clustername = DEFAULT_CLUSTER_NAME;
231            return clustername;
232        }

234        protected void connect(URI uri) throws IOException {
235            String hostname = uri.getHost();
236            int port = uri.getPort();
237            boolean newaddresses = false;
238            if (!"es".equals(uri.getScheme())) {
239                logger.warn("please specify URI scheme 'es'");
240            }
241            if ("hostname".equals(hostname)) {
242                InetSocketTransportAddress address = new InetSocketTransportAddress(InetAddress.getLocalHost().getHostName(), port);
243                if (!addresses.contains(address)) {
244                    logger.info("adding hostname address for transport client: {}", address);
245                    client.addTransportAddress(address);
246                    addresses.add(address);
247                    newaddresses = true;
248                }
249            } else if ("interfaces".equals(hostname)) {
250                Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
251                for (NetworkInterface netint : Collections.list(nets)) {
252                    logger.info("checking network interface = {}", netint.getName());
253                    Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
254                    for (InetAddress addr : Collections.list(inetAddresses)) {
255                        logger.info("checking address = {}", addr.getHostAddress());
256                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
257                        if (!addresses.contains(address)) {
258                            logger.info("adding address to transport client: {}", address);
259                            client.addTransportAddress(address);
260                            addresses.add(address);
261                            newaddresses = true;
262                        }
263                    }
264                }
265            } else if ("inet4".equals(hostname)) {
266                Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
267                for (NetworkInterface netint : Collections.list(nets)) {
268                    logger.info("checking network interface = {}", netint.getName());
269                    Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
270                    for (InetAddress addr : Collections.list(inetAddresses)) {
271                        if (addr instanceof Inet4Address) {
272                            logger.info("checking address = {}", addr.getHostAddress());
273                            InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
274                            if (!addresses.contains(address)) {
275                                logger.info("adding address for transport client: {}", address);
276                                client.addTransportAddress(address);
277                                addresses.add(address);
278                                newaddresses = true;
279                            }
280                        }
281                    }
282                }
283            } else if ("inet6".equals(hostname)) {
284                Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
285                for (NetworkInterface netint : Collections.list(nets)) {
286                    logger.info("checking network interface = {}", netint.getName());
287                    Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
288                    for (InetAddress addr : Collections.list(inetAddresses)) {
289                        if (addr instanceof Inet6Address) {
290                            logger.info("checking address = {}", addr.getHostAddress());
291                            InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
292                            if (!addresses.contains(address)) {
293                                logger.info("adding address for transport client: {}", address);
294                                client.addTransportAddress(address);
295                                addresses.add(address);
296                                newaddresses = true;
297                            }
298                        }
299                    }
300                }
301            } else {
302                InetSocketTransportAddress address = new InetSocketTransportAddress(hostname, port);
303                if (!addresses.contains(address)) {
304                    logger.info("adding custom address for transport client: {}", address);
305                    client.addTransportAddress(address);
306                    addresses.add(address);
307                    newaddresses = true;
308                }
309            }
310            logger.info("configured addresses to connect: {}", addresses);
311            if (client.connectedNodes() != null) {
312                List<DiscoveryNode> nodes = client.connectedNodes().asList();
313                logger.info("connected nodes = {}", nodes);
314                if (newaddresses) {
315                    for (DiscoveryNode node : nodes) {
316                        logger.info("new connection to {}", node);
317                    }
318                    if (!nodes.isEmpty()) {
319                        try {
320                            connectMore();
321                        } catch (Exception e) {
322                            logger.error("error while connecting to more nodes", e);
323                        }
324                    }
325                }
326            }
327        }
328    
329        private void connectMore() throws IOException {
330            logger.info("trying to discover more nodes...");
331            ClusterStateResponse clusterStateResponse = client.admin().cluster().state(new ClusterStateRequest()).actionGet();
332            DiscoveryNodes nodes = clusterStateResponse.getState().getNodes();
333            for (DiscoveryNode node : nodes) {
334                logger.info("adding discovered node {}", node);
335                try {
336                    client.addTransportAddress(node.address());
337                } catch (Exception e) {
338                    logger.warn("can't add node " + node, e);
339                }
340            }
341            logger.info("... discovery done");
342        }

344        private Map<String, String> parseQueryString(URI uri, String encoding)
345                throws UnsupportedEncodingException {
346            Map<String, String> m = new HashMap<String, String>();
347            if (uri == null) {
348                throw new IllegalArgumentException();
349            }
350            if (uri.getRawQuery() == null) {
351                return m;
352            }
353            // getRawQuery() because we do our decoding by ourselves
354            StringTokenizer st = new StringTokenizer(uri.getRawQuery(), "&");
355            while (st.hasMoreTokens()) {
356                String pair = st.nextToken();
357                int pos = pair.indexOf('=');
358                if (pos < 0) {
359                    m.put(pair, null);
360                } else {
361                    m.put(pair.substring(0, pos), decode(pair.substring(pos + 1, pair.length()), encoding));
362                }
363            }
364            return m;
365        }

367        private String decode(String s, String encoding) {
368            StringBuilder sb = new StringBuilder();
369            boolean fragment = false;
370            for (int i = 0; i < s.length(); i++) {
371                char ch = s.charAt(i);
372                switch (ch) {
373                    case '+':
374                        sb.append(' ');
375                        break;
376                    case '#':
377                        sb.append(ch);
378                        fragment = true;
379                        break;
380                    case '%':
381                        if (!fragment) {
382                            // fast hex decode
383                            sb.append((char) ((Character.digit(s.charAt(++i), 16) << 4)
384                                    | Character.digit(s.charAt(++i), 16)));
385                        } else {
386                            sb.append(ch);
387                        }
388                        break;
389                    default:
390                        sb.append(ch);
391                        break;
392                }
393            }
394            try {
395                // URL default encoding is ISO-8859-1
396                return new String(sb.toString().getBytes("ISO-8859-1"), encoding);
397            } catch (UnsupportedEncodingException e) {
398                throw new Error("encoding " + encoding + " not supported");
399            }
400        }
