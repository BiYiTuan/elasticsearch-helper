002package org.xbib.elasticsearch.support.client.search;
003
004import org.elasticsearch.common.settings.Settings;
005import org.xbib.elasticsearch.action.search.support.BasicSearchRequest;
006import org.elasticsearch.common.logging.ESLogger;
007import org.elasticsearch.common.logging.ESLoggerFactory;
008import org.elasticsearch.common.transport.InetSocketTransportAddress;
009
010import java.io.IOException;
011import java.net.Inet4Address;
012import java.net.Inet6Address;
013import java.net.InetAddress;
014import java.net.NetworkInterface;
015import java.net.URI;
016import java.util.Collections;
017import java.util.Enumeration;
018import java.util.Set;
019
020import static org.elasticsearch.common.collect.Sets.newHashSet;
021import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
022
023/**
024 * Search client mock. Do not perform actions on a real cluster.
025 */
026public class MockSearchClient extends SearchClient {
027
028    private final ESLogger logger = ESLoggerFactory.getLogger(MockSearchClient.class.getName());
029
030    private final Set<InetSocketTransportAddress> addresses = newHashSet();
031
032    public MockSearchClient newClient() {
033        super.newClient();
034        return this;
035    }
036
037    public MockSearchClient newClient(URI uri) {
038        return this.newClient(uri, settingsBuilder()
039                .put("cluster.name", findClusterName(uri))
040                .build());
041    }
042
043    public MockSearchClient newClient(URI uri, Settings settings) {
044        super.newClient(uri, settings);
045        return this;
046    }
047
048    public synchronized void shutdown() {
049    }
050
051    public BasicSearchRequest newRequest() {
052        return new BasicSearchRequest();
053    }
054
055    /**
056     * Configure addresses, but do not connect.
057     *
058     * @param uri
059     * @throws IOException
060     */
061    protected void connect(URI uri) throws IOException {
062        String hostname = uri.getHost();
063        int port = uri.getPort();
064        if (!"es".equals(uri.getScheme())) {
065            logger.warn("please specify URI scheme 'es'");
066        }
067        if ("hostname".equals(hostname)) {
068            InetSocketTransportAddress address = new InetSocketTransportAddress(InetAddress.getLocalHost().getHostName(), port);
069            if (!addresses.contains(address)) {
070                logger.info("adding hostname address for transport client = {}", address);
071                addresses.add(address);
072            }
073        } else if ("inet4".equals(hostname)) {
074            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
075            for (NetworkInterface netint : Collections.list(nets)) {
076                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
077                for (InetAddress addr : Collections.list(inetAddresses)) {
078                    if (addr instanceof Inet4Address) {
079                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
080                        if (!addresses.contains(address)) {
081                            logger.info("adding interface address for transport client = {}", address);
082                            client.addTransportAddress(address);
083                            addresses.add(address);
084                        }
085                    }
086                }
087            }
088        } else if ("inet6".equals(hostname)) {
089            Enumeration<NetworkInterface> nets = NetworkInterface.getNetworkInterfaces();
090            for (NetworkInterface netint : Collections.list(nets)) {
091                Enumeration<InetAddress> inetAddresses = netint.getInetAddresses();
092                for (InetAddress addr : Collections.list(inetAddresses)) {
093                    if (addr instanceof Inet6Address) {
094                        InetSocketTransportAddress address = new InetSocketTransportAddress(addr, port);
095                        if (!addresses.contains(address)) {
096                            logger.info("adding interface address for transport client = {}", address);
097                            client.addTransportAddress(address);
098                            addresses.add(address);
099                        }
100                    }
101                }
102            }
103        } else {
104            InetSocketTransportAddress address = new InetSocketTransportAddress(hostname, port);
105            if (!addresses.contains(address)) {
106                logger.info("adding custom address for transport client = {}", address);
107                client.addTransportAddress(address);
108                addresses.add(address);
109            }
110        }
111        logger.info("configured addresses to connect = {}", addresses);
112    }
113
114    public MockSearchClient waitForHealthyCluster() throws IOException {
115        return this;
116    }
117
118}
