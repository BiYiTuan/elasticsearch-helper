001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
004import org.elasticsearch.action.admin.indices.create.CreateIndexRequestBuilder;
005import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequestBuilder;
006import org.elasticsearch.common.logging.ESLogger;
007import org.elasticsearch.common.logging.ESLoggerFactory;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.common.unit.TimeValue;
010
011import java.io.IOException;
012import java.io.InputStream;
013import java.util.Map;
014
015public abstract class BaseIngestTransportClient extends BaseTransportClient
016        implements Ingest {
017
018    private final static ESLogger logger = ESLoggerFactory.getLogger(BaseIngestTransportClient.class.getSimpleName());
019
020    @Override
021    public BaseIngestTransportClient shards(int shards) {
022        super.addSetting("index.number_of_shards", shards);
023        return this;
024    }
025
026    @Override
027    public BaseIngestTransportClient replica(int replica) {
028        super.addSetting("index.number_of_replicas", replica);
029        return this;
030    }
031
032    @Override
033    public BaseIngestTransportClient newIndex(String index) {
034        return newIndex(index, (Settings)null, (Map<String,String>)null);
035    }
036
037    public BaseIngestTransportClient newIndex(String index, String type, InputStream settings, InputStream mappings) throws IOException {
038        configHelper.reset();
039        configHelper.setting(settings);
040        configHelper.mapping(type, mappings);
041        return newIndex(index, configHelper.settings(), configHelper.mappings());
042    }
043
044    public BaseIngestTransportClient newIndex(String index, Settings settings, Map<String,String> mappings) {
045        if (client == null) {
046            logger.warn("no client for create index");
047            return this;
048        }
049        if (index == null) {
050            logger.warn("no index name given to create index");
051            return this;
052        }
053        CreateIndexRequestBuilder createIndexRequestBuilder =
054                new CreateIndexRequestBuilder(client.admin().indices()).setIndex(index);
055        Settings concreteSettings;
056        if (settings == null && getSettings() != null) {
057            concreteSettings = getSettings();
058        } else if (settings != null) {
059            concreteSettings = settings;
060        } else {
061            concreteSettings = null;
062        }
063        if (concreteSettings != null) {
064            createIndexRequestBuilder.setSettings(getSettings());
065        }
066        if (mappings == null && getMappings() != null) {
067            for (String type : getMappings().keySet()) {
068                createIndexRequestBuilder.addMapping(type, getMappings().get(type));
069            }
070        } else if (mappings != null) {
071            for (String type : mappings.keySet()) {
072                createIndexRequestBuilder.addMapping(type, mappings.get(type));
073            }
074        }
075        createIndexRequestBuilder.execute().actionGet();
076        logger.info("index {} created with settings {} and {} mappings", index,
077                concreteSettings != null ? concreteSettings.getAsMap() : "",
078                mappings != null ? mappings.size() : 0);
079        return this;
080    }
081
082    @Override
083    public synchronized BaseIngestTransportClient deleteIndex(String index) {
084        if (client == null) {
085            logger.warn("no client for delete index");
086            return this;
087        }
088        if (index == null) {
089            logger.warn("no index name given to delete index");
090            return this;
091        }
092        DeleteIndexRequestBuilder deleteIndexRequestBuilder =
093                    new DeleteIndexRequestBuilder(client.admin().indices(), index);
094        deleteIndexRequestBuilder.execute().actionGet();
095        return this;
096    }
097
098    public BaseIngestTransportClient putMapping(String index) {
099        if (client == null) {
100            logger.warn("no client for put mapping");
101            return this;
102        }
103        configHelper.putMapping(client, index);
104        return this;
105    }
106
107    public BaseIngestTransportClient deleteMapping(String index, String type) {
108        if (client == null) {
109            logger.warn("no client for delete mapping");
110            return this;
111        }
112        configHelper.deleteMapping(client, index, type);
113        return this;
114    }
115
116    @Override
117    public BaseIngestTransportClient waitForCluster(ClusterHealthStatus status, TimeValue timeValue) throws IOException {
118        ClientHelper.waitForCluster(client, status, timeValue);
119        return this;
120    }
121
122}
