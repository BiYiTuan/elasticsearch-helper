001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
004import org.elasticsearch.action.admin.indices.create.CreateIndexAction;
005import org.elasticsearch.action.admin.indices.create.CreateIndexRequestBuilder;
006import org.elasticsearch.action.admin.indices.delete.DeleteIndexAction;
007import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequestBuilder;
008import org.elasticsearch.action.admin.indices.mapping.put.PutMappingAction;
009import org.elasticsearch.action.admin.indices.mapping.put.PutMappingRequestBuilder;
010import org.elasticsearch.common.logging.ESLogger;
011import org.elasticsearch.common.logging.ESLoggerFactory;
012import org.elasticsearch.common.settings.Settings;
013import org.elasticsearch.common.unit.TimeValue;
014
015import java.io.IOException;
016import java.io.InputStream;
017import java.util.Map;
018
019public abstract class BaseIngestTransportClient extends BaseTransportClient
020        implements Ingest {
021
022    private final static ESLogger logger = ESLoggerFactory.getLogger(BaseIngestTransportClient.class.getSimpleName());
023
024    protected Metric metric;
025
026    public Ingest newClient(Settings settings) throws IOException {
027        super.createClient(settings);
028        if (metric == null) {
029            this.metric = new Metric();
030            metric.start();
031        }
032        return this;
033    }
034
035    @Override
036    public BaseIngestTransportClient newIndex(String index) {
037        return newIndex(index, null, null);
038    }
039
040    @Override
041    public BaseIngestTransportClient newIndex(String index, String type, InputStream settings, InputStream mappings) throws IOException {
042        configHelper.reset();
043        configHelper.setting(settings);
044        configHelper.mapping(type, mappings);
045        return newIndex(index, configHelper.settings(), configHelper.mappings());
046    }
047
048    @Override
049    public BaseIngestTransportClient newIndex(String index, Settings settings, Map<String,String> mappings) {
050        if (client == null) {
051            logger.warn("no client for create index");
052            return this;
053        }
054        if (index == null) {
055            logger.warn("no index name given to create index");
056            return this;
057        }
058        CreateIndexRequestBuilder createIndexRequestBuilder =
059                new CreateIndexRequestBuilder(client(), CreateIndexAction.INSTANCE).setIndex(index);
060        if (settings != null) {
061            logger.info("settings = {}", settings.getAsStructuredMap());
062            createIndexRequestBuilder.setSettings(settings);
063        }
064        if (mappings != null ) {
065            for (String type : mappings.keySet()) {
066                logger.info("found mapping for {}", type);
067                createIndexRequestBuilder.addMapping(type, mappings.get(type));
068            }
069        }
070        createIndexRequestBuilder.execute().actionGet();
071        logger.info("index {} created", index);
072        return this;
073    }
074
075    @Override
076    public BaseIngestTransportClient newMapping(String index, String type, Map<String,Object> mapping) {
077        PutMappingRequestBuilder putMappingRequestBuilder =
078                new PutMappingRequestBuilder(client(), PutMappingAction.INSTANCE)
079                        .setIndices(index)
080                        .setType(type)
081                        .setSource(mapping);
082        putMappingRequestBuilder.execute().actionGet();
083        logger.info("mapping created for index {} and type {}", index, type);
084        return this;
085    }
086
087    @Override
088    public synchronized BaseIngestTransportClient deleteIndex(String index) {
089        if (client == null) {
090            logger.warn("no client for delete index");
091            return this;
092        }
093        if (index == null) {
094            logger.warn("no index name given to delete index");
095            return this;
096        }
097        DeleteIndexRequestBuilder deleteIndexRequestBuilder =
098                new DeleteIndexRequestBuilder(client(), DeleteIndexAction.INSTANCE, index);
099        deleteIndexRequestBuilder.execute().actionGet();
100        return this;
101    }
102
103    public BaseIngestTransportClient putMapping(String index) {
104        if (client == null) {
105            logger.warn("no client for put mapping");
106            return this;
107        }
108        configHelper.putMapping(client, index);
109        return this;
110    }
111
112    @Override
113    public BaseIngestTransportClient waitForCluster(ClusterHealthStatus status, TimeValue timeValue) throws IOException {
114        ClientHelper.waitForCluster(client, status, timeValue);
115        return this;
116    }
117
118    @Override
119    public BaseIngestTransportClient startBulk(String index, long startRefreshIntervalMillis, long stopRefreshItervalMillis) throws IOException {
120        if (metric == null) {
121            return this;
122        }
123        if (!metric.isBulk(index)) {
124            metric.setupBulk(index, startRefreshIntervalMillis, stopRefreshItervalMillis);
125            ClientHelper.updateIndexSetting(client, index, "refresh_interval", startRefreshIntervalMillis + "ms");
126        }
127        return this;
128    }
129
130    @Override
131    public BaseIngestTransportClient stopBulk(String index) throws IOException {
132        if (metric == null) {
133            return this;
134        }
135        if (metric.isBulk(index)) {
136            ClientHelper.updateIndexSetting(client, index, "refresh_interval", metric.getStopBulkRefreshIntervals().get(index) + "ms");
137            metric.removeBulk(index);
138        }
139        return this;
140    }
141
142}
