001    
002    package org.xbib.elasticsearch.support.client;
003    
004    import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
005    import org.elasticsearch.common.unit.ByteSizeValue;
006    import org.elasticsearch.common.unit.TimeValue;
007    
008    import java.io.IOException;
009    import java.io.InputStream;
010    import java.net.URI;
011    
012    /**
013     * Interface for providing convenient ingest methods.
014     */
015    public interface Ingest extends Feeder {
016    
017        Ingest newClient();
018    
019        Ingest newClient(URI uri);
020    
021        Ingest setIndex(String index);
022    
023        Ingest setType(String type);
024    
025        Ingest dateDetection(boolean b);
026    
027        /**
028         * Wait for healthy cluster
029         *
030         * @return this ingest
031         * @throws IOException
032         */
033        Ingest waitForCluster() throws IOException;
034    
035        Ingest waitForCluster(ClusterHealthStatus status, TimeValue timevalue) throws IOException;
036    
037        /**
038         * Set the maximum number of actions per bulk request
039         *
040         * @param maxActions maximum number of bulk actions
041         * @return this ingest
042         */
043        Ingest maxActionsPerBulkRequest(int maxActions);
044    
045        /**
046         * Set the maximum concurent bulk requests
047         *
048         * @param maxConcurentBulkRequests maximum number of concurrent bulk requests
049         * @return this Ingest
050         */
051        Ingest maxConcurrentBulkRequests(int maxConcurentBulkRequests);
052    
053        /**
054         * Set the maximum volume for bulk request before flush
055         * @param maxVolume maximum volume
056         * @return this ingest
057         */
058        Ingest maxVolumePerBulkRequest(ByteSizeValue maxVolume);
059    
060        Ingest shards(int shards);
061    
062        Ingest replica(int replica);
063    
064        Ingest resetSettings();
065    
066        Ingest setting(String key, String value);
067    
068        Ingest setting(String key, Boolean value);
069    
070        Ingest setting(String key, Integer value);
071    
072        Ingest setting(InputStream in) throws IOException;
073    
074        Ingest mapping(String type, InputStream in) throws IOException;
075    
076        Ingest mapping(String type, String mapping);
077    
078        /**
079         * Start bulk mode
080         *
081         * @return this ingest
082         */
083        Ingest startBulk() throws IOException;
084    
085        /**
086         * Stops bulk mode. Enables refresh.
087         *
088         * @return this Ingest
089         */
090        Ingest stopBulk() throws IOException;
091    
092        /**
093         * Create a new index
094         *
095         * @return this ingest
096         */
097        Ingest newIndex();
098    
099        /**
100         * Delete index
101         *
102         * @return this ingest
103         */
104        Ingest deleteIndex();
105    
106        /**
107         * Refresh the index.
108         *
109         * @return this ingest
110         */
111        Ingest refresh();
112    
113        /**
114         * Add replica level.
115         *
116         * @param level the replica level
117         * @return number of shards after updating replica level
118         */
119        int updateReplicaLevel(int level) throws IOException;
120    
121        int waitForRecovery();
122    
123        long getTotalBulkRequests();
124    
125        long getTotalBulkRequestTime();
126    
127        long getTotalDocuments();
128    
129        /**
130         * Get the total ingested data size in bytes so far.
131         *
132         * @return the total size in bytes
133         */
134        long getTotalSizeInBytes();
135    
136        boolean hasErrors();
137    
138        Throwable getThrowable();
139    }
