001    
002    package org.xbib.elasticsearch.support.client;
003    
004    import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
005    import org.elasticsearch.common.unit.ByteSizeValue;
006    import org.elasticsearch.common.unit.TimeValue;
007    
008    import java.io.IOException;
009    import java.io.InputStream;
010    
011    /**
012     * Interface for providing convenient administrative methods for ingesting data into Elasticsearch.
013     */
014    public interface Ingest extends ClientBuilder, Feeder {
015    
016        /**
017         * Set index name
018         * @param index the index name
019         * @return this
020         */
021        Ingest setIndex(String index);
022    
023        /**
024         * Set type name
025         * @param type the type
026         * @return this
027         */
028        Ingest setType(String type);
029    
030        /**
031         * Set the maximum number of actions per bulk request
032         *
033         * @param maxActions maximum number of bulk actions
034         * @return this ingest
035         */
036        Ingest maxActionsPerBulkRequest(int maxActions);
037    
038        /**
039         * Set the maximum concurent bulk requests
040         *
041         * @param maxConcurentBulkRequests maximum number of concurrent bulk requests
042         * @return this Ingest
043         */
044        Ingest maxConcurrentBulkRequests(int maxConcurentBulkRequests);
045    
046        /**
047         * Set the maximum volume for bulk request before flush
048         * @param maxVolume maximum volume
049         * @return this ingest
050         */
051        Ingest maxVolumePerBulkRequest(ByteSizeValue maxVolume);
052    
053        /**
054         * The number of shards for index creation
055         * @param shards the number of shards
056         * @return this
057         */
058        Ingest shards(int shards);
059    
060        /**
061         * The number of replica for index creation
062         * @param replica the number of replica
063         * @return this
064         */
065        Ingest replica(int replica);
066    
067        /**
068         * Clear settings
069         * @return this
070         */
071        Ingest resetSettings();
072    
073        /**
074         * Create a key/value in the settings
075         * @param key the key
076         * @param value the value
077         * @return this
078         */
079        Ingest setting(String key, String value);
080    
081        /**
082         * Create a key/value in the settings
083         * @param key the key
084         * @param value the value
085         * @return this
086         */
087        Ingest setting(String key, Boolean value);
088    
089        /**
090         * Create a key/value in the settings
091         * @param key the key
092         * @param value the value
093         * @return this
094         */
095        Ingest setting(String key, Integer value);
096    
097        /**
098         * Create a key/value in the settings
099         * @param in the input stream with settings
100         * @return this
101         */
102        Ingest setting(InputStream in) throws IOException;
103    
104        /**
105         * Enable date format detection in strings in the settings
106         * @param b true if  date format detection should be enabled
107         * @return this
108         */
109        Ingest dateDetection(boolean b);
110    
111        Ingest mapping(String type, InputStream in) throws IOException;
112    
113        Ingest mapping(String type, String mapping);
114    
115        /**
116         * Start bulk mode
117         *
118         * @return this ingest
119         */
120        Ingest startBulk() throws IOException;
121    
122        /**
123         * Stops bulk mode. Enables refresh.
124         *
125         * @return this Ingest
126         */
127        Ingest stopBulk() throws IOException;
128    
129        /**
130         * Create a new index
131         *
132         * @return this ingest
133         */
134        Ingest newIndex();
135    
136        /**
137         * Delete index
138         *
139         * @return this ingest
140         */
141        Ingest deleteIndex();
142    
143        /**
144         * Flush ingest, move all pending documents to the bulk indexer
145         * @return this
146         */
147        Ingest flush();
148    
149        /**
150         * Refresh the index.
151         *
152         * @return this ingest
153         */
154        Ingest refresh();
155    
156        /**
157         * Add replica level.
158         *
159         * @param level the replica level
160         * @return number of shards after updating replica level
161         */
162        int updateReplicaLevel(int level) throws IOException;
163    
164        /**
165         * Wait for cluster being healthy.
166         * @throws IOException
167         */
168        void waitForCluster() throws IOException;
169    
170        /**
171         * Wait for index recovery (after replica change)
172         * @return number of shards found
173         */
174        int waitForRecovery();
175    
176        /**
177         * Get total bulk request count
178         * @return the total bulk request
179         */
180        long getTotalBulkRequests();
181    
182        /**
183         * Get total bulk request time spent
184         * @return the total bulk request time
185         */
186        long getTotalBulkRequestTime();
187    
188        /**
189         * Get total documents that have been ingested
190         * @return the total number of documents
191         */
192        long getTotalDocuments();
193    
194        /**
195         * Get the total ingested data size in bytes so far.
196         *
197         * @return the total size in bytes
198         */
199        long getTotalSizeInBytes();
200    
201        boolean hasErrors();
202    
203        /**
204         * Return last throwable if exists.
205         * @return last throwable
206         */
207        Throwable getThrowable();
208    
209        /**
210         * Shutdown the ingesting
211         */
212        void shutdown();
213    }
