002    package org.xbib.elasticsearch.support.metrics;
003    
004    /**
005     */
006    public class MeanMetric implements Metric {
007    
008        private final LongAdder counter = new LongAdder();
009        private final LongAdder sum = new LongAdder();
010    
011        public void inc(long n) {
012            counter.increment();
013            sum.add(n);
014        }
015    
016        public void dec(long n) {
017            counter.decrement();
018            sum.add(-n);
019        }
020    
021        public long count() {
022            return counter.sum();
023        }
024    
025        public long sum() {
026            return sum.sum();
027        }
028    
029        public double mean() {
030            long count = count();
031            if (count > 0) {
032                return sum.sum() / (double) count;
033            }
034            return 0.0;
035        }
036    
037        public void clear() {
038            counter.reset();
039            sum.reset();
040        }
041    }
