002    package org.xbib.elasticsearch.support.metrics;
003    
004    import java.util.concurrent.TimeUnit;
005    
006    /**
007     * An exponentially-weighted moving average.
008     *
009     * @see <a href="http://www.teamquest.com/pdfs/whitepaper/ldavg1.pdf">UNIX Load Average Part 1: How It Works</a>
010     * @see <a href="http://www.teamquest.com/pdfs/whitepaper/ldavg2.pdf">UNIX Load Average Part 2: Not Your Average Average</a>
011     */
012    public class ExpWeightedMovingAverage {
013        private static final double M1_ALPHA = 1 - Math.exp(-5 / 60.0);
014        private static final double M5_ALPHA = 1 - Math.exp(-5 / 60.0 / 5);
015        private static final double M15_ALPHA = 1 - Math.exp(-5 / 60.0 / 15);
016    
017        private volatile boolean initialized = false;
018        private volatile double rate = 0.0;
019    
020        private final LongAdder uncounted = new LongAdder();
021        private final double alpha, interval;
022    
023        /**
024         * Creates a new EWMA which is equivalent to the UNIX one minute load average and which expects to be ticked every
025         * 5 seconds.
026         *
027         * @return a one-minute EWMA
028         */
029        public static ExpWeightedMovingAverage oneMinuteEWMA() {
030            return new ExpWeightedMovingAverage(M1_ALPHA, 5, TimeUnit.SECONDS);
031        }
032    
033        /**
034         * Creates a new EWMA which is equivalent to the UNIX five minute load average and which expects to be ticked every
035         * 5 seconds.
036         *
037         * @return a five-minute EWMA
038         */
039        public static ExpWeightedMovingAverage fiveMinuteEWMA() {
040            return new ExpWeightedMovingAverage(M5_ALPHA, 5, TimeUnit.SECONDS);
041        }
042    
043        /**
044         * Creates a new EWMA which is equivalent to the UNIX fifteen minute load average and which expects to be ticked
045         * every 5 seconds.
046         *
047         * @return a fifteen-minute EWMA
048         */
049        public static ExpWeightedMovingAverage fifteenMinuteEWMA() {
050            return new ExpWeightedMovingAverage(M15_ALPHA, 5, TimeUnit.SECONDS);
051        }
052    
053        /**
054         * Create a new EWMA with a specific smoothing constant.
055         *
056         * @param alpha        the smoothing constant
057         * @param interval     the expected tick interval
058         * @param intervalUnit the time unit of the tick interval
059         */
060        public ExpWeightedMovingAverage(double alpha, long interval, TimeUnit intervalUnit) {
061            this.interval = intervalUnit.toNanos(interval);
062            this.alpha = alpha;
063        }
064    
065        /**
066         * Update the moving average with a new value.
067         *
068         * @param n the new value
069         */
070        public void update(long n) {
071            uncounted.add(n);
072        }
073    
074        /**
075         * Mark the passage of time and decay the current rate accordingly.
076         */
077        public void tick() {
078            final long count = uncounted.sumThenReset();
079            double instantRate = count / interval;
080            if (initialized) {
081                rate += (alpha * (instantRate - rate));
082            } else {
083                rate = instantRate;
084                initialized = true;
085            }
086        }
087    
088        /**
089         * Returns the rate in the given units of time.
090         *
091         * @param rateUnit the unit of time
092         * @return the rate
093         */
094        public double rate(TimeUnit rateUnit) {
095            return rate * (double) rateUnit.toNanos(1);
096        }
097    }
