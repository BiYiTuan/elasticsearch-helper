002    package org.xbib.elasticsearch.support.metrics;
003    
004    import java.util.concurrent.ScheduledExecutorService;
005    import java.util.concurrent.ScheduledFuture;
006    import java.util.concurrent.TimeUnit;
007    
008    /**
009     * A meter metric which measures mean throughput and one-, five-, and
010     * fifteen-minute exponentially-weighted moving average throughputs.
011     *
012     * @see <a href="http://en.wikipedia.org/wiki/Moving_average#Exponential_moving_average">EMA</a>
013     */
014    public class MeterMetric implements Metric {
015        private static final long INTERVAL = 5; // seconds
016    
017        private final ExpWeightedMovingAverage m1Rate = ExpWeightedMovingAverage.oneMinuteEWMA();
018        private final ExpWeightedMovingAverage m5Rate = ExpWeightedMovingAverage.fiveMinuteEWMA();
019        private final ExpWeightedMovingAverage m15Rate = ExpWeightedMovingAverage.fifteenMinuteEWMA();
020    
021        private final LongAdder count = new LongAdder();
022        private final long startTime = System.nanoTime();
023        private final TimeUnit rateUnit;
024        private final ScheduledFuture<?> future;
025    
026        public MeterMetric(ScheduledExecutorService tickThread, TimeUnit rateUnit) {
027            this.rateUnit = rateUnit;
028            this.future = tickThread.scheduleAtFixedRate(new Runnable() {
029                @Override
030                public void run() {
031                    tick();
032                }
033            }, INTERVAL, INTERVAL, TimeUnit.SECONDS);
034        }
035    
036        public TimeUnit rateUnit() {
037            return rateUnit;
038        }
039    
040        /**
041         * Updates the moving averages.
042         */
043        void tick() {
044            m1Rate.tick();
045            m5Rate.tick();
046            m15Rate.tick();
047        }
048    
049        /**
050         * Mark the occurrence of an event.
051         */
052        public void mark() {
053            mark(1);
054        }
055    
056        /**
057         * Mark the occurrence of a given number of events.
058         *
059         * @param n the number of events
060         */
061        public void mark(long n) {
062            count.add(n);
063            m1Rate.update(n);
064            m5Rate.update(n);
065            m15Rate.update(n);
066        }
067    
068        public long count() {
069            return count.sum();
070        }
071    
072        public double fifteenMinuteRate() {
073            return m15Rate.rate(rateUnit);
074        }
075    
076        public double fiveMinuteRate() {
077            return m5Rate.rate(rateUnit);
078        }
079    
080        public double meanRate() {
081            long count = count();
082            if (count == 0) {
083                return 0.0;
084            } else {
085                final long elapsed = (System.nanoTime() - startTime);
086                return convertNsRate(count / (double) elapsed);
087            }
088        }
089    
090        public double oneMinuteRate() {
091            return m1Rate.rate(rateUnit);
092        }
093    
094        private double convertNsRate(double ratePerNs) {
095            return ratePerNs * (double) rateUnit.toNanos(1);
096        }
097    
098        public void stop() {
099            future.cancel(false);
100        }
101    }
