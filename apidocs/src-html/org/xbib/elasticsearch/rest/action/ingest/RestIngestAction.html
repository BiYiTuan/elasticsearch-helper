001package org.xbib.elasticsearch.rest.action.ingest;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.Loggers;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.common.unit.ByteSizeValue;
009import org.elasticsearch.common.unit.TimeValue;
010import org.elasticsearch.common.xcontent.XContentBuilder;
011import org.elasticsearch.rest.BaseRestHandler;
012import org.elasticsearch.rest.BytesRestResponse;
013import org.elasticsearch.rest.RestChannel;
014import org.elasticsearch.rest.RestController;
015import org.elasticsearch.rest.RestRequest;
016import org.xbib.elasticsearch.action.ingest.IngestActionFailure;
017import org.xbib.elasticsearch.action.ingest.IngestProcessor;
018import org.xbib.elasticsearch.action.ingest.IngestRequest;
019import org.xbib.elasticsearch.action.ingest.IngestResponse;
020
021import java.io.IOException;
022import java.util.concurrent.CountDownLatch;
023import java.util.concurrent.TimeUnit;
024import java.util.concurrent.atomic.AtomicLong;
025
026import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
027import static org.elasticsearch.rest.RestRequest.Method.POST;
028import static org.elasticsearch.rest.RestRequest.Method.PUT;
029import static org.elasticsearch.rest.RestStatus.BAD_REQUEST;
030import static org.elasticsearch.rest.RestStatus.OK;
031
032/**
033 * <pre>
034 * { "index" : { "_index" : "test", "_type" : "type1", "_id" : "1" }
035 * { "type1" : { "field1" : "value1" } }
036 * { "delete" : { "_index" : "test", "_type" : "type1", "_id" : "2" } }
037 * { "create" : { "_index" : "test", "_type" : "type1", "_id" : "1" }
038 * { "type1" : { "field1" : "value1" } }
039 * </pre>
040 */
041
042public class RestIngestAction extends BaseRestHandler {
043
044    private final static ESLogger logger = Loggers.getLogger(RestIngestAction.class);
045    /**
046     * Count the volume
047     */
048    public final static AtomicLong volumeCounter = new AtomicLong();
049    /**
050     * The IngestProcessor
051     */
052    private final IngestProcessor ingestProcessor;
053
054    @Inject
055    public RestIngestAction(Settings settings, Client client, RestController controller) {
056        super(settings, client);
057
058        controller.registerHandler(POST, "/_ingest", this);
059        controller.registerHandler(PUT, "/_ingest", this);
060        controller.registerHandler(POST, "/{index}/_ingest", this);
061        controller.registerHandler(PUT, "/{index}/_ingest", this);
062        controller.registerHandler(POST, "/{index}/{type}/_ingest", this);
063        controller.registerHandler(PUT, "/{index}/{type}/_ingest", this);
064
065        int actions = settings.getAsInt("action.ingest.maxactions", 1000);
066        int concurrency = settings.getAsInt("action.ingest.maxconcurrency", Runtime.getRuntime().availableProcessors() * 4);
067        ByteSizeValue volume = settings.getAsBytesSize("action.ingest.maxvolume", ByteSizeValue.parseBytesSizeValue("10m"));
068        TimeValue maxwait = settings.getAsTime("action.ingest.maxwait", TimeValue.timeValueSeconds(60));
069
070        this.ingestProcessor = new IngestProcessor(client)
071                .maxActions(actions)
072                .maxConcurrentRequests(concurrency)
073                .maxVolumePerRequest(volume)
074                .maxWaitForResponses(maxwait);
075    }
076
077    @Override
078    public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
079        final IngestIdHolder idHolder = new IngestIdHolder();
080        final CountDownLatch latch = new CountDownLatch(1);
081        IngestProcessor.IngestListener ingestListener = new IngestProcessor.IngestListener() {
082            @Override
083            public void onRequest(int concurrency, IngestRequest ingestRequest) {
084                long v = volumeCounter.addAndGet(ingestRequest.estimatedSizeInBytes());
085                if (logger.isDebugEnabled()) {
086                    logger.debug("ingest request [{}] of {} items, {} bytes, {} concurrent requests",
087                            ingestRequest.ingestId(), ingestRequest.numberOfActions(), v, concurrency);
088                }
089                idHolder.ingestId(ingestRequest.ingestId());
090                latch.countDown();
091            }
092
093            @Override
094            public void onResponse(int concurrency, IngestResponse response) {
095                if (logger.isDebugEnabled()) {
096                    logger.debug("ingest response [{}] [{} succeeded] [{} failed] [{}ms]",
097                            response.ingestId(),
098                            response.successSize(),
099                            response.getFailures().size(),
100                            response.tookInMillis());
101                }
102                if (!response.getFailures().isEmpty()) {
103                    for (IngestActionFailure f : response.getFailures()) {
104                        logger.error("ingest [{}] failure, reason: {}", response.ingestId(), f.message());
105                    }
106                }
107            }
108
109            @Override
110            public void onFailure(int concurrency, long ingestId, Throwable failure) {
111                logger.error("ingest [{}] error", ingestId, failure);
112            }
113        };
114        try {
115            long t0 = System.currentTimeMillis();
116            ingestProcessor.add(request.content(),
117                    request.contentUnsafe(),
118                    request.param("index"),
119                    request.param("type"),
120                    ingestListener);
121            // estimation, should be enough time to wait for an ID
122            boolean b = latch.await(100, TimeUnit.MILLISECONDS);
123            long t1 = System.currentTimeMillis();
124            XContentBuilder builder = jsonBuilder();
125            builder.startObject();
126            builder.field("took", t1 - t0);
127            // got ID?
128            if (b) {
129                builder.field("id", idHolder.ingestId());
130            }
131            builder.endObject();
132            channel.sendResponse(new BytesRestResponse(OK, builder));
133        } catch (Exception e) {
134            try {
135                XContentBuilder builder = jsonBuilder();
136                builder.startObject().field("error", e.getMessage()).endObject();
137                channel.sendResponse(new BytesRestResponse(BAD_REQUEST, builder));
138            } catch (IOException e1) {
139                logger.error("Failed to send failure response", e1);
140            }
141        }
142    }
143
144    static final class IngestIdHolder {
145        private long ingestId;
146
147        public void ingestId(long ingestId) {
148            this.ingestId = ingestId;
149        }
150
151        public long ingestId() {
152            return ingestId;
153        }
154    }
155
156}
