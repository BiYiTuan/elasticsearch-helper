001package org.xbib.elasticsearch.rest.action.ingest;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.Loggers;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.common.unit.ByteSizeValue;
009import org.elasticsearch.common.unit.TimeValue;
010import org.elasticsearch.common.xcontent.XContentBuilder;
011import org.elasticsearch.common.xcontent.XContentBuilderString;
012import org.elasticsearch.rest.BaseRestHandler;
013import org.elasticsearch.rest.RestChannel;
014import org.elasticsearch.rest.RestController;
015import org.elasticsearch.rest.RestRequest;
016import org.xbib.elasticsearch.action.ingest.IngestActionFailure;
017import org.xbib.elasticsearch.action.ingest.IngestProcessor;
018import org.xbib.elasticsearch.action.ingest.IngestRequest;
019import org.xbib.elasticsearch.action.ingest.IngestResponse;
020import org.xbib.elasticsearch.rest.action.support.XContentRestResponse;
021
022import java.io.IOException;
023import java.util.concurrent.CountDownLatch;
024import java.util.concurrent.TimeUnit;
025import java.util.concurrent.atomic.AtomicLong;
026
027import static org.elasticsearch.rest.RestRequest.Method.POST;
028import static org.elasticsearch.rest.RestRequest.Method.PUT;
029import static org.elasticsearch.rest.RestStatus.BAD_REQUEST;
030import static org.elasticsearch.rest.RestStatus.OK;
031import static org.xbib.elasticsearch.rest.action.support.RestXContentBuilder.restContentBuilder;
032
033/**
034 * <pre>
035 * { "index" : { "_index" : "test", "_type" : "type1", "_id" : "1" }
036 * { "type1" : { "field1" : "value1" } }
037 * { "delete" : { "_index" : "test", "_type" : "type1", "_id" : "2" } }
038 * { "create" : { "_index" : "test", "_type" : "type1", "_id" : "1" }
039 * { "type1" : { "field1" : "value1" } }
040 * </pre>
041 */
042
043public class RestIngestAction extends BaseRestHandler {
044
045    private final static ESLogger logger = Loggers.getLogger(RestIngestAction.class);
046    /**
047     * Count the volume
048     */
049    public final static AtomicLong volumeCounter = new AtomicLong();
050    /**
051     * The IngestProcessor
052     */
053    private final IngestProcessor ingestProcessor;
054
055    @Inject
056    public RestIngestAction(Settings settings, Client client, RestController controller) {
057        super(settings, client);
058
059        controller.registerHandler(POST, "/_ingest", this);
060        controller.registerHandler(PUT, "/_ingest", this);
061        controller.registerHandler(POST, "/{index}/_ingest", this);
062        controller.registerHandler(PUT, "/{index}/_ingest", this);
063        controller.registerHandler(POST, "/{index}/{type}/_ingest", this);
064        controller.registerHandler(PUT, "/{index}/{type}/_ingest", this);
065
066        int actions = settings.getAsInt("action.ingest.maxactions", 1000);
067        int concurrency = settings.getAsInt("action.ingest.maxconcurrency", Runtime.getRuntime().availableProcessors() * 4);
068        ByteSizeValue volume = settings.getAsBytesSize("action.ingest.maxvolume", ByteSizeValue.parseBytesSizeValue("10m"));
069        TimeValue waitingTime = settings.getAsTime("action.ingest.waitingtime", TimeValue.timeValueSeconds(60));
070
071        this.ingestProcessor = new IngestProcessor(client)
072                .maxActions(actions)
073                .maxConcurrentRequests(concurrency)
074                .maxVolumePerRequest(volume)
075                .maxWaitForResponses(waitingTime);
076    }
077
078    @Override
079    public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
080
081        final IngestIdHolder idHolder = new IngestIdHolder();
082        final CountDownLatch latch = new CountDownLatch(1);
083
084        IngestProcessor.IngestListener ingestListener = new IngestProcessor.IngestListener() {
085            @Override
086            public void onRequest(int concurrency, IngestRequest ingestRequest) {
087                long v = volumeCounter.addAndGet(ingestRequest.estimatedSizeInBytes());
088                if (logger.isDebugEnabled()) {
089                    logger.debug("ingest request [{}] of {} items, {} bytes, {} concurrent requests",
090                            ingestRequest.ingestId(), ingestRequest.numberOfActions(), v, concurrency);
091                }
092                idHolder.ingestId(ingestRequest.ingestId());
093                latch.countDown();
094            }
095
096            @Override
097            public void onResponse(int concurrency, IngestResponse response) {
098                if (logger.isDebugEnabled()) {
099                    logger.debug("ingest response [{}] [{} succeeded] [{} failed] [{}ms]",
100                            response.ingestId(),
101                            response.successSize(),
102                            response.getFailures().size(),
103                            response.tookInMillis());
104                }
105                if (!response.getFailures().isEmpty()) {
106                    for (IngestActionFailure f : response.getFailures()) {
107                        logger.error("ingest [{}] failure, reason: {}", response.ingestId(), f.message());
108                    }
109                }
110            }
111
112            @Override
113            public void onFailure(int concurrency, long ingestId, Throwable failure) {
114                logger.error("ingest [{}] error", ingestId, failure);
115            }
116        };
117        try {
118            long t0 = System.currentTimeMillis();
119            ingestProcessor.add(request.content(),
120                    request.contentUnsafe(),
121                    request.param("index"),
122                    request.param("type"),
123                    ingestListener);
124            // estimation, should be enough time to wait for an ID
125            boolean b = latch.await(100, TimeUnit.MILLISECONDS);
126            long t1 = System.currentTimeMillis();
127
128            XContentBuilder builder = restContentBuilder(request);
129            builder.startObject();
130            builder.field(Fields.TOOK, t1 - t0);
131            // got ID?
132            if (b) {
133                builder.field(Fields.ID, idHolder.ingestId());
134            }
135            builder.endObject();
136            channel.sendResponse(new XContentRestResponse(request, OK, builder));
137        } catch (Exception e) {
138            try {
139                XContentBuilder builder = restContentBuilder(request);
140                channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder.startObject().field("error", e.getMessage()).endObject()));
141            } catch (IOException e1) {
142                logger.error("Failed to send failure response", e1);
143            }
144        }
145    }
146
147    static final class Fields {
148        static final XContentBuilderString TOOK = new XContentBuilderString("took");
149        static final XContentBuilderString ID = new XContentBuilderString("id");
150    }
151
152    static final class IngestIdHolder {
153        private long ingestId;
154
155        public void ingestId(long ingestId) {
156            this.ingestId = ingestId;
157        }
158
159        public long ingestId() {
160            return ingestId;
161        }
162    }
163
164}
