001package org.xbib.elasticsearch.rest.action.river.state;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.common.xcontent.ToXContent;
008import org.elasticsearch.common.xcontent.XContentBuilder;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestController;
011import org.elasticsearch.rest.RestRequest;
012import org.elasticsearch.rest.RestStatus;
013import org.xbib.elasticsearch.action.river.state.RiverState;
014import org.xbib.elasticsearch.action.river.state.RiverStateAction;
015import org.xbib.elasticsearch.action.river.state.RiverStateRequest;
016import org.xbib.elasticsearch.action.river.state.RiverStateResponse;
017import org.xbib.elasticsearch.rest.action.support.AbstractRestRiverAction;
018import org.xbib.elasticsearch.rest.action.support.RestXContentBuilder;
019import org.xbib.elasticsearch.rest.action.support.XContentRestResponse;
020import org.xbib.elasticsearch.rest.action.support.XContentThrowableRestResponse;
021
022import java.io.IOException;
023
024public class RestRiverStateAction extends AbstractRestRiverAction {
025
026    @Inject
027    public RestRiverStateAction(Settings settings, Client client, RestController controller, Injector injector) {
028        super(settings, client, injector);
029        controller.registerHandler(RestRequest.Method.GET, "/_river/{riverType}/{riverName}/_state", this);
030    }
031
032    @Override
033    public void handleRequest(RestRequest request, RestChannel channel, Client client) {
034        try {
035            String riverName = request.param("riverName");
036            String riverType = request.param("riverType");
037            RiverStateRequest riverStateRequest = new RiverStateRequest()
038                    .setRiverName(riverName)
039                    .setRiverType(riverType);
040            RiverStateResponse riverStateResponse = client.admin().cluster()
041                    .execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
042            XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
043            builder.startObject();
044            builder.startArray("state");
045            for (RiverState state : riverStateResponse.getStates()) {
046                state.toXContent(builder, ToXContent.EMPTY_PARAMS);
047            }
048            builder.endArray().endObject();
049            channel.sendResponse(new XContentRestResponse(request, RestStatus.OK, builder));
050        } catch (IOException ioe) {
051            try {
052                channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
053            } catch (IOException e) {
054                logger.error("unable to send response to client");
055            }
056        }
057    }
058
059}
