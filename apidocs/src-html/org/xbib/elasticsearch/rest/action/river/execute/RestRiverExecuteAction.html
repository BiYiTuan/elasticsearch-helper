001package org.xbib.elasticsearch.rest.action.river.execute;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.rest.RestChannel;
008import org.elasticsearch.rest.RestController;
009import org.elasticsearch.rest.RestRequest;
010import org.elasticsearch.rest.RestStatus;
011import org.xbib.elasticsearch.action.river.execute.RiverExecuteAction;
012import org.xbib.elasticsearch.action.river.execute.RiverExecuteRequest;
013import org.xbib.elasticsearch.action.river.execute.RiverExecuteResponse;
014import org.xbib.elasticsearch.rest.action.support.AbstractRestRiverAction;
015
016/**
017 * Run a river. The river can be executed once with such a call. Example:
018 *
019 * curl -XPOST 'localhost:9200/_river/my_jdbc_river/_execute'
020 */
021public class RestRiverExecuteAction extends AbstractRestRiverAction {
022
023    @Inject
024    public RestRiverExecuteAction(Settings settings, Client client, RestController controller, Injector injector) {
025        super(settings, client, injector);
026        controller.registerHandler(RestRequest.Method.POST, "/_river/{riverType}/{riverName}/_execute", this);
027    }
028
029    @Override
030    public void handleRequest(final RestRequest request, RestChannel channel) {
031        String riverName = request.param("riverName");
032        if (riverName == null || riverName.isEmpty()) {
033            respond(false, request, channel, "parameter 'riverName' is required", RestStatus.BAD_REQUEST);
034            return;
035        }
036        String riverType = request.param("riverType");
037        RiverExecuteRequest riverExecuteRequest = new RiverExecuteRequest()
038                .setRiverName(riverName)
039                .setRiverType(riverType);
040        RiverExecuteResponse riverExecuteResponse = client
041                .execute(RiverExecuteAction.INSTANCE, riverExecuteRequest).actionGet();
042        boolean isExecuted = false;
043        for (int i = 0; i < riverExecuteResponse.isExecuted().length; i++) {
044            isExecuted = isExecuted || riverExecuteResponse.isExecuted()[i];
045        }
046        respond(isExecuted, request, channel, "ok", RestStatus.OK);
047    }
048
049}
