001package org.xbib.elasticsearch.plugin.feeder;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.settings.Settings;
005import org.xbib.elasticsearch.action.river.state.RiverState;
006import org.xbib.pipeline.Pipeline;
007import org.xbib.pipeline.PipelineProvider;
008import org.xbib.pipeline.PipelineRequest;
009
010import java.io.Closeable;
011import java.io.IOException;
012import java.util.Map;
013
014/**
015 * A feeder is a helper for executing feeds of documents to Elastisearch, either in push or pull mode.
016 *
017 * Rivers use the pull mode, standalone programs operates usually in the the push mode.
018 *
019 * The feed is executing by help of multiple threads with configurable concurrency.
020 * Each thread starts a pipeline which iterates over the feed sources. Each pipeline
021 * processes specifications that describe how the feed source is used.
022 *
023 * @param <T> the pipeline element
024 * @param <R> the pipeline request
025 * @param <P> the pipeline exception
026 */
027public interface Feeder <T, R extends PipelineRequest, P extends Pipeline<T, R>>
028    extends Tool, Runnable, Closeable {
029
030    /**
031     * Set the type of the feeder
032     * @param type the type
033     * @return the feeder
034     */
035    Feeder<T, R, P> setType(String type);
036
037    /**
038     * Set the specification for the feeder. The specification contains information
039     * about a feeding step.
040     * @param spec the specification
041     * @return the specification
042     */
043    Feeder<T, R, P> setSpec(Map<String, Object> spec);
044
045    /**
046     * Get the specification of this feeder
047     * @return the specification
048     */
049    Map<String,Object> getSpec();
050
051    /**
052     * Set settings for the feeder. The settings control the behaviour of the feeder.
053     * @param settings the settings
054     * @return the feeder
055     */
056    Feeder<T, R, P> setSettings(Settings settings);
057
058    /**
059     * Get the feeder settings
060     * @return the settings
061     */
062    Settings getSettings();
063
064    /**
065     * Set the client for the feeder
066     * @param client the client
067     * @return the feeder
068     */
069    Feeder<T, R, P> setClient(Client client);
070
071    /**
072     * Get the client of the feeder.
073     * @return the client
074     */
075    Client getClient();
076
077    /**
078     * Get the pipeline provider. The pipeline provider is called once per thread.
079     *
080     * @return the pipeline provider
081     */
082    PipelineProvider<P> pipelineProvider();
083
084    /**
085     * Prepare the feeder. This is the first phase of the feeder run. It is called before feeder threads
086     * are started.
087     * @return the feeder
088     * @throws IOException if prepare fails
089     */
090    Feeder<T, R, P> beforeRun() throws IOException;
091
092    /**
093     * Executing a task
094     * @param parameters the parameters for the task execution
095     * @throws Exception if processing fails
096     */
097    void executeTask(Map<String, Object> parameters) throws Exception;
098
099    /**
100     * Clean up the feeder. This is the last phase of the feeder run. It is called after feeder threads
101     * have ended.
102     * @return this feeder
103     * @throws IOException if cleanup fails
104     */
105    Feeder<T, R, P> afterRun() throws IOException;
106
107    /**
108     * Set the state of the feeder if the feeder runs in a river.
109     * @param riverState the river state
110     * @return the feeder
111     */
112    Feeder<T, R, P> setRiverState(RiverState riverState);
113
114    /**
115     * Get the feeder state if the feeder runs in a river.
116     * @return the river state
117     */
118    RiverState getRiverState();
119
120    /**
121     * Interrupt the feeder. The interruption request is delegated to all threads.
122     * @param state true if feeder should be interrupted
123     */
124    void setInterrupted(boolean state);
125
126}
