002package org.xbib.elasticsearch.action.search.support;
003
004import org.elasticsearch.action.get.GetResponse;
005import org.elasticsearch.action.search.SearchResponse;
006import org.elasticsearch.common.xcontent.ToXContent;
007import org.elasticsearch.common.xcontent.XContentBuilder;
008import org.elasticsearch.common.xcontent.json.JsonXContent;
009
010import java.io.IOException;
011import java.io.OutputStream;
012
013/**
014 * Helper class for Elasticsearch responses
015 */
016public class BasicResponse {
017
018    private SearchResponse searchResponse;
019
020    private GetResponse getResponse;
021
022    public BasicResponse searchResponse(SearchResponse response) {
023        this.searchResponse = response;
024        return this;
025    }
026
027    public BasicResponse getResponse(GetResponse response) {
028        this.getResponse = response;
029        return this;
030    }
031
032    public long tookInMillis() {
033        return searchResponse.getTookInMillis();
034    }
035
036    public long totalHits() {
037        return searchResponse.getHits().getTotalHits();
038    }
039
040    public boolean exists() {
041        return getResponse.isExists();
042    }
043
044    public BasicResponse toJson(OutputStream out) throws IOException {
045        if (out == null) {
046            return this;
047        }
048        if (searchResponse == null) {
049            out.write(jsonErrorMessage("no response yet"));
050            return this;
051        }
052        XContentBuilder jsonBuilder = new XContentBuilder(JsonXContent.jsonXContent, out);
053        jsonBuilder.startObject();
054        searchResponse.toXContent(jsonBuilder, ToXContent.EMPTY_PARAMS);
055        jsonBuilder.endObject();
056        jsonBuilder.close();
057        return this;
058    }
059
060    private static byte[] jsonEmptyMessage(String message) {
061        return ("{\"error\":404,\"message\":\"" + message + "\"}").getBytes();
062    }
063
064
065    private static byte[] jsonErrorMessage(String message) {
066        return ("{\"error\":500,\"message\":\"" + message + "\"}").getBytes();
067    }
068
069
070}
