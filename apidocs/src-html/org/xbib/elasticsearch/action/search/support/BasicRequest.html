002package org.xbib.elasticsearch.action.search.support;
003
004import org.elasticsearch.action.get.GetRequestBuilder;
005import org.elasticsearch.action.search.SearchRequestBuilder;
006import org.elasticsearch.common.logging.ESLogger;
007import org.elasticsearch.common.logging.ESLoggerFactory;
008import org.elasticsearch.common.unit.TimeValue;
009
010import java.io.IOException;
011
012/**
013 * Helper class for Elasticsearch search/get requests
014 *
015 */
016public class BasicRequest {
017
018    private final ESLogger logger = ESLoggerFactory.getLogger(BasicRequest.class.getName());
019
020    private SearchRequestBuilder searchRequestBuilder;
021
022    private GetRequestBuilder getRequestBuilder;
023
024    private String[] index;
025
026    private String[] type;
027
028    private String id;
029
030    private String query;
031
032    public BasicRequest newSearchRequest(SearchRequestBuilder searchRequestBuilder) {
033        this.searchRequestBuilder = searchRequestBuilder;
034        return this;
035    }
036
037    public SearchRequestBuilder searchRequestBuilder() {
038        return searchRequestBuilder;
039    }
040
041    public BasicRequest newGetRequest(GetRequestBuilder getRequestBuilder) {
042        this.getRequestBuilder = getRequestBuilder;
043        return this;
044    }
045
046    public GetRequestBuilder getRequestBuilder() {
047        return getRequestBuilder;
048    }
049
050    public BasicRequest index(String index) {
051        if (index != null && !"*".equals(index)) {
052            this.index = new String[]{index};
053        }
054        return this;
055    }
056
057    public BasicRequest index(String... index) {
058        this.index = index;
059        return this;
060    }
061
062    public String index() {
063        return index[0];
064    }
065
066    public BasicRequest type(String type) {
067        if (type != null && !"*".equals(type)) {
068            this.type = new String[]{type};
069        }
070        return this;
071    }
072
073    public BasicRequest type(String... type) {
074        this.type = type;
075        return this;
076    }
077
078    public String type() {
079        return type[0];
080    }
081
082    public BasicRequest id(String id) {
083        this.id = id;
084        return this;
085    }
086
087    public String id() {
088        return id;
089    }
090
091    public BasicRequest from(int from) {
092        searchRequestBuilder.setFrom(from);
093        return this;
094    }
095
096    public BasicRequest size(int size) {
097        searchRequestBuilder.setSize(size);
098        return this;
099    }
100
101    public BasicRequest filter(String filter) {
102        searchRequestBuilder.setFilter(filter);
103        return this;
104    }
105
106    public BasicRequest facets(String facets) {
107        searchRequestBuilder.setFacets(facets.getBytes());
108        return this;
109    }
110
111    public BasicRequest timeout(TimeValue timeout) {
112        searchRequestBuilder.setTimeout(timeout);
113        return this;
114    }
115
116    public BasicRequest query(String query) {
117        this.query = query == null || query.trim().length() == 0 ? "{\"query\":{\"match_all\":{}}}" : query;
118        return this;
119    }
120
121    public BasicResponse execute()
122            throws IOException {
123        BasicResponse response = new BasicResponse();
124        if (searchRequestBuilder == null) {
125            return response;
126        }
127        if (query == null) {
128            return response;
129        }
130        if (hasIndex(index)) {
131            searchRequestBuilder.setIndices(fixIndexName(index));
132        }
133        if (hasType(type)) {
134            searchRequestBuilder.setTypes(type);
135        }
136        long t0 = System.currentTimeMillis();
137        response.searchResponse(searchRequestBuilder.setExtraSource(query)
138                .execute().actionGet());
139        long t1 = System.currentTimeMillis();
140        logger.info(" [{}] [{}ms] [{}ms] [{}] [{}]",
141                formatIndexType(), t1 - t0, response.tookInMillis(), response.totalHits(), query);
142        return response;
143    }
144
145    public BasicResponse executeGet() throws IOException {
146        BasicResponse response = new BasicResponse();
147        if (getRequestBuilder == null) {
148            return response;
149        }
150        getRequestBuilder
151                .setIndex(index[0])
152                .setType(type[0])
153                .setId(id);
154        long t0 = System.currentTimeMillis();
155        response.getResponse(getRequestBuilder.execute().actionGet());
156        long t1 = System.currentTimeMillis();
157        logger.info(" get complete: {}/{}/{} [{}ms] {}",
158                getRequestBuilder.request().index(),
159                getRequestBuilder.request().type(),
160                getRequestBuilder.request().id(),
161                (t1 - t0), response.exists());
162        return response;
163    }
164
165    private boolean hasIndex(String[] s) {
166        if (s == null) {
167            return false;
168        }
169        if (s.length == 0) {
170            return false;
171        }
172        if (s[0] == null) {
173            return false;
174        }
175        return true;
176    }
177
178    private boolean hasType(String[] s) {
179        if (s == null) {
180            return false;
181        }
182        if (s.length == 0) {
183            return false;
184        }
185        if (s[0] == null) {
186            return false;
187        }
188        return true;
189    }
190
191    private String[] fixIndexName(String[] s) {
192        if (s == null) {
193            return new String[]{"*"};
194        }
195        if (s.length == 0) {
196            return new String[]{"*"};
197        }
198        for (int i = 0; i < s.length; i++) {
199            if (s[i] == null || s[i].length() == 0) {
200                s[i] = "*";
201            }
202        }
203        return s;
204    }
205
206    private String formatIndexType() {
207        StringBuilder indexes = new StringBuilder();
208        if (index != null) {
209            for (String s : index) {
210                if (s != null && s.length() > 0) {
211                    if (indexes.length() > 0) {
212                        indexes.append(',');
213                    }
214                    indexes.append(s);
215                }
216            }
217        }
218        if (indexes.length() == 0) {
219            indexes.append('*');
220        }
221        StringBuilder types = new StringBuilder();
222        if (type != null) {
223            for (String s : type) {
224                if (s != null && s.length() > 0) {
225                    if (types.length() > 0) {
226                        types.append(',');
227                    }
228                    types.append(s);
229                }
230            }
231        }
232        if (types.length() == 0) {
233            types.append('*');
234        }
235        return indexes.append("/").append(types).toString();
236    }

238}
