001package org.xbib.elasticsearch.action.delete;
002
003import org.elasticsearch.action.ActionRequestValidationException;
004import org.elasticsearch.common.Nullable;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007import org.elasticsearch.common.lucene.uid.Versions;
008import org.elasticsearch.index.VersionType;
009import org.xbib.elasticsearch.action.support.replication.leader.LeaderShardOperationRequest;
010
011import java.io.IOException;
012
013import static org.elasticsearch.action.ValidateActions.addValidationError;
014
015public class DeleteRequest extends LeaderShardOperationRequest<DeleteRequest> {
016
017    private String type;
018    private String id;
019    @Nullable
020    private String routing;
021    private boolean refresh;
022    private long version = Versions.MATCH_ANY;
023    private VersionType versionType = VersionType.INTERNAL;
024
025    public DeleteRequest() {
026    }
027
028    public DeleteRequest(String index) {
029        this.index = index;
030    }
031
032    public DeleteRequest(String index, String type, String id) {
033        this.index = index;
034        this.type = type;
035        this.id = id;
036    }
037
038    public DeleteRequest(DeleteRequest request) {
039        super(request);
040        this.type = request.type();
041        this.id = request.id();
042        this.routing = request.routing();
043        this.refresh = request.refresh();
044        this.version = request.version();
045        this.versionType = request.versionType();
046    }
047
048    public DeleteRequest(org.elasticsearch.action.delete.DeleteRequest indexRequest) {
049        this.index = indexRequest.index();
050        this.type = indexRequest.type();
051        this.id = indexRequest.id();
052        this.routing = indexRequest.routing();
053        this.version = indexRequest.version();
054        this.versionType = indexRequest.versionType();
055        this.refresh = indexRequest.refresh();
056    }
057    @Override
058    public ActionRequestValidationException validate() {
059        ActionRequestValidationException validationException = super.validate();
060        if (type == null) {
061            validationException = addValidationError("type is missing", validationException);
062        }
063        if (id == null) {
064            validationException = addValidationError("id is missing", validationException);
065        }
066        if (!versionType.validateVersionForWrites(version)) {
067            validationException = addValidationError("illegal version value [" + version + "] for version type [" + versionType.name() + "]", validationException);
068        }
069        return validationException;
070    }
071
072    public String type() {
073        return type;
074    }
075
076    public DeleteRequest type(String type) {
077        this.type = type;
078        return this;
079    }
080
081    public String id() {
082        return id;
083    }
084
085    public DeleteRequest id(String id) {
086        this.id = id;
087        return this;
088    }
089
090    public DeleteRequest parent(String parent) {
091        if (routing == null) {
092            routing = parent;
093        }
094        return this;
095    }
096
097    public DeleteRequest routing(String routing) {
098        if (routing != null && routing.length() == 0) {
099            this.routing = null;
100        } else {
101            this.routing = routing;
102        }
103        return this;
104    }
105
106    public String routing() {
107        return this.routing;
108    }
109
110    public DeleteRequest refresh(boolean refresh) {
111        this.refresh = refresh;
112        return this;
113    }
114
115    public boolean refresh() {
116        return this.refresh;
117    }
118
119    public DeleteRequest version(long version) {
120        this.version = version;
121        return this;
122    }
123
124    public long version() {
125        return this.version;
126    }
127
128    public DeleteRequest versionType(VersionType versionType) {
129        this.versionType = versionType;
130        return this;
131    }
132
133    public VersionType versionType() {
134        return this.versionType;
135    }
136
137    @Override
138    public void readFrom(StreamInput in) throws IOException {
139        super.readFrom(in);
140        type = in.readString();
141        id = in.readString();
142        routing = in.readOptionalString();
143        refresh = in.readBoolean();
144        version = Versions.readVersion(in);
145        versionType = VersionType.fromValue(in.readByte());
146    }
147
148    @Override
149    public void writeTo(StreamOutput out) throws IOException {
150        super.writeTo(out);
151        out.writeString(type);
152        out.writeString(id);
153        out.writeOptionalString(routing());
154        out.writeBoolean(refresh);
155        Versions.writeVersion(version, out);
156        out.writeByte(versionType.getValue());
157    }
158
159    @Override
160    public String toString() {
161        return "delete {[" + index + "][" + type + "][" + id + "]}";
162    }
163}
