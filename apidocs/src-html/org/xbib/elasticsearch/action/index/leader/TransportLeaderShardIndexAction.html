001package org.xbib.elasticsearch.action.index.leader;
002
003import org.elasticsearch.ExceptionsHelper;
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.RoutingMissingException;
006import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
007import org.elasticsearch.action.admin.indices.create.CreateIndexResponse;
008import org.elasticsearch.action.admin.indices.create.TransportCreateIndexAction;
009import org.elasticsearch.action.support.AutoCreateIndex;
010import org.elasticsearch.cluster.ClusterService;
011import org.elasticsearch.cluster.ClusterState;
012import org.elasticsearch.cluster.action.index.MappingUpdatedAction;
013import org.elasticsearch.cluster.action.shard.ShardStateAction;
014import org.elasticsearch.cluster.block.ClusterBlockException;
015import org.elasticsearch.cluster.block.ClusterBlockLevel;
016import org.elasticsearch.cluster.metadata.IndexMetaData;
017import org.elasticsearch.cluster.metadata.MappingMetaData;
018import org.elasticsearch.cluster.metadata.MetaData;
019import org.elasticsearch.cluster.routing.ShardIterator;
020import org.elasticsearch.common.inject.Inject;
021import org.elasticsearch.common.settings.Settings;
022import org.elasticsearch.index.engine.Engine;
023import org.elasticsearch.index.mapper.SourceToParse;
024import org.elasticsearch.index.shard.service.IndexShard;
025import org.elasticsearch.indices.IndexAlreadyExistsException;
026import org.elasticsearch.indices.IndicesService;
027import org.elasticsearch.threadpool.ThreadPool;
028import org.elasticsearch.transport.TransportService;
029import org.xbib.elasticsearch.action.index.IndexAction;
030import org.xbib.elasticsearch.action.index.IndexRequest;
031import org.xbib.elasticsearch.action.index.IndexResponse;
032import org.xbib.elasticsearch.action.support.replication.leader.TransportLeaderShardOperationAction;
033
034public class TransportLeaderShardIndexAction extends TransportLeaderShardOperationAction<IndexRequest, IndexResponse> {
035
036    private final AutoCreateIndex autoCreateIndex;
037
038    private final boolean allowIdGeneration;
039
040    private final TransportCreateIndexAction createIndexAction;
041
042    private final MappingUpdatedAction mappingUpdatedAction;
043
044    @Inject
045    public TransportLeaderShardIndexAction(Settings settings, TransportService transportService, ClusterService clusterService,
046                                           IndicesService indicesService, ThreadPool threadPool, ShardStateAction shardStateAction,
047                                           TransportCreateIndexAction createIndexAction, MappingUpdatedAction mappingUpdatedAction) {
048        super(settings, IndexAction.NAME, transportService, clusterService, indicesService, threadPool, shardStateAction);
049        this.createIndexAction = createIndexAction;
050        this.mappingUpdatedAction = mappingUpdatedAction;
051        this.autoCreateIndex = new AutoCreateIndex(settings);
052        this.allowIdGeneration = settings.getAsBoolean("action.allow_id_generation", true);
053    }
054
055    @Override
056    protected void doExecute(final IndexRequest request, final ActionListener<IndexResponse> listener) {
057        if (autoCreateIndex.shouldAutoCreate(request.index(), clusterService.state())) {
058            request.beforeLocalFork();
059            createIndexAction.execute(new CreateIndexRequest(request.index()).cause("auto(index api)").masterNodeTimeout(request.timeout()), new ActionListener<CreateIndexResponse>() {
060                @Override
061                public void onResponse(CreateIndexResponse result) {
062                    innerExecute(request, listener);
063                }
064
065                @Override
066                public void onFailure(Throwable e) {
067                    if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
068                        try {
069                            innerExecute(request, listener);
070                        } catch (Throwable e1) {
071                            listener.onFailure(e1);
072                        }
073                    } else {
074                        listener.onFailure(e);
075                    }
076                }
077            });
078        } else {
079            innerExecute(request, listener);
080        }
081    }
082
083    @Override
084    protected boolean resolveRequest(ClusterState state, IndexRequest request, ActionListener<IndexResponse> indexResponseActionListener) {
085        MetaData metaData = clusterService.state().metaData();
086        String aliasOrIndex = request.index();
087        request.index(metaData.concreteSingleIndex(request.index()));
088        MappingMetaData mappingMd = null;
089        if (metaData.hasIndex(request.index())) {
090            mappingMd = metaData.index(request.index()).mappingOrDefault(request.type());
091        }
092        request.process(metaData, aliasOrIndex, mappingMd, allowIdGeneration);
093        return true;
094    }
095
096    private void innerExecute(final IndexRequest request, final ActionListener<IndexResponse> listener) {
097        super.doExecute(request, listener);
098    }
099
100    @Override
101    protected IndexRequest newRequestInstance() {
102        return new IndexRequest();
103    }
104
105    @Override
106    protected IndexResponse newResponseInstance() {
107        return new IndexResponse();
108    }
109
110    @Override
111    protected String transportAction() {
112        return IndexAction.NAME;
113    }
114
115    @Override
116    protected String executor() {
117        return ThreadPool.Names.INDEX;
118    }
119
120    @Override
121    protected ClusterBlockException checkGlobalBlock(ClusterState state, IndexRequest request) {
122        return state.blocks().globalBlockedException(ClusterBlockLevel.WRITE);
123    }
124
125    @Override
126    protected ClusterBlockException checkRequestBlock(ClusterState state, IndexRequest request) {
127        return state.blocks().indexBlockedException(ClusterBlockLevel.WRITE, request.index());
128    }
129
130    @Override
131    protected ShardIterator shards(ClusterState clusterState, IndexRequest request) {
132        return clusterService.operationRouting()
133                .indexShards(clusterService.state(), request.index(), request.type(), request.id(), request.routing());
134    }
135
136    @Override
137    protected IndexResponse shardOperationOnLeader(ClusterState clusterState, int replicaLevel, LeaderOperationRequest shardRequest) {
138        final IndexRequest request = shardRequest.request();
139        IndexMetaData indexMetaData = clusterState.metaData().index(request.index());
140        MappingMetaData mappingMd = indexMetaData.mappingOrDefault(request.type());
141        if (mappingMd != null && mappingMd.routing().required()) {
142            if (request.routing() == null) {
143                throw new RoutingMissingException(request.index(), request.type(), request.id());
144            }
145        }
146        IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request().index()).shardSafe(shardRequest.shardId());
147        SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, request.source()).type(request.type()).id(request.id())
148                .routing(request.routing()).parent(request.parent()).timestamp(request.timestamp()).ttl(request.ttl());
149        long version;
150        if (request.opType() == IndexRequest.OpType.INDEX) {
151            Engine.Index index = indexShard.prepareIndex(sourceToParse, request.version(), request.versionType(), Engine.Operation.Origin.PRIMARY, false);
152            if (index.parsedDoc().mappingsModified()) {
153                mappingUpdatedAction.updateMappingOnMaster(request.index(), index.docMapper(), indexMetaData.getUUID());
154            }
155            indexShard.index(index);
156            version = index.version();
157        } else {
158            Engine.Create create = indexShard.prepareCreate(sourceToParse,
159                    request.version(), request.versionType(), Engine.Operation.Origin.PRIMARY, false, request.autoGeneratedId());
160            if (create.parsedDoc().mappingsModified()) {
161                mappingUpdatedAction.updateMappingOnMaster(request.index(), create.docMapper(), indexMetaData.getUUID());
162            }
163            indexShard.create(create);
164            version = create.version();
165        }
166        if (request.refresh()) {
167            try {
168                indexShard.refresh(new Engine.Refresh("refresh_flag_index").force(false));
169            } catch (Exception e) {
170                // ignore?
171            }
172        }
173        request.version(version);
174        request.versionType(request.versionType().versionTypeForReplicationAndRecovery());
175
176        assert request.versionType().validateVersionForWrites(request.version());
177
178        int quorumShards = findQuorum(clusterState, shards(clusterState, request), request);
179
180        return new IndexResponse()
181            .setIndex(request.index())
182            .setType(request.type())
183            .setId(request.id())
184            .setVersion(version)
185            .setQuorumShards(quorumShards);
186    }
