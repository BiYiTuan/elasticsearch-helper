001package org.xbib.elasticsearch.action.index;
002
003import org.elasticsearch.action.ClientAction;
004import org.elasticsearch.client.Client;
005
006public class IndexAction extends ClientAction<IndexRequest, IndexResponse, IndexRequestBuilder> {
007
008    public static final IndexAction INSTANCE = new IndexAction();
009    public static final String NAME = "index";
010
011    private IndexAction() {
012        super(NAME);
013    }
014
015    @Override
016    public IndexResponse newResponse() {
017        return new IndexResponse();
018    }
019
020    @Override
021    public IndexRequestBuilder newRequestBuilder(Client client) {
022        return new IndexRequestBuilder(client);
023    }
024}
