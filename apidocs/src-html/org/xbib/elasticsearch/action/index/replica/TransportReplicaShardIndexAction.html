001package org.xbib.elasticsearch.action.index.replica;
002
003import org.elasticsearch.cluster.ClusterService;
004import org.elasticsearch.cluster.ClusterState;
005import org.elasticsearch.cluster.action.shard.ShardStateAction;
006import org.elasticsearch.cluster.block.ClusterBlockException;
007import org.elasticsearch.cluster.block.ClusterBlockLevel;
008import org.elasticsearch.cluster.routing.ShardIterator;
009import org.elasticsearch.common.inject.Inject;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.index.engine.Engine;
012import org.elasticsearch.index.mapper.SourceToParse;
013import org.elasticsearch.index.shard.service.IndexShard;
014import org.elasticsearch.indices.IndicesService;
015import org.elasticsearch.threadpool.ThreadPool;
016import org.elasticsearch.transport.TransportService;
017import org.xbib.elasticsearch.action.index.IndexAction;
018import org.xbib.elasticsearch.action.index.IndexRequest;
019import org.xbib.elasticsearch.action.support.replication.replica.TransportReplicaShardOperationAction;
020
021public class TransportReplicaShardIndexAction
022        extends TransportReplicaShardOperationAction<IndexReplicaShardRequest, IndexReplicaShardResponse, TransportReplicaShardOperationAction.ReplicaOperationResponse> {
023
024    @Inject
025    public TransportReplicaShardIndexAction(Settings settings, TransportService transportService, ClusterService clusterService,
026                                             IndicesService indicesService, ThreadPool threadPool, ShardStateAction shardStateAction) {
027        super(settings, IndexAction.NAME, transportService, clusterService, indicesService, threadPool, shardStateAction);
028    }
029
030    @Override
031    protected IndexReplicaShardRequest newRequestInstance() {
032        return new IndexReplicaShardRequest();
033    }
034
035    @Override
036    protected IndexReplicaShardResponse newResponseInstance() {
037        return new IndexReplicaShardResponse();
038    }
039
040    @Override
041    protected String transportAction() {
042        return IndexAction.NAME;
043    }
044
045    @Override
046    protected String executor() {
047        return ThreadPool.Names.INDEX;
048    }
049
050    @Override
051    protected ClusterBlockException checkGlobalBlock(ClusterState state, IndexReplicaShardRequest request) {
052        return state.blocks().globalBlockedException(ClusterBlockLevel.WRITE);
053    }
054
055    @Override
056    protected ClusterBlockException checkRequestBlock(ClusterState state, IndexReplicaShardRequest request) {
057        return state.blocks().indexBlockedException(ClusterBlockLevel.WRITE, request.index());
058    }
059
060    @Override
061    protected ShardIterator shards(ClusterState clusterState, IndexReplicaShardRequest request) {
062        return clusterService.operationRouting()
063                .indexShards(clusterService.state(), request.request().index(), request.request().type(),
064                        request.request().id(), request.request().routing());
065    }
066
067    @Override
068    protected ReplicaOperationResponse newReplicaResponseInstance() {
069        return new ReplicaOperationResponse();
070    }
071
072    @Override
073    protected IndexReplicaShardResponse shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
074        final long t0 = shardRequest.startTime();
075        IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request().index()).shardSafe(shardRequest.shardId());
076        IndexRequest request = shardRequest.request().request();
077        SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.REPLICA, request.source()).type(request.type()).id(request.id())
078                .routing(request.routing()).parent(request.parent()).timestamp(request.timestamp()).ttl(request.ttl());
079        if (request.opType() == IndexRequest.OpType.INDEX) {
080            Engine.Index index = indexShard.prepareIndex(sourceToParse, request.version(), request.versionType(), Engine.Operation.Origin.REPLICA, false);
081            indexShard.index(index);
082        } else {
083            Engine.Create create = indexShard.prepareCreate(sourceToParse,
084                    request.version(), request.versionType(), Engine.Operation.Origin.REPLICA, false, request.autoGeneratedId());
085            indexShard.create(create);
086        }
087        if (request.refresh()) {
088            try {
089                indexShard.refresh(new Engine.Refresh("refresh_flag_index").force(false));
090            } catch (Exception e) {
091                // ignore?
092            }
093        }
094        return new IndexReplicaShardResponse(shardRequest.request().shardId(), shardRequest.replicaId(),
095                System.currentTimeMillis() - t0);
096    }
097}
