002package org.xbib.elasticsearch.action.ingest.create;
003
004import org.elasticsearch.ExceptionsHelper;
005import org.elasticsearch.action.ActionListener;
006import org.elasticsearch.action.ActionRequest;
007import org.elasticsearch.action.index.IndexRequest;
008import org.elasticsearch.action.support.TransportAction;
009import org.elasticsearch.cluster.ClusterService;
010import org.elasticsearch.cluster.ClusterState;
011import org.elasticsearch.cluster.block.ClusterBlockLevel;
012import org.elasticsearch.cluster.metadata.MappingMetaData;
013import org.elasticsearch.cluster.metadata.MetaData;
014import org.elasticsearch.common.inject.Inject;
015import org.elasticsearch.common.settings.Settings;
016import org.elasticsearch.index.shard.ShardId;
017import org.elasticsearch.threadpool.ThreadPool;
018import org.elasticsearch.transport.BaseTransportRequestHandler;
019import org.elasticsearch.transport.TransportChannel;
020import org.elasticsearch.transport.TransportService;
021
022import org.xbib.elasticsearch.action.ingest.IngestItemFailure;
023import org.xbib.elasticsearch.action.ingest.IngestResponse;
024
025import java.util.List;
026import java.util.Map;
027import java.util.concurrent.atomic.AtomicInteger;
028
029import static org.elasticsearch.common.collect.Lists.newLinkedList;
030import static org.elasticsearch.common.collect.Maps.newHashMap;
031
032public class TransportIngestCreateAction extends TransportAction<IngestCreateRequest, IngestResponse> {
033
034    private final boolean allowIdGeneration;
035
036    private final ClusterService clusterService;
037
038    private final TransportShardIngestCreateAction shardBulkAction;
039
040    @Inject
041    public TransportIngestCreateAction(Settings settings, ThreadPool threadPool, TransportService transportService, ClusterService clusterService,
042                                       TransportShardIngestCreateAction shardBulkAction) {
043        super(settings, threadPool);
044        this.clusterService = clusterService;
045        this.shardBulkAction = shardBulkAction;
046
047        this.allowIdGeneration = componentSettings.getAsBoolean("action.allow_id_generation", true);
048
049        transportService.registerHandler(IngestCreateAction.NAME, new IngestTransportHandler());
050    }
051
052    @Override
053    protected void doExecute(final IngestCreateRequest ingestRequest, final ActionListener<IngestResponse> listener) {
054        final long startTime = System.currentTimeMillis();
055        executeBulk(ingestRequest, startTime, listener);
056    }
057
058    private void executeBulk(final IngestCreateRequest ingestRequest, final long startTime, final ActionListener<IngestResponse> listener) {
059        ClusterState clusterState = clusterService.state();
060        // TODO use timeout to wait here if its blocked...
061        clusterState.blocks().globalBlockedRaiseException(ClusterBlockLevel.WRITE);
062
063        MetaData metaData = clusterState.metaData();
064        for (ActionRequest request : ingestRequest.requests()) {
065            IndexRequest indexRequest = (IndexRequest) request;
066            String aliasOrIndex = indexRequest.index();
067            indexRequest.index(clusterState.metaData().concreteIndex(indexRequest.index()));
068
069            MappingMetaData mappingMd = null;
070            if (metaData.hasIndex(indexRequest.index())) {
071                mappingMd = metaData.index(indexRequest.index()).mappingOrDefault(indexRequest.type());
072            }
073            indexRequest.process(metaData, aliasOrIndex, mappingMd, allowIdGeneration);
074        }
075
076        // first, go over all the requests and create a ShardId -> Operations mapping
077        Map<ShardId, List<IngestCreateItemRequest>> requestsByShard = newHashMap();
078        int i = 0;
079        for (ActionRequest request : ingestRequest.requests()) {
080            IndexRequest indexRequest = (IndexRequest) request;
081            ShardId shardId = clusterService.operationRouting().indexShards(clusterState, indexRequest.index(), indexRequest.type(), indexRequest.id(), indexRequest.routing()).shardId();
082            List<IngestCreateItemRequest> list = requestsByShard.get(shardId);
083            if (list == null) {
084                list = newLinkedList();
085                requestsByShard.put(shardId, list);
086            }
087            list.add(new IngestCreateItemRequest(i, request));
088            i++;
089        }
090
091        final AtomicInteger successSize = new AtomicInteger(0);
092        final List<IngestItemFailure> failure = newLinkedList();
093
094        if (requestsByShard.isEmpty()) {
095            listener.onResponse(new IngestResponse(0, failure, System.currentTimeMillis() - startTime));
096            return;
097        }
098
099        final AtomicInteger counter = new AtomicInteger(requestsByShard.size());
100        for (Map.Entry<ShardId, List<IngestCreateItemRequest>> entry : requestsByShard.entrySet()) {
101            final ShardId shardId = entry.getKey();
102            final List<IngestCreateItemRequest> requests = entry.getValue();
103            IngestCreateShardRequest ingestShardRequest = new IngestCreateShardRequest(shardId.index().name(), shardId.id(), requests);
104            ingestShardRequest.replicationType(ingestRequest.replicationType());
105            ingestShardRequest.consistencyLevel(ingestRequest.consistencyLevel());
106            shardBulkAction.execute(ingestShardRequest, new ActionListener<IngestCreateShardResponse>() {
107                @Override
108                public void onResponse(IngestCreateShardResponse ingestShardResponse) {
109                    successSize.addAndGet(ingestShardResponse.successSize());
110                    if (counter.decrementAndGet() == 0) {
111                        finishHim();
112                    }
113                }
114
115                @Override
116                public void onFailure(Throwable e) {
117                    // create failures for all relevant requests
118                    String message = ExceptionsHelper.detailedMessage(e);
119                    synchronized (failure) {
120                        for (IngestCreateItemRequest request : requests) {
121                            failure.add(new IngestItemFailure(request.id(), message));
122                        }
123                    }
124                    if (counter.decrementAndGet() == 0) {
125                        finishHim();
126                    }
127                }
128
129                private void finishHim() {
130                    listener.onResponse(new IngestResponse(successSize.get(), failure, System.currentTimeMillis() - startTime));
131                }
132            });
133        }
134    }
135
136    class IngestTransportHandler extends BaseTransportRequestHandler<IngestCreateRequest> {
137
138        @Override
139        public IngestCreateRequest newInstance() {
140            return new IngestCreateRequest();
141        }
142
143        @Override
144        public void messageReceived(final IngestCreateRequest request, final TransportChannel channel) throws Exception {
145            // no need to use threaded listener, since we just send a response
146            request.listenerThreaded(false);
147            execute(request, new ActionListener<IngestResponse>() {
148                @Override
149                public void onResponse(IngestResponse result) {
150                    try {
151                        channel.sendResponse(result);
152                    } catch (Exception e) {
153                        onFailure(e);
154                    }
155                }
156
157                @Override
158                public void onFailure(Throwable e) {
159                    try {
160                        channel.sendResponse(e);
161                    } catch (Exception e1) {
162                        logger.warn("Failed to send error response for request [" + request + "]", e1);
163                    }
164                }
165            });
166        }
167
168        @Override
169        public String executor() {
170            return ThreadPool.Names.SAME;
171        }
172    }
173}
