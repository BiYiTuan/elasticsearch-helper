126    protected IngestLeaderShardResponse shardOperationOnLeader(ClusterState clusterState, int replicaLevel, LeaderOperationRequest shardRequest) {
127        final long t0 = shardRequest.startTime();
128        final IngestLeaderShardRequest request = shardRequest.request();
129        int successCount = 0;
130        List<IngestActionFailure> failures = new LinkedList<>();
131        int size = request.getActionRequests().size();
132        long[] versions = new long[size];
133        IndexService indexService = indicesService.indexServiceSafe(request.index());
134        for (int i = 0; i < size; i++) {
135            ActionRequest actionRequest = request.getActionRequests().get(i);
136            if (actionRequest instanceof IndexRequest) {
137                try {
138                    IndexRequest indexRequest = (IndexRequest) actionRequest;
139                    long version = indexOperationOnLeader(clusterState, indexRequest, request);
140                    versions[i] = indexRequest.version();
141                    indexRequest.version(indexRequest.version() == Versions.MATCH_ANY ? Versions.MATCH_ANY : version);
142                    successCount++;
143                } catch (Throwable e) {
144                    if (retryLeaderException(e)) {
145                        for (int j = 0; j < i; j++) {
146                            ((IndexRequest) request.getActionRequests().get(j)).version(versions[j]);
147                        }
148                        logger.error(e.getMessage(), e);
149                        throw new ElasticsearchException(e.getMessage(), e);
150                    }
151                    logger.error("[{}][{}] failed to execute ingest (index) {}", e, request.index(), shardRequest.shardId(), actionRequest);
152                    failures.add(new IngestActionFailure(request.getIngestId(), request.getShardId(), ExceptionsHelper.detailedMessage(e)));
153                    request.getActionRequests().set(i, null);
154                }
155            } else if (actionRequest instanceof DeleteRequest) {
156                try {
157                    IndexShard indexShard = indexService.shardSafe(shardRequest.shardId());
158                    DeleteRequest deleteRequest = (DeleteRequest) actionRequest;
159                    Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version(), deleteRequest.versionType(), Engine.Operation.Origin.PRIMARY);
160                    indexShard.delete(delete);
161                    deleteRequest.version(deleteRequest.version() == Versions.MATCH_ANY ? Versions.MATCH_ANY : delete.version());
162                    successCount++;
163                } catch (Throwable e) {
164                    if (retryLeaderException(e)) {
165                        for (int j = 0; j < i; j++) {
166                            ((DeleteRequest) request.getActionRequests().get(j)).version(versions[j]);
167                        }
168                        logger.error(e.getMessage(), e);
169                        throw new ElasticsearchException(e.getMessage(), e);
170                    }
171                    logger.error("[{}][{}] failed to execute ingest (delete) {}", e, request.index(), shardRequest.shardId(), actionRequest);
172                    failures.add(new IngestActionFailure(request.getIngestId(), request.getShardId(), ExceptionsHelper.detailedMessage(e)));
173                    request.getActionRequests().set(i, null);
174                }
175            }
176        }
177        int quorumShards = findQuorum(clusterState, shards(clusterState, request), request);
178        return new IngestLeaderShardResponse()
179                .setTookInMillis(System.currentTimeMillis() - t0)
180                .setIngestId(request.getIngestId())
181                .setShardId(request.getShardId())
182                .setSuccessCount(successCount)
183                .setQuorumShards(quorumShards)
184                .setActionRequests(request.getActionRequests())
185                .setFailures(failures);
186    }

188    private long indexOperationOnLeader(ClusterState clusterState, IndexRequest indexRequest,
189                                        IngestLeaderShardRequest request) {
190            MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
191            if (mappingMd != null && mappingMd.routing().required() && indexRequest.routing() == null) {
192                throw new RoutingMissingException(indexRequest.index(), indexRequest.type(), indexRequest.id());
193            }
194            SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, indexRequest.source())
195                    .type(indexRequest.type())
196                    .id(indexRequest.id())
197                    .routing(indexRequest.routing())
198                    .parent(indexRequest.parent())
199                    .timestamp(indexRequest.timestamp())
200                    .ttl(indexRequest.ttl());
201            IndexShard indexShard = indicesService.indexServiceSafe(request.index()).shardSafe(request.getShardId().id());
202            if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
203                Engine.Index index = indexShard.prepareIndex(sourceToParse,
204                        indexRequest.version(),
205                        indexRequest.versionType(),
206                        Engine.Operation.Origin.PRIMARY,
207                        false);
208                Mapping mapping = index.parsedDoc().dynamicMappingsUpdate();
209                if (mapping != null) {
210                    try {
211                        mappingUpdatedAction.updateMappingOnMasterSynchronously(indexRequest.index(), indexRequest.type(), mapping);
212                    } catch (Throwable t) {
213                        logger.error("mapping update on master failed on {}/{}", indexRequest.index(), indexRequest.type());
214                    }
215                }
216                indexShard.index(index);
217                return index.version();
218            } else if (indexRequest.opType() == IndexRequest.OpType.CREATE) {
219                Engine.Create create = indexShard.prepareCreate(sourceToParse,
220                        indexRequest.version(),
221                        indexRequest.versionType(),
222                        Engine.Operation.Origin.PRIMARY,
223                        false,
224                        indexRequest.autoGeneratedId());
225                Mapping mapping = create.parsedDoc().dynamicMappingsUpdate();
226                if (mapping != null) {
227                    try {
228                        mappingUpdatedAction.updateMappingOnMasterSynchronously(indexRequest.index(), indexRequest.type(), mapping);
229                    } catch (Throwable t) {
230                        logger.error("mapping update on master failed on {}/{}", indexRequest.index(), indexRequest.type());
231                    }
232                }
233                indexShard.create(create);
234                return create.version();
235            } else {
236                logger.error("unknown op type " + indexRequest.opType());
237                return 0;
238            }

256    protected class AsyncShardOperationAction {
257
258        private final ActionListener<IngestLeaderShardResponse> listener;
259
260        private final IngestLeaderShardRequest request;
261
262        private volatile ShardIterator shardIt;
263
264        private final AtomicBoolean leaderOperationStarted = new AtomicBoolean();
265
266        private volatile ClusterStateObserver observer;
267
268        AsyncShardOperationAction(IngestLeaderShardRequest request, ActionListener<IngestLeaderShardResponse> listener) {
269            this.request = request;
270            this.listener = listener;
271        }
272
273        public void start() {
274            observer = new ClusterStateObserver(clusterService, request.timeout(), logger);
275            doStart();
276        }
277
278        protected boolean doStart() throws ElasticsearchException {
279            try {
280                ClusterBlockException blockException = checkGlobalBlock(observer.observedState(), request);
281                if (blockException != null) {
282                    if (blockException.retryable()) {
283                        if (logger.isTraceEnabled()) {
284                            logger.trace("cluster is blocked ({}), scheduling a retry", blockException.getMessage());
285                        }
286                        retry(blockException);
287                        return false;
288                    } else {
289                        throw blockException;
290                    }
291                }
292                if (!resolveRequest(observer.observedState(), request, listener)) {
293                    return true;
294                }
295                blockException = checkRequestBlock(observer.observedState(), request);
296                if (blockException != null) {
297                    if (blockException.retryable()) {
298                        if (logger.isTraceEnabled()) {
299                            logger.trace("cluster is blocked ({}), scheduling a retry", blockException.getMessage());
300                        }
301                        retry(blockException);
302                        return false;
303                    } else {
304                        throw blockException;
305                    }
306                }
307                shardIt = shards(observer.observedState(), request);
308            } catch (Throwable e) {
309                logger.error(e.getMessage(), e);
310                listener.onFailure(e);
311                return true;
312            }
313            if (shardIt.size() == 0) {
314                if (logger.isTraceEnabled()) {
315                    logger.trace("no shard instances known for shard [{}], scheduling a retry", shardIt.shardId());
316                }
317                retry(null);
318                return false;
319            }
320            final int quorum = findQuorum(observer.observedState(), shardIt, request);
321            if (quorum < 0) {
322                if (logger.isTraceEnabled()) {
323                    logger.trace("quorum not fulfilled for index {}, scheduling a retry", request.index());
324                }
325                retry(null);
326                return false;
327            }
328            shardIt.reset();
329            boolean foundLeader = false;
330            ShardRouting shardRouting;
331            while ((shardRouting = shardIt.nextOrNull()) != null) {
332                final ShardRouting shard = shardRouting;
333                if (!shard.primary()) {
334                    continue;
335                }
336                if (!shard.active() || !observer.observedState().nodes().nodeExists(shard.currentNodeId())) {
337                    if (logger.isTraceEnabled()) {
338                        logger.trace("leader shard [{}] is not yet active or we do not know the node it is assigned to [{}], scheduling a retry",
339                                shard.shardId(), shard.currentNodeId());
340                    }
341                    retry(null);
342                    return false;
343                }
344                if (!leaderOperationStarted.compareAndSet(false, true)) {
345                    return true;
346                }
347                foundLeader = true;
348                if (shard.currentNodeId().equals(observer.observedState().nodes().localNodeId())) {
349                    try {
350                        if (request.operationThreaded()) {
351                            threadPool.executor(executor).execute(new Runnable() {
352                                @Override
353                                public void run() {
354                                    try {
355                                        performOnLeader(shard, quorum, observer.observedState());
356                                    } catch (Throwable t) {
357                                        logger.error(t.getMessage(), t);
358                                        listener.onFailure(t);
359                                    }
360                                }
361                            });
362                        } else {
363                            performOnLeader(shard, quorum, observer.observedState());
364                        }
365                    } catch (Throwable t) {
366                        logger.error(t.getMessage(), t);
367                        listener.onFailure(t);
368                    }
369                } else {
370                    DiscoveryNode node = observer.observedState().nodes().get(shard.currentNodeId());
371                    transportService.sendRequest(node, transportAction, request, transportOptions, new BaseTransportResponseHandler<IngestLeaderShardResponse>() {
372
373                        @Override
374                        public IngestLeaderShardResponse newInstance() {
375                            return newResponseInstance();
376                        }
377
378                        @Override
379                        public String executor() {
380                            return ThreadPool.Names.SAME;
381                        }
382
383                        @Override
384                        public void handleResponse(IngestLeaderShardResponse response) {
385                            listener.onResponse(response);
386                        }
387
388                        @Override
389                        public void handleException(TransportException exp) {
390                            logger.error(exp.getMessage(), exp);
391                            if (exp.unwrapCause() instanceof ConnectTransportException
392                                    || exp.unwrapCause() instanceof NodeClosedException
393                                    || retryLeaderException(exp)) {
394                                leaderOperationStarted.set(false);
395                                if (logger.isTraceEnabled()) {
396                                    logger.trace("received an error from node the leader was assigned to ({}), scheduling a retry", exp.getMessage());
397                                }
398                                retry(null);
399                            } else {
400                                listener.onFailure(exp);
401                            }
402                        }
403                    });
404                }
405                break;
406            }
407            if (!foundLeader) {
408                if (logger.isTraceEnabled()) {
409                    logger.trace("couldn't find a leader shard, scheduling for retry");
410                }
411                retry(null);
412                return false;
413            }
414            return true;
415        }
416
417        private void retry(@Nullable final Throwable failure) {
418            if (observer.isTimedOut()) {
419                listener.onFailure(failure);
420                return;
421            }
422            request.operationThreaded(true);
423            observer.waitForNextChange(new ClusterStateObserver.Listener() {
424                @Override
425                public void onNewClusterState(ClusterState state) {
426                    doStart();
427                }
428
429                @Override
430                public void onClusterServiceClose() {
431                    listener.onFailure(new NodeClosedException(clusterService.localNode()));
432                }
433
434                @Override
435                public void onTimeout(TimeValue timeout) {
436                    if (doStart()) {
437                        return;
438                    }
439                    raiseTimeoutFailure(timeout, failure);
440                }
441            });
442        }
443
444        private void raiseTimeoutFailure(TimeValue timeout, @Nullable Throwable failure) {
445            if (failure == null) {
446                if (shardIt == null) {
447                    failure = new UnavailableShardsException(null, "no available shards, timeout waiting for [" + timeout + "], request: " + request.toString());
448                } else {
449                    failure = new UnavailableShardsException(shardIt.shardId(), "[" + shardIt.size() + "] shardIt, [" + shardIt.sizeActive() + "] active : Timeout waiting for [" + timeout + "], request: " + request.toString());
450                }
451            }
452            listener.onFailure(failure);
453        }
454
455        private void performOnLeader(final ShardRouting shard, int quorum, ClusterState clusterState) {
456            try {
457                LeaderOperationRequest leaderOperationRequest = new LeaderOperationRequest(System.currentTimeMillis(), shard.id(), request);
458                IngestLeaderShardResponse response = shardOperationOnLeader(clusterState, quorum, leaderOperationRequest);
459                listener.onResponse(response);
460            } catch (Throwable e) {
461                logger.error(e.getMessage(), e);
462                if (retryLeaderException(e)) {
463                    leaderOperationStarted.set(false);
464                    if (logger.isTraceEnabled()) {
465                        logger.trace("error while performing operation on leader {}, scheduling a retry", e);
466                    }
467                    retry(e);
468                    return;
469                }
470                if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
471                    if (logger.isTraceEnabled()) {
472                        logger.trace(shard.shortSummary() + ": failed to execute [" + request + "]", e);
473                    }
474                } else {
475                    if (logger.isDebugEnabled()) {
476                        logger.debug(shard.shortSummary() + ": failed to execute [" + request + "]", e);
477                    }
478                }
479                listener.onFailure(e);
480            }
481        }
482    }

484    public int findQuorum(ClusterState clusterState, ShardIterator shardIt, IngestLeaderShardRequest request) {
485        if (request.requiredConsistency() == Consistency.IGNORE) {
486            return 0;
487        }
488        int numberOfDataNodes = 0;
489        for (DiscoveryNode node : clusterState.getNodes()) {
490            if (node.isDataNode()) {
491                numberOfDataNodes++;
492            }
493        }
494        // single node, do not care about replica
495        if (numberOfDataNodes == 1) {
496            return 0;
497        }
498        int replicaLevelOfIndex = clusterState.metaData().index(request.index()).getNumberOfReplicas();
499        // no replica defined, so nothing to check
500        if (replicaLevelOfIndex == 0) {
501            return 0;
502        }
503        int replicaLevel = findReplicaLevel(shardIt) + 1;
504        switch (request.requiredConsistency()) {
505            case ONE:
506                if (replicaLevel >= 1 && replicaLevelOfIndex >= 1) {
507                    return 1;
508                }
509                break;
510            case QUORUM:
511                int quorum = (replicaLevelOfIndex / 2) + 1;
512                if (replicaLevel >= quorum) {
513                    return quorum;
514                }
515                break;
516            case ALL:
517                if (replicaLevel == replicaLevelOfIndex) {
518                    return replicaLevel;
519                }
520                break;
521        }
522        // quorum not matched - we have a problem
523        return -1;
524    }
525
526    private int findReplicaLevel(ShardIterator shardIt) {
527        int replicaLevel = 0;
528        shardIt.reset();
529        ShardRouting shard;
530        while ((shard = shardIt.nextOrNull()) != null) {
531            if (shard.unassigned()) {
532                continue;
533            }
534            boolean doOnlyOnRelocating = false;
535            if (shard.primary()) {
536                if (shard.relocating()) {
537                    doOnlyOnRelocating = true;
538                } else {
539                    continue;
540                }
541            }
542            String nodeId = !doOnlyOnRelocating ? shard.currentNodeId() : shard.relocating() ? shard.relocatingNodeId() : null;
543            if (nodeId == null) {
544                continue;
545            }
546            replicaLevel++;
547        }
548        return replicaLevel;
549    }

552    protected class LeaderOperationRequest implements Streamable {
553
554        private long startTime;
555
556        private int shardId;
557
558        private IngestLeaderShardRequest request;
559
560        public LeaderOperationRequest(long startTime, int shardId, IngestLeaderShardRequest request) {
561            this.startTime = startTime;
562            this.shardId = shardId;
563            this.request = request;
564        }
565
566        public long startTime() {
567            return startTime;
568        }
569
570        public int shardId() {
571            return shardId;
572        }
573
574        public IngestLeaderShardRequest request() {
575            return request;
576        }
577
578        @Override
579        public void readFrom(StreamInput in) throws IOException {
580            shardId = in.readVInt();
581            request = newRequestInstance();
582            request.readFrom(in);
583        }
584
585        @Override
586        public void writeTo(StreamOutput out) throws IOException {
587            out.writeVInt(shardId);
588            request.writeTo(out);
589        }
590    }

592    class LeaderOperationTransportHandler implements TransportRequestHandler<IngestLeaderShardRequest> {
593
594        @Override
595        public void messageReceived(final IngestLeaderShardRequest request, final TransportChannel channel) throws Exception {
596            execute(request, new ActionListener<IngestLeaderShardResponse>() {
597                @Override
598                public void onResponse(IngestLeaderShardResponse result) {
599                    try {
600                        channel.sendResponse(result);
601                    } catch (Throwable e) {
602                        onFailure(e);
603                    }
604                }
605
606                @Override
607                public void onFailure(Throwable e) {
608                    try {
609                        channel.sendResponse(e);
610                        logger.error(e.getMessage(), e);
611                    } catch (Throwable e1) {
612                        logger.warn("failed to send response for " + transportAction, e1);
613                    }
614                }
615            });
616        }
