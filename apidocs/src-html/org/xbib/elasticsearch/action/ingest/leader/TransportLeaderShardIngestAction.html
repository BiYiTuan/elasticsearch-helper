041public class TransportLeaderShardIngestAction extends TransportLeaderShardOperationAction<IngestLeaderShardRequest, IngestLeaderShardResponse> {
042
043    private final static ESLogger logger = ESLoggerFactory.getLogger(TransportLeaderShardIngestAction.class.getSimpleName());
044
045    private final MappingUpdatedAction mappingUpdatedAction;
046
047    @Inject
048    public TransportLeaderShardIngestAction(Settings settings, TransportService transportService, ClusterService clusterService,
049                                            IndicesService indicesService, ThreadPool threadPool, ShardStateAction shardStateAction,
050                                            MappingUpdatedAction mappingUpdatedAction) {
051        super(settings, IngestAction.NAME, transportService, clusterService, indicesService, threadPool, shardStateAction);
052        this.mappingUpdatedAction = mappingUpdatedAction;
053    }
054
055    @Override
056    protected String executor() {
057        return ThreadPool.Names.BULK;
058    }
059
060    @Override
061    protected TransportRequestOptions transportOptions() {
062        return IngestAction.INSTANCE.transportOptions(settings);
063    }
064
065    @Override
066    protected IngestLeaderShardRequest newRequestInstance() {
067        return new IngestLeaderShardRequest();
068    }
069
070    @Override
071    protected IngestLeaderShardResponse newResponseInstance() {
072        return new IngestLeaderShardResponse();
073    }
074
075    @Override
076    protected String transportAction() {
077        return IngestAction.NAME + ".shard.leader";
078    }
079
080    @Override
081    protected ClusterBlockException checkGlobalBlock(ClusterState state, IngestLeaderShardRequest request) {
082        return state.blocks().globalBlockedException(ClusterBlockLevel.WRITE);
083    }
084
085    @Override
086    protected ClusterBlockException checkRequestBlock(ClusterState state, IngestLeaderShardRequest request) {
087        return state.blocks().indexBlockedException(ClusterBlockLevel.WRITE, request.index());
088    }
089
090    @Override
091    protected ShardIterator shards(ClusterState clusterState, IngestLeaderShardRequest request) {
092        return clusterState.routingTable().index(request.index()).shard(request.getShardId().id()).shardsIt();
093    }
094
095    @Override
096    protected IngestLeaderShardResponse shardOperationOnLeader(ClusterState clusterState, int replicaLevel, LeaderOperationRequest shardRequest) {
097        final long t0 = shardRequest.startTime();
098        final IngestLeaderShardRequest request = shardRequest.request();
099        int successCount = 0;
100        List<IngestActionFailure> failures = newLinkedList();
101        int size = request.getActionRequests().size();
102        long[] versions = new long[size];
103        Set<String> mappingsToUpdate = newHashSet();
104        IndexService indexService = indicesService.indexServiceSafe(request.index());
105        for (int i = 0; i < size; i++) {
106            ActionRequest actionRequest = request.getActionRequests().get(i);
107            if (actionRequest instanceof IndexRequest) {
108                try {
109                    IndexRequest indexRequest = (IndexRequest) actionRequest;
110                    long version = indexOperationOnLeader(clusterState, indexRequest, request, mappingsToUpdate);
111                    versions[i] = indexRequest.version();
112                    indexRequest.version(indexRequest.version() == Versions.MATCH_ANY ? Versions.MATCH_ANY : version);
113                    successCount++;
114                } catch (Throwable e) {
115                    if (retryLeaderException(e)) {
116                        for (int j = 0; j < i; j++) {
117                            ((IndexRequest) request.getActionRequests().get(j)).version(versions[j]);
118                        }
119                        logger.error(e.getMessage(), e);
120                        throw new ElasticsearchException(e.getMessage(), e);
121                    }
122                    logger.error("[{}][{}] failed to execute ingest (index) {}", e, request.index(), shardRequest.shardId(), actionRequest);
123                    failures.add(new IngestActionFailure(request.getIngestId(), request.getShardId(), ExceptionsHelper.detailedMessage(e)));
124                    request.getActionRequests().set(i, null);
125                }
126            } else if (actionRequest instanceof DeleteRequest) {
127                try {
128                    IndexShard indexShard = indexService.shardSafe(shardRequest.shardId());
129                    DeleteRequest deleteRequest = (DeleteRequest) actionRequest;
130                    Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version(), deleteRequest.versionType(), Engine.Operation.Origin.PRIMARY);
131                    indexShard.delete(delete);
132                    deleteRequest.version(deleteRequest.version() == Versions.MATCH_ANY ? Versions.MATCH_ANY : delete.version());
133                    successCount++;
134                } catch (Throwable e) {
135                    if (retryLeaderException(e)) {
136                        for (int j = 0; j < i; j++) {
137                            ((DeleteRequest) request.getActionRequests().get(j)).version(versions[j]);
138                        }
139                        logger.error(e.getMessage(), e);
140                        throw new ElasticsearchException(e.getMessage(), e);
141                    }
142                    logger.error("[{}][{}] failed to execute ingest (delete) {}", e, request.index(), shardRequest.shardId(), actionRequest);
143                    failures.add(new IngestActionFailure(request.getIngestId(), request.getShardId(), ExceptionsHelper.detailedMessage(e)));
144                    request.getActionRequests().set(i, null);
145                }
146            }
147        }
148        if (!mappingsToUpdate.isEmpty()) {
149            for (String mappingToUpdate : mappingsToUpdate) {
150                DocumentMapper docMapper = indexService.mapperService().documentMapper(mappingToUpdate);
151                mappingUpdatedAction.updateMappingOnMaster(indexService.index().name(), docMapper, indexService.indexUUID());
152            }
153        }
154        int quorumShards = findQuorum(clusterState, shards(clusterState, request), request);
155        return new IngestLeaderShardResponse()
156                .setTookInMillis(System.currentTimeMillis() - t0)
157                .setIngestId(request.getIngestId())
158                .setShardId(request.getShardId())
159                .setSuccessCount(successCount)
160                .setQuorumShards(quorumShards)
161                .setActionRequests(request.getActionRequests())
162                .setFailures(failures);
163    }
164
165    private long indexOperationOnLeader(ClusterState clusterState, IndexRequest indexRequest,
166                                        IngestLeaderShardRequest request, Set<String> mappingsToUpdate) {
167        boolean mappingsModified = false;
168        try {
169            MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
170            if (mappingMd != null && mappingMd.routing().required() && indexRequest.routing() == null) {
171                throw new RoutingMissingException(indexRequest.index(), indexRequest.type(), indexRequest.id());
172            }
173            SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, indexRequest.source())
174                    .type(indexRequest.type())
175                    .id(indexRequest.id())
176                    .routing(indexRequest.routing())
177                    .parent(indexRequest.parent())
178                    .timestamp(indexRequest.timestamp())
179                    .ttl(indexRequest.ttl());
180            IndexShard indexShard = indicesService.indexServiceSafe(request.index()).shardSafe(request.getShardId().id());
181            if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
182                Engine.Index index = indexShard.prepareIndex(sourceToParse,
183                        indexRequest.version(),
184                        indexRequest.versionType(),
185                        Engine.Operation.Origin.PRIMARY,
186                        false);
187                mappingsModified = index.parsedDoc().mappingsModified();
188                indexShard.index(index);
189                return index.version();
190            } else if (indexRequest.opType() == IndexRequest.OpType.CREATE) {
191                Engine.Create create = indexShard.prepareCreate(sourceToParse,
192                        indexRequest.version(),
193                        indexRequest.versionType(),
194                        Engine.Operation.Origin.PRIMARY,
195                        false,
196                        indexRequest.autoGeneratedId());
197                mappingsModified = create.parsedDoc().mappingsModified();
198                indexShard.create(create);
199                return create.version();
200            } else {
201                logger.error("unknown op type " + indexRequest.opType());
202                return 0;
203            }
204        } finally {
205            if (mappingsModified) {
206                mappingsToUpdate.add(indexRequest.type());
207            }
208        }
209    }
