002    package org.xbib.elasticsearch.action.ingest;
003    
004    import org.elasticsearch.action.Action;
005    import org.elasticsearch.client.Client;
006    import org.elasticsearch.common.settings.Settings;
007    import org.elasticsearch.transport.TransportRequestOptions;
008    
009    public class IngestAction extends Action<IngestRequest, IngestResponse, IngestRequestBuilder> {
010    
011        public static final IngestAction INSTANCE = new IngestAction();
012    
013        public static final String NAME = "org.xbib.elasticsearch.action.ingest";
014    
015        private IngestAction() {
016            super(NAME);
017        }
018    
019        @Override
020        public IngestResponse newResponse() {
021            return new IngestResponse();
022        }
023    
024        @Override
025        public IngestRequestBuilder newRequestBuilder(Client client) {
026            return new IngestRequestBuilder(client);
027        }
028    
029        @Override
030        public TransportRequestOptions transportOptions(Settings settings) {
031            return TransportRequestOptions.options()
032                    .withType(TransportRequestOptions.Type.fromString(settings.get("action.ingest.transport.type",
033                            TransportRequestOptions.Type.LOW.toString())))
034                    .withCompress(settings.getAsBoolean("action.ingest.compress", true));
035        }
036    }
