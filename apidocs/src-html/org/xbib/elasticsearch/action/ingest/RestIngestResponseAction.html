002package org.xbib.elasticsearch.action.ingest;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.common.xcontent.XContentBuilder;
008import org.elasticsearch.common.xcontent.XContentBuilderString;
009import org.elasticsearch.rest.BaseRestHandler;
010import org.elasticsearch.rest.RestChannel;
011import org.elasticsearch.rest.RestController;
012import org.elasticsearch.rest.RestRequest;
013import org.elasticsearch.rest.XContentRestResponse;
014
015import java.io.IOException;
016import java.util.Map;
017
018import static org.elasticsearch.rest.RestRequest.Method.GET;
019import static org.elasticsearch.rest.RestStatus.BAD_REQUEST;
020import static org.elasticsearch.rest.RestStatus.OK;
021import static org.elasticsearch.rest.action.support.RestXContentBuilder.restContentBuilder;
022
023public class RestIngestResponseAction extends BaseRestHandler {
024
025    @Inject
026    public RestIngestResponseAction(Settings settings, Client client, RestController controller) {
027        super(settings, client);
028
029        controller.registerHandler(GET, "/_ingest", this);
030    }
031
032    @Override
033    public void handleRequest(final RestRequest request, final RestChannel channel) {
034        try {
035            XContentBuilder builder = restContentBuilder(request);
036            builder.startObject();
037            builder.field(Fields.OK, true);
038            builder.startArray(Fields.RESPONSES);
039            synchronized (RestIngestAction.responses) {
040                for (Map.Entry<Long, Object> me : RestIngestAction.responses.entrySet()) {
041                    builder.startObject();
042                    builder.field(Fields.ID, me.getKey());
043                    if (me.getValue() instanceof IngestResponse) {
044                        IngestResponse response = (IngestResponse) me.getValue();
045                        builder.startObject();
046                        builder.field(Fields.TOOK, response.tookInMillis());
047                        builder.startArray(Fields.SUCCESS);
048                        for (IngestItemSuccess success : response.success()) {
049                            builder.field(Fields.ID, success.id());
050                        }
051                        builder.endArray();
052                        builder.startArray(Fields.FAILURE);
053                        for (IngestItemFailure failure : response.failure()) {
054                            builder.field(Fields.ID, failure.id());
055                        }
056                        builder.endArray();
057                        builder.endObject();
058                    } else if (me.getValue() instanceof Throwable) {
059                        Throwable t = (Throwable)me.getValue();
060                        builder.field(Fields.ERROR, t.getMessage());
061                    }
062                    builder.endObject();
063                }
064                RestIngestAction.responses.clear(); // write each response only once
065            }
066            builder.endArray();
067            builder.endObject();
068            channel.sendResponse(new XContentRestResponse(request, OK, builder));
069        } catch (Exception e) {
070            try {
071                XContentBuilder builder = restContentBuilder(request);
072                channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder.startObject().field("error", e.getMessage()).endObject()));
073            } catch (IOException e1) {
074                logger.error("Failed to send failure response", e1);
075            }
076        }
077    }
078
079    static final class Fields {
080        static final XContentBuilderString RESPONSES = new XContentBuilderString("responses");
081        static final XContentBuilderString SUCCESS = new XContentBuilderString("success");
082        static final XContentBuilderString FAILURE = new XContentBuilderString("failure");
083        static final XContentBuilderString ID = new XContentBuilderString("id");
084        static final XContentBuilderString ERROR = new XContentBuilderString("error");
085        static final XContentBuilderString OK = new XContentBuilderString("ok");
086        static final XContentBuilderString TOOK = new XContentBuilderString("took");
087    }
088}
