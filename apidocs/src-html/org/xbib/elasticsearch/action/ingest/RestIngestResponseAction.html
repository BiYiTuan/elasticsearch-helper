002    package org.xbib.elasticsearch.action.ingest;
003    
004    import org.elasticsearch.client.Client;
005    import org.elasticsearch.common.inject.Inject;
006    import org.elasticsearch.common.settings.Settings;
007    import org.elasticsearch.common.xcontent.XContentBuilder;
008    import org.elasticsearch.common.xcontent.XContentBuilderString;
009    import org.elasticsearch.rest.BaseRestHandler;
010    import org.elasticsearch.rest.RestChannel;
011    import org.elasticsearch.rest.RestController;
012    import org.elasticsearch.rest.RestRequest;
013    import org.elasticsearch.rest.XContentRestResponse;
014    
015    import java.io.IOException;
016    import java.util.Map;
017    
018    import static org.elasticsearch.rest.RestRequest.Method.GET;
019    import static org.elasticsearch.rest.RestStatus.BAD_REQUEST;
020    import static org.elasticsearch.rest.RestStatus.OK;
021    import static org.elasticsearch.rest.action.support.RestXContentBuilder.restContentBuilder;
022    
023    public class RestIngestResponseAction extends BaseRestHandler {
024    
025        @Inject
026        public RestIngestResponseAction(Settings settings, Client client, RestController controller) {
027            super(settings, client);
028    
029            controller.registerHandler(GET, "/_ingest", this);
030        }
031    
032        @Override
033        public void handleRequest(final RestRequest request, final RestChannel channel) {
034            try {
035                XContentBuilder builder = restContentBuilder(request);
036                builder.startObject();
037                builder.field(Fields.OK, true);
038                builder.startArray(Fields.RESPONSES);
039                synchronized (RestIngestAction.responses) {
040                    for (Map.Entry<Long, Object> me : RestIngestAction.responses.entrySet()) {
041                        builder.startObject();
042                        builder.field(Fields.ID, me.getKey());
043                        if (me.getValue() instanceof IngestResponse) {
044                            IngestResponse response = (IngestResponse) me.getValue();
045                            builder.startObject();
046                            builder.field(Fields.TOOK, response.tookInMillis());
047                            builder.field(Fields.SUCCESS, response.successSize());
048                            builder.startArray(Fields.FAILURE);
049                            for (IngestItemFailure failure : response.failure()) {
050                                builder.field(Fields.ID, failure.id());
051                            }
052                            builder.endArray();
053                            builder.endObject();
054                        } else if (me.getValue() instanceof Throwable) {
055                            Throwable t = (Throwable)me.getValue();
056                            builder.field(Fields.ERROR, t.getMessage());
057                        }
058                        builder.endObject();
059                    }
060                    RestIngestAction.responses.clear(); // write each response only once
061                }
062                builder.endArray();
063                builder.endObject();
064                channel.sendResponse(new XContentRestResponse(request, OK, builder));
065            } catch (Exception e) {
066                try {
067                    XContentBuilder builder = restContentBuilder(request);
068                    channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder.startObject().field("error", e.getMessage()).endObject()));
069                } catch (IOException e1) {
070                    logger.error("Failed to send failure response", e1);
071                }
072            }
073        }
074    
075        static final class Fields {
076            static final XContentBuilderString RESPONSES = new XContentBuilderString("responses");
077            static final XContentBuilderString SUCCESS = new XContentBuilderString("success");
078            static final XContentBuilderString FAILURE = new XContentBuilderString("failure");
079            static final XContentBuilderString ID = new XContentBuilderString("id");
080            static final XContentBuilderString ERROR = new XContentBuilderString("error");
081            static final XContentBuilderString OK = new XContentBuilderString("ok");
082            static final XContentBuilderString TOOK = new XContentBuilderString("took");
083        }
084    }
