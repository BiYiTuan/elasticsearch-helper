143        @Override
144        protected PrimaryResponse<IngestShardResponse, IngestShardRequest> shardOperationOnPrimary(ClusterState clusterState, PrimaryOperationRequest shardRequest) {
145            final IngestShardRequest request = shardRequest.request;
146            IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
147            Set<Tuple<String, String>> mappingsToUpdate = null;
148    
149            int successSize = 0;
150            List<IngestItemFailure> failure = newLinkedList();
151            int size = request.items().size();
152            long[] versions = new long[size];
153            for (int i = 0; i < size; i++) {
154                IngestItemRequest item = request.items().get(i);
155                if (item.request() instanceof IndexRequest) {
156                    IndexRequest indexRequest = (IndexRequest) item.request();
157                    try {
158                        // validate, if routing is required, that we got routing
159                        MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
160                        if (mappingMd != null && mappingMd.routing().required()) {
161                            if (indexRequest.routing() == null) {
162                                throw new RoutingMissingException(indexRequest.index(), indexRequest.type(), indexRequest.id());
163                            }
164                        }
165    
166                        SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
167                                .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
168    
169                        long version;
170                        Engine.IndexingOperation op;
171                        if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
172                            Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).versionType(indexRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
173                            indexShard.index(index);
174                            version = index.version();
175                            op = index;
176                        } else {
177                            Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).versionType(indexRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
178                            indexShard.create(create);
179                            version = create.version();
180                            op = create;
181                        }
182                        versions[i] = indexRequest.version();
183                        // update the version on request so it will happen on the replicas
184                        indexRequest.version(version);
185    
186                        // update mapping on master if needed, we won't update changes to the same type, since once its changed, it won't have mappers added
187                        if (op.parsedDoc().mappingsModified()) {
188                            if (mappingsToUpdate == null) {
189                                mappingsToUpdate = newHashSet();
190                            }
191                            mappingsToUpdate.add(Tuple.tuple(indexRequest.index(), indexRequest.type()));
192                        }
193    
194                        successSize++;
195    
196                    } catch (Exception e) {
197                        // rethrow the failure if we are going to retry on primary and let parent failure to handle it
198                        if (retryPrimaryException(e)) {
199                            // restore updated versions...
200                            for (int j = 0; j < i; j++) {
201                                applyVersion(request.items().get(j), versions[j]);
202                            }
203                            throw (ElasticSearchException) e;
204                        }
205                        if (e instanceof ElasticSearchException && ((ElasticSearchException) e).status() == RestStatus.CONFLICT) {
206                            logger.trace("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
207                        } else {
208                            logger.debug("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
209                        }
210                        failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
211                        // nullify the request so it won't execute on the replicas
212                        request.items().set(i, null);
213                    }
214                } else if (item.request() instanceof DeleteRequest) {
215                    DeleteRequest deleteRequest = (DeleteRequest) item.request();
216                    try {
217                        Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version()).versionType(deleteRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
218                        indexShard.delete(delete);
219                        // update the request with teh version so it will go to the replicas
220                        deleteRequest.version(delete.version());
221    
222                        successSize++;
223                    } catch (Exception e) {
224                        // rethrow the failure if we are going to retry on primary and let parent failure to handle it
225                        if (retryPrimaryException(e)) {
226                            // restore updated versions...
227                            for (int j = 0; j < i; j++) {
228                                applyVersion(request.items().get(j), versions[j]);
229                            }
230                            throw (ElasticSearchException) e;
231                        }
232                        if (e instanceof ElasticSearchException && ((ElasticSearchException) e).status() == RestStatus.CONFLICT) {
233                            logger.trace("[{}][{}] failed to execute bulk item (delete) {}", e, shardRequest.request.index(), shardRequest.shardId, deleteRequest);
234                        } else {
235                            logger.debug("[{}][{}] failed to execute bulk item (delete) {}", e, shardRequest.request.index(), shardRequest.shardId, deleteRequest);
236                        }
237                        failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
238                        // nullify the request so it won't execute on the replicas
239                        request.items().set(i, null);
240                    }
241                }
242            }
243    
244            if (mappingsToUpdate != null) {
245                for (Tuple<String, String> mappingToUpdate : mappingsToUpdate) {
246                    updateMappingOnMaster(mappingToUpdate.v1(), mappingToUpdate.v2());
247                }
248            }
249    
250            IngestShardResponse response = new IngestShardResponse(new ShardId(request.index(), request.shardId()), successSize, failure);
251            return new PrimaryResponse<IngestShardResponse, IngestShardRequest>(shardRequest.request, response, null);
252        }

254        @Override
255        protected void shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
256            IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
257            final IngestShardRequest request = shardRequest.request;
258            int size = request.items().size();
259            for (int i = 0; i < size; i++) {
260                IngestItemRequest item = request.items().get(i);
261                if (item == null) {
262                    continue;
263                }
264                if (item.request() instanceof IndexRequest) {
265                    IndexRequest indexRequest = (IndexRequest) item.request();
266                    try {
267                        SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.REPLICA, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
268                                .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
269    
270                        if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
271                            Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
272                            indexShard.index(index);
273                        } else {
274                            Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
275                            indexShard.create(create);
276                        }
277                    } catch (Exception e) {
278                        // ignore, we are on backup
279                    }
280                } else if (item.request() instanceof DeleteRequest) {
281                    DeleteRequest deleteRequest = (DeleteRequest) item.request();
282                    try {
283                        Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version()).origin(Engine.Operation.Origin.REPLICA);
284                        indexShard.delete(delete);
285                    } catch (Exception e) {
286                        // ignore, we are on backup
287                    }
288                }
289            }
290        }

292        private void updateMappingOnMaster(final String index, final String type) {
293            try {
294                MapperService mapperService = indicesService.indexServiceSafe(index).mapperService();
295                final DocumentMapper documentMapper = mapperService.documentMapper(type);
296                if (documentMapper == null) { // should not happen
297                    return;
298                }
299                documentMapper.refreshSource();
300    
301                IndexMetaData metaData = clusterService.state().metaData().index(index);
302    
303                final MappingUpdatedAction.MappingUpdatedRequest request = new MappingUpdatedAction.MappingUpdatedRequest(index, metaData.uuid(), type, documentMapper.mappingSource());
304                mappingUpdatedAction.execute(request, new ActionListener<MappingUpdatedAction.MappingUpdatedResponse>() {
305                    @Override
306                    public void onResponse(MappingUpdatedAction.MappingUpdatedResponse mappingUpdatedResponse) {
307                        // all is well
308                    }
309    
310                    @Override
311                    public void onFailure(Throwable e) {
312                        try {
313                            logger.warn("failed to update master on updated mapping for index [{}], type [{}] and source [{}]", e, index, type, documentMapper.mappingSource().string());
314                        } catch (IOException e1) {
315                            // ignore
316                        }
317                    }
318                });
319            } catch (Exception e) {
320                logger.warn("failed to update master on updated mapping for index [{}], type [{}]", e, index, type);
321            }
322        }

324        private void applyVersion(IngestItemRequest item, long version) {
325            if (item.request() instanceof IndexRequest) {
326                ((IndexRequest) item.request()).version(version);
327            } else if (item.request() instanceof DeleteRequest) {
328                ((DeleteRequest) item.request()).version(version);
329            }
330        }
