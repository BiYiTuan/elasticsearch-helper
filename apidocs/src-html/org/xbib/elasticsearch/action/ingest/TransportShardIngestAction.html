106    @Override
107    protected PrimaryResponse<IngestShardResponse, IngestShardRequest> shardOperationOnPrimary(ClusterState clusterState, PrimaryOperationRequest shardRequest) {
108        final IngestShardRequest request = shardRequest.request;
109        IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
110
111        Engine.IndexingOperation[] ops = null;
112
113        Set<Tuple<String, String>> mappingsToUpdate = null;
114
115        List<IngestItemSuccess> success = Lists.newLinkedList();
116        List<IngestItemFailure> failure = Lists.newLinkedList();
117
118        int size = request.items().size();
119        long[] versions = new long[size];
120        for (int i = 0; i < size; i++) {
121            IngestItemRequest item = request.items().get(i);
122            if (item.request() instanceof IndexRequest) {
123                IndexRequest indexRequest = (IndexRequest) item.request();
124                try {
125
126                    // validate, if routing is required, that we got routing
127                    MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
128                    if (mappingMd != null && mappingMd.routing().required()) {
129                        if (indexRequest.routing() == null) {
130                            throw new RoutingMissingException(indexRequest.index(), indexRequest.type(), indexRequest.id());
131                        }
132                    }
133
134                    SourceToParse sourceToParse = SourceToParse.source(indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
135                            .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
136
137                    long version;
138                    Engine.IndexingOperation op;
139                    if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
140                        Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).versionType(indexRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
141                        indexShard.index(index);
142                        version = index.version();
143                        op = index;
144                    } else {
145                        Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).versionType(indexRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
146                        indexShard.create(create);
147                        version = create.version();
148                        op = create;
149                    }
150                    versions[i] = indexRequest.version();
151                    // update the version on request so it will happen on the replicas
152                    indexRequest.version(version);
153
154                    // update mapping on master if needed, we won't update changes to the same type, since once its changed, it won't have mappers added
155                    if (op.parsedDoc().mappingsModified()) {
156                        if (mappingsToUpdate == null) {
157                            mappingsToUpdate = Sets.newHashSet();
158                        }
159                        mappingsToUpdate.add(Tuple.tuple(indexRequest.index(), indexRequest.type()));
160                    }
161
162                    // if we are going to percolate, then we need to keep this op for the postPrimary operation
163                    if (Strings.hasLength(indexRequest.percolate())) {
164                        if (ops == null) {
165                            ops = new Engine.IndexingOperation[size];
166                        }
167                        ops[i] = op;
168                    }
169
170                    success.add(new IngestItemSuccess(item.id()));
171                } catch (Exception e) {
172                    // rethrow the failure if we are going to retry on primary and let parent failure to handle it
173                    if (retryPrimaryException(e)) {
174                        // restore updated versions...
175                        for (int j = 0; j < i; j++) {
176                            applyVersion(request.items().get(j), versions[j]);
177                        }
178                        throw (ElasticSearchException) e;
179                    }
180                    if (e instanceof ElasticSearchException && ((ElasticSearchException) e).status() == RestStatus.CONFLICT) {
181                        logger.trace("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
182                    } else {
183                        logger.debug("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
184                    }
185                    failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
186                    // nullify the request so it won't execute on the replicas
187                    request.items().set(i, null);
188                }
189            } else if (item.request() instanceof DeleteRequest) {
190                DeleteRequest deleteRequest = (DeleteRequest) item.request();
191                try {
192                    Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version()).versionType(deleteRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
193                    indexShard.delete(delete);
194                    // update the request with teh version so it will go to the replicas
195                    deleteRequest.version(delete.version());
196
197                    // add the response
198                    success.add(new IngestItemSuccess(item.id()));
199                } catch (Exception e) {
200                    // rethrow the failure if we are going to retry on primary and let parent failure to handle it
201                    if (retryPrimaryException(e)) {
202                        // restore updated versions...
203                        for (int j = 0; j < i; j++) {
204                            applyVersion(request.items().get(j), versions[j]);
205                        }
206                        throw (ElasticSearchException) e;
207                    }
208                    if (e instanceof ElasticSearchException && ((ElasticSearchException) e).status() == RestStatus.CONFLICT) {
209                        logger.trace("[{}][{}] failed to execute bulk item (delete) {}", e, shardRequest.request.index(), shardRequest.shardId, deleteRequest);
210                    } else {
211                        logger.debug("[{}][{}] failed to execute bulk item (delete) {}", e, shardRequest.request.index(), shardRequest.shardId, deleteRequest);
212                    }
213                    failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
214                    // nullify the request so it won't execute on the replicas
215                    request.items().set(i, null);
216                }
217            }
218        }
219
220        if (mappingsToUpdate != null) {
221            for (Tuple<String, String> mappingToUpdate : mappingsToUpdate) {
222                updateMappingOnMaster(mappingToUpdate.v1(), mappingToUpdate.v2());
223            }
224        }
225
226        IngestShardResponse response = new IngestShardResponse(new ShardId(request.index(), request.shardId()), success, failure);
227        return new PrimaryResponse<IngestShardResponse, IngestShardRequest>(shardRequest.request, response, ops);
228    }

235    @Override
236    protected void shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
237        IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
238        final IngestShardRequest request = shardRequest.request;
239        for (int i = 0; i < request.items().size(); i++) {
240            IngestItemRequest item = request.items().get(i);
241            if (item == null) {
242                continue;
243            }
244            if (item.request() instanceof IndexRequest) {
245                IndexRequest indexRequest = (IndexRequest) item.request();
246                try {
247                    SourceToParse sourceToParse = SourceToParse.source(indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
248                            .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
249
250                    if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
251                        Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
252                        indexShard.index(index);
253                    } else {
254                        Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
255                        indexShard.create(create);
256                    }
257                } catch (Exception e) {
258                    // ignore, we are on backup
259                }
260            } else if (item.request() instanceof DeleteRequest) {
261                DeleteRequest deleteRequest = (DeleteRequest) item.request();
262                try {
263                    Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version()).origin(Engine.Operation.Origin.REPLICA);
264                    indexShard.delete(delete);
265                } catch (Exception e) {
266                    // ignore, we are on backup
267                }
268            }
269        }

271    }

273    private void updateMappingOnMaster(final String index, final String type) {
274        try {
275            MapperService mapperService = indicesService.indexServiceSafe(index).mapperService();
276            final DocumentMapper documentMapper = mapperService.documentMapper(type);
277            if (documentMapper == null) { // should not happen
278                return;
279            }
280            documentMapper.refreshSource();
281
282            mappingUpdatedAction.execute(new MappingUpdatedAction.MappingUpdatedRequest(index, type, documentMapper.mappingSource()), new ActionListener<MappingUpdatedAction.MappingUpdatedResponse>() {
283                @Override
284                public void onResponse(MappingUpdatedAction.MappingUpdatedResponse mappingUpdatedResponse) {
285                    // all is well
286                }
287
288                @Override
289                public void onFailure(Throwable e) {
290                    try {
291                        logger.warn("failed to update master on updated mapping for index [{}], type [{}] and source [{}]", e, index, type, documentMapper.mappingSource().string());
292                    } catch (IOException e1) {
293                        // ignore
294                    }
295                }
296            });
297        } catch (Exception e) {
298            logger.warn("failed to update master on updated mapping for index [{}], type [{}]", e, index, type);
299        }
300    }
301
302    private void applyVersion(IngestItemRequest item, long version) {
303        if (item == null) {
304            return;
305        }
306        if (item.request() instanceof IndexRequest) {
307            ((IndexRequest) item.request()).version(version);
308        } else if (item.request() instanceof DeleteRequest) {
309            ((DeleteRequest) item.request()).version(version);
310        } else {
311            // log?
312        }
313    }
