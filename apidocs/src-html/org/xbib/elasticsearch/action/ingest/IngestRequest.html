051        /**
052         * Adds a list of requests to be executed. Either index or delete requests.
053         */
054        public IngestRequest add(ActionRequest... requests) {
055            for (ActionRequest request : requests) {
056                add(request);
057            }
058            return this;
059        }
060    
061        public IngestRequest add(ActionRequest request) {
062            if (request instanceof IndexRequest) {
063                add((IndexRequest) request);
064            } else if (request instanceof DeleteRequest) {
065                add((DeleteRequest) request);
066            } else {
067                throw new ElasticsearchIllegalArgumentException("no support for request [" + request + "]");
068            }
069            return this;
070        }

072        /**
073         * Adds a list of requests to be executed. Either index or delete requests.
074         */
075        public IngestRequest add(Iterable<ActionRequest> requests) {
076            for (ActionRequest request : requests) {
077                if (request instanceof IndexRequest) {
078                    add((IndexRequest) request);
079                } else if (request instanceof DeleteRequest) {
080                    add((DeleteRequest) request);
081                } else {
082                    throw new ElasticsearchIllegalArgumentException("no support for request [" + request + "]");
083                }
084            }
085            return this;
086        }

088        /**
089         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows
090         * the same behavior of {@link org.elasticsearch.action.index.IndexRequest} (for example, if no id is
091         * provided, one will be generated, or usage of the create flag).
092         */
093        public IngestRequest add(IndexRequest request) {
094            request.beforeLocalFork();
095            return internalAdd(request);
096        }
097    
098        IngestRequest internalAdd(IndexRequest request) {
099            requests.offer(request);
100            long length = request.source() != null ? request.source().length() + REQUEST_OVERHEAD : REQUEST_OVERHEAD;
101            sizeInBytes.addAndGet(length);
102            return this;
103        }
104    
105        /**
106         * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
107         */
108        public IngestRequest add(DeleteRequest request) {
109            requests.offer(request);
110            sizeInBytes.addAndGet(REQUEST_OVERHEAD);
111            return this;
112        }

143        /**
144         * Adds a framed data in binary format
145         */
146        public IngestRequest add(BytesReference data, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
147            XContent xContent = XContentFactory.xContent(data);
148            int from = 0;
149            int length = data.length();
150            byte marker = xContent.streamSeparator();
151            while (true) {
152                int nextMarker = findNextMarker(marker, from, data, length);
153                if (nextMarker == -1) {
154                    break;
155                }
156                // now parse the move
157                XContentParser parser = xContent.createParser(data.slice(from, nextMarker - from));
158    
159                try {
160                    // move pointers
161                    from = nextMarker + 1;
162    
163                    // Move to START_OBJECT
164                    XContentParser.Token token = parser.nextToken();
165                    if (token == null) {
166                        continue;
167                    }
168                    assert token == XContentParser.Token.START_OBJECT;
169                    // Move to FIELD_NAME, that's the move
170                    token = parser.nextToken();
171                    assert token == XContentParser.Token.FIELD_NAME;
172                    String action = parser.currentName();
173    
174                    String index = defaultIndex;
175                    String type = defaultType;
176                    String id = null;
177                    String routing = null;
178                    String parent = null;
179                    String timestamp = null;
180                    Long ttl = null;
181                    String opType = null;
182                    long version = 0;
183                    VersionType versionType = VersionType.INTERNAL;
184    
185                    // at this stage, next token can either be END_OBJECT (and use default index and type, with auto generated id)
186                    // or START_OBJECT which will have another set of parameters
187    
188                    String currentFieldName = null;
189                    while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
190                        if (token == XContentParser.Token.FIELD_NAME) {
191                            currentFieldName = parser.currentName();
192                        } else if (token.isValue()) {
193                            if ("_index".equals(currentFieldName)) {
194                                index = parser.text();
195                            } else if ("_type".equals(currentFieldName)) {
196                                type = parser.text();
197                            } else if ("_id".equals(currentFieldName)) {
198                                id = parser.text();
199                            } else if ("_routing".equals(currentFieldName) || "routing".equals(currentFieldName)) {
200                                routing = parser.text();
201                            } else if ("_parent".equals(currentFieldName) || "parent".equals(currentFieldName)) {
202                                parent = parser.text();
203                            } else if ("_timestamp".equals(currentFieldName) || "timestamp".equals(currentFieldName)) {
204                                timestamp = parser.text();
205                            } else if ("_ttl".equals(currentFieldName) || "ttl".equals(currentFieldName)) {
206                                if (parser.currentToken() == XContentParser.Token.VALUE_STRING) {
207                                    ttl = TimeValue.parseTimeValue(parser.text(), null).millis();
208                                } else {
209                                    ttl = parser.longValue();
210                                }
211                            } else if ("op_type".equals(currentFieldName) || "opType".equals(currentFieldName)) {
212                                opType = parser.text();
213                            } else if ("_version".equals(currentFieldName) || "version".equals(currentFieldName)) {
214                                version = parser.longValue();
215                            } else if ("_version_type".equals(currentFieldName) || "_versionType".equals(currentFieldName) || "version_type".equals(currentFieldName) || "versionType".equals(currentFieldName)) {
216                                versionType = VersionType.fromString(parser.text());
217                            }
218                        }
219                    }
220    
221                    if ("delete".equals(action)) {
222                        add(new DeleteRequest(index, type, id).parent(parent).version(version).versionType(versionType).routing(routing));
223                    } else {
224                        nextMarker = findNextMarker(marker, from, data, length);
225                        if (nextMarker == -1) {
226                            break;
227                        }
228                        // order is important, we set parent after routing, so routing will be set to parent if not set explicitly
229                        // we use internalAdd so we don't fork here, this allows us not to copy over the big byte array to small chunks
230                        // of index request. All index requests are still unsafe if applicable.
231                        if ("index".equals(action)) {
232                            if (opType == null) {
233                                internalAdd(new IndexRequest(index, type, id).routing(routing).parent(parent).timestamp(timestamp).ttl(ttl).version(version).versionType(versionType)
234                                        .source(data.slice(from, nextMarker - from), contentUnsafe));
235                            } else {
236                                internalAdd(new IndexRequest(index, type, id).routing(routing).parent(parent).timestamp(timestamp).ttl(ttl).version(version).versionType(versionType)
237                                        .create("create".equals(opType))
238                                        .source(data.slice(from, nextMarker - from), contentUnsafe));
239                            }
240                        } else if ("create".equals(action)) {
241                            internalAdd(new IndexRequest(index, type, id).routing(routing).parent(parent).timestamp(timestamp).ttl(ttl).version(version).versionType(versionType)
242                                    .create(true)
243                                    .source(data.slice(from, nextMarker - from), contentUnsafe));
244                        }
245                        // move pointers
246                        from = nextMarker + 1;
247                    }
248                } finally {
249                    parser.close();
250                }
251            }
252            return this;
253        }

292        /**
293         * Take all requests from queue. This method is thread safe.
294         *
295         * @return a bulk request
296         */
297        public IngestRequest takeAll() {
298            IngestRequest request = new IngestRequest();
299            while (!requests.isEmpty()) {
300                ActionRequest actionRequest = requests.poll();
301                request.add(actionRequest);
302                if (actionRequest instanceof IndexRequest) {
303                    IndexRequest indexRequest = (IndexRequest)actionRequest;
304                    long length = indexRequest.source() != null ? indexRequest.source().length() + REQUEST_OVERHEAD : REQUEST_OVERHEAD;
305                    sizeInBytes.addAndGet(-length);
306                } else if (actionRequest instanceof DeleteRequest) {
307                    sizeInBytes.addAndGet(REQUEST_OVERHEAD);
308                }
309            }
310            return request;
311        }

313        /**
314         * Take a number of requests from the bulk request queue.
315         *
316         * This method is thread safe.
317         *
318         * @param numRequests number of requests
319         * @return a partial bulk request
320         */
321        public IngestRequest take(int numRequests) {
322            IngestRequest request = new IngestRequest();
323            for (int i = 0; i < numRequests; i++) {
324                ActionRequest actionRequest = requests.poll();
325                request.add(actionRequest);
326                if (actionRequest instanceof IndexRequest) {
327                    IndexRequest indexRequest = (IndexRequest)actionRequest;
328                    long length = indexRequest.source() != null ? indexRequest.source().length() + REQUEST_OVERHEAD : REQUEST_OVERHEAD;
329                    sizeInBytes.addAndGet(-length);
330                } else if (actionRequest instanceof DeleteRequest) {
331                    sizeInBytes.addAndGet(REQUEST_OVERHEAD);
332                }
333            }
334            return request;
335        }
336    
337        private int findNextMarker(byte marker, int from, BytesReference data, int length) {
338            for (int i = from; i < length; i++) {
339                if (data.get(i) == marker) {
340                    return i;
341                }
342            }
343            return -1;
344        }

346        @Override
347        public ActionRequestValidationException validate() {
348            ActionRequestValidationException validationException = null;
349            if (requests.isEmpty()) {
350                validationException = addValidationError("no requests added", null);
351            }
352            for (ActionRequest request : requests) {
353                ActionRequestValidationException ex = request.validate();
354                if (ex != null) {
355                    if (validationException == null) {
356                        validationException = new ActionRequestValidationException();
357                    }
358                    validationException.addValidationErrors(ex.validationErrors());
359                }
360            }
361            return validationException;
362        }

364        @Override
365        public void readFrom(StreamInput in) throws IOException {
366            replicationType = ReplicationType.fromId(in.readByte());
367            consistencyLevel = WriteConsistencyLevel.fromId(in.readByte());
368            timeout = TimeValue.readTimeValue(in);
369            int size = in.readVInt();
370            for (int i = 0; i < size; i++) {
371                byte type = in.readByte();
372                if (type == 0) {
373                    IndexRequest request = new IndexRequest();
374                    request.readFrom(in);
375                    requests.add(request);
376                } else if (type == 1) {
377                    DeleteRequest request = new DeleteRequest();
378                    request.readFrom(in);
379                    requests.add(request);
380                }
381            }
382        }
383    
384        @Override
385        public void writeTo(StreamOutput out) throws IOException {
386            out.writeByte(replicationType.id());
387            out.writeByte(consistencyLevel.id());
388            timeout.writeTo(out);
389            out.writeVInt(requests.size());
390            for (ActionRequest request : requests) {
391                if (request instanceof IndexRequest) {
392                    out.writeByte((byte) 0);
393                } else if (request instanceof DeleteRequest) {
394                    out.writeByte((byte) 1);
395                }
396                request.writeTo(out);
397            }
398        }
