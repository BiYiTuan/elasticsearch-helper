001package org.xbib.elasticsearch.action.ingest.replica;
002
003import org.elasticsearch.ElasticsearchIllegalStateException;
004import org.elasticsearch.action.ActionRequest;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007import org.elasticsearch.index.shard.ShardId;
008import org.xbib.elasticsearch.action.delete.DeleteRequest;
009import org.xbib.elasticsearch.action.index.IndexRequest;
010import org.xbib.elasticsearch.action.support.replication.replica.ReplicaShardOperationRequest;
011
012import java.io.IOException;
013import java.util.List;
014
015import static org.elasticsearch.common.collect.Lists.newLinkedList;
016
017public class IngestReplicaShardRequest extends ReplicaShardOperationRequest<IngestReplicaShardRequest> {
018
019    private long ingestId;
020
021    private ShardId shardId;
022
023    private List<ActionRequest> actionRequests = newLinkedList();
024
025    public IngestReplicaShardRequest() {
026    }
027
028    public IngestReplicaShardRequest(long ingestId, ShardId shardId, List<ActionRequest> actionRequests) {
029        this.index = shardId.index().name();
030        this.ingestId = ingestId;
031        this.shardId = shardId;
032        this.actionRequests = actionRequests;
033    }
034
035    public long ingestId() {
036        return ingestId;
037    }
038
039    public ShardId shardId() {
040        return shardId;
041    }
042
043    public List<ActionRequest> actionRequests() {
044        return actionRequests;
045    }
046
047    @Override
048    public void writeTo(StreamOutput out) throws IOException {
049        super.writeTo(out);
050        out.writeString(index);
051        out.writeLong(ingestId);
052        shardId.writeTo(out);
053        out.writeVInt(actionRequests.size());
054        for (ActionRequest actionRequest : actionRequests) {
055            if (actionRequest == null) {
056                out.writeBoolean(false);
057                continue;
058            }
059            out.writeBoolean(true);
060            if (actionRequest instanceof IndexRequest) {
061                out.writeBoolean(true);
062            } else if (actionRequest instanceof DeleteRequest) {
063                out.writeBoolean(false);
064            } else {
065                throw new ElasticsearchIllegalStateException("action request not supported: " + actionRequest.getClass().getName());
066            }
067            actionRequest.writeTo(out);
068        }
069    }
070
071    @Override
072    public void readFrom(StreamInput in) throws IOException {
073        super.readFrom(in);
074        index = in.readString();
075        ingestId = in.readLong();
076        shardId = ShardId.readShardId(in);
077        int size = in.readVInt();
078        actionRequests = newLinkedList();
079        for (int i = 0; i < size; i++) {
080            boolean exists = in.readBoolean();
081            if (exists) {
082                boolean b = in.readBoolean();
083                if (b) {
084                    IndexRequest indexRequest = new IndexRequest();
085                    indexRequest.readFrom(in);
086                    actionRequests.add(indexRequest);
087                } else {
088                    DeleteRequest deleteRequest = new DeleteRequest();
089                    deleteRequest.readFrom(in);
090                    actionRequests.add(deleteRequest);
091                }
092            } else {
093                actionRequests.add(null);
094            }
095        }
096    }
097}
