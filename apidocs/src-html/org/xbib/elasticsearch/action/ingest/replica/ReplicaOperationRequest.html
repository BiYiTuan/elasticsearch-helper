001package org.xbib.elasticsearch.action.ingest.replica;
002
003import org.elasticsearch.common.io.stream.StreamInput;
004import org.elasticsearch.common.io.stream.StreamOutput;
005import org.elasticsearch.common.io.stream.Streamable;
006import org.elasticsearch.transport.TransportRequest;
007
008import java.io.IOException;
009
010public class ReplicaOperationRequest extends TransportRequest implements Streamable {
011
012    private long startTime;
013
014    private String index;
015
016    private int shardId;
017
018    private int replicaId;
019
020    private IngestReplicaShardRequest request;
021
022    public ReplicaOperationRequest() {
023    }
024
025    public ReplicaOperationRequest(long startTime, String index, int shardId, int replicaId, IngestReplicaShardRequest request) {
026        this.startTime = startTime;
027        this.index = index;
028        this.shardId = shardId;
029        this.replicaId = replicaId;
030        this.request = request;
031    }
032
033    public long startTime() {
034        return startTime;
035    }
036
037    public String index() {
038        return index;
039    }
040
041    public int shardId() {
042        return shardId;
043    }
044
045    public int replicaId() {
046        return replicaId;
047    }
048
049    public IngestReplicaShardRequest request() {
050        return request;
051    }
052
053    @Override
054    public void readFrom(StreamInput in) throws IOException {
055        super.readFrom(in);
056        startTime = in.readLong();
057        index = in.readString();
058        shardId = in.readVInt();
059        replicaId = in.readVInt();
060        request = new IngestReplicaShardRequest();
061        request.readFrom(in);
062    }
063
064    @Override
065    public void writeTo(StreamOutput out) throws IOException {
066        super.writeTo(out);
067        out.writeLong(startTime);
068        out.writeString(index);
069        out.writeVInt(shardId);
070        out.writeVInt(replicaId);
071        request.writeTo(out);
072    }
073}
