002package org.xbib.elasticsearch.action.ingest;
003
004import org.elasticsearch.action.ActionResponse;
005import org.elasticsearch.common.collect.Lists;
006import org.elasticsearch.common.io.stream.StreamInput;
007import org.elasticsearch.common.io.stream.StreamOutput;
008import org.elasticsearch.index.shard.ShardId;
009
010import java.io.IOException;
011import java.util.List;
012
013public class IngestShardResponse extends ActionResponse {
014
015    private ShardId shardId;
016
017    private int successSize;
018
019    private List<IngestItemFailure> failure;
020
021    public IngestShardResponse() {
022        this.failure = Lists.newArrayList();
023    }
024
025    public IngestShardResponse(ShardId shardId, int successSize, List<IngestItemFailure> failure) {
026        this.shardId = shardId;
027        this.successSize = successSize;
028        this.failure = failure;
029    }
030
031    public ShardId shardId() {
032        return shardId;
033    }
034
035    public int successSize() {
036        return successSize;
037    }
038
039    public List<IngestItemFailure> failure() {
040        return failure;
041    }
042
043    @Override
044    public void readFrom(StreamInput in) throws IOException {
045        shardId = ShardId.readShardId(in);
046        successSize = in.readVInt();
047        failure = Lists.newLinkedList();
048        for (int i = 0; i < in.readVInt(); i++) {
049            failure.add(new IngestItemFailure(in.readVInt(), in.readString()));
050        }
051    }
052
053    @Override
054    public void writeTo(StreamOutput out) throws IOException {
055        shardId.writeTo(out);
056        out.writeVInt(successSize);
057        out.writeVInt(failure.size());
058        for (IngestItemFailure f : failure) {
059            out.writeVInt(f.pos());
060            out.writeString(f.message());
061        }
062    }
063}
