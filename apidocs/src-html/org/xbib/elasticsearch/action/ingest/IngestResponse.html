001package org.xbib.elasticsearch.action.ingest;
002
003import org.elasticsearch.action.ActionResponse;
004import org.elasticsearch.common.io.stream.StreamInput;
005import org.elasticsearch.common.io.stream.StreamOutput;
006import org.xbib.elasticsearch.action.ingest.leader.IngestLeaderShardResponse;
007import org.xbib.elasticsearch.action.ingest.replica.IngestReplicaShardResponse;
008
009import java.io.IOException;
010import java.util.Collections;
011import java.util.LinkedList;
012import java.util.List;
013
014import static org.elasticsearch.common.collect.Lists.newLinkedList;
015
016public class IngestResponse extends ActionResponse {
017
018    protected long ingestId;
019
020    protected int successSize;
021
022    protected IngestLeaderShardResponse leaderResponse = new IngestLeaderShardResponse();
023
024    @SuppressWarnings("unchecked")
025    protected List<IngestReplicaShardResponse> replicaResponses = Collections.synchronizedList(new LinkedList());
026
027    @SuppressWarnings("unchecked")
028    protected List<IngestActionFailure> failures = Collections.synchronizedList(new LinkedList());
029
030    protected long tookInMillis;
031
032    public IngestResponse() {
033    }
034
035    public IngestResponse setIngestId(long ingestId) {
036        this.ingestId = ingestId;
037        return this;
038    }
039
040    public long ingestId() {
041        return ingestId;
042    }
043
044    public IngestResponse setSuccessSize(int successSize) {
045        this.successSize = successSize;
046        return this;
047    }
048
049    public int successSize() {
050        return successSize;
051    }
052
053    public IngestResponse setLeaderResponse(IngestLeaderShardResponse leaderResponse) {
054        this.leaderResponse = leaderResponse;
055        return this;
056    }
057
058    public IngestLeaderShardResponse leaderShardResponse() {
059        return leaderResponse;
060    }
061
062    public IngestResponse addReplicaResponses(List<IngestReplicaShardResponse> response) {
063        this.replicaResponses.addAll(response);
064        return this;
065    }
066
067    public List<IngestReplicaShardResponse> replicaShardResponses() {
068        return replicaResponses;
069    }
070
071    public IngestResponse addFailure(IngestActionFailure failure) {
072        this.failures.add(failure);
073        return this;
074    }
075
076    public List<IngestActionFailure> getFailures() {
077        return failures;
078    }
079
080    public IngestResponse setTookInMillis(long tookInMillis) {
081        this.tookInMillis = tookInMillis;
082        return this;
083    }
084
085    public long tookInMillis() {
086        return tookInMillis;
087    }
088
089    @Override
090    public void readFrom(StreamInput in) throws IOException {
091        super.readFrom(in);
092        successSize = in.readVInt();
093        failures = newLinkedList();
094        int size = in.readVInt();
095        for (int i = 0; i < size; i++) {
096            failures.add(IngestActionFailure.from(in));
097        }
098        tookInMillis = in.readVLong();
099        leaderResponse = new IngestLeaderShardResponse();
100        leaderResponse.readFrom(in);
101        replicaResponses = newLinkedList();
102        size = in.readVInt();
103        for (int i = 0; i < size; i++) {
104            IngestReplicaShardResponse r = new IngestReplicaShardResponse();
105            r.readFrom(in);
106            replicaResponses.add(r);
107        }
108    }
109
110    @Override
111    public void writeTo(StreamOutput out) throws IOException {
112        super.writeTo(out);
113        out.writeVInt(successSize);
114        out.writeVInt(failures.size());
115        for (IngestActionFailure f : failures) {
116            f.writeTo(out);
117        }
118        out.writeVLong(tookInMillis);
119        leaderResponse.writeTo(out);
120        out.writeVInt(replicaResponses.size());
121        for (IngestReplicaShardResponse r : replicaResponses) {
122            r.writeTo(out);
123        }
124    }
125}
