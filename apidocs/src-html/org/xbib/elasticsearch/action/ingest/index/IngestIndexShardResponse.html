001    
002    package org.xbib.elasticsearch.action.ingest.index;
003    
004    import org.elasticsearch.action.ActionResponse;
005    import org.elasticsearch.common.io.stream.StreamInput;
006    import org.elasticsearch.common.io.stream.StreamOutput;
007    import org.elasticsearch.index.shard.ShardId;
008    
009    import org.xbib.elasticsearch.action.ingest.IngestItemFailure;
010    
011    import java.io.IOException;
012    import java.util.List;
013    
014    import static org.elasticsearch.common.collect.Lists.newLinkedList;
015    
016    public class IngestIndexShardResponse extends ActionResponse {
017    
018        private ShardId shardId;
019    
020        private int successSize;
021    
022        private List<IngestItemFailure> failure;
023    
024        public IngestIndexShardResponse() {
025            this.failure = newLinkedList();
026        }
027    
028        public IngestIndexShardResponse(ShardId shardId, int successSize, List<IngestItemFailure> failure) {
029            this.shardId = shardId;
030            this.successSize = successSize;
031            this.failure = failure;
032        }
033    
034        public ShardId shardId() {
035            return shardId;
036        }
037    
038        public int successSize() {
039            return successSize;
040        }
041    
042        public List<IngestItemFailure> failure() {
043            return failure;
044        }
045    
046        @Override
047        public void readFrom(StreamInput in) throws IOException {
048            shardId = ShardId.readShardId(in);
049            successSize = in.readVInt();
050            failure = newLinkedList();
051            for (int i = 0; i < in.readVInt(); i++) {
052                failure.add(new IngestItemFailure(in.readVInt(), in.readString()));
053            }
054        }
055    
056        @Override
057        public void writeTo(StreamOutput out) throws IOException {
058            shardId.writeTo(out);
059            out.writeVInt(successSize);
060            out.writeVInt(failure.size());
061            for (IngestItemFailure f : failure) {
062                out.writeVInt(f.pos());
063                out.writeString(f.message());
064            }
065        }
066    }
