001    
002    package org.xbib.elasticsearch.action.ingest.index;
003    
004    import org.elasticsearch.action.ActionRequest;
005    import org.elasticsearch.action.ActionRequestValidationException;
006    import org.elasticsearch.action.WriteConsistencyLevel;
007    import org.elasticsearch.action.index.IndexRequest;
008    import org.elasticsearch.action.support.replication.ReplicationType;
009    import org.elasticsearch.common.io.stream.StreamInput;
010    import org.elasticsearch.common.io.stream.StreamOutput;
011    import org.elasticsearch.common.unit.TimeValue;
012    
013    import java.io.IOException;
014    import java.util.Queue;
015    import java.util.concurrent.atomic.AtomicLong;
016    
017    import static org.elasticsearch.action.ValidateActions.addValidationError;
018    import static org.elasticsearch.common.collect.Queues.newConcurrentLinkedQueue;
019    
020    public class IngestIndexRequest extends ActionRequest {
021    
022        private static final int REQUEST_OVERHEAD = 50;
023    
024        private final Queue<IndexRequest> requests = newQueue();
025    
026        private final AtomicLong sizeInBytes = new AtomicLong();
027    
028        private ReplicationType replicationType = ReplicationType.DEFAULT;
029    
030        private WriteConsistencyLevel consistencyLevel = WriteConsistencyLevel.DEFAULT;
031    
032        private TimeValue timeout = IngestIndexShardRequest.DEFAULT_TIMEOUT;
033    
034        private String defaultIndex;
035    
036        private String defaultType;
037    
038        protected Queue<IndexRequest> newQueue() {
039            return newConcurrentLinkedQueue();
040        }
041    
042        public IngestIndexRequest setIndex(String index) {
043            this.defaultIndex = index;
044            return this;
045        }
046    
047        public String getIndex() {
048            return defaultIndex;
049        }
050    
051        public IngestIndexRequest setType(String type) {
052            this.defaultType = type;
053            return this;
054        }
055    
056        public String getType() {
057            return defaultType;
058        }
059    
060        public Queue<IndexRequest> requests() {
061            return requests;
062        }
063    
064        public IngestIndexRequest add(IndexRequest... requests) {
065            for (IndexRequest request : requests) {
066                add(request);
067            }
068            return this;
069        }
070    
071        /**
072         * Adds a list of requests to be executed. Either index or delete requests.
073         */
074        public IngestIndexRequest add(Iterable<IndexRequest> requests) {
075            for (IndexRequest request : requests) {
076                add(request);
077            }
078            return this;
079        }
080    
081        /**
082         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows
083         * the same behavior of {@link org.elasticsearch.action.index.IndexRequest} (for example, if no id is
084         * provided, one will be generated, or usage of the create flag).
085         */
086        public IngestIndexRequest add(IndexRequest request) {
087            request.beforeLocalFork();
088            return internalAdd(request);
089        }
090    
091        IngestIndexRequest internalAdd(IndexRequest request) {
092            requests.offer(request);
093            sizeInBytes.addAndGet(request.source().length() + REQUEST_OVERHEAD);
094            return this;
095        }
096    
097        /**
098         * The number of actions in the bulk request.
099         */
100        public int numberOfActions() {
101            // for ConcurrentLinkedQueue, this call is not O(n), and may not be the size of the current list
102            return requests.size();
103        }
104    
105        /**
106         * The estimated size in bytes of the bulk request.
107         */
108        public long estimatedSizeInBytes() {
109            return sizeInBytes.longValue();
110        }
111    
112        /**
113         * Sets the consistency level of write. Defaults to
114         * {@link org.elasticsearch.action.WriteConsistencyLevel#DEFAULT}
115         */
116        public IngestIndexRequest consistencyLevel(WriteConsistencyLevel consistencyLevel) {
117            this.consistencyLevel = consistencyLevel;
118            return this;
119        }
120    
121        public WriteConsistencyLevel consistencyLevel() {
122            return this.consistencyLevel;
123        }
124    
125        /**
126         * Set the replication type for this operation.
127         */
128        public IngestIndexRequest replicationType(ReplicationType replicationType) {
129            this.replicationType = replicationType;
130            return this;
131        }
132    
133        public ReplicationType replicationType() {
134            return this.replicationType;
135        }
136    
137        /**
138         * Set the timeout for this operation.
139         */
140        public IngestIndexRequest timeout(TimeValue timeout) {
141            this.timeout = timeout;
142            return this;
143        }
144    
145        public TimeValue timeout() {
146            return this.timeout;
147        }
148    
149        /**
150         * Take all requests out of this bulk request.
151         * This method is thread safe.
152         *
153         * @return another bulk request
154         */
155        public IngestIndexRequest takeAll() {
156            IngestIndexRequest request = new IngestIndexRequest();
157            while (!requests.isEmpty()) {
158                IndexRequest indexRequest = requests.poll();
159                request.add(indexRequest);
160                long length = indexRequest.source() != null ? indexRequest.source().length() + REQUEST_OVERHEAD : REQUEST_OVERHEAD;
161                sizeInBytes.addAndGet(-length);
162            }
163            return request;
164        }
165    
166        /**
167         * Take a number of requests out of this bulk request and put them
168         * into an array list.
169         *
170         * This method is thread safe.
171         *
172         * @param numRequests number of requests
173         * @return a partial bulk request
174         */
175        public IngestIndexRequest take(int numRequests) {
176            IngestIndexRequest request = new IngestIndexRequest();
177            for (int i = 0; i < numRequests; i++) {
178                IndexRequest indexRequest = requests.poll();
179                request.add(indexRequest);
180                long length = indexRequest.source() != null ? indexRequest.source().length() + REQUEST_OVERHEAD : REQUEST_OVERHEAD;
181                sizeInBytes.addAndGet(-length);
182            }
183            return request;
184        }
185    
186        @Override
187        public ActionRequestValidationException validate() {
188            ActionRequestValidationException validationException = null;
189            if (requests.isEmpty()) {
190                validationException = addValidationError("no requests added", null);
191            }
192            for (ActionRequest request : requests) {
193                ActionRequestValidationException ex = request.validate();
194                if (ex != null) {
195                    if (validationException == null) {
196                        validationException = new ActionRequestValidationException();
197                    }
198                    validationException.addValidationErrors(ex.validationErrors());
199                }
200            }
201            return validationException;
202        }
203    
204        @Override
205        public void readFrom(StreamInput in) throws IOException {
206            replicationType = ReplicationType.fromId(in.readByte());
207            consistencyLevel = WriteConsistencyLevel.fromId(in.readByte());
208            timeout = TimeValue.readTimeValue(in);
209            int size = in.readVInt();
210            for (int i = 0; i < size; i++) {
211                IndexRequest request = new IndexRequest();
212                request.readFrom(in);
213                requests.add(request);
214            }
215        }
216    
217        @Override
218        public void writeTo(StreamOutput out) throws IOException {
219            out.writeByte(replicationType.id());
220            out.writeByte(consistencyLevel.id());
221            timeout.writeTo(out);
222            out.writeVInt(requests.size());
223            for (ActionRequest request : requests) {
224                request.writeTo(out);
225            }
226        }
227    }
