105    @Override
106    protected PrimaryResponse<IngestIndexShardResponse, IngestIndexShardRequest> shardOperationOnPrimary(ClusterState clusterState, PrimaryOperationRequest shardRequest) {
107        final IngestIndexShardRequest request = shardRequest.request;
108        IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
109        Set<Tuple<String, String>> mappingsToUpdate = null;
110        int successSize = 0;
111        List<IngestItemFailure> failure = Lists.newLinkedList();
112        int size = request.items().size();
113        long[] versions = new long[size];
114        for (int i = 0; i < size; i++) {
115            IngestIndexItemRequest item = request.items().get(i);
116            IndexRequest indexRequest = (IndexRequest) item.request();
117            Engine.IndexingOperation op = null;
118            try {
119                // validate, if routing is required, that we got routing
120                MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
121                if (mappingMd != null && mappingMd.routing().required()) {
122                    if (indexRequest.routing() == null) {
123                        throw new RoutingMissingException(indexRequest.index(), indexRequest.type(), indexRequest.id());
124                    }
125                }
126                SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
127                        .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
128                long version;
129                if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
130                    Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
131                    op = index;
132                    indexShard.index(index);
133                    version = index.version();
134                } else {
135                    Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
136                    op = create;
137                    indexShard.create(create);
138                    version = create.version();
139                }
140                versions[i] = indexRequest.version();
141                // update the version on request so it will happen on the replicas
142                indexRequest.version(version);
143                successSize++;
144            } catch (Throwable e) {
145                // rethrow the failure if we are going to retry on primary and let parent failure to handle it
146                if (retryPrimaryException(e)) {
147                    // restore updated versions...
148                    for (int j = 0; j < i; j++) {
149                        ((IndexRequest) request.items().get(j).request()).version(versions[j]);
150                    }
151                    logger.error(e.getMessage(), e);
152                    throw new ElasticsearchException(e.getMessage());
153                }
154                if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
155                    logger.trace("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
156                } else {
157                    logger.debug("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
158                }
159                failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
160                // nullify the request so it won't execute on the replicas
161                request.items().set(i, null);
162            } finally {
163                // update mapping on master if needed, we won't update changes to the same type, since once its changed, it won't have mappers added
164                if (op != null && op.parsedDoc().mappingsModified()) {
165                    if (mappingsToUpdate == null) {
166                        mappingsToUpdate = Sets.newHashSet();
167                    }
168                    mappingsToUpdate.add(Tuple.tuple(indexRequest.index(), indexRequest.type()));
169                }
170            }
171        }
172        if (mappingsToUpdate != null) {
173            for (Tuple<String, String> mappingToUpdate : mappingsToUpdate) {
174                updateMappingOnMaster(mappingToUpdate.v1(), mappingToUpdate.v2());
175            }
176        }
177        IngestIndexShardResponse response = new IngestIndexShardResponse(new ShardId(request.index(), request.shardId()), successSize, failure);
178        return new PrimaryResponse<IngestIndexShardResponse, IngestIndexShardRequest>(shardRequest.request, response, null);
179    }

181    @Override
182    protected void shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
183        IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
184        final IngestIndexShardRequest request = shardRequest.request;
185        int size = request.items().size();
186        for (int i = 0; i < size; i++) {
187            IngestIndexItemRequest item = request.items().get(i);
188            if (item == null) {
189                continue;
190            }
191            IndexRequest indexRequest = (IndexRequest) item.request();
192            try {
193                SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.REPLICA, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
194                        .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
195
196                if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
197                    Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
198                    indexShard.index(index);
199                } else {
200                    Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
201                    indexShard.create(create);
202                }
203            } catch (Throwable e) {
204                // ignore, we are on backup
205            }
206        }
207    }

209    private void updateMappingOnMaster(final String index, final String type) {
210        try {
211            MapperService mapperService = indicesService.indexServiceSafe(index).mapperService();
212            final DocumentMapper documentMapper = mapperService.documentMapper(type);
213            if (documentMapper == null) { // should not happen
214                return;
215            }
216            IndexMetaData metaData = clusterService.state().metaData().index(index);
217            if (metaData == null) {
218                return;
219            }
220            long orderId = mappingUpdatedAction.generateNextMappingUpdateOrder();
221            documentMapper.refreshSource();
222            DiscoveryNode node = clusterService.localNode();
223            final MappingUpdatedAction.MappingUpdatedRequest request = new MappingUpdatedAction.MappingUpdatedRequest(index, metaData.uuid(), type, documentMapper.mappingSource(), orderId, node != null ? node.id() : null);
224            mappingUpdatedAction.execute(request, new ActionListener<MappingUpdatedAction.MappingUpdatedResponse>() {
225                @Override
226                public void onResponse(MappingUpdatedAction.MappingUpdatedResponse mappingUpdatedResponse) {
227                    // all is well
228                }
229
230                @Override
231                public void onFailure(Throwable e) {
232                    try {
233                        logger.warn("failed to update master on updated mapping for index [{}], type [{}] and source [{}]", e, index, type, documentMapper.mappingSource().string());
234                    } catch (IOException e1) {
235                        // ignore
236                    }
237                }
238            });
239        } catch (Throwable e) {
240            logger.warn("failed to update master on updated mapping for index [{}], type [{}]", e, index, type);
241        }
242    }
