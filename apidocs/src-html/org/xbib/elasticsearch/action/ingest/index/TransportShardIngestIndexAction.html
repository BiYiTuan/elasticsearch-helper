105        @Override
106        protected PrimaryResponse<IngestIndexShardResponse, IngestIndexShardRequest> shardOperationOnPrimary(ClusterState clusterState, PrimaryOperationRequest shardRequest) {
107            final IngestIndexShardRequest request = shardRequest.request;
108            IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
109            Set<Tuple<String, String>> mappingsToUpdate = null;
110            int successSize = 0;
111            List<IngestItemFailure> failure = Lists.newLinkedList();
112            int size = request.items().size();
113            long[] versions = new long[size];
114            for (int i = 0; i < size; i++) {
115                IngestIndexItemRequest item = request.items().get(i);
116                IndexRequest indexRequest = (IndexRequest) item.request();
117                try {
118                    // validate, if routing is required, that we got routing
119                    MappingMetaData mappingMd = clusterState.metaData().index(request.index()).mappingOrDefault(indexRequest.type());
120                    if (mappingMd != null && mappingMd.routing().required()) {
121                        if (indexRequest.routing() == null) {
122                            throw new RoutingMissingException(indexRequest.index(), indexRequest.type(), indexRequest.id());
123                        }
124                    }
125                    SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.PRIMARY, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
126                            .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
127                    long version;
128                    Engine.IndexingOperation op;
129                    if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
130                        Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
131                        indexShard.index(index);
132                        version = index.version();
133                        op = index;
134                    } else {
135                        Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
136                        indexShard.create(create);
137                        version = create.version();
138                        op = create;
139                    }
140                    versions[i] = indexRequest.version();
141                    // update the version on request so it will happen on the replicas
142                    indexRequest.version(version);
143                    // update mapping on master if needed, we won't update changes to the same type, since once its changed, it won't have mappers added
144                    if (op.parsedDoc().mappingsModified()) {
145                        if (mappingsToUpdate == null) {
146                            mappingsToUpdate = Sets.newHashSet();
147                        }
148                        mappingsToUpdate.add(Tuple.tuple(indexRequest.index(), indexRequest.type()));
149                    }
150                    successSize++;
151                } catch (Throwable e) {
152                    // rethrow the failure if we are going to retry on primary and let parent failure to handle it
153                    if (retryPrimaryException(e)) {
154                        // restore updated versions...
155                        for (int j = 0; j < i; j++) {
156                            ((IndexRequest) request.items().get(j).request()).version(versions[j]);
157                        }
158                        logger.error(e.getMessage(), e);
159                        throw new ElasticsearchException(e.getMessage());
160                    }
161                    if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
162                        logger.trace("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
163                    } else {
164                        logger.debug("[{}][{}] failed to execute bulk item (index) {}", e, shardRequest.request.index(), shardRequest.shardId, indexRequest);
165                    }
166                    failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
167                    // nullify the request so it won't execute on the replicas
168                    request.items().set(i, null);
169                }
170            }
171            if (mappingsToUpdate != null) {
172                for (Tuple<String, String> mappingToUpdate : mappingsToUpdate) {
173                    updateMappingOnMaster(mappingToUpdate.v1(), mappingToUpdate.v2());
174                }
175            }
176            IngestIndexShardResponse response = new IngestIndexShardResponse(new ShardId(request.index(), request.shardId()), successSize, failure);
177            return new PrimaryResponse<IngestIndexShardResponse, IngestIndexShardRequest>(shardRequest.request, response, null);
178        }

180        @Override
181        protected void shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
182            IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
183            final IngestIndexShardRequest request = shardRequest.request;
184            int size = request.items().size();
185            for (int i = 0; i < size; i++) {
186                IngestIndexItemRequest item = request.items().get(i);
187                if (item == null) {
188                    continue;
189                }
190                IndexRequest indexRequest = (IndexRequest) item.request();
191                try {
192                    SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.REPLICA, indexRequest.source()).type(indexRequest.type()).id(indexRequest.id())
193                            .routing(indexRequest.routing()).parent(indexRequest.parent()).timestamp(indexRequest.timestamp()).ttl(indexRequest.ttl());
194    
195                    if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
196                        Engine.Index index = indexShard.prepareIndex(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
197                        indexShard.index(index);
198                    } else {
199                        Engine.Create create = indexShard.prepareCreate(sourceToParse).version(indexRequest.version()).origin(Engine.Operation.Origin.REPLICA);
200                        indexShard.create(create);
201                    }
202                } catch (Throwable e) {
203                    // ignore, we are on backup
204                }
205            }
206        }

208        private void updateMappingOnMaster(final String index, final String type) {
209            try {
210                MapperService mapperService = indicesService.indexServiceSafe(index).mapperService();
211                final DocumentMapper documentMapper = mapperService.documentMapper(type);
212                if (documentMapper == null) { // should not happen
213                    return;
214                }
215                IndexMetaData metaData = clusterService.state().metaData().index(index);
216                if (metaData == null) {
217                    return;
218                }
219                long orderId = mappingUpdatedAction.generateNextMappingUpdateOrder();
220                documentMapper.refreshSource();
221                DiscoveryNode node = clusterService.localNode();
222                final MappingUpdatedAction.MappingUpdatedRequest request = new MappingUpdatedAction.MappingUpdatedRequest(index, metaData.uuid(), type, documentMapper.mappingSource(), orderId, node != null ? node.id() : null);
223                mappingUpdatedAction.execute(request, new ActionListener<MappingUpdatedAction.MappingUpdatedResponse>() {
224                    @Override
225                    public void onResponse(MappingUpdatedAction.MappingUpdatedResponse mappingUpdatedResponse) {
226                        // all is well
227                    }
228    
229                    @Override
230                    public void onFailure(Throwable e) {
231                        try {
232                            logger.warn("failed to update master on updated mapping for index [{}], type [{}] and source [{}]", e, index, type, documentMapper.mappingSource().string());
233                        } catch (IOException e1) {
234                            // ignore
235                        }
236                    }
237                });
238            } catch (Throwable e) {
239                logger.warn("failed to update master on updated mapping for index [{}], type [{}]", e, index, type);
240            }
241        }
