002package org.xbib.elasticsearch.action.ingest.index;
003
004import org.elasticsearch.ElasticsearchIllegalStateException;
005import org.elasticsearch.action.ActionListener;
006import org.elasticsearch.action.WriteConsistencyLevel;
007import org.elasticsearch.action.index.IndexRequest;
008import org.elasticsearch.action.support.replication.ReplicationType;
009import org.elasticsearch.client.Client;
010import org.elasticsearch.common.unit.ByteSizeValue;
011import org.elasticsearch.common.unit.ByteSizeUnit;
012import org.elasticsearch.common.unit.TimeValue;
013
014import org.xbib.elasticsearch.action.ingest.IngestResponse;
015
016import java.util.concurrent.Semaphore;
017import java.util.concurrent.TimeUnit;
018import java.util.concurrent.atomic.AtomicLong;
019
020public class IngestIndexProcessor {
021
022    private final Client client;
023
024    private final int concurrency;
025
026    private final int actions;
027
028    private final ByteSizeValue maxVolume;
029
030    private final TimeValue waitForResponses;
031
032    private final Semaphore semaphore;
033
034    private final AtomicLong bulkId;
035
036    private final IngestIndexRequest ingestRequest;
037
038    private Listener listener;
039
040    private volatile boolean closed = false;
041
042    public IngestIndexProcessor(Client client, Integer concurrency, Integer actions,
043                                ByteSizeValue maxVolume, TimeValue waitForResponses) {
044        this.client = client;
045        this.concurrency = concurrency != null ?
046                Math.min(Math.abs(concurrency), 256) :
047                Runtime.getRuntime().availableProcessors() * 4;
048        this.actions = actions != null ? Math.min(actions, 32768) : 1000;
049        this.maxVolume = maxVolume != null ?
050                new ByteSizeValue(Math.max(maxVolume.bytes(), 1024), ByteSizeUnit.BYTES) :
051                new ByteSizeValue(10, ByteSizeUnit.MB);
052        this.waitForResponses = waitForResponses != null ?
053                new TimeValue(Math.max(waitForResponses.millis(), 1000), TimeUnit.MILLISECONDS) :
054                new TimeValue(60, TimeUnit.SECONDS);
055        this.semaphore = new Semaphore(this.concurrency);
056        this.bulkId = new AtomicLong(0L);
057        this.ingestRequest = new IngestIndexRequest();
058    }
059
060    public IngestIndexProcessor listener(Listener listener) {
061        this.listener = listener;
062        return this;
063    }
064
065    public IngestIndexProcessor listenerThreaded(boolean threaded) {
066        ingestRequest.listenerThreaded(threaded);
067        return this;
068    }
069
070    public IngestIndexProcessor replicationType(ReplicationType type) {
071        ingestRequest.replicationType(type);
072        return this;
073    }
074
075    public IngestIndexProcessor consistencyLevel(WriteConsistencyLevel level) {
076        ingestRequest.consistencyLevel(level);
077        return this;
078    }
079
080    /**
081     * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list
082     * of actions to execute.
083     */
084    public IngestIndexProcessor add(IndexRequest request) {
085        ingestRequest.add(request);
086        flushIfNeeded(listener);
087        return this;
088    }
089
090    /**
091     * Closes the processor.
092     * Any remaining actions are flushed, and for the bulk responses is being waited.
093     */
094    public void close() throws InterruptedException {
095        if (closed) {
096            throw new ElasticsearchIllegalStateException("processor already closed");
097        }
098        closed = true;
099        flush();
100        waitForResponses(waitForResponses);
101    }
102
103    /**
104     * Flush this processor, write all requests.
105     */
106    public synchronized void flush() {
107        if (ingestRequest.numberOfActions() > 0) {
108            process(ingestRequest.takeAll(), listener);
109        }
110    }
111
112    /**
113     * Critical phase, check if flushing condition is met and
114     * push the part of the bulk requests that is required to process
115     */
116    private synchronized void flushIfNeeded(Listener listener) {
117        if (closed) {
118            throw new ElasticsearchIllegalStateException("processor already closed");
119        }
120        if (actions > 0) {
121            while (ingestRequest.numberOfActions() >= actions) {
122                process(ingestRequest.take(actions), listener);
123            }
124        } else {
125            while (ingestRequest.numberOfActions() > 0
126                && maxVolume.bytesAsInt() > 0
127                && ingestRequest.estimatedSizeInBytes() > maxVolume.bytesAsInt()) {
128                process(ingestRequest.takeAll(), listener);
129            }
130        }
131    }
132
133    /**
134     * Process the ingest index request.
135     *
136     * @param request the ingest request
137     * @param listener the listener
138     */
139    private void process(final IngestIndexRequest request, final Listener listener) {
140        if (request.numberOfActions() == 0) {
141            return;
142        }
143        if (listener == null) {
144            return;
145        }
146        final long id = bulkId.incrementAndGet();
147        boolean done = false;
148        try {
149            semaphore.acquire();
150            listener.beforeBulk(id, concurrency - semaphore.availablePermits(), request);
151            client.execute(IngestIndexAction.INSTANCE, request, new ActionListener<IngestResponse>() {
152                @Override
153                public void onResponse(IngestResponse response) {
154                    try {
155                        listener.afterBulk(id, concurrency - semaphore.availablePermits(), response);
156                    } finally {
157                        semaphore.release();
158                    }
159                }
160
161                @Override
162                public void onFailure(Throwable e) {
163                    try {
164                        listener.afterBulk(id, concurrency - semaphore.availablePermits(), e);
165                    } finally {
166                        semaphore.release();
167                    }
168                }
169            });
170            done = true;
171        } catch (InterruptedException e) {
172            // semaphore not acquired
173            Thread.currentThread().interrupt();
174            listener.afterBulk(id, concurrency - semaphore.availablePermits(), e);
175        } finally {
176            if (!done) {
177               semaphore.release();
178            }
179        }
180    }
181
182    /**
183     * Wait for responses of outstanding requests.
184     *
185     * @return true if all requests answered within the waiting time, false if not
186     * @throws InterruptedException
187     */
188    public boolean waitForResponses(TimeValue maxWait) throws InterruptedException {
189        semaphore.tryAcquire(concurrency, maxWait.getMillis(), TimeUnit.MILLISECONDS);
190        return semaphore.availablePermits() == concurrency;
191    }
192
193    /**
194     * A listener for the execution.
195     */
196    public static interface Listener {
197
198        /**
199         * Callback before the bulk is executed.
200         */
201        void beforeBulk(long bulkId, int concurrency, IngestIndexRequest request);
202
203        /**
204         * Callback after a successful execution of bulk request.
205         */
206        void afterBulk(long bulkId, int concurrency, IngestResponse response);
207
208        /**
209         * Callback after a failed execution of bulk request.
210         */
211        void afterBulk(long bulkId, int concurrency, Throwable failure);
212    }
213
