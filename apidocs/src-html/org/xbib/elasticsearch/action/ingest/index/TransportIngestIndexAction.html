002    package org.xbib.elasticsearch.action.ingest.index;
003    
004    import org.elasticsearch.ExceptionsHelper;
005    import org.elasticsearch.action.ActionListener;
006    import org.elasticsearch.action.ActionRequest;
007    import org.elasticsearch.action.index.IndexRequest;
008    import org.elasticsearch.action.support.TransportAction;
009    import org.elasticsearch.cluster.ClusterService;
010    import org.elasticsearch.cluster.ClusterState;
011    import org.elasticsearch.cluster.block.ClusterBlockLevel;
012    import org.elasticsearch.cluster.metadata.MappingMetaData;
013    import org.elasticsearch.cluster.metadata.MetaData;
014    import org.elasticsearch.common.inject.Inject;
015    import org.elasticsearch.common.settings.Settings;
016    import org.elasticsearch.index.shard.ShardId;
017    import org.elasticsearch.threadpool.ThreadPool;
018    import org.elasticsearch.transport.BaseTransportRequestHandler;
019    import org.elasticsearch.transport.TransportChannel;
020    import org.elasticsearch.transport.TransportService;
021    
022    import org.xbib.elasticsearch.action.ingest.IngestItemFailure;
023    import org.xbib.elasticsearch.action.ingest.IngestResponse;
024    
025    import java.util.List;
026    import java.util.Map;
027    import java.util.concurrent.atomic.AtomicInteger;
028    
029    import static org.elasticsearch.common.collect.Lists.newLinkedList;
030    import static org.elasticsearch.common.collect.Maps.newHashMap;
031    
032    public class TransportIngestIndexAction extends TransportAction<IngestIndexRequest, IngestResponse> {
033    
034        private final boolean allowIdGeneration;
035    
036        private final ClusterService clusterService;
037    
038        private final TransportShardIngestIndexAction shardBulkAction;
039    
040        @Inject
041        public TransportIngestIndexAction(Settings settings,
042                                          ThreadPool threadPool,
043                                          TransportService transportService,
044                                          ClusterService clusterService,
045                                          TransportShardIngestIndexAction shardBulkAction) {
046            super(settings, threadPool);
047            this.clusterService = clusterService;
048            this.shardBulkAction = shardBulkAction;
049    
050            this.allowIdGeneration = componentSettings.getAsBoolean("action.allow_id_generation", true);
051    
052            transportService.registerHandler(IngestIndexAction.NAME, new IngestTransportHandler());
053        }
054    
055        @Override
056        protected void doExecute(final IngestIndexRequest ingestRequest, final ActionListener<IngestResponse> listener) {
057            executeBulk(ingestRequest, listener);
058        }
059    
060        private void executeBulk(final IngestIndexRequest ingestRequest, final ActionListener<IngestResponse> listener) {
061            final long startTime = System.currentTimeMillis();
062            ClusterState clusterState = clusterService.state();
063            // TODO use timeout to wait here if its blocked...
064            clusterState.blocks().globalBlockedRaiseException(ClusterBlockLevel.WRITE);
065            MetaData metaData = clusterState.metaData();
066            for (ActionRequest request : ingestRequest.requests()) {
067                IndexRequest indexRequest = (IndexRequest) request;
068                String aliasOrIndex = indexRequest.index();
069                indexRequest.index(clusterState.metaData().concreteIndex(indexRequest.index()));
070                MappingMetaData mappingMd = null;
071                if (metaData.hasIndex(indexRequest.index())) {
072                    mappingMd = metaData.index(indexRequest.index()).mappingOrDefault(indexRequest.type());
073                }
074                indexRequest.process(metaData, aliasOrIndex, mappingMd, allowIdGeneration);
075            }
076            // first, go over all the requests and create a ShardId -> Operations mapping
077            Map<ShardId, List<IngestIndexItemRequest>> requestsByShard = newHashMap();
078            int i = 0;
079            for (ActionRequest request : ingestRequest.requests()) {
080                IndexRequest indexRequest = (IndexRequest) request;
081                ShardId shardId = clusterService.operationRouting().indexShards(clusterState, indexRequest.index(), indexRequest.type(), indexRequest.id(), indexRequest.routing()).shardId();
082                List<IngestIndexItemRequest> list = requestsByShard.get(shardId);
083                if (list == null) {
084                    list = newLinkedList();
085                    requestsByShard.put(shardId, list);
086                }
087                list.add(new IngestIndexItemRequest(i, request));
088                i++;
089            }
090            final AtomicInteger successSize = new AtomicInteger(0);
091            final List<IngestItemFailure> failure = newLinkedList();
092            if (requestsByShard.isEmpty()) {
093                listener.onResponse(new IngestResponse(0, failure, System.currentTimeMillis() - startTime));
094                return;
095            }
096            final AtomicInteger counter = new AtomicInteger(requestsByShard.size());
097            for (Map.Entry<ShardId, List<IngestIndexItemRequest>> entry : requestsByShard.entrySet()) {
098                final ShardId shardId = entry.getKey();
099                final List<IngestIndexItemRequest> requests = entry.getValue();
100                IngestIndexShardRequest ingestShardRequest = new IngestIndexShardRequest(shardId.index().name(), shardId.id(), requests);
101                ingestShardRequest.replicationType(ingestRequest.replicationType());
102                ingestShardRequest.consistencyLevel(ingestRequest.consistencyLevel());
103                ingestShardRequest.timeout(ingestRequest.timeout());
104                shardBulkAction.execute(ingestShardRequest, new ActionListener<IngestIndexShardResponse>() {
105                    @Override
106                    public void onResponse(IngestIndexShardResponse ingestShardResponse) {
107                        successSize.addAndGet(ingestShardResponse.successSize());
108                        if (counter.decrementAndGet() == 0) {
109                            finishHim();
110                        }
111                    }
112    
113                    @Override
114                    public void onFailure(Throwable e) {
115                        // create failures for all relevant requests
116                        String message = ExceptionsHelper.detailedMessage(e);
117                        synchronized (failure) {
118                            for (IngestIndexItemRequest request : requests) {
119                                failure.add(new IngestItemFailure(request.id(), message));
120                            }
121                        }
122                        if (counter.decrementAndGet() == 0) {
123                            finishHim();
124                        }
125                    }
126    
127                    private void finishHim() {
128                        listener.onResponse(new IngestResponse(successSize.get(), failure, System.currentTimeMillis() - startTime));
129                    }
130                });
131            }
132        }
133    
134        class IngestTransportHandler extends BaseTransportRequestHandler<IngestIndexRequest> {
135    
136            @Override
137            public IngestIndexRequest newInstance() {
138                return new IngestIndexRequest();
139            }
140    
141            @Override
142            public void messageReceived(final IngestIndexRequest request, final TransportChannel channel) throws Exception {
143                // no need to use threaded listener, since we just send a response
144                request.listenerThreaded(false);
145                execute(request, new ActionListener<IngestResponse>() {
146                    @Override
147                    public void onResponse(IngestResponse result) {
148                        try {
149                            channel.sendResponse(result);
150                        } catch (Exception e) {
151                            onFailure(e);
152                        }
153                    }
154    
155                    @Override
156                    public void onFailure(Throwable e) {
157                        try {
158                            channel.sendResponse(e);
159                        } catch (Exception e1) {
160                            logger.warn("Failed to send error response for request [" + request + "]", e1);
161                        }
162                    }
163                });
164            }
165    
166            @Override
167            public String executor() {
168                return ThreadPool.Names.SAME;
169            }
170        }
171    }
