002    package org.xbib.elasticsearch.action.ingest.index;
003    
004    import org.elasticsearch.action.ActionRequest;
005    import org.elasticsearch.common.io.stream.StreamInput;
006    import org.elasticsearch.common.io.stream.StreamOutput;
007    import org.elasticsearch.common.io.stream.Streamable;
008    
009    import java.io.IOException;
010    
011    public class IngestIndexItemRequest implements Streamable {
012    
013        private int id;
014    
015        private ActionRequest request;
016    
017        IngestIndexItemRequest() {
018        }
019    
020        public IngestIndexItemRequest(int id, ActionRequest request) {
021            this.id = id;
022            this.request = request;
023        }
024    
025        public int id() {
026            return id;
027        }
028    
029        public ActionRequest request() {
030            return request;
031        }
032    
033        public static IngestIndexItemRequest readBulkItem(StreamInput in) throws IOException {
034            IngestIndexItemRequest item = new IngestIndexItemRequest();
035            item.readFrom(in);
036            return item;
037        }
038    
039        @Override
040        public void readFrom(StreamInput in) throws IOException {
041            id = in.readVInt();
042            request = new IngestIndexRequest();
043            request.readFrom(in);
044        }
045    
046        @Override
047        public void writeTo(StreamOutput out) throws IOException {
048            out.writeVInt(id);
049            request.writeTo(out);
050        }
051    }
