002    package org.xbib.elasticsearch.action.ingest.index;
003    
004    import org.elasticsearch.action.ActionListener;
005    import org.elasticsearch.action.ActionRequestBuilder;
006    import org.elasticsearch.action.ListenableActionFuture;
007    import org.elasticsearch.action.WriteConsistencyLevel;
008    import org.elasticsearch.action.index.IndexRequest;
009    import org.elasticsearch.action.index.IndexRequestBuilder;
010    import org.elasticsearch.action.support.PlainListenableActionFuture;
011    import org.elasticsearch.action.support.replication.ReplicationType;
012    import org.elasticsearch.client.Client;
013    import org.elasticsearch.client.internal.InternalClient;
014    
015    import org.elasticsearch.common.unit.TimeValue;
016    import org.xbib.elasticsearch.action.ingest.IngestResponse;
017    
018    /**
019     * A bulk request holds ordered {@link org.xbib.elasticsearch.action.ingest.index.IngestIndexRequest}s
020     * and allows to executes it in a single batch.
021     */
022    public class IngestIndexRequestBuilder extends ActionRequestBuilder<IngestIndexRequest, IngestResponse, IngestIndexRequestBuilder> {
023    
024        public IngestIndexRequestBuilder(Client client) {
025            super((InternalClient)client, new IngestIndexRequest());
026        }
027    
028        public IngestIndexRequest request() {
029            return this.request;
030        }
031    
032        @Override
033        public ListenableActionFuture<IngestResponse> execute() {
034            PlainListenableActionFuture<IngestResponse> future = new PlainListenableActionFuture<IngestResponse>(request.listenerThreaded(), client.threadPool());
035            execute(future);
036            return future;
037        }
038    
039        @Override
040        public void execute(ActionListener<IngestResponse> listener) {
041            doExecute(listener);
042        }
043    
044        public IngestIndexRequestBuilder setIndex(String index) {
045            this.request.setIndex(index);
046            return this;
047        }
048    
049        public IngestIndexRequestBuilder setType(String index) {
050            this.request.setIndex(index);
051            return this;
052        }
053    
054        /**
055         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute.
056         * Follows the same behavior of {@link org.elasticsearch.action.index.IndexRequest}
057         * (for example, if no id is provided, one will be generated, or usage of the create flag).
058         */
059        public IngestIndexRequestBuilder add(IndexRequest request) {
060            this.request.add(request);
061            return this;
062        }
063    
064        /**
065         * Adds an {@link org.elasticsearch.action.index.IndexRequestBuilder} to the list of actions to execute.
066         * Follows the same behavior of {@link org.elasticsearch.action.index.IndexRequest}
067         * (for example, if no id is provided, one will be generated, or usage of the create flag).
068         */
069        public IngestIndexRequestBuilder add(IndexRequestBuilder request) {
070            this.request.add(request.request());
071            return this;
072        }
073    
074        /**
075         * Set the replication type for this operation.
076         */
077        public IngestIndexRequestBuilder setReplicationType(ReplicationType replicationType) {
078            this.request.replicationType(replicationType);
079            return this;
080        }
081    
082        /**
083         * Sets the consistency level. Defaults to {@link org.elasticsearch.action.WriteConsistencyLevel#DEFAULT}.
084         */
085        public IngestIndexRequestBuilder setConsistencyLevel(WriteConsistencyLevel consistencyLevel) {
086            request.consistencyLevel(consistencyLevel);
087            return this;
088        }
089    
090        /**
091         * Sets the timeout. Defaults to {@link org.xbib.elasticsearch.action.ingest.IngestShardRequest#DEFAULT_TIMEOUT}.
092         */
093        public IngestIndexRequestBuilder setTimeout(TimeValue timeout) {
094            request.timeout(timeout);
095            return this;
096        }
097    
098        /**
099         * The number of actions currently in the bulk.
100         */
101        public int numberOfActions() {
102            return request.numberOfActions();
103        }
104    
105        protected void doExecute(ActionListener<IngestResponse> listener) {
106            ((InternalClient)client).execute(IngestIndexAction.INSTANCE, request, listener);
107        }
108    }
