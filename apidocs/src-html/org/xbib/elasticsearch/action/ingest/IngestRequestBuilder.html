001
002package org.xbib.elasticsearch.action.ingest;
003
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.ActionRequestBuilder;
006import org.elasticsearch.action.ListenableActionFuture;
007import org.elasticsearch.action.WriteConsistencyLevel;
008import org.elasticsearch.action.delete.DeleteRequest;
009import org.elasticsearch.action.delete.DeleteRequestBuilder;
010import org.elasticsearch.action.index.IndexRequest;
011import org.elasticsearch.action.index.IndexRequestBuilder;
012import org.elasticsearch.action.support.PlainListenableActionFuture;
013import org.elasticsearch.action.support.replication.ReplicationType;
014import org.elasticsearch.client.Client;
015import org.elasticsearch.client.internal.InternalClient;
016import org.elasticsearch.common.Nullable;
017
018/**
019 * A bulk request holds an ordered {@link org.elasticsearch.action.index.IndexRequest}s
020 * and {@link org.elasticsearch.action.delete.DeleteRequest}s and allows to executes
021 * it in a single batch.
022 */
023public class IngestRequestBuilder extends ActionRequestBuilder<IngestRequest, IngestResponse, IngestRequestBuilder> {
024
025
026    public IngestRequestBuilder(Client client) {
027        this(client, new IngestRequest());
028    }
029
030    public IngestRequestBuilder(Client client, IngestRequest request) {
031        super((InternalClient) client, request);
032    }
033
034    @Override
035    public IngestRequest request() {
036        return this.request;
037    }
038
039    @Override
040    public ListenableActionFuture<IngestResponse> execute() {
041        PlainListenableActionFuture<IngestResponse> future = new PlainListenableActionFuture<IngestResponse>(request.listenerThreaded(), client.threadPool());
042        execute(future);
043        return future;
044    }
045
046    @Override
047    public void execute(ActionListener<IngestResponse> listener) {
048        doExecute(listener);
049    }
050
051    /**
052     * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows the same behavior of {@link org.elasticsearch.action.index.IndexRequest}
053     * (for example, if no id is provided, one will be generated, or usage of the create flag).
054     */
055    public IngestRequestBuilder add(IndexRequest request) {
056        this.request.add(request);
057        return this;
058    }
059
060    /**
061     * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows the same behavior of {@link org.elasticsearch.action.index.IndexRequest}
062     * (for example, if no id is provided, one will be generated, or usage of the create flag).
063     */
064    public IngestRequestBuilder add(IndexRequestBuilder request) {
065        this.request.add(request.request());
066        return this;
067    }
068
069    /**
070     * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
071     */
072    public IngestRequestBuilder add(DeleteRequest request) {
073        this.request.add(request);
074        return this;
075    }
076
077    /**
078     * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
079     */
080    public IngestRequestBuilder add(DeleteRequestBuilder request) {
081        this.request.add(request.request());
082        return this;
083    }
084
085    /**
086     * Adds a framed data in binary format
087     */
088    public IngestRequestBuilder add(byte[] data, int from, int length, boolean contentUnsafe) throws Exception {
089        this.request.add(data, from, length, contentUnsafe, null, null);
090        return this;
091    }
092
093    /**
094     * Adds a framed data in binary format
095     */
096    public IngestRequestBuilder add(byte[] data, int from, int length, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
097        this.request.add(data, from, length, contentUnsafe, defaultIndex, defaultType);
098        return this;
099    }
100
101    /**
102     * Set the replication type for this operation.
103     */
104    public IngestRequestBuilder setReplicationType(ReplicationType replicationType) {
105        this.request.replicationType(replicationType);
106        return this;
107    }
108
109    /**
110     * Sets the consistency level. Defaults to {@link org.elasticsearch.action.WriteConsistencyLevel#DEFAULT}.
111     */
112    public IngestRequestBuilder setConsistencyLevel(WriteConsistencyLevel consistencyLevel) {
113        request.consistencyLevel(consistencyLevel);
114        return this;
115    }
116
117    /**
118     * The number of actions currently in the bulk.
119     */
120    public int numberOfActions() {
121        return request.numberOfActions();
122    }
123
124    protected void doExecute(ActionListener<IngestResponse> listener) {
125        ((InternalClient)client).execute(IngestAction.INSTANCE, request, listener);
126    }
127}
