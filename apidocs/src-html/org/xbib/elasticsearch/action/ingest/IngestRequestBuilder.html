002    package org.xbib.elasticsearch.action.ingest;
003    
004    import org.elasticsearch.action.ActionListener;
005    import org.elasticsearch.action.ActionRequestBuilder;
006    import org.elasticsearch.action.ListenableActionFuture;
007    import org.elasticsearch.action.WriteConsistencyLevel;
008    import org.elasticsearch.action.delete.DeleteRequest;
009    import org.elasticsearch.action.delete.DeleteRequestBuilder;
010    import org.elasticsearch.action.index.IndexRequest;
011    import org.elasticsearch.action.index.IndexRequestBuilder;
012    import org.elasticsearch.action.support.PlainListenableActionFuture;
013    import org.elasticsearch.action.support.replication.ReplicationType;
014    import org.elasticsearch.client.Client;
015    import org.elasticsearch.client.internal.InternalClient;
016    import org.elasticsearch.common.Nullable;
017    
018    /**
019     * A bulk request holds an ordered {@link org.elasticsearch.action.index.IndexRequest}s
020     * and {@link org.elasticsearch.action.delete.DeleteRequest}s and allows to executes
021     * it in a single batch.
022     */
023    public class IngestRequestBuilder extends ActionRequestBuilder<IngestRequest, IngestResponse, IngestRequestBuilder> {
024    
025        protected IngestRequestBuilder(Client client) {
026            super((InternalClient) client, new IngestRequest());
027        }
028    
029        public IngestRequest request() {
030            return this.request;
031        }
032    
033        @Override
034        public ListenableActionFuture<IngestResponse> execute() {
035            PlainListenableActionFuture<IngestResponse> future = new PlainListenableActionFuture<IngestResponse>(request.listenerThreaded(), client.threadPool());
036            execute(future);
037            return future;
038        }
039    
040        @Override
041        public void execute(ActionListener<IngestResponse> listener) {
042            doExecute(listener);
043        }
044    
045        /**
046         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows the same behavior of {@link org.elasticsearch.action.index.IndexRequest}
047         * (for example, if no id is provided, one will be generated, or usage of the create flag).
048         */
049        public IngestRequestBuilder add(IndexRequest request) {
050            this.request.add(request);
051            return this;
052        }
053    
054        /**
055         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows the same behavior of {@link org.elasticsearch.action.index.IndexRequest}
056         * (for example, if no id is provided, one will be generated, or usage of the create flag).
057         */
058        public IngestRequestBuilder add(IndexRequestBuilder request) {
059            this.request.add(request.request());
060            return this;
061        }
062    
063        /**
064         * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
065         */
066        public IngestRequestBuilder add(DeleteRequest request) {
067            this.request.add(request);
068            return this;
069        }
070    
071        /**
072         * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
073         */
074        public IngestRequestBuilder add(DeleteRequestBuilder request) {
075            this.request.add(request.request());
076            return this;
077        }
078    
079        /**
080         * Adds a framed data in binary format
081         */
082        public IngestRequestBuilder add(byte[] data, int from, int length, boolean contentUnsafe) throws Exception {
083            this.request.add(data, from, length, contentUnsafe, null, null);
084            return this;
085        }
086    
087        /**
088         * Adds a framed data in binary format
089         */
090        public IngestRequestBuilder add(byte[] data, int from, int length, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
091            this.request.add(data, from, length, contentUnsafe, defaultIndex, defaultType);
092            return this;
093        }
094    
095        /**
096         * Set the replication type for this operation.
097         */
098        public IngestRequestBuilder setReplicationType(ReplicationType replicationType) {
099            this.request.replicationType(replicationType);
100            return this;
101        }
102    
103        /**
104         * Sets the consistency level. Defaults to {@link org.elasticsearch.action.WriteConsistencyLevel#DEFAULT}.
105         */
106        public IngestRequestBuilder setConsistencyLevel(WriteConsistencyLevel consistencyLevel) {
107            request.consistencyLevel(consistencyLevel);
108            return this;
109        }
110    
111        /**
112         * The number of actions currently in the bulk.
113         */
114        public int numberOfActions() {
115            return request.numberOfActions();
116        }
117    
118        protected void doExecute(ActionListener<IngestResponse> listener) {
119            ((InternalClient)client).execute(IngestAction.INSTANCE, request, listener);
120        }
121    }
