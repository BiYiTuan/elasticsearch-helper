001package org.xbib.elasticsearch.action.ingest;
002
003import org.elasticsearch.action.ActionRequestBuilder;
004import org.elasticsearch.action.delete.DeleteRequest;
005import org.elasticsearch.action.index.IndexRequest;
006import org.elasticsearch.client.ElasticsearchClient;
007import org.elasticsearch.common.Nullable;
008import org.elasticsearch.common.unit.TimeValue;
009
010public class IngestRequestBuilder extends ActionRequestBuilder<IngestRequest, IngestResponse, IngestRequestBuilder> {
011
012    public IngestRequestBuilder(ElasticsearchClient client, IngestAction action) {
013        super(client, action, new IngestRequest());
014    }
015
016    public IngestRequestBuilder add(IndexRequest request) {
017        this.request.add(request);
018        return this;
019    }
020
021    public IngestRequestBuilder add(DeleteRequest request) {
022        this.request.add(request);
023        return this;
024    }
025
026    public IngestRequestBuilder add(byte[] data, int from, int length) throws Exception {
027        this.request.add(data, from, length, null, null);
028        return this;
029    }
030
031    public IngestRequestBuilder add(byte[] data, int from, int length, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
032        this.request.add(data, from, length, defaultIndex, defaultType);
033        return this;
034    }
035
036    public IngestRequestBuilder setTimeout(TimeValue timeout) {
037        request.timeout(timeout);
038        return this;
039    }
040
041    public int numberOfActions() {
042        return request.numberOfActions();
043    }
044}
