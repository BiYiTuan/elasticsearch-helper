019        public void resolve(SourceToParse source) throws IOException {
020            XContentParser parser = source.parser();
021            try {
022                if (parser == null) {
023                    parser = XContentHelper.createParser(source.source());
024                }
025                XContentParser.Token token = parser.nextToken();
026                if (token != XContentParser.Token.START_OBJECT) {
027                    throw new MapperParsingException("Malformed content, must start with an object");
028                }
029                String currentFieldName = parser.currentName();
030                while (token != XContentParser.Token.END_OBJECT) {
031                    if (token == XContentParser.Token.START_OBJECT) {
032                        //serializeObject(context, currentFieldName);
033                    } else if (token == XContentParser.Token.START_ARRAY) {
034                        //serializeArray(context, currentFieldName);
035                    } else if (token == XContentParser.Token.FIELD_NAME) {
036                        currentFieldName = parser.currentName();
037                    } else if (token == XContentParser.Token.VALUE_NULL) {
038                        //serializeNullValue(context, currentFieldName);
039                    } else if (token == null) {
040                        throw new MapperParsingException("object mapping for [" + source.type() + "] tried to parse as object, but got EOF, has a concrete value been provided to it?");
041                    } else if (token.isValue()) {
042                        //serializeValue(context, currentFieldName, token);
043                    }
044                    token = parser.nextToken();
045                }
046    
047            } finally {
048                // only close the parser when its not provided externally
049                if (source.parser() == null && parser != null) {
050                    parser.close();
051                }
052            }
053        }
054    
055

057    
058            String index;
059            String type;
060            String id;
061    
062            Doc(String index, String type, String id) {
063                this.index = index;
064                this.type = type;
065                this.id = id;
066            }
067    
068        }
