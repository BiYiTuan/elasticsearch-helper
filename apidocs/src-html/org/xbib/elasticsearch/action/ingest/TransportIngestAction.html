001package org.xbib.elasticsearch.action.ingest;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.ExceptionsHelper;
005import org.elasticsearch.action.ActionListener;
006import org.elasticsearch.action.ActionRequest;
007import org.elasticsearch.action.delete.DeleteRequest;
008import org.elasticsearch.action.index.IndexRequest;
009import org.elasticsearch.action.support.ActionFilters;
010import org.elasticsearch.cluster.ClusterService;
011import org.elasticsearch.cluster.ClusterState;
012import org.elasticsearch.cluster.block.ClusterBlockException;
013import org.elasticsearch.cluster.block.ClusterBlockLevel;
014import org.elasticsearch.cluster.metadata.MappingMetaData;
015import org.elasticsearch.cluster.metadata.MetaData;
016import org.elasticsearch.cluster.routing.GroupShardsIterator;
017import org.elasticsearch.cluster.routing.ShardIterator;
018import org.elasticsearch.common.inject.Inject;
019import org.elasticsearch.common.settings.Settings;
020import org.elasticsearch.index.shard.ShardId;
021import org.elasticsearch.threadpool.ThreadPool;
022import org.elasticsearch.transport.TransportService;
023import org.xbib.elasticsearch.action.ingest.leader.IngestLeaderShardRequest;
024import org.xbib.elasticsearch.action.ingest.leader.IngestLeaderShardResponse;
025import org.xbib.elasticsearch.action.ingest.leader.TransportLeaderShardIngestAction;
026import org.xbib.elasticsearch.action.ingest.replica.IngestReplicaShardRequest;
027import org.xbib.elasticsearch.action.ingest.replica.TransportReplicaShardIngestAction;
028
029import java.util.HashMap;
030import java.util.LinkedList;
031import java.util.List;
032import java.util.Map;
033import java.util.concurrent.atomic.AtomicInteger;
034
035public class TransportIngestAction extends HandledTransportAction<IngestRequest, IngestResponse> {
036
037    private final boolean allowIdGeneration;
038
039    private final ClusterService clusterService;
040
041    private final TransportLeaderShardIngestAction leaderShardIngestAction;
042
043    private final TransportReplicaShardIngestAction replicaShardIngestAction;
044
045    @Inject
046    public TransportIngestAction(Settings settings, ThreadPool threadPool,
047                                 TransportService transportService, ClusterService clusterService,
048                                 TransportLeaderShardIngestAction leaderShardIngestAction,
049                                 TransportReplicaShardIngestAction replicaShardIngestAction,
050                                 ActionFilters actionFilters) {
051        super(settings, IngestAction.NAME, threadPool, transportService, actionFilters, IngestRequest.class);
052        this.clusterService = clusterService;
053        this.leaderShardIngestAction = leaderShardIngestAction;
054        this.replicaShardIngestAction = replicaShardIngestAction;
055        this.allowIdGeneration = this.settings.getAsBoolean("action.allow_id_generation", true);
056    }
057
058    @Override
059    protected void doExecute(final IngestRequest ingestRequest, final ActionListener<IngestResponse> listener) {
060        final long startTime = System.currentTimeMillis();
061        final IngestResponse ingestResponse = new IngestResponse();
062        ingestResponse.setIngestId(ingestRequest.ingestId());
063        ClusterState clusterState = clusterService.state();
064        try {
065            clusterState.blocks().globalBlockedRaiseException(ClusterBlockLevel.WRITE);
066        } catch (ClusterBlockException e) {
067            listener.onFailure(e);
068            return;
069        }
070        MetaData metaData = clusterState.metaData();
071        final List<ActionRequest> requests = new LinkedList<ActionRequest>();
072        // first, iterate over all requests and parse them for mapping, filter out erraneous requests
073        for (ActionRequest request : ingestRequest.requests()) {
074            if (request instanceof IndexRequest) {
075                try {
076                    IndexRequest indexRequest = (IndexRequest) request;
077                    indexRequest.index(clusterState.metaData().concreteSingleIndex(indexRequest.index(), indexRequest.indicesOptions()));
078                    MappingMetaData mappingMd = null;
079                    if (metaData.hasIndex(indexRequest.index())) {
080                        mappingMd = metaData.index(indexRequest.index()).mappingOrDefault(indexRequest.type());
081                    }
082                    indexRequest.process(metaData, mappingMd, allowIdGeneration, indexRequest.index());
083                    requests.add(indexRequest);
084                } catch (Throwable e) {
085                    logger.error(e.getMessage(), e);
086                    ingestResponse.addFailure(new IngestActionFailure(-1L, null, ExceptionsHelper.detailedMessage(e)));
087                }
088            } else if (request instanceof DeleteRequest) {
089                DeleteRequest deleteRequest = (DeleteRequest) request;
090                deleteRequest.routing(clusterState.metaData().resolveIndexRouting(deleteRequest.routing(), deleteRequest.index()));
091                deleteRequest.index(clusterState.metaData().concreteSingleIndex(deleteRequest.index(), deleteRequest.indicesOptions()));
092                requests.add(deleteRequest);
093            } else {
094                throw new ElasticsearchException("action request not known: " + request.getClass().getName());
095            }
096        }
097        // second, go over all the requests and create a shard request map
098        Map<ShardId, List<ActionRequest>> requestsByShard = new HashMap<ShardId, List<ActionRequest>>();
099        for (ActionRequest request : requests) {
100            if (request instanceof IndexRequest) {
101                IndexRequest indexRequest = (IndexRequest) request;
102                ShardId shardId = clusterService.operationRouting().indexShards(clusterState, indexRequest.index(), indexRequest.type(), indexRequest.id(), indexRequest.routing()).shardId();
103                List<ActionRequest> list = requestsByShard.get(shardId);
104                if (list == null) {
105                    list = new LinkedList<ActionRequest>();
106                    requestsByShard.put(shardId, list);
107                }
108                list.add(request);
109            } else if (request instanceof DeleteRequest) {
110                DeleteRequest deleteRequest = (DeleteRequest) request;
111                MappingMetaData mappingMd = clusterState.metaData().index(deleteRequest.index()).mappingOrDefault(deleteRequest.type());
112                if (mappingMd != null && mappingMd.routing().required() && deleteRequest.routing() == null) {
113                    GroupShardsIterator groupShards = clusterService.operationRouting().broadcastDeleteShards(clusterState, deleteRequest.index());
114                    for (ShardIterator shardIt : groupShards) {
115                        List<ActionRequest> list = requestsByShard.get(shardIt.shardId());
116                        if (list == null) {
117                            list = new LinkedList<ActionRequest>();
118                            requestsByShard.put(shardIt.shardId(), list);
119                        }
120                        list.add(deleteRequest);
121                    }
122                } else {
123                    ShardId shardId = clusterService.operationRouting().deleteShards(clusterState, deleteRequest.index(), deleteRequest.type(), deleteRequest.id(), deleteRequest.routing()).shardId();
124                    List<ActionRequest> list = requestsByShard.get(shardId);
125                    if (list == null) {
126                        list = new LinkedList<ActionRequest>();
127                        requestsByShard.put(shardId, list);
128                    }
129                    list.add(request);
130                }
131            }
132        }
133        if (requestsByShard.isEmpty()) {
134            logger.error("no shards to execute ingest");
135            ingestResponse.setSuccessSize(0)
136                    .addFailure(new IngestActionFailure(-1L, null, "no shards to execute ingest"))
137                    .setTookInMillis(System.currentTimeMillis() - startTime);
138            listener.onResponse(ingestResponse);
139            return;
140        }
141        // third, for each shard, execute leader/replica action
142        final AtomicInteger successCount = new AtomicInteger(0);
143        final AtomicInteger responseCounter = new AtomicInteger(requestsByShard.size());
144        for (Map.Entry<ShardId, List<ActionRequest>> entry : requestsByShard.entrySet()) {
145            final ShardId shardId = entry.getKey();
146            final List<ActionRequest> actionRequests = entry.getValue();
147            final IngestLeaderShardRequest ingestLeaderShardRequest = new IngestLeaderShardRequest()
148                    .setIngestId(ingestRequest.ingestId())
149                    .setShardId(shardId)
150                    .setActionRequests(actionRequests)
151                    .timeout(ingestRequest.timeout())
152                    .requiredConsistency(ingestRequest.requiredConsistency());
153            leaderShardIngestAction.execute(ingestLeaderShardRequest, new ActionListener<IngestLeaderShardResponse>() {
154                @Override
155                public void onResponse(IngestLeaderShardResponse ingestLeaderShardResponse) {
156                    long millis = System.currentTimeMillis() - startTime;
157                    ingestResponse.setIngestId(ingestRequest.ingestId());
158                    ingestResponse.setLeaderResponse(ingestLeaderShardResponse);
159                    successCount.addAndGet(ingestLeaderShardResponse.getSuccessCount());
160                    int quorumShards = ingestLeaderShardResponse.getQuorumShards();
161                    if (quorumShards < 0) {
162                        ingestResponse.addFailure(new IngestActionFailure(ingestRequest.ingestId(), shardId, "quorum not reached for shard " + shardId));
163                    } else if (quorumShards > 0) {
164                        responseCounter.incrementAndGet();
165                        final IngestReplicaShardRequest ingestReplicaShardRequest =
166                                new IngestReplicaShardRequest(ingestLeaderShardRequest.getIngestId(),
167                                        ingestLeaderShardRequest.getShardId(),
168                                        ingestLeaderShardRequest.getActionRequests());
169                        ingestReplicaShardRequest.timeout(ingestRequest.timeout());
170                        replicaShardIngestAction.execute(ingestReplicaShardRequest, new ActionListener<TransportReplicaShardIngestAction.ReplicaOperationResponse>() {
171                            @Override
172                            public void onResponse(TransportReplicaShardIngestAction.ReplicaOperationResponse response) {
173                                long millis = Math.max(1, System.currentTimeMillis() - startTime);
174                                ingestResponse.addReplicaResponses(response.responses());
175                                if (responseCounter.decrementAndGet() == 0) {
176                                    ingestResponse.setSuccessSize(successCount.get())
177                                            .setTookInMillis(millis);
178                                    listener.onResponse(ingestResponse);
179                                }
180                            }
181
182                            @Override
183                            public void onFailure(Throwable e) {
184                                long millis = Math.max(1, System.currentTimeMillis() - startTime);
185                                logger.error(e.getMessage(), e);
186                                ingestResponse.addFailure(new IngestActionFailure(ingestRequest.ingestId(), shardId, ExceptionsHelper.detailedMessage(e)));
187                                if (responseCounter.decrementAndGet() == 0) {
188                                    ingestResponse.setSuccessSize(successCount.get())
189                                            .setTookInMillis(millis);
190                                    listener.onResponse(ingestResponse);
191                                }
192                            }
193                        });
194                    }
195                    if (responseCounter.decrementAndGet() == 0) {
196                        ingestResponse.setSuccessSize(successCount.get())
197                                .setTookInMillis(millis);
198                        listener.onResponse(ingestResponse);
199                    }
200                }
201
202                @Override
203                public void onFailure(Throwable e) {
204                    long millis = System.currentTimeMillis() - startTime;
205                    logger.error(e.getMessage(), e);
206                    ingestResponse.addFailure(new IngestActionFailure(-1L, shardId, ExceptionsHelper.detailedMessage(e)));
207                    if (responseCounter.decrementAndGet() == 0) {
208                        ingestResponse.setSuccessSize(successCount.get())
209                                .setTookInMillis(millis);
210                        listener.onResponse(ingestResponse);
211                    }
212                }
213
214            });
215        }
216    }
217
218}
