003    
004    import org.elasticsearch.action.WriteConsistencyLevel;
005    import org.elasticsearch.action.support.replication.ReplicationType;
006    import org.elasticsearch.client.Client;

010    import org.elasticsearch.common.settings.Settings;
011    import org.elasticsearch.common.unit.ByteSizeValue;
012    import org.elasticsearch.common.xcontent.XContentBuilder;
013    import org.elasticsearch.common.xcontent.XContentBuilderString;
014    import org.elasticsearch.rest.BaseRestHandler;
015    import org.elasticsearch.rest.RestChannel;
016    import org.elasticsearch.rest.RestController;
017    import org.elasticsearch.rest.RestRequest;
018    import org.elasticsearch.rest.XContentRestResponse;
019    
020    import java.io.IOException;
021    import java.util.LinkedHashMap;
022    import java.util.Map;
023    import java.util.concurrent.CountDownLatch;
024    import java.util.concurrent.TimeUnit;
025    import java.util.concurrent.atomic.AtomicLong;
026    
027    import static org.elasticsearch.rest.RestRequest.Method.POST;
028    import static org.elasticsearch.rest.RestRequest.Method.PUT;
029    import static org.elasticsearch.rest.RestStatus.BAD_REQUEST;
030    import static org.elasticsearch.rest.RestStatus.OK;
031    import static org.elasticsearch.rest.action.support.RestXContentBuilder.restContentBuilder;
032    
033    /**
034     * <pre>
035     * { "index" : { "_index" : "test", "_type" : "type1", "_id" : "1" }
036     * { "type1" : { "field1" : "value1" } }
037     * { "delete" : { "_index" : "test", "_type" : "type1", "_id" : "2" } }
038     * { "create" : { "_index" : "test", "_type" : "type1", "_id" : "1" }
039     * { "type1" : { "field1" : "value1" } }
040     * </pre>
041     */
042    
043    public class RestIngestAction extends BaseRestHandler {
044    
045        private final static ESLogger logger = Loggers.getLogger(RestIngestAction.class);
046        /**
047         * The outstanding requests
048         */
049        public final static AtomicLong outstandingRequests = new AtomicLong();
050        /**
051         * Count the volume
052         */
053        public final static AtomicLong volumeCounter = new AtomicLong();
054        /**
055         * The responses
056         */
057        public final static Map<Long,Object> responses = new LRUHashMap(1000);
058        /**
059         * The IngestProcessor
060         */
061        private final IngestProcessor ingestProcessor;
062    
063        @Inject
064        public RestIngestAction(Settings settings, Client client, RestController controller) {
065            super(settings, client);
066    
067            controller.registerHandler(POST, "/_ingest", this);
068            controller.registerHandler(PUT, "/_ingest", this);
069            controller.registerHandler(POST, "/{index}/_ingest", this);
070            controller.registerHandler(PUT, "/{index}/_ingest", this);
071            controller.registerHandler(POST, "/{index}/{type}/_ingest", this);
072            controller.registerHandler(PUT, "/{index}/{type}/_ingest", this);
073    
074            int actions = settings.getAsInt("action.ingest.maxactions", 100);
075            int concurrency = settings.getAsInt("action.ingest.maxconcurrency", 30);
076            ByteSizeValue volume = settings.getAsBytesSize("action.ingest.maxvolume", ByteSizeValue.parseBytesSizeValue("5m"));
077    
078            this.ingestProcessor = IngestProcessor.builder(client)
079                    .actions(actions)
080                    .concurrency(concurrency)
081                    .maxVolume(volume)
082                    .build();
083        }
084    
085        @Override
086        public void handleRequest(final RestRequest request, final RestChannel channel) {
087    
088            final ExecutionIdHolder idHolder = new ExecutionIdHolder();
089            final CountDownLatch latch = new CountDownLatch(1);
090    
091            IngestProcessor.Listener listener = new IngestProcessor.Listener() {
092                @Override
093                public void beforeBulk(long executionId, IngestRequest ingestRequest) {
094                    long l = outstandingRequests.getAndIncrement();
095                    long v = volumeCounter.addAndGet(ingestRequest.estimatedSizeInBytes());
096                    logger.info("new bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
097                            executionId, ingestRequest.numberOfActions(), v, l);
098                    idHolder.executionId(executionId);
099                    latch.countDown();
100                }
101    
102                @Override
103                public void afterBulk(long executionId, IngestResponse response) {
104                    long l = outstandingRequests.decrementAndGet();
105                    logger.info("bulk [{}] [{} items succeeded] [{} items failed] [{}ms]",
106                            executionId, response.successSize(), response.failureSize(), response.took().millis());
107                    if (!response.failure().isEmpty()) {
108                        for (IngestItemFailure f: response.failure()) {
109                            logger.error("bulk [{}] [{} failure reason: {}", executionId, f.id(), f.message());
110                        }
111                    }
112                }
113    
114                @Override
115                public void afterBulk(long executionId, Throwable failure) {
116                    long l = outstandingRequests.decrementAndGet();
117                    logger.error("bulk [{}] error", executionId, failure);
118                    synchronized (responses) {
119                        responses.put(executionId, failure);
120                    }
121                }
122            };
123    
124            String replicationType = request.param("replication");
125            if (replicationType != null) {
126                ingestProcessor.replicationType(ReplicationType.fromString(replicationType));
127            }
128            String consistencyLevel = request.param("consistency");
129            if (consistencyLevel != null) {
130                ingestProcessor.consistencyLevel(WriteConsistencyLevel.fromString(consistencyLevel));
131            }
132            try {
133                long t0 = System.currentTimeMillis();
134                ingestProcessor.add(request.content(), request.contentUnsafe(),
135                        request.param("index"), request.param("type"), listener);
136                // estimation, should be enough time to wait for an ID
137                boolean b = latch.await(100, TimeUnit.MILLISECONDS);
138                long t1 = System.currentTimeMillis();
139    
140                XContentBuilder builder = restContentBuilder(request);
141                builder.startObject();
142                builder.field(Fields.OK, true);
143                builder.field(Fields.TOOK, t1 - t0);
144                // got ID?
145                if (b) {
146                    builder.field(Fields.ID, idHolder.executionId());
147                }
148                builder.endObject();
149                channel.sendResponse(new XContentRestResponse(request, OK, builder));
150            } catch (Exception e) {
151                try {
152                    XContentBuilder builder = restContentBuilder(request);
153                    channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder.startObject().field("error", e.getMessage()).endObject()));
154                } catch (IOException e1) {
155                    logger.error("Failed to send failure response", e1);
156                }
157            }
158        }
159    
160        static final class Fields {
161            static final XContentBuilderString OK = new XContentBuilderString("ok");
162            static final XContentBuilderString TOOK = new XContentBuilderString("took");
163            static final XContentBuilderString ID = new XContentBuilderString("id");
164        }
165    
166        static final class ExecutionIdHolder {
167            private long executionId;
168            public void executionId(long executionId) {
169                this.executionId = executionId;
170            }
171            public long executionId() {
172                return executionId;
173            }
174        }
175    
176        static final class LRUHashMap<K, V> extends LinkedHashMap<K, V> {
177    
178            private final int maxSize;
179    
180            /**
181             * Constructor.
182             *
183             * @param initialSize initial cache size.
184             * @param maxSize maximum cache size.
185             */
186            public LRUHashMap(final int initialSize, final int maxSize) {
187                super(initialSize);
188                this.maxSize = maxSize;
189            }
190    
191            /**
192             * Constructor.
193             * Equivalent to calling LRUHashMap(size, size);
194             *
195             * @param size initial and maximum cache size.
196             */
197            public LRUHashMap(final int size) {
198                this(size, size);
199            }
200    
201            @Override
202            protected boolean removeEldestEntry(final Map.Entry<K, V> eldest) {
203                return size() >= this.maxSize;
204            }
205        }
