002    package org.xbib.elasticsearch.action.ingest.delete;
003    
004    import org.elasticsearch.action.Action;
005    import org.elasticsearch.client.Client;
006    import org.elasticsearch.common.settings.Settings;
007    import org.elasticsearch.common.unit.TimeValue;
008    import org.elasticsearch.transport.TransportRequestOptions;
009    
010    import org.xbib.elasticsearch.action.ingest.IngestResponse;
011    
012    public class IngestDeleteAction extends Action<IngestDeleteRequest, IngestResponse, IngestDeleteRequestBuilder> {
013    
014        public static final IngestDeleteAction INSTANCE = new IngestDeleteAction();
015    
016        public static final String NAME = "org.xbib.elasticsearch.action.ingest.delete";
017    
018        private IngestDeleteAction() {
019            super(NAME);
020        }
021    
022        @Override
023        public IngestResponse newResponse() {
024            return new IngestResponse();
025        }
026    
027        @Override
028        public IngestDeleteRequestBuilder newRequestBuilder(Client client) {
029            return new IngestDeleteRequestBuilder(client);
030        }
031    
032        @Override
033        public TransportRequestOptions transportOptions(Settings settings) {
034            return TransportRequestOptions.options()
035                    .withType(TransportRequestOptions.Type.BULK)
036                    .withTimeout(settings.getAsTime("action.ingest.timeout", TimeValue.timeValueMinutes(1)))
037                    .withCompress(settings.getAsBoolean("action.ingest.compress", true));
038        }
039    }
