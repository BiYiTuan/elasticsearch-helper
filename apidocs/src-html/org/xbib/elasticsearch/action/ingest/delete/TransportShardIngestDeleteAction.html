002    package org.xbib.elasticsearch.action.ingest.delete;
003    
004    import org.elasticsearch.ElasticsearchException;
005    import org.elasticsearch.ExceptionsHelper;
006    import org.elasticsearch.action.delete.DeleteRequest;
007    import org.elasticsearch.action.support.replication.TransportShardReplicationOperationAction;
008    import org.elasticsearch.cluster.ClusterService;
009    import org.elasticsearch.cluster.ClusterState;
010    import org.elasticsearch.cluster.action.shard.ShardStateAction;
011    import org.elasticsearch.cluster.block.ClusterBlockException;
012    import org.elasticsearch.cluster.block.ClusterBlockLevel;
013    import org.elasticsearch.cluster.routing.ShardIterator;

015    import org.elasticsearch.common.settings.Settings;
016    import org.elasticsearch.index.engine.Engine;
017    import org.elasticsearch.index.shard.ShardId;
018    import org.elasticsearch.index.shard.service.IndexShard;
019    import org.elasticsearch.indices.IndicesService;
020    import org.elasticsearch.rest.RestStatus;
021    import org.elasticsearch.threadpool.ThreadPool;
022    import org.elasticsearch.transport.TransportRequestOptions;
023    import org.elasticsearch.transport.TransportService;
024    
025    import org.xbib.elasticsearch.action.ingest.IngestItemFailure;
026    
027    import java.util.List;
028    
029    import static org.elasticsearch.common.collect.Lists.newLinkedList;
030    
031    
032    public class TransportShardIngestDeleteAction extends TransportShardReplicationOperationAction<IngestDeleteShardRequest, IngestDeleteShardRequest, IngestDeleteShardResponse> {
033    
034        @Inject
035        public TransportShardIngestDeleteAction(Settings settings, TransportService transportService, ClusterService clusterService,
036                                                IndicesService indicesService, ThreadPool threadPool, ShardStateAction shardStateAction) {
037            super(settings, transportService, clusterService, indicesService, threadPool, shardStateAction);
038        }
039    
040        @Override
041        protected String executor() {
042            return ThreadPool.Names.BULK;
043        }
044    
045        @Override
046        protected boolean checkWriteConsistency() {
047            return true;
048        }
049    
050        @Override
051        protected TransportRequestOptions transportOptions() {
052            return IngestDeleteAction.INSTANCE.transportOptions(settings);
053        }
054    
055        @Override
056        protected IngestDeleteShardRequest newRequestInstance() {
057            return new IngestDeleteShardRequest();
058        }
059    
060        @Override
061        protected IngestDeleteShardRequest newReplicaRequestInstance() {
062            return new IngestDeleteShardRequest();
063        }
064    
065        @Override
066        protected IngestDeleteShardResponse newResponseInstance() {
067            return new IngestDeleteShardResponse();
068        }
069    
070        @Override
071        protected String transportAction() {
072            return IngestDeleteAction.NAME + ".shard.delete";
073        }
074    
075        @Override
076        protected ClusterBlockException checkGlobalBlock(ClusterState state, IngestDeleteShardRequest request) {
077            return state.blocks().globalBlockedException(ClusterBlockLevel.WRITE);
078        }
079    
080        @Override
081        protected ClusterBlockException checkRequestBlock(ClusterState state, IngestDeleteShardRequest request) {
082            return state.blocks().indexBlockedException(ClusterBlockLevel.WRITE, request.index());
083        }
084    
085        @Override
086        protected ShardIterator shards(ClusterState clusterState, IngestDeleteShardRequest request) {
087            return clusterState.routingTable().index(request.index()).shard(request.shardId()).shardsIt();
088        }
089    
090        @Override
091        protected PrimaryResponse<IngestDeleteShardResponse, IngestDeleteShardRequest> shardOperationOnPrimary(ClusterState clusterState, PrimaryOperationRequest shardRequest) {
092            final IngestDeleteShardRequest request = shardRequest.request;
093            IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
094    
095            int successSize = 0;
096            List<IngestItemFailure> failure = newLinkedList();
097            int size = request.items().size();
098            long[] versions = new long[size];
099            for (int i = 0; i < size; i++) {
100                IngestDeleteItemRequest item = request.items().get(i);
101                    DeleteRequest deleteRequest = (DeleteRequest) item.request();
102                    try {
103                        Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version()).versionType(deleteRequest.versionType()).origin(Engine.Operation.Origin.PRIMARY);
104                        indexShard.delete(delete);
105                        versions[i] = delete.version();
106                        // update the request with the version so it will go to the replicas
107                        deleteRequest.version(delete.version());
108                        successSize++;
109                    } catch (Throwable e) {
110                        // rethrow the failure if we are going to retry on primary and let parent failure to handle it
111                        if (retryPrimaryException(e)) {
112                            // restore updated versions...
113                            for (int j = 0; j < i; j++) {
114                                ((DeleteRequest)request.items().get(j).request()).version(versions[j]);
115                            }
116                            logger.error(e.getMessage(), e);
117                            throw new ElasticsearchException(e.getMessage());
118                        }
119                        if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
120                            logger.trace("[{}][{}] failed to execute bulk item (delete) {}", e, shardRequest.request.index(), shardRequest.shardId, deleteRequest);
121                        } else {
122                            logger.debug("[{}][{}] failed to execute bulk item (delete) {}", e, shardRequest.request.index(), shardRequest.shardId, deleteRequest);
123                        }
124                        failure.add(new IngestItemFailure(item.id(), ExceptionsHelper.detailedMessage(e)));
125                        // nullify the request so it won't execute on the replicas
126                        request.items().set(i, null);
127                    }
128            }
129            IngestDeleteShardResponse response = new IngestDeleteShardResponse(new ShardId(request.index(), request.shardId()), successSize, failure);
130            return new PrimaryResponse<IngestDeleteShardResponse, IngestDeleteShardRequest>(shardRequest.request, response, null);
131        }
132    
133        @Override
134        protected void shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
135            IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request.index()).shardSafe(shardRequest.shardId);
136            final IngestDeleteShardRequest request = shardRequest.request;
137            int size = request.items().size();
138            for (int i = 0; i < size; i++) {
139                IngestDeleteItemRequest item = request.items().get(i);
140                if (item == null) {
141                    continue;
142                }
143                DeleteRequest deleteRequest = (DeleteRequest) item.request();
144                try {
145                    Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(), deleteRequest.id(), deleteRequest.version()).origin(Engine.Operation.Origin.REPLICA);
146                    indexShard.delete(delete);
147                } catch (Throwable e) {
148                    // ignore, we are on backup
149                }
150            }
151    
152        }
153    
154    }
