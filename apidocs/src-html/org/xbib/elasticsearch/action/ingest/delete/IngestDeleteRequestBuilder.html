002    package org.xbib.elasticsearch.action.ingest.delete;
003    
004    import org.elasticsearch.action.ActionListener;
005    import org.elasticsearch.action.ActionRequestBuilder;
006    import org.elasticsearch.action.ListenableActionFuture;
007    import org.elasticsearch.action.WriteConsistencyLevel;
008    import org.elasticsearch.action.delete.DeleteRequest;
009    import org.elasticsearch.action.delete.DeleteRequestBuilder;
010    import org.elasticsearch.action.support.PlainListenableActionFuture;
011    import org.elasticsearch.action.support.replication.ReplicationType;
012    import org.elasticsearch.client.Client;
013    import org.elasticsearch.client.internal.InternalClient;
014    
015    import org.elasticsearch.common.unit.TimeValue;
016    import org.xbib.elasticsearch.action.ingest.IngestResponse;
017    
018    /**
019     * A bulk request holds ordered {@link org.elasticsearch.action.delete.DeleteRequest}s
020     * and allows to executes it in a single batch.
021     */
022    public class IngestDeleteRequestBuilder extends ActionRequestBuilder<IngestDeleteRequest, IngestResponse, IngestDeleteRequestBuilder> {
023    
024        public IngestDeleteRequestBuilder(Client client) {
025            super((InternalClient)client, new IngestDeleteRequest());
026        }
027    
028        public IngestDeleteRequest request() {
029            return this.request;
030        }
031    
032        @Override
033        public ListenableActionFuture<IngestResponse> execute() {
034            PlainListenableActionFuture<IngestResponse> future = new PlainListenableActionFuture<IngestResponse>(request.listenerThreaded(), client.threadPool());
035            execute(future);
036            return future;
037        }
038    
039        @Override
040        public void execute(ActionListener<IngestResponse> listener) {
041            doExecute(listener);
042        }
043    
044        /**
045         * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
046         */
047        public IngestDeleteRequestBuilder add(DeleteRequest request) {
048            this.request.add(request);
049            return this;
050        }
051    
052        /**
053         * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list of actions to execute.
054         */
055        public IngestDeleteRequestBuilder add(DeleteRequestBuilder request) {
056            this.request.add(request.request());
057            return this;
058        }
059    
060        /**
061         * Set the replication type for this operation.
062         */
063        public IngestDeleteRequestBuilder setReplicationType(ReplicationType replicationType) {
064            request.replicationType(replicationType);
065            return this;
066        }
067    
068        /**
069         * Sets the consistency level. Defaults to {@link org.elasticsearch.action.WriteConsistencyLevel#DEFAULT}.
070         */
071        public IngestDeleteRequestBuilder setConsistencyLevel(WriteConsistencyLevel consistencyLevel) {
072            request.consistencyLevel(consistencyLevel);
073            return this;
074        }
075    
076        /**
077         * Sets the timeout. Defaults to {@link org.xbib.elasticsearch.action.ingest.delete.IngestDeleteShardRequest#DEFAULT_TIMEOUT}.
078         */
079        public IngestDeleteRequestBuilder setTimeout(TimeValue timeout) {
080            request.timeout(timeout);
081            return this;
082        }
083    
084        /**
085         * The number of actions currently in the bulk.
086         */
087        public int numberOfActions() {
088            return request.numberOfActions();
089        }
090    
091        protected void doExecute(ActionListener<IngestResponse> listener) {
092            ((InternalClient)client).execute(IngestDeleteAction.INSTANCE, request, listener);
093        }
094    }
