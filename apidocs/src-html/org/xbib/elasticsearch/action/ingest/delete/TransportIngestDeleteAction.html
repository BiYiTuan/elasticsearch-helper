002package org.xbib.elasticsearch.action.ingest.delete;
003
004import org.elasticsearch.ExceptionsHelper;
005import org.elasticsearch.action.ActionListener;
006import org.elasticsearch.action.ActionRequest;
007import org.elasticsearch.action.delete.DeleteRequest;
008import org.elasticsearch.action.support.TransportAction;
009import org.elasticsearch.cluster.ClusterService;
010import org.elasticsearch.cluster.ClusterState;
011import org.elasticsearch.cluster.block.ClusterBlockLevel;
012import org.elasticsearch.cluster.metadata.MappingMetaData;
013import org.elasticsearch.cluster.routing.GroupShardsIterator;
014import org.elasticsearch.cluster.routing.ShardIterator;
015import org.elasticsearch.common.inject.Inject;
016import org.elasticsearch.common.settings.Settings;
017import org.elasticsearch.index.shard.ShardId;
018import org.elasticsearch.threadpool.ThreadPool;
019import org.elasticsearch.transport.BaseTransportRequestHandler;
020import org.elasticsearch.transport.TransportChannel;
021import org.elasticsearch.transport.TransportService;
022
023import org.xbib.elasticsearch.action.ingest.IngestItemFailure;
024import org.xbib.elasticsearch.action.ingest.IngestResponse;
025
026import java.util.List;
027import java.util.Map;
028import java.util.concurrent.atomic.AtomicInteger;
029
030import static org.elasticsearch.common.collect.Lists.newLinkedList;
031import static org.elasticsearch.common.collect.Maps.newHashMap;
032
033public class TransportIngestDeleteAction extends TransportAction<IngestDeleteRequest, IngestResponse> {
034
035    private final ClusterService clusterService;
036
037    private final TransportShardIngestDeleteAction shardBulkAction;
038
039    @Inject
040    public TransportIngestDeleteAction(Settings settings, ThreadPool threadPool,
041                                       TransportService transportService,
042                                       ClusterService clusterService,
043                                       TransportShardIngestDeleteAction shardBulkAction) {
044        super(settings, threadPool);
045        this.clusterService = clusterService;
046        this.shardBulkAction = shardBulkAction;
047
048        transportService.registerHandler(IngestDeleteAction.NAME, new IngestTransportHandler());
049    }
050
051    @Override
052    protected void doExecute(final IngestDeleteRequest ingestRequest, final ActionListener<IngestResponse> listener) {
053        executeBulk(ingestRequest, listener);
054    }
055
056    private void executeBulk(final IngestDeleteRequest ingestRequest, final ActionListener<IngestResponse> listener) {
057        final long startTime = System.currentTimeMillis();
058        ClusterState clusterState = clusterService.state();
059        // TODO use timeout to wait here if its blocked...
060        clusterState.blocks().globalBlockedRaiseException(ClusterBlockLevel.WRITE);
061
062        for (ActionRequest request : ingestRequest.requests()) {
063            DeleteRequest deleteRequest = (DeleteRequest) request;
064            deleteRequest.routing(clusterState.metaData().resolveIndexRouting(deleteRequest.routing(), deleteRequest.index()));
065            deleteRequest.index(clusterState.metaData().concreteIndex(deleteRequest.index()));
066        }
067
068        // first, go over all the requests and create a ShardId -> Operations mapping
069        Map<ShardId, List<IngestDeleteItemRequest>> requestsByShard = newHashMap();
070        int i = 0;
071        for (ActionRequest request : ingestRequest.requests()) {
072            DeleteRequest deleteRequest = (DeleteRequest) request;
073            MappingMetaData mappingMd = clusterState.metaData().index(deleteRequest.index()).mappingOrDefault(deleteRequest.type());
074            if (mappingMd != null && mappingMd.routing().required() && deleteRequest.routing() == null) {
075                // if routing is required, and no routing on the delete request, we need to broadcast it....
076                GroupShardsIterator groupShards = clusterService.operationRouting().broadcastDeleteShards(clusterState, deleteRequest.index());
077                for (ShardIterator shardIt : groupShards) {
078                    List<IngestDeleteItemRequest> list = requestsByShard.get(shardIt.shardId());
079                    if (list == null) {
080                        list = newLinkedList();
081                        requestsByShard.put(shardIt.shardId(), list);
082                    }
083                    list.add(new IngestDeleteItemRequest(i, deleteRequest));
084                }
085            } else {
086                ShardId shardId = clusterService.operationRouting().deleteShards(clusterState, deleteRequest.index(), deleteRequest.type(), deleteRequest.id(), deleteRequest.routing()).shardId();
087                List<IngestDeleteItemRequest> list = requestsByShard.get(shardId);
088                if (list == null) {
089                    list = newLinkedList();
090                    requestsByShard.put(shardId, list);
091                }
092                list.add(new IngestDeleteItemRequest(i, request));
093            }
094            i++;
095        }
096
097        final AtomicInteger successSize = new AtomicInteger(0);
098        final List<IngestItemFailure> failure = newLinkedList();
099
100        if (requestsByShard.isEmpty()) {
101            listener.onResponse(new IngestResponse(0, failure, System.currentTimeMillis() - startTime));
102            return;
103        }
104
105        final AtomicInteger counter = new AtomicInteger(requestsByShard.size());
106        for (Map.Entry<ShardId, List<IngestDeleteItemRequest>> entry : requestsByShard.entrySet()) {
107            final ShardId shardId = entry.getKey();
108            final List<IngestDeleteItemRequest> requests = entry.getValue();
109            IngestDeleteShardRequest ingestShardRequest = new IngestDeleteShardRequest(shardId.index().name(), shardId.id(), requests);
110            ingestShardRequest.replicationType(ingestRequest.replicationType());
111            ingestShardRequest.consistencyLevel(ingestRequest.consistencyLevel());
112            ingestShardRequest.timeout(ingestRequest.timeout());
113            shardBulkAction.execute(ingestShardRequest, new ActionListener<IngestDeleteShardResponse>() {
114                @Override
115                public void onResponse(IngestDeleteShardResponse ingestShardResponse) {
116                    successSize.addAndGet(ingestShardResponse.successSize());
117                    if (counter.decrementAndGet() == 0) {
118                        finishHim();
119                    }
120                }
121
122                @Override
123                public void onFailure(Throwable e) {
124                    // create failures for all relevant requests
125                    String message = ExceptionsHelper.detailedMessage(e);
126                    synchronized (failure) {
127                        for (IngestDeleteItemRequest request : requests) {
128                            failure.add(new IngestItemFailure(request.id(), message));
129                        }
130                    }
131                    if (counter.decrementAndGet() == 0) {
132                        finishHim();
133                    }
134                }
135
136                private void finishHim() {
137                    listener.onResponse(new IngestResponse(successSize.get(), failure, System.currentTimeMillis() - startTime));
138                }
139            });
140        }
141    }
142
143    class IngestTransportHandler extends BaseTransportRequestHandler<IngestDeleteRequest> {
144
145        @Override
146        public IngestDeleteRequest newInstance() {
147            return new IngestDeleteRequest();
148        }
149
150        @Override
151        public void messageReceived(final IngestDeleteRequest request, final TransportChannel channel) throws Exception {
152            // no need to use threaded listener, since we just send a response
153            request.listenerThreaded(false);
154            execute(request, new ActionListener<IngestResponse>() {
155                @Override
156                public void onResponse(IngestResponse result) {
157                    try {
158                        channel.sendResponse(result);
159                    } catch (Exception e) {
160                        onFailure(e);
161                    }
162                }
163
164                @Override
165                public void onFailure(Throwable e) {
166                    try {
167                        channel.sendResponse(e);
168                    } catch (Exception e1) {
169                        logger.warn("Failed to send error response for request [" + request + "]", e1);
170                    }
171                }
172            });
173        }
174
175        @Override
176        public String executor() {
177            return ThreadPool.Names.SAME;
178        }
179    }
180}
