001package org.xbib.elasticsearch.action.river.state;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.action.admin.cluster.node.info.NodeInfo;
005import org.elasticsearch.action.support.nodes.TransportNodesOperationAction;
006import org.elasticsearch.cluster.ClusterName;
007import org.elasticsearch.cluster.ClusterService;
008import org.elasticsearch.common.inject.Inject;
009import org.elasticsearch.common.inject.Injector;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.node.service.NodeService;
012import org.elasticsearch.river.River;
013import org.elasticsearch.river.RiverName;
014import org.elasticsearch.threadpool.ThreadPool;
015import org.elasticsearch.transport.TransportService;
016import org.xbib.elasticsearch.support.river.RiverHelper;
017
018import java.util.Map;
019import java.util.concurrent.atomic.AtomicReferenceArray;
020
021public class TransportRiverStateAction extends TransportNodesOperationAction<RiverStateRequest, RiverStateResponse, NodeRiverStateRequest, NodeRiverStateResponse> {
022
023    private final NodeService nodeService;
024
025    private final Injector injector;
026
027    @Inject
028    public TransportRiverStateAction(Settings settings, ClusterName clusterName, ThreadPool threadPool,
029                                     ClusterService clusterService, TransportService transportService,
030                                     NodeService nodeService, Injector injector) {
031        super(settings, clusterName, threadPool, clusterService, transportService);
032        this.nodeService = nodeService;
033        this.injector = injector;
034    }
035
036    @Override
037    protected String transportAction() {
038        return RiverStateAction.NAME;
039    }
040
041    @Override
042    protected String executor() {
043        return ThreadPool.Names.GENERIC;
044    }
045
046    @Override
047    protected NodeRiverStateResponse nodeOperation(NodeRiverStateRequest request) throws ElasticsearchException {
048        try {
049            NodeInfo nodeInfo = nodeService.info(false, true, false, true, false, false, true, false, true);
050            String riverName = request.getRiverName();
051            String riverType = request.getRiverType();
052            NodeRiverStateResponse nodeRiverStateResponse = new NodeRiverStateResponse(nodeInfo.getNode());
053            for (Map.Entry<RiverName, River> entry : RiverHelper.rivers(injector).entrySet()) {
054                RiverName name = entry.getKey();
055                if ((riverName == null || "*".equals(riverName) || name.getName().equals(riverName))
056                        && (riverType == null || "*".equals(riverType) || name.getType().equals(riverType))
057                        && entry.getValue() instanceof StatefulRiver) {
058                    StatefulRiver river = (StatefulRiver) entry.getValue();
059                    nodeRiverStateResponse.addState(river.getRiverState());
060                }
061            }
062            return nodeRiverStateResponse;
063        } catch (Throwable t) {
064            logger.error(t.getMessage(), t);
065        }
066        throw new ElasticsearchException("error in node operation");
067    }
068
069    @Override
070    protected RiverStateRequest newRequest() {
071        return new RiverStateRequest();
072    }
073
074    @Override
075    protected RiverStateResponse newResponse(RiverStateRequest request, AtomicReferenceArray nodeResponses) {
076        RiverStateResponse riverStateResponse = new RiverStateResponse();
077        for (int i = 0; i < nodeResponses.length(); i++) {
078            Object nodeResponse = nodeResponses.get(i);
079            if (nodeResponse instanceof NodeRiverStateResponse) {
080                NodeRiverStateResponse nodeRiverStateResponse = (NodeRiverStateResponse) nodeResponse;
081                if (nodeRiverStateResponse.getStates() != null) {
082                    for (RiverState riverState : nodeRiverStateResponse.getStates()) {
083                        riverStateResponse.addState(riverState);
084                    }
085                }
086            }
087        }
088        return riverStateResponse;
089    }
090
091    @Override
092    protected NodeRiverStateRequest newNodeRequest() {
093        return new NodeRiverStateRequest();
094    }
095
096    @Override
097    protected NodeRiverStateRequest newNodeRequest(String nodeId, RiverStateRequest request) {
098        return new NodeRiverStateRequest(nodeId, request);
099    }
100
101    @Override
102    protected NodeRiverStateResponse newNodeResponse() {
103        return new NodeRiverStateResponse();
104    }
105
106    @Override
107    protected boolean accumulateExceptions() {
108        return false;
109    }
110
111}
