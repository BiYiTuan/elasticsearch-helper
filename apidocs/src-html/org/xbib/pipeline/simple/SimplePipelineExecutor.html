001package org.xbib.pipeline.simple;
002
003import org.xbib.pipeline.Pipeline;
004import org.xbib.pipeline.PipelineExecutor;
005import org.xbib.pipeline.PipelineProvider;
006import org.xbib.pipeline.PipelineRequest;
007import org.xbib.pipeline.PipelineSink;
008
009import java.io.IOException;
010import java.util.Collection;
011import java.util.LinkedList;
012import java.util.List;
013import java.util.concurrent.Callable;
014import java.util.concurrent.ExecutionException;
015import java.util.concurrent.ExecutorService;
016import java.util.concurrent.Executors;
017import java.util.concurrent.Future;
018import java.util.concurrent.TimeUnit;
019
020/**
021 * A simple pipeline executor.
022 *
023 * @param <T> the pipeline result type
024 * @param <R> the pipeline request type
025 * @param <P> the pipeline type
026 */
027public class SimplePipelineExecutor<T, R extends PipelineRequest, P extends Pipeline<T, R>>
028        implements PipelineExecutor<T, R, P> {
029
030    private ExecutorService executorService;
031
032    private Collection<P> pipelines;
033
034    private Collection<Future<T>> futures;
035
036    private PipelineProvider<P> provider;
037
038    private PipelineSink<T> sink;
039
040    private List<Throwable> exceptions;
041
042    private int concurrency;
043
044    @Override
045    public SimplePipelineExecutor<T, R, P> setConcurrency(int concurrency) {
046        this.concurrency = concurrency;
047        return this;
048    }
049
050    @Override
051    public SimplePipelineExecutor<T, R, P> setPipelineProvider(PipelineProvider<P> provider) {
052        this.provider = provider;
053        return this;
054    }
055
056    @Override
057    public SimplePipelineExecutor<T, R, P> setSink(PipelineSink<T> sink) {
058        this.sink = sink;
059        return this;
060    }
061
062    @Override
063    public SimplePipelineExecutor<T, R, P> prepare() {
064        if (provider == null) {
065            throw new IllegalStateException("no provider set");
066        }
067        if (executorService == null) {
068            this.executorService = Executors.newFixedThreadPool(concurrency);
069        }
070        this.pipelines = new LinkedList<P>();
071        if (concurrency < 1) {
072            concurrency = 1;
073        }
074        for (int i = 0; i < Math.min(concurrency, 256); i++) {
075            pipelines.add(provider.get());
076        }
077        return this;
078    }
079
080    /**
081     * Execute pipelines
082     *
083     * @return this executor
084     */
085    @Override
086    public SimplePipelineExecutor<T, R, P> execute() {
087        if (pipelines == null) {
088            prepare();
089        }
090        if (pipelines.isEmpty()) {
091            throw new IllegalStateException("pipelines empty");
092        }
093        if (executorService == null) {
094            this.executorService = Executors.newFixedThreadPool(concurrency);
095        }
096        futures = new LinkedList<Future<T>>();
097        for (Callable<T> pipeline : pipelines) {
098            futures.add(executorService.submit(pipeline));
099        }
100        return this;
101    }
102
103    /**
104     * Wait for all results of the executions.
105     *
106     * @return this pipeline executor
107     * @throws InterruptedException
108     * @throws java.util.concurrent.ExecutionException
109     */
110    @Override
111    public SimplePipelineExecutor<T, R, P> waitFor()
112            throws InterruptedException, ExecutionException {
113        if (executorService == null || pipelines == null || futures == null || futures.isEmpty()) {
114            return this;
115        }
116        exceptions = new LinkedList<Throwable>();
117        for (Future<T> future : futures) {
118            T t = future.get();
119            if (sink != null && !future.isCancelled()) {
120                try {
121                    sink.write(t);
122                } catch (IOException e) {
123                    exceptions.add(e);
124                }
125            }
126        }
127        return this;
128    }
129
130    @Override
131    public void shutdown() throws InterruptedException, IOException {
132        if (executorService == null) {
133            return;
134        }
135        executorService.shutdown();
136        if (!executorService.awaitTermination(15, TimeUnit.SECONDS)) {
137            executorService.shutdownNow();
138            if (!executorService.awaitTermination(15, TimeUnit.SECONDS)) {
139                throw new IOException("pool did not terminate");
140            }
141        }
142    }
143
144    /**
145     * Get the pipelines of this executor.
146     *
147     * @return the pipelines
148     */
149    @Override
150    public Collection<P> getPipelines() {
151        return pipelines;
152    }
153
154    /**
155     * Get the collected I/O exceptions that were thrown by the pipelines.
156     *
157     * @return list of exceptions
158     */
159    public List<Throwable> getExceptions() {
160        return exceptions;
161    }
162
163}
