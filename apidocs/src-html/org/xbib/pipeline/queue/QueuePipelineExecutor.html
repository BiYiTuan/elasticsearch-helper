002package org.xbib.pipeline.queue;
003
004import org.xbib.pipeline.Pipeline;
005import org.xbib.pipeline.PipelineExecutor;
006import org.xbib.pipeline.PipelineProvider;
007import org.xbib.pipeline.PipelineRequest;
008import org.xbib.pipeline.PipelineSink;
009import org.xbib.pipeline.element.PipelineElement;
010
011import java.io.IOException;
012import java.util.Collection;
013import java.util.HashSet;
014import java.util.LinkedList;
015import java.util.Set;
016import java.util.concurrent.BlockingQueue;
017import java.util.concurrent.CountDownLatch;
018import java.util.concurrent.ExecutionException;
019import java.util.concurrent.ExecutorService;
020import java.util.concurrent.Executors;
021import java.util.concurrent.Future;
022import java.util.concurrent.SynchronousQueue;
023import java.util.concurrent.TimeUnit;
024
025/**
026 * An Queue Pipeline setExecutor. This setExecutor can execute pipelines in parallel
027 * and manage a queue that have to be processed by the pipelines.
028 *
029 * By doing this, the concurrency works on archive entry level, not URI level.
030 *
031 * Pipelines are created by a pipeline provider.
032 * The maximum number of concurrent pipelines is 256.
033 * Each pipeline can receive archive entries, which are put into a blocking queue by
034 * this setExecutor.
035 *
036 * @param <T> the result type
037 * @param <R> the pipeline request type
038 * @param <P> the pipeline type
039 * @param <E> the element type
040 */
041public class QueuePipelineExecutor<T, R extends PipelineRequest, P extends Pipeline<T,R>, E extends PipelineElement>
042     implements PipelineExecutor<T,R,P> {
043
044    private ExecutorService executorService;
045
046    private PipelineProvider<P> provider;
047
048    private Set<P> pipelines;
049
050    private Collection<Future<T>> futures;
051
052    private PipelineSink<T> sink;
053
054    private CountDownLatch latch;
055
056    private BlockingQueue<E> queue;
057
058    private int concurrency;
059
060    private volatile boolean closed;
061
062    @Override
063    public QueuePipelineExecutor<T,R,P,E> setConcurrency(int concurrency) {
064        this.concurrency = concurrency;
065        return this;
066    }
067
068    @Override
069    public QueuePipelineExecutor<T,R,P,E> setPipelineProvider(PipelineProvider<P> provider) {
070        this.provider = provider;
071        return this;
072    }
073
074    @Override
075    public QueuePipelineExecutor<T,R,P,E> setSink(PipelineSink<T> sink) {
076        this.sink = sink;
077        return this;
078    }
079
080    @Override
081    public QueuePipelineExecutor<T,R,P,E> prepare() {
082        if (provider == null) {
083            throw new IllegalArgumentException("no provider set");
084        }
085        if (executorService == null) {
086            this.executorService = Executors.newFixedThreadPool(concurrency);
087        }
088        if (pipelines == null) {
089            this.pipelines = new HashSet<P>();
090        }
091        if (concurrency < 1) {
092            concurrency = 1;
093        }
094        this.queue = new SynchronousQueue<E>(true); // true means fair queue
095        this.latch = new CountDownLatch(concurrency);
096        for (int i = 0; i < Math.min(concurrency, 256); i++) {
097            P pipeline = provider.get();
098            pipelines.add(pipeline);
099        }
100        return this;
101    }
102
103    /**
104     * Usually, the pipelines receive elements from the setExecutor.
105     * Here we ensure that all pipelines that receive elements are invoked.
106     * This method returns immediately without waiting for the completion of pipelines.
107     *
108     * @return this setExecutor
109     */
110    @Override
111    public QueuePipelineExecutor<T,R,P,E> execute() {
112        futures = new LinkedList<Future<T>>();
113        for (P pipeline : pipelines) {
114            futures.add(executorService.submit(pipeline));
115        }
116        return this;
117    }
118
119    /**
120     * Wait for all the pipelines executed.
121     * @return this setExecutor
122     * @throws InterruptedException
123     * @throws java.util.concurrent.ExecutionException
124     */
125    public QueuePipelineExecutor<T,R,P,E> waitFor() throws InterruptedException, ExecutionException, IOException {
126        for (Future<T> future : futures) {
127            T t = future.get();
128            if (sink != null)  {
129                sink.write(t);
130            }
131        }
132        return this;
133    }
134
135    @Override
136    public void shutdown() throws InterruptedException, IOException {
137        if (executorService == null) {
138            return;
139        }
140        executorService.shutdown();
141        if (!executorService.awaitTermination(15, TimeUnit.SECONDS)) {
142            executorService.shutdownNow();
143            if (!executorService.awaitTermination(15, TimeUnit.SECONDS)) {
144                throw new IOException("pool did not terminate");
145            }
146        }
147    }
148
149    public void shutdown(E poisonElement) throws InterruptedException, ExecutionException, IOException {
150        if (closed) {
151            return;
152        }
153        closed = true;
154        // how many pipelines are still running?
155        int active = 0;
156        for (P pipeline : pipelines) {
157            if (pipeline.getMetric().stopped() == 0L) {
158                active++;
159            }
160        }
161        for (int i = 0; i < active; i++) {
162            queue.offer(poisonElement, 30, TimeUnit.SECONDS);
163        }
164        waitFor();
165        shutdown();
166    }
167
168    /**
169     * Get queue
170     * @return the queue
171     */
172    public BlockingQueue<E> queue() {
173        return queue;
174    }
175
176    /**
177     * Get pipelines
178     * @return the pipelines
179     */
180    public Set<P> getPipelines() {
181        return pipelines;
182    }
183
184    /**
185     * Count down the latch. Decreases the number of active pipelines.
186     * Called from a pipeline when it terminates.
187     * @return this setExecutor
188     */
189    public QueuePipelineExecutor countDown() {
190        latch.countDown();
191        return this;
192    }
193
194    /**
195     * Returns the number of pipelines.  If this pipeline setExecutor can receive requests,
196     * the returned number is greater than 0.
197     * @return number of pipelines ready to receive requests
198     */
199    public long canReceive() {
200        return latch.getCount();
201    }
202
203}
