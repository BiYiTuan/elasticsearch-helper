001package org.xbib.pipeline;
002
003import java.io.IOException;
004import java.util.Collection;
005import java.util.concurrent.ExecutionException;
006
007/**
008 * The Pipeline Executor can execute provided pipelines.
009 * If the concurrency level is set to higher than one, more than one pipeline is executed in parallel.
010 *
011 * @param <T> the result type
012 * @param <R> the request type
013 * @param <P> the pipeline type
014 */
015public interface PipelineExecutor<T, R extends PipelineRequest, P extends Pipeline<T, R>> {
016
017    /**
018     * Set the concurrency of this pipeline setExecutor
019     *
020     * @param concurrency the concurrency, must be a positive integer
021     * @return this setExecutor
022     */
023    PipelineExecutor<T, R, P> setConcurrency(int concurrency);
024
025    /**
026     * Set the provider of this pipeline setExecutor
027     *
028     * @param provider the pipeline provider
029     * @return this setExecutor
030     */
031    PipelineExecutor<T, R, P> setPipelineProvider(PipelineProvider<P> provider);
032
033    /**
034     * Set pipeline sink
035     *
036     * @param sink the pipeline sink
037     * @return this setExecutor
038     */
039    PipelineExecutor<T, R, P> setSink(PipelineSink<T> sink);
040
041    /**
042     * Prepare the pipeline execution.
043     *
044     * @return this setExecutor
045     */
046    PipelineExecutor<T, R, P> prepare();
047
048    /**
049     * Execute the pipelines.
050     *
051     * @return this setExecutor
052     */
053    PipelineExecutor<T, R, P> execute();
054
055    /**
056     * Execute the pipelines.
057     *
058     * @return this setExecutor
059     * @throws InterruptedException
060     * @throws java.util.concurrent.ExecutionException
061     * @throws java.io.IOException
062     */
063    PipelineExecutor<T, R, P> waitFor() throws InterruptedException, ExecutionException, IOException;
064
065    /**
066     * Shut down this pipeline executor.
067     *
068     * @throws InterruptedException
069     * @throws java.util.concurrent.ExecutionException
070     * @throws java.io.IOException
071     */
072    void shutdown() throws InterruptedException, ExecutionException, IOException;
073
074    /**
075     * Return pipelines
076     *
077     * @return the pipelines
078     */
079    Collection<P> getPipelines();
080}
