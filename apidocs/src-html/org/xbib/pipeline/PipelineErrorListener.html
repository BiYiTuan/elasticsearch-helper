001package org.xbib.pipeline;
002
003public interface PipelineErrorListener<T, R extends PipelineRequest, E extends PipelineException> {
004
005    /**

008     * @param pipeline the pipeline

012    void error(Pipeline<T, R> pipeline, R request, E error) throws PipelineException;
013}
